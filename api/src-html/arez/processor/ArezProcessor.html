076  @Nonnull
077  private ObservableDescriptor addObservable( @Nonnull final ComponentDescriptor component,
078                                              @Nonnull final AnnotationMirror annotation,
079                                              @Nonnull final ExecutableElement method,
080                                              @Nonnull final ExecutableType methodType )
081    throws ProcessorException
082  {
083    MemberChecks.mustBeOverridable( component.getElement(),
084                                    Constants.COMPONENT_ANNOTATION_CLASSNAME,
085                                    Constants.OBSERVABLE_ANNOTATION_CLASSNAME,
086                                    method );
087
088    final String declaredName = AnnotationsUtil.getAnnotationValue( annotation, "name" );
089    final boolean expectSetter = AnnotationsUtil.getAnnotationValue( annotation, "expectSetter" );
090    final VariableElement readOutsideTransaction =
091      AnnotationsUtil.getAnnotationValue( annotation, "readOutsideTransaction" );
092    final VariableElement writeOutsideTransaction =
093      AnnotationsUtil.getAnnotationValue( annotation, "writeOutsideTransaction" );
094    final boolean setterAlwaysMutates = AnnotationsUtil.getAnnotationValue( annotation, "setterAlwaysMutates" );
095    final Boolean requireInitializer = isInitializerRequired( method );
096
097    final TypeMirror returnType = method.getReturnType();
098    final String methodName = method.getSimpleName().toString();
099    String name;
100    final boolean setter;
101    if ( TypeKind.VOID == returnType.getKind() )
102    {
103      setter = true;
104      //Should be a setter
105      if ( 1 != method.getParameters().size() )
106      {
107        throw new ProcessorException( "@Observable target should be a setter or getter", method );
108      }
109
110      name = ArezUtils.deriveName( method, SETTER_PATTERN, declaredName );
111      if ( null == name )
112      {
113        name = methodName;
114      }
115    }
116    else
117    {
118      setter = false;
119      //Must be a getter
120      if ( 0 != method.getParameters().size() )
121      {
122        throw new ProcessorException( "@Observable target should be a setter or getter", method );
123      }
124      name = getPropertyAccessorName( method, declaredName );
125    }
126    // Override name if supplied by user
127    if ( !Constants.SENTINEL.equals( declaredName ) )
128    {
129      name = declaredName;
130      if ( !SourceVersion.isIdentifier( name ) )
131      {
132        throw new ProcessorException( "@Observable target specified an invalid name '" + name + "'. The " +
133                                      "name must be a valid java identifier.", method );
134      }
135      else if ( SourceVersion.isKeyword( name ) )
136      {
137        throw new ProcessorException( "@Observable target specified an invalid name '" + name + "'. The " +
138                                      "name must not be a java keyword.", method );
139      }
140    }
141    checkNameUnique( component, name, method, Constants.OBSERVABLE_ANNOTATION_CLASSNAME );
142
143    if ( setter && !expectSetter )
144    {
145      throw new ProcessorException( "Method annotated with @Observable is a setter but defines " +
146                                    "expectSetter = false for observable named " + name, method );
147    }
148
149    final ObservableDescriptor observable = component.findOrCreateObservable( name );
150    observable.setReadOutsideTransaction( readOutsideTransaction.getSimpleName().toString() );
151    observable.setWriteOutsideTransaction( writeOutsideTransaction.getSimpleName().toString() );
152    if ( !setterAlwaysMutates )
153    {
154      observable.setSetterAlwaysMutates( false );
155    }
156    if ( !expectSetter )
157    {
158      observable.setExpectSetter( false );
159    }
160    if ( !observable.expectSetter() )
161    {
162      if ( observable.hasSetter() )
163      {
164        throw new ProcessorException( "Method annotated with @Observable defines expectSetter = false but a " +
165                                      "setter exists named " + observable.getSetter().getSimpleName() +
166                                      "for observable named " + name, method );
167      }
168    }
169    if ( setter )
170    {
171      if ( observable.hasSetter() )
172      {
173        throw new ProcessorException( "Method annotated with @Observable defines duplicate setter for " +
174                                      "observable named " + name, method );
175      }
176      if ( !observable.expectSetter() )
177      {
178        throw new ProcessorException( "Method annotated with @Observable defines expectSetter = false but a " +
179                                      "setter exists for observable named " + name, method );
180      }
181      observable.setSetter( method, methodType );
182    }
183    else
184    {
185      if ( observable.hasGetter() )
186      {
187        throw new ProcessorException( "Method annotated with @Observable defines duplicate getter for " +
188                                      "observable named " + name, method );
189      }
190      observable.setGetter( method, methodType );
191    }
192    if ( null != requireInitializer )
193    {
194      if ( !method.getModifiers().contains( Modifier.ABSTRACT ) )
195      {
196        throw new ProcessorException( "@Observable target set initializer parameter to ENABLED but " +
197                                      "method is not abstract.", method );
198      }
199      final Boolean existing = observable.getInitializer();
200      if ( null == existing )
201      {
202        observable.setInitializer( requireInitializer );
203      }
204      else if ( existing != requireInitializer )
205      {
206        throw new ProcessorException( "@Observable target set initializer parameter to value that differs from " +
207                                      "the paired observable method.", method );
208      }
209    }
210    return observable;
211  }

213  private void addObservableValueRef( @Nonnull final ComponentDescriptor component,
214                                      @Nonnull final AnnotationMirror annotation,
215                                      @Nonnull final ExecutableElement method,
216                                      @Nonnull final ExecutableType methodType )
217    throws ProcessorException
218  {
219    ArezUtils.mustBeStandardRefMethod( processingEnv,
220                                       component,
221                                       method,
222                                       Constants.OBSERVABLE_VALUE_REF_ANNOTATION_CLASSNAME );
223
224    final TypeMirror returnType = methodType.getReturnType();
225    if ( TypeKind.DECLARED != returnType.getKind() ||
226         !ElementsUtil.toRawType( returnType ).toString().equals( "arez.ObservableValue" ) )
227    {
228      throw new ProcessorException( "Method annotated with @ObservableValueRef must return an instance of " +
229                                    "arez.ObservableValue", method );
230    }
231
232    final String declaredName = AnnotationsUtil.getAnnotationValue( annotation, "name" );
233    final String name;
234    if ( Constants.SENTINEL.equals( declaredName ) )
235    {
236      name = ArezUtils.deriveName( method, OBSERVABLE_REF_PATTERN, declaredName );
237      if ( null == name )
238      {
239        throw new ProcessorException( "Method annotated with @ObservableValueRef should specify name or be " +
240                                      "named according to the convention get[Name]ObservableValue", method );
241      }
242    }
243    else
244    {
245      name = declaredName;
246      if ( !SourceVersion.isIdentifier( name ) )
247      {
248        throw new ProcessorException( "@ObservableValueRef target specified an invalid name '" + name + "'. The " +
249                                      "name must be a valid java identifier.", method );
250      }
251      else if ( SourceVersion.isKeyword( name ) )
252      {
253        throw new ProcessorException( "@ObservableValueRef target specified an invalid name '" + name + "'. The " +
254                                      "name must not be a java keyword.", method );
255      }
256    }
257
258    component.findOrCreateObservable( name ).addRefMethod( method, methodType );
259  }

341  private void addOnActivate( @Nonnull final ComponentDescriptor component,
342                              @Nonnull final AnnotationMirror annotation,
343                              @Nonnull final ExecutableElement method )
344    throws ProcessorException
345  {
346    final String name = deriveHookName( component, method,
347                                        ON_ACTIVATE_PATTERN,
348                                        "Activate",
349                                        AnnotationsUtil.getAnnotationValue( annotation, "name" ) );
350    MemberChecks.mustNotBeAbstract( Constants.ON_ACTIVATE_ANNOTATION_CLASSNAME, method );
351    MemberChecks.mustBeSubclassCallable( component.getElement(),
352                                         Constants.COMPONENT_ANNOTATION_CLASSNAME,
353                                         Constants.ON_ACTIVATE_ANNOTATION_CLASSNAME,
354                                         method );
355    final List<? extends VariableElement> parameters = method.getParameters();
356
357    if (
358      !(
359        parameters.isEmpty() ||
360        ( 1 == parameters.size() &&
361          Constants.COMPUTABLE_VALUE_CLASSNAME.equals( ElementsUtil.toRawType( parameters.get( 0 ).asType() )
362                                                         .toString() ) )
363      )
364    )
365    {
366      throw new ProcessorException( "@OnActivate target must not have any parameters or must have a single " +
367                                    "parameter of type arez.ComputableValue", method );
368    }
369
370    MemberChecks.mustNotReturnAnyValue( Constants.ON_ACTIVATE_ANNOTATION_CLASSNAME, method );
371    MemberChecks.mustNotThrowAnyExceptions( Constants.ON_ACTIVATE_ANNOTATION_CLASSNAME, method );
372    ArezUtils.shouldBeInternalHookMethod( processingEnv,
373                                          component,
374                                          method,
375                                          Constants.ON_ACTIVATE_ANNOTATION_CLASSNAME );
376
377    component.findOrCreateMemoize( name ).setOnActivate( method );
378  }

401  @Nonnull
402  private String deriveHookName( @Nonnull final ComponentDescriptor component,
403                                 @Nonnull final ExecutableElement method,
404                                 @Nonnull final Pattern pattern,
405                                 @Nonnull final String type,
406                                 @Nonnull final String name )
407    throws ProcessorException
408  {
409    final String value = ArezUtils.deriveName( method, pattern, name );
410    if ( null == value )
411    {
412      throw new ProcessorException( "Unable to derive name for @On" + type + " as does not match " +
413                                    "on[Name]" + type + " pattern. Please specify name.", method );
414    }
415    else if ( !SourceVersion.isIdentifier( value ) )
416    {
417      throw new ProcessorException( "@On" + type + " target specified an invalid name '" + value + "'. The " +
418                                    "name must be a valid java identifier.", component.getElement() );
419    }
420    else if ( SourceVersion.isKeyword( value ) )
421    {
422      throw new ProcessorException( "@On" + type + " target specified an invalid name '" + value + "'. The " +
423                                    "name must not be a java keyword.", component.getElement() );
424    }
425    else
426    {
427      return value;
428    }
429  }

588  private void linkUnAnnotatedObservables( @Nonnull final ComponentDescriptor component,
589                                           @Nonnull final Map<String, CandidateMethod> getters,
590                                           @Nonnull final Map<String, CandidateMethod> setters )
591    throws ProcessorException
592  {
593    for ( final ObservableDescriptor observable : component.getObservables().values() )
594    {
595      if ( !observable.hasSetter() && !observable.hasGetter() )
596      {
597        throw new ProcessorException( "@ObservableValueRef target unable to be associated with an " +
598                                      "Observable property", observable.getRefMethods().get( 0 ).getMethod() );
599      }
600      else if ( !observable.hasSetter() && observable.expectSetter() )
601      {
602        final CandidateMethod candidate = setters.remove( observable.getName() );
603        if ( null != candidate )
604        {
605          MemberChecks.mustBeOverridable( component.getElement(),
606                                          Constants.COMPONENT_ANNOTATION_CLASSNAME,
607                                          Constants.OBSERVABLE_ANNOTATION_CLASSNAME,
608                                          candidate.getMethod() );
609          observable.setSetter( candidate.getMethod(), candidate.getMethodType() );
610        }
611        else if ( observable.hasGetter() )
612        {
613          throw new ProcessorException( "@Observable target defined getter but no setter was defined and no " +
614                                        "setter could be automatically determined", observable.getGetter() );
615        }
616      }
617      else if ( !observable.hasGetter() )
618      {
619        final CandidateMethod candidate = getters.remove( observable.getName() );
620        if ( null != candidate )
621        {
622          MemberChecks.mustBeOverridable( component.getElement(),
623                                          Constants.COMPONENT_ANNOTATION_CLASSNAME,
624                                          Constants.OBSERVABLE_ANNOTATION_CLASSNAME,
625                                          candidate.getMethod() );
626          observable.setGetter( candidate.getMethod(), candidate.getMethodType() );
627        }
628        else
629        {
630          throw new ProcessorException( "@Observable target defined setter but no getter was defined and no " +
631                                        "getter could be automatically determined", observable.getSetter() );
632        }
633      }
634    }
635
636    // Find pairs of un-annotated abstract setter/getter pairs and treat them as if they
637    // are annotated with @Observable
638    for ( final Map.Entry<String, CandidateMethod> entry : new ArrayList<>( getters.entrySet() ) )
639    {
640      final CandidateMethod getter = entry.getValue();
641      if ( getter.getMethod().getModifiers().contains( Modifier.ABSTRACT ) )
642      {
643        final String name = entry.getKey();
644        final CandidateMethod setter = setters.remove( name );
645        if ( null != setter && setter.getMethod().getModifiers().contains( Modifier.ABSTRACT ) )
646        {
647          final ObservableDescriptor observable = component.findOrCreateObservable( name );
648          observable.setGetter( getter.getMethod(), getter.getMethodType() );
649          observable.setSetter( setter.getMethod(), setter.getMethodType() );
650          getters.remove( name );
651        }
652      }
653    }
654  }

792  private void validate( final boolean allowEmpty, @Nonnull final ComponentDescriptor component )
793    throws ProcessorException
794  {
795    component.getCascadeDisposes().values().forEach( CascadeDisposableDescriptor::validate );
796    component.getObservables().values().forEach( ObservableDescriptor::validate );
797    component.getMemoizes().values().forEach( e -> e.validate( processingEnv ) );
798    component.getObserves().values().forEach( ObserveDescriptor::validate );
799    component.getDependencies().values().forEach( DependencyDescriptor::validate );
800    component.getReferences().values().forEach( ReferenceDescriptor::validate );
801    component.getInverses().values().forEach( e -> e.validate( processingEnv ) );
802
803    final boolean hasReactiveElements =
804      component.getObservables().isEmpty() &&
805      component.getActions().isEmpty() &&
806      component.getMemoizes().isEmpty() &&
807      component.getDependencies().isEmpty() &&
808      component.getCascadeDisposes().isEmpty() &&
809      component.getReferences().isEmpty() &&
810      component.getInverses().isEmpty() &&
811      component.getObserves().isEmpty();
812
813    if ( null != component.getDefaultPriority() &&
814         component.getMemoizes().isEmpty() &&
815         component.getObserves().isEmpty() &&
816         isWarningNotSuppressed( component.getElement(), Constants.WARNING_UNNECESSARY_DEFAULT_PRIORITY ) )
817    {
818      final String message =
819        MemberChecks.toSimpleName( Constants.COMPONENT_ANNOTATION_CLASSNAME ) + " target should not specify " +
820        "the defaultPriority parameter unless it contains methods annotated with either the " +
821        MemberChecks.toSimpleName( Constants.MEMOIZE_ANNOTATION_CLASSNAME ) + " annotation or the " +
822        MemberChecks.toSimpleName( Constants.OBSERVE_ANNOTATION_CLASSNAME ) + " annotation. " +
823        suppressedBy( Constants.WARNING_UNNECESSARY_DEFAULT_PRIORITY );
824      processingEnv.getMessager().printMessage( WARNING, message, component.getElement() );
825    }
826    if ( !allowEmpty && hasReactiveElements )
827    {
828      throw new ProcessorException( "@ArezComponent target has no methods annotated with @Action, " +
829                                    "@CascadeDispose, @Memoize, @Observable, @Inverse, " +
830                                    "@Reference, @ComponentDependency or @Observe", component.getElement() );
831    }
832    else if ( allowEmpty &&
833              !hasReactiveElements &&
834              isWarningNotSuppressed( component.getElement(), Constants.WARNING_UNNECESSARY_ALLOW_EMPTY ) )
835    {
836      final String message =
837        "@ArezComponent target has specified allowEmpty = true but has methods " +
838        "annotated with @Action, @CascadeDispose, @Memoize, @Observable, @Inverse, " +
839        "@Reference, @ComponentDependency or @Observe. " +
840        suppressedBy( Constants.WARNING_UNNECESSARY_ALLOW_EMPTY );
841      processingEnv.getMessager().printMessage( WARNING, message, component.getElement() );
842    }
843
844    if ( component.isDeferSchedule() && !component.requiresSchedule() )
845    {
846      throw new ProcessorException( "@ArezComponent target has specified the deferSchedule = true " +
847                                    "annotation parameter but has no methods annotated with @Observe, " +
848                                    "@ComponentDependency or @Memoize(keepAlive=true)", component.getElement() );
849    }
850    for ( final ExecutableElement componentIdRef : component.getComponentIdRefs() )
851    {
852      if ( null != component.getComponentId() &&
853           !processingEnv.getTypeUtils()
854             .isSameType( component.getComponentId().getReturnType(), componentIdRef.getReturnType() ) )
855      {
856        throw new ProcessorException( "@ComponentIdRef target has a return type " + componentIdRef.getReturnType() +
857                                      " and a @ComponentId annotated method with a return type " +
858                                      componentIdRef.getReturnType() + ". The types must match.",
859                                      component.getElement() );
860      }
861      else if ( null == component.getComponentId() &&
862                !processingEnv.getTypeUtils()
863                  .isSameType( processingEnv.getTypeUtils().getPrimitiveType( TypeKind.INT ),
864                               componentIdRef.getReturnType() ) )
865      {
866        throw new ProcessorException( "@ComponentIdRef target has a return type " + componentIdRef.getReturnType() +
867                                      " but no @ComponentId annotated method. The type is expected to be of " +
868                                      "type int.", component.getElement() );
869      }
870    }
871    if ( InjectMode.NONE != component.getInjectMode() )
872    {
873      for ( final ExecutableElement constructor : ElementsUtil.getConstructors( component.getElement() ) )
874      {
875        // The annotation processor engine can not distinguish between a "default constructor"
876        // synthesized by the compiler and one written by a user that has the same signature.
877        // So our check just skips scenarios where the constructor could be synthetic.
878        if ( constructor.getModifiers().contains( Modifier.PUBLIC ) &&
879             !( constructor.getParameters().isEmpty() && constructor.getThrownTypes().isEmpty() ) )
880        {
881          throw new ProcessorException( "@ArezComponent target has a public constructor but the inject parameter " +
882                                        "does not resolve to NONE. Public constructors are not necessary when " +
883                                        "the instantiation of the component is managed by the injection framework.",
884                                        constructor );
885        }
886      }
887      if ( InjectMode.PROVIDE == component.getInjectMode() &&
888           component.isDagger() &&
889           !component.getElement().getModifiers().contains( Modifier.PUBLIC ) )
890      {
891        throw new ProcessorException( "@ArezComponent target is not public but is configured as inject = PROVIDE " +
892                                      "using the dagger injection framework. Due to constraints within the " +
893                                      "dagger framework the type needs to made public.",
894                                      component.getElement() );
895      }
896    }
897    if ( null != component.getDeclaredDefaultReadOutsideTransaction() &&
898         component.getObservables().isEmpty() &&
899         component.getMemoizes().isEmpty() &&
900         isWarningNotSuppressed( component.getElement(), Constants.WARNING_UNNECESSARY_DEFAULT ) )
901    {
902      final String message =
903        "@ArezComponent target has specified a value for the defaultReadOutsideTransaction parameter but does not " +
904        "contain any methods annotated with either @Memoize or @Observable. " +
905        suppressedBy( Constants.WARNING_UNNECESSARY_ALLOW_EMPTY );
906      processingEnv.getMessager().printMessage( WARNING, message, component.getElement() );
907    }
908    if ( null != component.getDeclaredDefaultWriteOutsideTransaction() &&
909         component.getObservables().isEmpty() &&
910         isWarningNotSuppressed( component.getElement(), Constants.WARNING_UNNECESSARY_DEFAULT ) )
911    {
912      final String message =
913        "@ArezComponent target has specified a value for the defaultWriteOutsideTransaction parameter but does not " +
914        "contain any methods annotated with @Observable. " +
915        suppressedBy( Constants.WARNING_UNNECESSARY_ALLOW_EMPTY );
916      processingEnv.getMessager().printMessage( WARNING, message, component.getElement() );
917    }
918  }

1800          final AnnotationMirror a = AnnotationsUtil.findAnnotationByType( m, Constants.INVERSE_ANNOTATION_CLASSNAME );
1801          if ( null == a )
1802          {
1803            return null;
1804          }
1805          final String inverseName = getInverseName( a, m );
1806          if ( !descriptor.hasInverse() && inverseName.equals( defaultInverseName ) )
1807          {
1808            throw new ProcessorException( "@Reference target has not configured an inverse but there is an " +
1809                                          "associated @Inverse annotated method named '" + m.getSimpleName() +
1810                                          "' on type '" + element.getQualifiedName() + "'.",
1811                                          descriptor.getMethod() );
1812          }
1813          if ( descriptor.hasInverse() && inverseName.equals( descriptor.getInverseName() ) )
1814          {
1815            final TypeElement target = getInverseManyTypeTarget( processingEnv, m );
1816            if ( null != target )
1817            {
1818              ensureTargetTypeAligns( component, descriptor, target.asType() );
1819              return Multiplicity.MANY;
1820            }
1821            else
1822            {
1823              ensureTargetTypeAligns( component, descriptor, m.getReturnType() );
1824              return AnnotationsUtil.hasNonnullAnnotation( m ) ? Multiplicity.ONE : Multiplicity.ZERO_OR_ONE;
1825            }
1826          }
1827          else
1828          {
1829            return null;
1830          }
1831        } )
1832        .filter( Objects::nonNull )
1833        .findAny()
1834        .orElse( null );
1835
1836    if ( descriptor.hasInverse() )
1837    {
1838      if ( null == multiplicity )
1839      {
1840        throw new ProcessorException( "@Reference target expected to find an associated @Inverse annotation " +
1841                                      "with a name parameter equal to '" + descriptor.getInverseName() + "' on " +
1842                                      "class " + descriptor.getMethod().getReturnType() + " but is unable to " +
1843                                      "locate a matching method.", descriptor.getMethod() );
1844      }
1845
1846      final Multiplicity inverseMultiplicity = descriptor.getInverseMultiplicity();
1847      if ( inverseMultiplicity != multiplicity )
1848      {
1849        throw new ProcessorException( "@Reference target has an inverseMultiplicity of " + inverseMultiplicity +
1850                                      " but that associated @Inverse has a multiplicity of " + multiplicity +
1851                                      ". The multiplicity must align.", descriptor.getMethod() );
1852      }
1853    }
1854  }

2010  @Nonnull
2011  private ComponentDescriptor parse( @Nonnull final TypeElement typeElement )
2012    throws ProcessorException
2013  {
2014    if ( ElementKind.CLASS != typeElement.getKind() && ElementKind.INTERFACE != typeElement.getKind() )
2015    {
2016      throw new ProcessorException( "@ArezComponent target must be a class or an interface", typeElement );
2017    }
2018    else if ( typeElement.getModifiers().contains( Modifier.FINAL ) )
2019    {
2020      throw new ProcessorException( "@ArezComponent target must not be final", typeElement );
2021    }
2022    else if ( NestingKind.TOP_LEVEL != typeElement.getNestingKind() &&
2023              !typeElement.getModifiers().contains( Modifier.STATIC ) )
2024    {
2025      throw new ProcessorException( "@ArezComponent target must not be a non-static nested class", typeElement );
2026    }
2027    final AnnotationMirror arezComponent =
2028      AnnotationsUtil.getAnnotationByType( typeElement, Constants.COMPONENT_ANNOTATION_CLASSNAME );
2029    final String declaredType = getAnnotationParameter( arezComponent, "name" );
2030    final boolean disposeOnDeactivate = getAnnotationParameter( arezComponent, "disposeOnDeactivate" );
2031    final boolean observableFlag = isComponentObservableRequired( arezComponent, typeElement, disposeOnDeactivate );
2032    final boolean disposeNotifierFlag = ArezUtils.isDisposableTrackableRequired( typeElement );
2033    final boolean allowEmpty = getAnnotationParameter( arezComponent, "allowEmpty" );
2034    final List<AnnotationMirror> scopeAnnotations =
2035      typeElement.getAnnotationMirrors().stream().filter( this::isScopeAnnotation ).collect( Collectors.toList() );
2036    final AnnotationMirror scopeAnnotation = scopeAnnotations.isEmpty() ? null : scopeAnnotations.get( 0 );
2037    final List<VariableElement> fields = ElementsUtil.getFields( typeElement );
2038    ensureNoFieldInjections( fields );
2039    ensureNoMethodInjections( typeElement );
2040    final VariableElement daggerParameter = getAnnotationParameter( arezComponent, "dagger" );
2041    final String daggerMode = daggerParameter.getSimpleName().toString();
2042    final AnnotationValue defaultReadOutsideTransaction =
2043      AnnotationsUtil.findAnnotationValueNoDefaults( arezComponent, "defaultReadOutsideTransaction" );
2044    final AnnotationValue defaultWriteOutsideTransaction =
2045      AnnotationsUtil.findAnnotationValueNoDefaults( arezComponent, "defaultWriteOutsideTransaction" );
2046
2047    final String injectMode = getInjectMode( arezComponent, typeElement, scopeAnnotation, daggerMode );
2048    final boolean dagger =
2049      "ENABLE".equals( daggerMode ) ||
2050      (
2051        "AUTODETECT".equals( daggerMode ) &&
2052        !"NONE".equals( injectMode ) &&
2053        null != processingEnv.getElementUtils().getTypeElement( Constants.DAGGER_MODULE_CLASSNAME )
2054      );
2055
2056    final boolean requireEquals = isEqualsRequired( arezComponent, typeElement );
2057    final boolean requireVerify = isVerifyRequired( arezComponent, typeElement );
2058    final boolean deferSchedule = getAnnotationParameter( arezComponent, "deferSchedule" );
2059
2060    if ( !typeElement.getModifiers().contains( Modifier.ABSTRACT ) )
2061    {
2062      throw new ProcessorException( "@ArezComponent target must be abstract", typeElement );
2063    }
2064
2065    final String type =
2066      Constants.SENTINEL.equals( declaredType ) ? typeElement.getSimpleName().toString() : declaredType;
2067
2068    if ( !SourceVersion.isIdentifier( type ) )
2069    {
2070      throw new ProcessorException( "@ArezComponent target specified an invalid type '" + type + "'. The " +
2071                                    "type must be a valid java identifier.", typeElement );
2072    }
2073    else if ( SourceVersion.isKeyword( type ) )
2074    {
2075      throw new ProcessorException( "@ArezComponent target specified an invalid type '" + type + "'. The " +
2076                                    "type must not be a java keyword.", typeElement );
2077    }
2078
2079    final List<ExecutableElement> constructors = ElementsUtil.getConstructors( typeElement );
2080    if ( !scopeAnnotations.isEmpty() && constructors.size() > 1 )
2081    {
2082      throw new ProcessorException( "@ArezComponent target has specified a scope annotation but has more than " +
2083                                    "one constructor and thus is not a candidate for injection",
2084                                    typeElement );
2085    }
2086
2087    if ( !"NONE".equals( injectMode ) && constructors.size() > 1 )
2088    {
2089      throw new ProcessorException( "@ArezComponent specified inject parameter but has more than one constructor",
2090                                    typeElement );
2091    }
2092
2093    for ( final ExecutableElement constructor : constructors )
2094    {
2095      if ( constructor.getModifiers().contains( Modifier.PROTECTED ) &&
2096           isWarningNotSuppressed( constructor, Constants.WARNING_PROTECTED_CONSTRUCTOR ) )
2097      {
2098        final String message =
2099          "@ArezComponent target has a protected constructor. The constructor should be public or package access. " +
2100          suppressedBy( Constants.WARNING_PROTECTED_CONSTRUCTOR );
2101        processingEnv.getMessager().printMessage( WARNING, message, constructor );
2102      }
2103    }
2104
2105    if ( scopeAnnotations.size() > 1 )
2106    {
2107      final List<String> scopes = scopeAnnotations.stream()
2108        .map( a -> processingEnv.getTypeUtils().asElement( a.getAnnotationType() ).asType().toString() )
2109        .collect( Collectors.toList() );
2110      throw new ProcessorException( "@ArezComponent target has specified multiple scope annotations: " + scopes,
2111                                    typeElement );
2112    }
2113    if ( !observableFlag && disposeOnDeactivate )
2114    {
2115      throw new ProcessorException( "@ArezComponent target has specified observable = DISABLE and " +
2116                                    "disposeOnDeactivate = true which is not a valid combination", typeElement );
2117    }
2118
2119    boolean generateFactory = false;
2120    if ( dagger )
2121    {
2122      final ExecutableElement ctor = constructors.get( 0 );
2123      assert null != ctor;
2124      final List<? extends VariableElement> perInstanceParameters = ctor.getParameters()
2125        .stream()
2126        .filter( f -> AnnotationsUtil.hasAnnotationOfType( f, Constants.PER_INSTANCE_ANNOTATION_CLASSNAME ) )
2127        .collect( Collectors.toList() );
2128      if ( !perInstanceParameters.isEmpty() )
2129      {
2130        if ( "PROVIDE".equals( injectMode ) )
2131        {
2132          throw new ProcessorException( "@ArezComponent target has specified at least one @PerInstance " +
2133                                        "parameter on the constructor but has set inject parameter to PROVIDE. " +
2134                                        "The component cannot be provided to other components if the invoker " +
2135                                        "must supply per-instance parameters so either change the inject " +
2136                                        "parameter to CONSUME or remove the @PerInstance parameter.",
2137                                        ctor );
2138        }
2139        generateFactory = true;
2140      }
2141    }
2142
2143    final List<ExecutableElement> methods =
2144      ElementsUtil.getMethods( typeElement, processingEnv.getElementUtils(), processingEnv.getTypeUtils() );
2145    final boolean generateToString = methods.stream().
2146      noneMatch( m -> m.getSimpleName().toString().equals( "toString" ) &&
2147                      m.getParameters().size() == 0 &&
2148                      !( m.getEnclosingElement().getSimpleName().toString().equals( "Object" ) &&
2149                         "java.lang".equals( processingEnv.getElementUtils().
2150                           getPackageOf( m.getEnclosingElement() ).getQualifiedName().toString() ) ) );
2151
2152    final String priority = getDefaultPriority( arezComponent );
2153    final Priority defaultPriority =
2154      null == priority ? null : "DEFAULT".equals( priority ) ? Priority.NORMAL : Priority.valueOf( priority );
2155
2156    final String defaultReadOutsideTransactionValue =
2157      null == defaultReadOutsideTransaction ?
2158      null :
2159      ( (VariableElement) defaultReadOutsideTransaction.getValue() ).getSimpleName().toString();
2160    final String defaultWriteOutsideTransactionValue =
2161      null == defaultWriteOutsideTransaction ?
2162      null :
2163      ( (VariableElement) defaultWriteOutsideTransaction.getValue() ).getSimpleName().toString();
2164
2165    final ComponentDescriptor descriptor =
2166      new ComponentDescriptor( type,
2167                               defaultPriority,
2168                               observableFlag,
2169                               disposeNotifierFlag,
2170                               disposeOnDeactivate,
2171                               injectMode,
2172                               dagger,
2173                               generateFactory,
2174                               requireEquals,
2175                               requireVerify,
2176                               scopeAnnotation,
2177                               deferSchedule,
2178                               generateToString,
2179                               typeElement,
2180                               defaultReadOutsideTransactionValue,
2181                               defaultWriteOutsideTransactionValue );
2182
2183    analyzeCandidateMethods( descriptor, methods, processingEnv.getTypeUtils() );
2184    validate( allowEmpty, descriptor );
2185
2186    for ( final ObservableDescriptor observable : descriptor.getObservables().values() )
2187    {
2188      if ( observable.expectSetter() )
2189      {
2190        final TypeMirror returnType = observable.getGetterType().getReturnType();
2191        final TypeMirror parameterType = observable.getSetterType().getParameterTypes().get( 0 );
2192        if ( !processingEnv.getTypeUtils().isSameType( parameterType, returnType ) &&
2193             !parameterType.toString().equals( returnType.toString() ) )
2194        {
2195          throw new ProcessorException( "@Observable property defines a setter and getter with different types." +
2196                                        " Getter type: " + returnType + " Setter type: " + parameterType + ".",
2197                                        observable.getGetter() );
2198        }
2199      }
2200    }
2201
2202    final AnnotationMirror repository =
2203      AnnotationsUtil.findAnnotationByType( typeElement, Constants.REPOSITORY_ANNOTATION_CLASSNAME );
2204    if ( null != repository )
2205    {
2206      final List<TypeElement> extensions =
2207        AnnotationsUtil.getTypeElementsAnnotationParameter( processingEnv,
2208                                                            typeElement,
2209                                                            Constants.REPOSITORY_ANNOTATION_CLASSNAME,
2210                                                            "extensions" );
2211      final String repositoryInjectConfig = getRepositoryInjectMode( repository );
2212      final String repositoryDaggerConfig = getRepositoryDaggerConfig( repository );
2213      final boolean shouldRepositoryDefineCreate = shouldRepositoryDefineCreate( descriptor );
2214      final boolean shouldRepositoryDefineAttach = shouldRepositoryDefineAttach( descriptor );
2215      final boolean shouldRepositoryDefineDestroy = shouldRepositoryDefineDestroy( descriptor );
2216      final boolean shouldRepositoryDefineDetach = shouldRepositoryDefineDetach( descriptor );
2217      for ( final TypeElement extension : extensions )
2218      {
2219        if ( ElementKind.INTERFACE != extension.getKind() )
2220        {
2221          throw new ProcessorException( "Class annotated with @Repository defined an extension that is " +
2222                                        "not an interface. Extension: " + extension.getQualifiedName(),
2223                                        descriptor.getElement() );
2224        }
2225
2226        for ( final Element enclosedElement : extension.getEnclosedElements() )
2227        {
2228          if ( ElementKind.METHOD == enclosedElement.getKind() )
2229          {
2230            final ExecutableElement method = (ExecutableElement) enclosedElement;
2231            if ( !method.isDefault() &&
2232                 !( method.getSimpleName().toString().equals( "self" ) && 0 == method.getParameters().size() ) )
2233            {
2234              throw new ProcessorException( "Class annotated with @Repository defined an extension that has " +
2235                                            "a non default method. Extension: " + extension.getQualifiedName() +
2236                                            " Method: " + method, descriptor.getElement() );
2237            }
2238          }
2239        }
2240      }
2241      descriptor.configureRepository( extensions,
2242                                      repositoryInjectConfig,
2243                                      repositoryDaggerConfig,
2244                                      shouldRepositoryDefineCreate,
2245                                      shouldRepositoryDefineAttach,
2246                                      shouldRepositoryDefineDestroy,
2247                                      shouldRepositoryDefineDetach );
2248    }
2249    if ( !observableFlag && descriptor.hasRepository() )
2250    {
2251      throw new ProcessorException( "@ArezComponent target has specified observable = DISABLE and " +
2252                                    "but is also annotated with the @Repository annotation which requires " +
2253                                    "that the observable != DISABLE.", typeElement );
2254    }
2255    if ( descriptor.hasRepository() &&
2256         AnnotationsUtil.hasAnnotationOfType( typeElement, Constants.SINGLETON_ANNOTATION_CLASSNAME ) )
2257    {
2258      throw new ProcessorException( "@ArezComponent target is annotated with both the " +
2259                                    "@arez.annotations.Repository annotation and the " +
2260                                    "javax.inject.Singleton annotation which is an invalid " +
2261                                    "combination.", typeElement );
2262    }
2263    if ( !descriptor.isDisposeNotifier() && descriptor.hasRepository() )
2264    {
2265      throw new ProcessorException( "@ArezComponent target has specified the disposeNotifier = DISABLE " +
2266                                    "annotation parameter but is also annotated with @Repository that " +
2267                                    "requires disposeNotifier = ENABLE.", typeElement );
2268    }
2269
2270    final boolean idRequired = isIdRequired( descriptor, arezComponent );
2271    descriptor.setIdRequired( idRequired );
2272    if ( !idRequired )
2273    {
2274      if ( descriptor.hasRepository() )
2275      {
2276        throw new ProcessorException( "@ArezComponent target has specified the idRequired = DISABLE " +
2277                                      "annotation parameter but is also annotated with @Repository that " +
2278                                      "requires idRequired = ENABLE.", typeElement );
2279      }
2280      if ( descriptor.hasComponentIdMethod() )
2281      {
2282        throw new ProcessorException( "@ArezComponent target has specified the idRequired = DISABLE " +
2283                                      "annotation parameter but also has annotated a method with @ComponentId " +
2284                                      "that requires idRequired = ENABLE.", typeElement );
2285      }
2286      if ( !descriptor.getComponentIdRefs().isEmpty() )
2287      {
2288        throw new ProcessorException( "@ArezComponent target has specified the idRequired = DISABLE " +
2289                                      "annotation parameter but also has annotated a method with @ComponentIdRef " +
2290                                      "that requires idRequired = ENABLE.", typeElement );
2291      }
2292    }
2293
2294    warnOnUnmanagedComponentReferences( descriptor, fields );
2295
2296    return descriptor;
2297  }

2325  private void analyzeCandidateMethods( @Nonnull final ComponentDescriptor componentDescriptor,
2326                                        @Nonnull final List<ExecutableElement> methods,
2327                                        @Nonnull final Types typeUtils )
2328    throws ProcessorException
2329  {
2330    for ( final ExecutableElement method : methods )
2331    {
2332      final String methodName = method.getSimpleName().toString();
2333      if ( AREZ_SPECIAL_METHODS.contains( methodName ) && method.getParameters().isEmpty() )
2334      {
2335        throw new ProcessorException( "Method defined on a class annotated by @ArezComponent uses a name " +
2336                                      "reserved by Arez", method );
2337      }
2338      else if ( methodName.startsWith( ComponentGenerator.FIELD_PREFIX ) ||
2339                methodName.startsWith( ComponentGenerator.OBSERVABLE_DATA_FIELD_PREFIX ) ||
2340                methodName.startsWith( ComponentGenerator.REFERENCE_FIELD_PREFIX ) ||
2341                methodName.startsWith( ComponentGenerator.FRAMEWORK_PREFIX ) )
2342      {
2343        throw new ProcessorException( "Method defined on a class annotated by @ArezComponent uses a name " +
2344                                      "with a prefix reserved by Arez", method );
2345      }
2346    }
2347    final Map<String, CandidateMethod> getters = new HashMap<>();
2348    final Map<String, CandidateMethod> setters = new HashMap<>();
2349    final Map<String, CandidateMethod> observes = new HashMap<>();
2350    final Map<String, CandidateMethod> onDepsChanges = new HashMap<>();
2351    for ( final ExecutableElement method : methods )
2352    {
2353      final ExecutableType methodType =
2354        (ExecutableType) typeUtils.asMemberOf( (DeclaredType) componentDescriptor.getElement().asType(), method );
2355      if ( !analyzeMethod( componentDescriptor, method, methodType ) )
2356      {
2357        /*
2358         * If we get here the method was not annotated so we can try to detect if it is a
2359         * candidate arez method in case some arez annotations are implied via naming conventions.
2360         */
2361        if ( method.getModifiers().contains( Modifier.STATIC ) )
2362        {
2363          continue;
2364        }
2365
2366        final CandidateMethod candidateMethod = new CandidateMethod( method, methodType );
2367        final boolean voidReturn = method.getReturnType().getKind() == TypeKind.VOID;
2368        final int parameterCount = method.getParameters().size();
2369        String name;
2370
2371        if ( !method.getModifiers().contains( Modifier.FINAL ) )
2372        {
2373          name = ArezUtils.deriveName( method, SETTER_PATTERN, Constants.SENTINEL );
2374          if ( voidReturn && 1 == parameterCount && null != name )
2375          {
2376            setters.put( name, candidateMethod );
2377            continue;
2378          }
2379          name = ArezUtils.deriveName( method, ISSER_PATTERN, Constants.SENTINEL );
2380          if ( !voidReturn && 0 == parameterCount && null != name )
2381          {
2382            getters.put( name, candidateMethod );
2383            continue;
2384          }
2385          name = ArezUtils.deriveName( method, GETTER_PATTERN, Constants.SENTINEL );
2386          if ( !voidReturn && 0 == parameterCount && null != name )
2387          {
2388            getters.put( name, candidateMethod );
2389            continue;
2390          }
2391        }
2392        name = ArezUtils.deriveName( method, ON_DEPS_CHANGE_PATTERN, Constants.SENTINEL );
2393        if ( voidReturn && null != name )
2394        {
2395          if ( 0 == parameterCount ||
2396               (
2397                 1 == parameterCount &&
2398                 Constants.OBSERVER_CLASSNAME.equals( method.getParameters().get( 0 ).asType().toString() )
2399               )
2400          )
2401          {
2402            onDepsChanges.put( name, candidateMethod );
2403            continue;
2404          }
2405        }
2406
2407        final String methodName = method.getSimpleName().toString();
2408        if ( !OBJECT_METHODS.contains( methodName ) )
2409        {
2410          observes.put( methodName, candidateMethod );
2411        }
2412      }
2413    }
2414
2415    linkUnAnnotatedObservables( componentDescriptor, getters, setters );
2416    linkUnAnnotatedObserves( componentDescriptor, observes, onDepsChanges );
2417    linkObserverRefs( componentDescriptor );
2418    linkCascadeDisposeObservables( componentDescriptor );
2419    linkCascadeDisposeReferences( componentDescriptor );
2420
2421    // CascadeDispose returned false but it was actually processed so lets remove them from getters set
2422
2423    componentDescriptor.getCascadeDisposes().keySet().forEach( method -> {
2424      for ( final Map.Entry<String, CandidateMethod> entry : new HashMap<>( getters ).entrySet() )
2425      {
2426        if ( method.equals( entry.getValue().getMethod() ) )
2427        {
2428          getters.remove( entry.getKey() );
2429        }
2430      }
2431    } );
2432
2433    linkDependencies( componentDescriptor, getters.values() );
2434
2435    autodetectObservableInitializers( componentDescriptor );
2436
2437    /*
2438     * All of the maps will have called remove() for all matching candidates.
2439     * Thus any left are the non-arez methods.
2440     */
2441
2442    ensureNoAbstractMethods( componentDescriptor, getters.values() );
2443    ensureNoAbstractMethods( componentDescriptor, setters.values() );
2444    ensureNoAbstractMethods( componentDescriptor, observes.values() );
2445    ensureNoAbstractMethods( componentDescriptor, onDepsChanges.values() );
2446
2447    processCascadeDisposeFields( componentDescriptor );
2448    processComponentDependencyFields( componentDescriptor );
2449  }

2905                           @Nonnull final AnnotationMirror annotation,
2906                           @Nonnull final ObservableDescriptor observable,
2907                           @Nonnull final ExecutableElement method )
2908  {
2909    MemberChecks.mustNotHaveAnyParameters( Constants.INVERSE_ANNOTATION_CLASSNAME, method );
2910    MemberChecks.mustBeSubclassCallable( descriptor.getElement(),
2911                                         Constants.COMPONENT_ANNOTATION_CLASSNAME,
2912                                         Constants.INVERSE_ANNOTATION_CLASSNAME,
2913                                         method );
2914    MemberChecks.mustNotThrowAnyExceptions( Constants.INVERSE_ANNOTATION_CLASSNAME, method );
2915    MemberChecks.mustReturnAValue( Constants.INVERSE_ANNOTATION_CLASSNAME, method );
2916    MemberChecks.mustBeAbstract( Constants.INVERSE_ANNOTATION_CLASSNAME, method );
2917
2918    final String name = getInverseName( annotation, method );
2919    final InverseDescriptor existing = descriptor.getInverses().get( name );
2920    if ( null != existing && existing.hasObservable() )
2921    {
2922      throw new ProcessorException( "@Inverse target defines duplicate inverse for name '" + name +
2923                                    "'. The other inverse is " + existing.getObservable().getGetter(),
2924                                    method );
2925    }
2926    else
2927    {
2928      final TypeMirror type = method.getReturnType();
2929
2930      final Multiplicity multiplicity;
2931      TypeElement targetType = getInverseManyTypeTarget( processingEnv, method );
2932      if ( null != targetType )
2933      {

2939             !AnnotationsUtil.hasAnnotationOfType( ( (DeclaredType) type ).asElement(),
2940                                                   Constants.COMPONENT_ANNOTATION_CLASSNAME ) )

2942          throw new ProcessorException( "@Inverse target expected to return a type annotated with " +
2943                                        Constants.COMPONENT_ANNOTATION_CLASSNAME, method );

2946        if ( AnnotationsUtil.hasNonnullAnnotation( method ) )
2947        {
2948          multiplicity = Multiplicity.ONE;
2949        }
2950        else if ( AnnotationsUtil.hasAnnotationOfType( method, GeneratorUtil.NULLABLE_ANNOTATION_CLASSNAME ) )

2959        }

2962

2964      inverse.setInverse( observable, referenceName, multiplicity, targetType );
2965      verifyMultiplicityOfAssociatedReferenceMethod( descriptor, inverse );
2966    }
