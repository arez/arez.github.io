033/**
034 * Annotation processor that analyzes Arez annotated source and generates models from the annotations.
035 */
036@SuppressWarnings( "Duplicates" )
037@AutoService( Processor.class )
038@SupportedAnnotationTypes( { "arez.annotations.*" } )
039@SupportedSourceVersion( SourceVersion.RELEASE_8 )
040public final class ArezProcessor
041  extends AbstractProcessor
042{
043  /**
044   * {@inheritDoc}
045   */
046  @Override
047  public boolean process( final Set<? extends TypeElement> annotations, final RoundEnvironment env )
048  {
049    final TypeElement annotation =
050      processingEnv.getElementUtils().getTypeElement( Constants.COMPONENT_ANNOTATION_CLASSNAME );
051    final Set<? extends Element> elements = env.getElementsAnnotatedWith( annotation );
052    processElements( elements );
053    return false;
054  }
055
056  private void processElements( @Nonnull final Set<? extends Element> elements )
057  {
058    for ( final Element element : elements )
059    {
060      try
061      {
062        process( element );
063      }
064      catch ( final IOException ioe )
065      {
066        processingEnv.getMessager().printMessage( ERROR, ioe.getMessage(), element );
067      }
068      catch ( final ArezProcessorException e )
069      {
070        processingEnv.getMessager().printMessage( ERROR, e.getMessage(), e.getElement() );
071      }
072      catch ( final Throwable e )
073      {
074        final StringWriter sw = new StringWriter();
075        e.printStackTrace( new PrintWriter( sw ) );
076        sw.flush();
077
078        final String message =
079          "Unexpected error will running the " + getClass().getName() + " processor. This has " +
080          "resulted in a failure to process the code and has left the compiler in an invalid " +
081          "state. Please report the failure to the developers so that it can be fixed.\n" +
082          " Report the error at: https://github.com/arez/arez/issues\n" +
083          "\n\n" +
084          sw.toString();
085        processingEnv.getMessager().printMessage( ERROR, message, element );
086      }
087    }
088  }
089
090  private void process( @Nonnull final Element element )
091    throws IOException, ArezProcessorException
092  {
093    final PackageElement packageElement = processingEnv.getElementUtils().getPackageOf( element );
094    final TypeElement typeElement = (TypeElement) element;
095    final ComponentDescriptor descriptor = parse( packageElement, typeElement );
096    emitTypeSpec( descriptor.getPackageName(), descriptor.buildType( processingEnv.getTypeUtils() ) );
097    if ( descriptor.shouldGenerateComponentDaggerModule() )
098    {
099      emitTypeSpec( descriptor.getPackageName(), descriptor.buildComponentDaggerModule() );
100    }
101    if ( descriptor.hasRepository() )
102    {
103      emitTypeSpec( descriptor.getPackageName(), descriptor.buildRepository( processingEnv.getTypeUtils() ) );
104    }
105  }
106
107  @SuppressWarnings( "unchecked" )
108  @Nonnull
109  private ComponentDescriptor parse( final PackageElement packageElement, final TypeElement typeElement )
110    throws ArezProcessorException
111  {
112    if ( ElementKind.CLASS != typeElement.getKind() )
113    {
114      throw new ArezProcessorException( "@ArezComponent target must be a class", typeElement );
115    }
116    else if ( typeElement.getModifiers().contains( Modifier.FINAL ) )
117    {
118      throw new ArezProcessorException( "@ArezComponent target must not be final", typeElement );
119    }
120    else if ( NestingKind.TOP_LEVEL != typeElement.getNestingKind() &&
121              !typeElement.getModifiers().contains( Modifier.STATIC ) )
122    {
123      throw new ArezProcessorException( "@ArezComponent target must not be a non-static nested class", typeElement );
124    }
125    final AnnotationMirror arezComponent =
126      ProcessorUtil.getAnnotationByType( typeElement, Constants.COMPONENT_ANNOTATION_CLASSNAME );
127    final String declaredType = getAnnotationParameter( arezComponent, "type" );
128    final AnnotationValue nameIncludesIdValue =
129      ProcessorUtil.findAnnotationValue( processingEnv.getElementUtils(), arezComponent, "nameIncludesId", false );
130    final AnnotationMirror singletonAnnotation =
131      ProcessorUtil.findAnnotationByType( typeElement, Constants.SINGLETON_ANNOTATION_CLASSNAME );
132    final boolean nameIncludesIdDefault = null == singletonAnnotation;
133    final boolean nameIncludesId =
134      null == nameIncludesIdValue ? nameIncludesIdDefault : (boolean) nameIncludesIdValue.getValue();
135    final boolean disposeOnDeactivate = getAnnotationParameter( arezComponent, "disposeOnDeactivate" );
136    final boolean allowConcrete = getAnnotationParameter( arezComponent, "allowConcrete" );
137    final boolean allowEmpty = getAnnotationParameter( arezComponent, "allowEmpty" );
138    final List<AnnotationMirror> scopeAnnotations =
139      typeElement.getAnnotationMirrors().stream().filter( this::isScopeAnnotation ).collect( Collectors.toList() );
140    final AnnotationMirror scopeAnnotation = scopeAnnotations.isEmpty() ? null : scopeAnnotations.get( 0 );
141    final boolean inject = isInjectionRequired( typeElement, scopeAnnotation );
142    final boolean dagger = isDaggerRequired( typeElement, scopeAnnotation );
143    final boolean requireEquals = isEqualsRequired( typeElement );
144    final boolean deferSchedule = getAnnotationParameter( arezComponent, "deferSchedule" );
145
146    if ( !typeElement.getModifiers().contains( Modifier.ABSTRACT ) && !allowConcrete )
147    {
148      throw new ArezProcessorException( "@ArezComponent target must be abstract unless the allowConcrete " +
149                                        "parameter is set to true", typeElement );
150    }
151
152    final String type =
153      ProcessorUtil.isSentinelName( declaredType ) ? typeElement.getSimpleName().toString() : declaredType;
154
155    if ( !SourceVersion.isIdentifier( type ) )
156    {
157      throw new ArezProcessorException( "@ArezComponent target specified an invalid type '" + type + "'. The " +
158                                        "type must be a valid java identifier.", typeElement );
159    }
160    else if ( SourceVersion.isKeyword( type ) )
161    {
162      throw new ArezProcessorException( "@ArezComponent target specified an invalid type '" + type + "'. The " +
163                                        "type must not be a java keyword.", typeElement );
164    }
165
166    if ( !scopeAnnotations.isEmpty() && ProcessorUtil.getConstructors( typeElement ).size() > 1 )
167    {
168      throw new ArezProcessorException( "@ArezComponent target has specified a scope annotation but has more than " +
169                                        "one constructor and thus is not a candidate for injection",
170                                        typeElement );
171    }
172
173    if ( inject && ProcessorUtil.getConstructors( typeElement ).size() > 1 )
174    {
175      throw new ArezProcessorException( "@ArezComponent specified inject parameter but has more than one constructor",
176                                        typeElement );
177    }
178
179    if ( scopeAnnotations.size() > 1 )
180    {
181      final List<String> scopes = scopeAnnotations.stream()
182        .map( a -> processingEnv.getTypeUtils().asElement( a.getAnnotationType() ).asType().toString() )
183        .collect( Collectors.toList() );
184      throw new ArezProcessorException( "@ArezComponent target has specified multiple scope annotations: " + scopes,
185                                        typeElement );
186    }
187
188    final List<ExecutableElement> methods =
189      ProcessorUtil.getMethods( typeElement, processingEnv.getElementUtils(), processingEnv.getTypeUtils() );
190    final boolean generateToString = methods.stream().
191      noneMatch( m -> m.getSimpleName().toString().equals( "toString" ) &&
192                      m.getParameters().size() == 0 &&
193                      !( m.getEnclosingElement().getSimpleName().toString().equals( "Object" ) &&
194                         "java.lang".equals( processingEnv.getElementUtils().
195                           getPackageOf( m.getEnclosingElement() ).getQualifiedName().toString() ) ) );
196
197    final boolean injectClassesPresent =
198      null != processingEnv.getElementUtils().getTypeElement( Constants.INJECT_ANNOTATION_CLASSNAME );
199
200    final ComponentDescriptor descriptor =
201      new ComponentDescriptor( processingEnv.getElementUtils(),
202                               type,
203                               nameIncludesId,
204                               allowEmpty,
205                               disposeOnDeactivate,
206                               injectClassesPresent,
207                               dagger || inject,
208                               dagger,
209                               requireEquals,
210                               scopeAnnotation,
211                               deferSchedule,
212                               generateToString,
213                               packageElement,
214                               typeElement );
215
216    descriptor.analyzeCandidateMethods( methods, processingEnv.getTypeUtils() );
217    descriptor.validate();
218
219    for ( final ObservableDescriptor observable : descriptor.getObservables() )
220    {
221      if ( observable.expectSetter() )
222      {
223        final TypeMirror returnType = observable.getGetterType().getReturnType();
224        final TypeMirror parameterType = observable.getSetterType().getParameterTypes().get( 0 );
225        if ( !processingEnv.getTypeUtils().isSameType( parameterType, returnType ) )
226        {
227          throw new ArezProcessorException( "@Observable property defines a setter and getter with different types." +
228                                            " Getter type: " + returnType + " Setter type: " + parameterType + ".",
229                                            observable.getGetter() );
230        }
231      }
232    }
233
234    final AnnotationMirror repository =
235      ProcessorUtil.findAnnotationByType( typeElement, Constants.REPOSITORY_ANNOTATION_CLASSNAME );
236    if ( null != repository )
237    {
238      final List<TypeElement> extensions =
239        ProcessorUtil.getTypeMirrorsAnnotationParameter( processingEnv.getElementUtils(),
240                                                         typeElement,
241                                                         Constants.REPOSITORY_ANNOTATION_CLASSNAME,
242                                                         "extensions" ).stream().
243          map( typeMirror -> (TypeElement) processingEnv.getTypeUtils().asElement( typeMirror ) ).
244          collect( Collectors.toList() );
245      final String name = getAnnotationParameter( repository, "name" );
246      final String repositoryInjectConfig = getRepositoryInjectConfig( typeElement );
247      final String repositoryDaggerConfig = getRepositoryDaggerConfig( typeElement );
248      descriptor.configureRepository( name, extensions, repositoryInjectConfig, repositoryDaggerConfig );
249    }
250
251    return descriptor;
252  }
253
254  private boolean isScopeAnnotation( @Nonnull final AnnotationMirror a )
255  {
256    final Element element = processingEnv.getTypeUtils().asElement( a.getAnnotationType() );
257    return null != ProcessorUtil.findAnnotationByType( element, Constants.SCOPE_ANNOTATION_CLASSNAME );
258  }
259
260  private boolean isInjectionRequired( @Nonnull final TypeElement typeElement,
261                                       @Nullable final AnnotationMirror scopeAnnotation )
262  {
263    final VariableElement injectParameter = (VariableElement)
264      ProcessorUtil.getAnnotationValue( processingEnv.getElementUtils(),
265                                        typeElement,
266                                        Constants.COMPONENT_ANNOTATION_CLASSNAME,
267                                        "inject" ).getValue();
268    switch ( injectParameter.getSimpleName().toString() )
269    {
270      case "ENABLE":
271        return true;
272      case "DISABLE":
273        return false;
274      default:
275        return null != scopeAnnotation ||
276               ProcessorUtil.getFieldElements( typeElement ).stream().anyMatch( this::hasInjectAnnotation ) ||
277               ProcessorUtil.getMethods( typeElement, processingEnv.getElementUtils(), processingEnv.getTypeUtils() ).
278                 stream().anyMatch( this::hasInjectAnnotation );
279    }
280  }
281
282  private boolean isDaggerRequired( @Nonnull final TypeElement typeElement,
283                                    @Nullable final AnnotationMirror scopeAnnotation )
284  {
285    final VariableElement daggerParameter = (VariableElement)
286      ProcessorUtil.getAnnotationValue( processingEnv.getElementUtils(),
287                                        typeElement,
288                                        Constants.COMPONENT_ANNOTATION_CLASSNAME,
289                                        "dagger" ).getValue();
290    switch ( daggerParameter.getSimpleName().toString() )
291    {
292      case "ENABLE":
293        return true;
294      case "DISABLE":
295        return false;
296      default:
297        return null != scopeAnnotation &&
298               null != processingEnv.getElementUtils().getTypeElement( Constants.DAGGER_MODULE_CLASSNAME );
299    }
300  }
301
302  private boolean isEqualsRequired( @Nonnull final TypeElement typeElement )
303  {
304    final VariableElement injectParameter = (VariableElement)
305      ProcessorUtil.getAnnotationValue( processingEnv.getElementUtils(),
306                                        typeElement,
307                                        Constants.COMPONENT_ANNOTATION_CLASSNAME,
308                                        "requireEquals" ).getValue();
309    switch ( injectParameter.getSimpleName().toString() )
310    {
311      case "ENABLE":
312        return true;
313      case "DISABLE":
314        return false;
315      default:
316        return null != ProcessorUtil.findAnnotationByType( typeElement, Constants.REPOSITORY_ANNOTATION_CLASSNAME );
317    }
318  }

320  @Nonnull
321  private String getRepositoryInjectConfig( @Nonnull final TypeElement typeElement )
322  {
323    final VariableElement injectParameter = (VariableElement)
324      ProcessorUtil.getAnnotationValue( processingEnv.getElementUtils(),
325                                        typeElement,
326                                        Constants.REPOSITORY_ANNOTATION_CLASSNAME,
327                                        "inject" ).getValue();
328    return injectParameter.getSimpleName().toString();
329  }
330
331  @Nonnull
332  private String getRepositoryDaggerConfig( @Nonnull final TypeElement typeElement )
333  {
334    final VariableElement daggerParameter = (VariableElement)
335      ProcessorUtil.getAnnotationValue( processingEnv.getElementUtils(),
336                                        typeElement,
337                                        Constants.REPOSITORY_ANNOTATION_CLASSNAME,
338                                        "dagger" ).getValue();
339    return daggerParameter.getSimpleName().toString();
340  }
341
342  private boolean hasInjectAnnotation( final Element method )
343  {
344    return null != ProcessorUtil.findAnnotationByType( method, Constants.INJECT_ANNOTATION_CLASSNAME );
345  }
346
347  @SuppressWarnings( "unchecked" )
348  private <T> T getAnnotationParameter( @Nonnull final AnnotationMirror annotation,
349                                        @Nonnull final String parameterName )
350  {
351    return (T) ProcessorUtil.getAnnotationValue( processingEnv.getElementUtils(),
352                                                 annotation,
353                                                 parameterName ).getValue();
354  }
355
356  private void emitTypeSpec( @Nonnull final String packageName, @Nonnull final TypeSpec typeSpec )
357    throws IOException
358  {
359    JavaFile.builder( packageName, typeSpec ).
360      skipJavaLangImports( true ).
361      build().
362      writeTo( processingEnv.getFiler() );
363  }
