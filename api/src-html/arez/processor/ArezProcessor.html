032/**
033 * Annotation processor that analyzes Arez annotated source and generates models from the annotations.
034 */
035@SuppressWarnings( "Duplicates" )
036@AutoService( Processor.class )
037@SupportedAnnotationTypes( { "arez.annotations.*", "javax.annotation.PostConstruct" } )
038@SupportedSourceVersion( SourceVersion.RELEASE_8 )
039public final class ArezProcessor
040  extends AbstractProcessor
041{
042  /**
043   * {@inheritDoc}
044   */
045  @Override
046  public boolean process( final Set<? extends TypeElement> annotations, final RoundEnvironment env )
047  {
048    final TypeElement annotation =
049      processingEnv.getElementUtils().getTypeElement( Constants.COMPONENT_ANNOTATION_CLASSNAME );
050    final Set<? extends Element> elements = env.getElementsAnnotatedWith( annotation );
051    processElements( elements );
052    return false;
053  }
054
055  private void processElements( @Nonnull final Set<? extends Element> elements )
056  {
057    for ( final Element element : elements )
058    {
059      try
060      {
061        process( element );
062      }
063      catch ( final IOException ioe )
064      {
065        processingEnv.getMessager().printMessage( ERROR, ioe.getMessage(), element );
066      }
067      catch ( final ArezProcessorException e )
068      {
069        processingEnv.getMessager().printMessage( ERROR, e.getMessage(), e.getElement() );
070      }
071      catch ( final Throwable e )
072      {
073        final StringWriter sw = new StringWriter();
074        e.printStackTrace( new PrintWriter( sw ) );
075        sw.flush();
076
077        final String message =
078          "Unexpected error will running the " + getClass().getName() + " processor. This has " +
079          "resulted in a failure to process the code and has left the compiler in an invalid " +
080          "state. Please report the failure to the developers so that it can be fixed.\n" +
081          " Report the error at: https://github.com/arez/arez/issues\n" +
082          "\n\n" +
083          sw.toString();
084        processingEnv.getMessager().printMessage( ERROR, message, element );
085      }
086    }
087  }
088
089  private void process( @Nonnull final Element element )
090    throws IOException, ArezProcessorException
091  {
092    final PackageElement packageElement = processingEnv.getElementUtils().getPackageOf( element );
093    final TypeElement typeElement = (TypeElement) element;
094    final ComponentDescriptor descriptor = parse( packageElement, typeElement );
095    emitTypeSpec( descriptor.getPackageName(), descriptor.buildType( processingEnv.getTypeUtils() ) );
096    if ( descriptor.shouldGenerateComponentDaggerModule() )
097    {
098      emitTypeSpec( descriptor.getPackageName(), descriptor.buildComponentDaggerModule() );
099    }
100    if ( descriptor.hasRepository() )
101    {
102      emitTypeSpec( descriptor.getPackageName(), descriptor.buildRepository( processingEnv.getTypeUtils() ) );
103    }
104  }
105
106  @SuppressWarnings( "unchecked" )
107  @Nonnull
108  private ComponentDescriptor parse( final PackageElement packageElement, final TypeElement typeElement )
109    throws ArezProcessorException
110  {
111    if ( ElementKind.CLASS != typeElement.getKind() )
112    {
113      throw new ArezProcessorException( "@ArezComponent target must be a class", typeElement );
114    }
115    else if ( typeElement.getModifiers().contains( Modifier.ABSTRACT ) )
116    {
117      throw new ArezProcessorException( "@ArezComponent target must not be abstract", typeElement );
118    }
119    else if ( typeElement.getModifiers().contains( Modifier.FINAL ) )
120    {
121      throw new ArezProcessorException( "@ArezComponent target must not be final", typeElement );
122    }
123    else if ( NestingKind.TOP_LEVEL != typeElement.getNestingKind() &&
124              !typeElement.getModifiers().contains( Modifier.STATIC ) )
125    {
126      throw new ArezProcessorException( "@ArezComponent target must not be a non-static nested class", typeElement );
127    }
128    final AnnotationMirror arezComponent =
129      ProcessorUtil.getAnnotationByType( typeElement, Constants.COMPONENT_ANNOTATION_CLASSNAME );
130    final String declaredType = getAnnotationParameter( arezComponent, "type" );
131    final boolean nameIncludesId = getAnnotationParameter( arezComponent, "nameIncludesId" );
132    final boolean allowEmpty = getAnnotationParameter( arezComponent, "allowEmpty" );
133    final List<AnnotationMirror> scopeAnnotations =
134      typeElement.getAnnotationMirrors().stream().filter( this::isScopeAnnotation ).collect( Collectors.toList() );
135    final AnnotationMirror scopeAnnotation = scopeAnnotations.isEmpty() ? null : scopeAnnotations.get( 0 );
136    final boolean inject = isInjectionRequired( typeElement, scopeAnnotation );
137    final boolean dagger = isDaggerRequired( typeElement, scopeAnnotation );
138    final boolean deferSchedule = getAnnotationParameter( arezComponent, "deferSchedule" );
139
140    final String type =
141      ProcessorUtil.isSentinelName( declaredType ) ? typeElement.getSimpleName().toString() : declaredType;
142
143    if ( !ProcessorUtil.isJavaIdentifier( type ) )
144    {
145      throw new ArezProcessorException( "@ArezComponent specified invalid type parameter", typeElement );
146    }
147
148    if ( !scopeAnnotations.isEmpty() && ProcessorUtil.getConstructors( typeElement ).size() > 1 )
149    {
150      throw new ArezProcessorException( "@ArezComponent target has specified a scope annotation but has more than " +
151                                        "one constructor and thus is not a candidate for injection",
152                                        typeElement );
153    }
154
155    if ( inject && ProcessorUtil.getConstructors( typeElement ).size() > 1 )
156    {
157      throw new ArezProcessorException( "@ArezComponent specified inject parameter but has more than one constructor",
158                                        typeElement );
159    }
160
161    if ( scopeAnnotations.size() > 1 )
162    {
163      final List<String> scopes = scopeAnnotations.stream()
164        .map( a -> processingEnv.getTypeUtils().asElement( a.getAnnotationType() ).asType().toString() )
165        .collect( Collectors.toList() );
166      throw new ArezProcessorException( "@ArezComponent target has specified multiple scope annotations: " + scopes,
167                                        typeElement );
168    }
169
170    final List<ExecutableElement> methods = ProcessorUtil.getMethods( typeElement, processingEnv.getTypeUtils() );
171    final boolean generateToString = methods.stream().
172      noneMatch( m -> m.getSimpleName().toString().equals( "toString" ) &&
173                      m.getParameters().size() == 0 &&
174                      !( m.getEnclosingElement().getSimpleName().toString().equals( "Object" ) &&
175                         "java.lang".equals( processingEnv.getElementUtils().
176                           getPackageOf( m.getEnclosingElement() ).getQualifiedName().toString() ) ) );
177
178    final boolean injectClassesPresent =
179      null != processingEnv.getElementUtils().getTypeElement( Constants.INJECT_ANNOTATION_CLASSNAME );
180
181    final ComponentDescriptor descriptor =
182      new ComponentDescriptor( processingEnv.getElementUtils(),
183                               type,
184                               nameIncludesId,
185                               allowEmpty,
186                               injectClassesPresent,
187                               dagger || inject,
188                               dagger,
189                               scopeAnnotation,
190                               deferSchedule,
191                               generateToString,
192                               packageElement,
193                               typeElement );
194
195    descriptor.analyzeCandidateMethods( methods, processingEnv.getTypeUtils() );
196    descriptor.validate();
197
198    for ( final ObservableDescriptor observable : descriptor.getObservables() )
199    {
200      if ( observable.expectSetter() )
201      {
202        final TypeMirror returnType = observable.getGetterType().getReturnType();
203        final TypeMirror parameterType = observable.getSetterType().getParameterTypes().get( 0 );
204        if ( !processingEnv.getTypeUtils().isSameType( parameterType, returnType ) )
205        {
206          throw new ArezProcessorException( "@Observable property defines a setter and getter with different types." +
207                                            " Getter type: " + returnType + " Setter type: " + parameterType + ".",
208                                            observable.getGetter() );
209        }
210      }
211    }
212
213    final AnnotationMirror repository =
214      ProcessorUtil.findAnnotationByType( typeElement, Constants.REPOSITORY_ANNOTATION_CLASSNAME );
215    if ( null != repository )
216    {
217      final List<TypeElement> extensions =
218        ProcessorUtil.getTypeMirrorsAnnotationParameter( processingEnv.getElementUtils(),
219                                                         typeElement,
220                                                         Constants.REPOSITORY_ANNOTATION_CLASSNAME,
221                                                         "extensions" ).stream().
222          map( typeMirror -> (TypeElement) processingEnv.getTypeUtils().asElement( typeMirror ) ).
223          collect( Collectors.toList() );
224      final String name = getAnnotationParameter( repository, "name" );
225      final String repositoryInjectConfig = getRepositoryInjectConfig( typeElement );
226      final String repositoryDaggerConfig = getRepositoryDaggerConfig( typeElement );
227      descriptor.configureRepository( name, extensions, repositoryInjectConfig, repositoryDaggerConfig );
228    }
229
230    return descriptor;
231  }

233  private boolean isScopeAnnotation( @Nonnull final AnnotationMirror a )
234  {
235    final Element element = processingEnv.getTypeUtils().asElement( a.getAnnotationType() );
236    return null != ProcessorUtil.findAnnotationByType( element, Constants.SCOPE_ANNOTATION_CLASSNAME );
237  }
238
239  private boolean isInjectionRequired( @Nonnull final TypeElement typeElement,
240                                       @Nullable final AnnotationMirror scopeAnnotation )
241  {
242    final VariableElement injectParameter = (VariableElement)
243      ProcessorUtil.getAnnotationValue( processingEnv.getElementUtils(),
244                                        typeElement,
245                                        Constants.COMPONENT_ANNOTATION_CLASSNAME,
246                                        "inject" ).getValue();
247    switch ( injectParameter.getSimpleName().toString() )
248    {
249      case "TRUE":
250        return true;
251      case "FALSE":
252        return false;
253      default:
254        return null != scopeAnnotation ||
255               ProcessorUtil.getFieldElements( typeElement ).stream().anyMatch( this::hasInjectAnnotation ) ||
256               ProcessorUtil.getMethods( typeElement, processingEnv.getTypeUtils() ).
257                 stream().anyMatch( this::hasInjectAnnotation );
258    }
259  }
260
261  private boolean isDaggerRequired( @Nonnull final TypeElement typeElement,
262                                    @Nullable final AnnotationMirror scopeAnnotation )
263  {
264    final VariableElement daggerParameter = (VariableElement)
265      ProcessorUtil.getAnnotationValue( processingEnv.getElementUtils(),
266                                        typeElement,
267                                        Constants.COMPONENT_ANNOTATION_CLASSNAME,
268                                        "dagger" ).getValue();
269    switch ( daggerParameter.getSimpleName().toString() )
270    {
271      case "TRUE":
272        return true;
273      case "FALSE":
274        return false;
275      default:
276        return null != scopeAnnotation &&
277               null != processingEnv.getElementUtils().getTypeElement( Constants.DAGGER_MODULE_CLASSNAME );
278    }
279  }

281  @Nonnull
282  private String getRepositoryInjectConfig( @Nonnull final TypeElement typeElement )
283  {
284    final VariableElement injectParameter = (VariableElement)
285      ProcessorUtil.getAnnotationValue( processingEnv.getElementUtils(),
286                                        typeElement,
287                                        Constants.REPOSITORY_ANNOTATION_CLASSNAME,
288                                        "inject" ).getValue();
289    return injectParameter.getSimpleName().toString();
290  }
291
292  @Nonnull
293  private String getRepositoryDaggerConfig( @Nonnull final TypeElement typeElement )
294  {
295    final VariableElement daggerParameter = (VariableElement)
296      ProcessorUtil.getAnnotationValue( processingEnv.getElementUtils(),
297                                        typeElement,
298                                        Constants.REPOSITORY_ANNOTATION_CLASSNAME,
299                                        "dagger" ).getValue();
300    return daggerParameter.getSimpleName().toString();
301  }
302
303  private boolean hasInjectAnnotation( final Element method )
304  {
305    return null != ProcessorUtil.findAnnotationByType( method, Constants.INJECT_ANNOTATION_CLASSNAME );
306  }
307
308  @SuppressWarnings( "unchecked" )
309  private <T> T getAnnotationParameter( @Nonnull final AnnotationMirror annotation,
310                                        @Nonnull final String parameterName )
311  {
312    return (T) ProcessorUtil.getAnnotationValue( processingEnv.getElementUtils(),
313                                                 annotation,
314                                                 parameterName ).getValue();
315  }
316
317  private void emitTypeSpec( @Nonnull final String packageName, @Nonnull final TypeSpec typeSpec )
318    throws IOException
319  {
320    JavaFile.builder( packageName, typeSpec ).
321      skipJavaLangImports( true ).
322      build().
323      writeTo( processingEnv.getFiler() );
324  }
