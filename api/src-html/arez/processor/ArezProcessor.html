039/**
040 * Annotation processor that analyzes Arez annotated source and generates models from the annotations.
041 */
042@AutoService( Processor.class )
043@SupportedAnnotationTypes( { "arez.annotations.*" } )
044@SupportedSourceVersion( SourceVersion.RELEASE_8 )
045@SupportedOptions( "arez.defer.unresolved" )
046public final class ArezProcessor
047  extends AbstractProcessor
048{
049  @Nonnull
050  private HashSet<TypeElement> _deferred = new HashSet<>();
051
052  /**
053   * {@inheritDoc}
054   */
055  @Override
056  public boolean process( final Set<? extends TypeElement> annotations, final RoundEnvironment env )
057  {
058    final TypeElement annotation =
059      processingEnv.getElementUtils().getTypeElement( Constants.COMPONENT_ANNOTATION_CLASSNAME );
060    final Set<? extends Element> elements = env.getElementsAnnotatedWith( annotation );
061
062    final Map<String, String> options = processingEnv.getOptions();
063    final String deferUnresolvedValue = options.get( "arez.defer.unresolved" );
064    final boolean deferUnresolved = null == deferUnresolvedValue || "true".equals( deferUnresolvedValue );
065
066    if ( deferUnresolved )
067    {
068      final Collection<Element> elementsToProcess = getElementsToProcess( elements );
069      processElements( elementsToProcess, env );
070      if ( env.getRootElements().isEmpty() && !_deferred.isEmpty() )
071      {
072        _deferred.forEach( this::processingErrorMessage );
073        _deferred.clear();
074      }
075    }
076    else
077    {
078      processElements( new ArrayList<>( elements ), env );
079    }
080    return true;
081  }
082
083  private void processingErrorMessage( @Nonnull final TypeElement target )
084  {
085    processingEnv
086      .getMessager()
087      .printMessage( ERROR,
088                     "ArezProcessor unable to process " + target.getQualifiedName() +
089                     " because not all of its dependencies could be resolved. Check for " +
090                     "compilation errors or a circular dependency with generated code.",
091                     target );
092  }
093
094  private void processElements( @Nonnull final Collection<Element> elements,
095                                @Nonnull final RoundEnvironment env )
096  {
097    for ( final Element element : elements )
098    {
099      try
100      {
101        process( element );
102      }
103      catch ( final IOException ioe )
104      {
105        processingEnv.getMessager().printMessage( ERROR, ioe.getMessage(), element );
106      }
107      catch ( final ArezProcessorException e )
108      {
109        final Element errorLocation = e.getElement();
110        final Element outerElement = getOuterElement( errorLocation );
111        if ( !env.getRootElements().contains( outerElement ) )
112        {
113          final String location;
114          if ( errorLocation instanceof ExecutableElement )
115          {
116            final ExecutableElement executableElement = (ExecutableElement) errorLocation;
117            final TypeElement typeElement = (TypeElement) executableElement.getEnclosingElement();
118            location = typeElement.getQualifiedName() + "." + executableElement.getSimpleName();
119          }
120          else if ( errorLocation instanceof VariableElement )
121          {
122            final VariableElement variableElement = (VariableElement) errorLocation;
123            final TypeElement typeElement = (TypeElement) variableElement.getEnclosingElement();
124            location = typeElement.getQualifiedName() + "." + variableElement.getSimpleName();
125          }
126          else
127          {
128            assert errorLocation instanceof TypeElement;
129            final TypeElement typeElement = (TypeElement) errorLocation;
130            location = typeElement.getQualifiedName().toString();
131          }
132
133          final StringWriter sw = new StringWriter();
134          processingEnv.getElementUtils().printElements( sw, errorLocation );
135          sw.flush();
136
137          final String message =
138            "An error was generated processing the element " + element.getSimpleName() +
139            " but the error was triggered by code not currently being compiled but inherited or " +
140            "implemented by the element and may not be highlighted by your tooling or IDE. The " +
141            "error occurred at " + location + " and may look like:\n" + sw.toString();
142
143          processingEnv.getMessager().printMessage( ERROR, e.getMessage(), element );
144          processingEnv.getMessager().printMessage( ERROR, message );
145        }
146        processingEnv.getMessager().printMessage( ERROR, e.getMessage(), e.getElement() );
147      }
148      catch ( final Throwable e )
149      {
150        final StringWriter sw = new StringWriter();
151        e.printStackTrace( new PrintWriter( sw ) );
152        sw.flush();
153
154        final String message =
155          "Unexpected error running the " + getClass().getName() + " processor. This has " +
156          "resulted in a failure to process the code and has left the compiler in an invalid " +
157          "state. Please report the failure to the developers so that it can be fixed.\n" +
158          " Report the error at: https://github.com/arez/arez/issues\n" +
159          "\n\n" +
160          sw.toString();
161        processingEnv.getMessager().printMessage( ERROR, message, element );
162      }
163    }
164  }

166  @Nonnull
167  private Collection<Element> getElementsToProcess( @Nonnull final Set<? extends Element> elements )
168  {
169    final List<TypeElement> deferred = _deferred
170      .stream()
171      .map( e -> processingEnv.getElementUtils().getTypeElement( e.getQualifiedName() ) )
172      .collect( Collectors.toList() );
173    _deferred = new HashSet<>();
174
175    final ArrayList<Element> elementsToProcess = new ArrayList<>();
176    collectElementsToProcess( elements, elementsToProcess );
177    collectElementsToProcess( deferred, elementsToProcess );
178    return elementsToProcess;
179  }
180
181  private void collectElementsToProcess( @Nonnull final Collection<? extends Element> elements,
182                                         @Nonnull final ArrayList<Element> elementsToProcess )
183  {
184    for ( final Element element : elements )
185    {
186      if ( SuperficialValidation.validateElement( element ) )
187      {
188        elementsToProcess.add( element );
189      }
190      else
191      {
192        _deferred.add( (TypeElement) element );
193      }
194    }
195  }

197  /**
198   * Return the outer enclosing element.
199   * This is either the top-level class, interface, enum, etc within a package.
200   * This helps identify the top level compilation units.
201   */
202  @Nonnull
203  private Element getOuterElement( @Nonnull final Element element )
204  {
205    Element result = element;
206    while ( !( result.getEnclosingElement() instanceof PackageElement ) )
207    {
208      result = result.getEnclosingElement();
209    }
210    return result;
211  }
212
213  private void process( @Nonnull final Element element )
214    throws IOException, ArezProcessorException
215  {
216    final PackageElement packageElement = processingEnv.getElementUtils().getPackageOf( element );
217    final TypeElement typeElement = (TypeElement) element;
218    final ComponentDescriptor descriptor = parse( packageElement, typeElement );
219    emitTypeSpec( descriptor.getPackageName(), descriptor.buildType( processingEnv.getTypeUtils() ) );
220    if ( descriptor.needsDaggerIntegration() )
221    {
222      if ( descriptor.needsDaggerComponentExtension() )
223      {
224        if ( ComponentDescriptor.InjectMode.PROVIDE == descriptor.getInjectMode() )
225        {
226          emitTypeSpec( descriptor.getPackageName(), Generator.buildProviderDaggerComponentExtension( descriptor ) );
227        }
228        else
229        {
230          emitTypeSpec( descriptor.getPackageName(), Generator.buildConsumerDaggerComponentExtension( descriptor ) );
231        }
232      }
233      else if ( descriptor.needsDaggerModule() )
234      {
235        emitTypeSpec( descriptor.getPackageName(), descriptor.buildComponentDaggerModule() );
236      }
237    }
238    if ( descriptor.hasRepository() )
239    {
240      emitTypeSpec( descriptor.getPackageName(), descriptor.buildRepository( processingEnv.getTypeUtils() ) );
241    }
242  }

244  @Nonnull
245  private ComponentDescriptor parse( @Nonnull final PackageElement packageElement,
246                                     @Nonnull final TypeElement typeElement )
247    throws ArezProcessorException
248  {
249    if ( ElementKind.CLASS != typeElement.getKind() )
250    {
251      throw new ArezProcessorException( "@ArezComponent target must be a class", typeElement );
252    }
253    else if ( typeElement.getModifiers().contains( Modifier.FINAL ) )
254    {
255      throw new ArezProcessorException( "@ArezComponent target must not be final", typeElement );
256    }
257    else if ( NestingKind.TOP_LEVEL != typeElement.getNestingKind() &&
258              !typeElement.getModifiers().contains( Modifier.STATIC ) )
259    {
260      throw new ArezProcessorException( "@ArezComponent target must not be a non-static nested class", typeElement );
261    }
262    // Is the component marked as generated
263    final boolean generated =
264      null != ProcessorUtil.findAnnotationByType( typeElement, Constants.GENERATED_ANNOTATION_CLASSNAME ) ||
265      null != ProcessorUtil.findAnnotationByType( typeElement, Constants.JAVA9_GENERATED_ANNOTATION_CLASSNAME );
266    final AnnotationMirror arezComponent =
267      ProcessorUtil.getAnnotationByType( typeElement, Constants.COMPONENT_ANNOTATION_CLASSNAME );
268    final String declaredType = getAnnotationParameter( arezComponent, "name" );
269    final AnnotationValue nameIncludesIdValue =
270      ProcessorUtil.findAnnotationValueNoDefaults( arezComponent, "nameIncludesId" );
271    final AnnotationMirror singletonAnnotation =
272      ProcessorUtil.findAnnotationByType( typeElement, Constants.SINGLETON_ANNOTATION_CLASSNAME );
273    final boolean nameIncludesIdDefault = null == singletonAnnotation;
274    final boolean nameIncludesId =
275      null == nameIncludesIdValue ? nameIncludesIdDefault : (boolean) nameIncludesIdValue.getValue();
276    final boolean disposeOnDeactivate = getAnnotationParameter( arezComponent, "disposeOnDeactivate" );
277    final boolean observableFlag = isComponentObservableRequired( arezComponent, typeElement, disposeOnDeactivate );
278    final boolean disposeTrackableFlag =
279      ProcessorUtil.isDisposableTrackableRequired( processingEnv.getElementUtils(), typeElement );
280    final boolean allowConcrete = getAnnotationParameter( arezComponent, "allowConcrete" );
281    final boolean allowEmpty = getAnnotationParameter( arezComponent, "allowEmpty" );
282    final List<AnnotationMirror> scopeAnnotations =
283      typeElement.getAnnotationMirrors().stream().filter( this::isScopeAnnotation ).collect( Collectors.toList() );
284    final AnnotationMirror scopeAnnotation = scopeAnnotations.isEmpty() ? null : scopeAnnotations.get( 0 );
285    final boolean fieldInjections =
286      ProcessorUtil.getFieldElements( typeElement ).stream().anyMatch( this::hasInjectAnnotation );
287    final boolean methodInjections =
288      ProcessorUtil.getMethods( typeElement, processingEnv.getElementUtils(), processingEnv.getTypeUtils() )
289        .stream()
290        .anyMatch( this::hasInjectAnnotation );
291    final boolean nonConstructorInjections = fieldInjections || methodInjections;
292    final VariableElement daggerParameter = getAnnotationParameter( arezComponent, "dagger" );
293    final String daggerMode = daggerParameter.getSimpleName().toString();
294
295    final String injectMode =
296      getInjectMode( arezComponent,
297                     typeElement,
298                     scopeAnnotation,
299                     daggerMode,
300                     fieldInjections,
301                     methodInjections );
302    final boolean dagger =
303      "ENABLE".equals( daggerMode ) ||
304      (
305        "AUTODETECT".equals( daggerMode ) &&
306        !"NONE".equals( injectMode ) &&
307        null != processingEnv.getElementUtils().getTypeElement( Constants.DAGGER_MODULE_CLASSNAME )
308      );
309
310    final boolean requireEquals = isEqualsRequired( arezComponent, typeElement );
311    final boolean requireVerify = isVerifyRequired( arezComponent, typeElement );
312    final boolean deferSchedule = getAnnotationParameter( arezComponent, "deferSchedule" );
313
314    final boolean isClassAbstract = typeElement.getModifiers().contains( Modifier.ABSTRACT );
315    if ( !isClassAbstract && !allowConcrete )
316    {
317      throw new ArezProcessorException( "@ArezComponent target must be abstract unless the allowConcrete " +
318                                        "parameter is set to true", typeElement );
319    }
320    else if ( isClassAbstract && allowConcrete )
321    {
322      throw new ArezProcessorException( "@ArezComponent target must be concrete if the allowConcrete " +
323                                        "parameter is set to true", typeElement );
324    }
325
326    final String type =
327      ProcessorUtil.isSentinelName( declaredType ) ? typeElement.getSimpleName().toString() : declaredType;
328
329    if ( !SourceVersion.isIdentifier( type ) )
330    {
331      throw new ArezProcessorException( "@ArezComponent target specified an invalid type '" + type + "'. The " +
332                                        "type must be a valid java identifier.", typeElement );
333    }
334    else if ( SourceVersion.isKeyword( type ) )
335    {
336      throw new ArezProcessorException( "@ArezComponent target specified an invalid type '" + type + "'. The " +
337                                        "type must not be a java keyword.", typeElement );
338    }
339
340    if ( !scopeAnnotations.isEmpty() && ProcessorUtil.getConstructors( typeElement ).size() > 1 )
341    {
342      throw new ArezProcessorException( "@ArezComponent target has specified a scope annotation but has more than " +
343                                        "one constructor and thus is not a candidate for injection",
344                                        typeElement );
345    }
346
347    if ( !"NONE".equals( injectMode ) && ProcessorUtil.getConstructors( typeElement ).size() > 1 )
348    {
349      throw new ArezProcessorException( "@ArezComponent specified inject parameter but has more than one constructor",
350                                        typeElement );
351    }
352
353    if ( scopeAnnotations.size() > 1 )
354    {
355      final List<String> scopes = scopeAnnotations.stream()
356        .map( a -> processingEnv.getTypeUtils().asElement( a.getAnnotationType() ).asType().toString() )
357        .collect( Collectors.toList() );
358      throw new ArezProcessorException( "@ArezComponent target has specified multiple scope annotations: " + scopes,
359                                        typeElement );
360    }
361    if ( !observableFlag && disposeOnDeactivate )
362    {
363      throw new ArezProcessorException( "@ArezComponent target has specified observable = DISABLE and " +
364                                        "disposeOnDeactivate = true which is not a valid combination", typeElement );
365    }
366
367    boolean generatesFactoryToInject = false;
368    if ( dagger )
369    {
370      final ExecutableElement ctor = ProcessorUtil.getConstructors( typeElement ).get( 0 );
371      assert null != ctor;
372      final List<? extends VariableElement> perInstanceParameters = ctor.getParameters()
373        .stream()
374        .filter( f -> null != ProcessorUtil.findAnnotationByType( f, Constants.PER_INSTANCE_ANNOTATION_CLASSNAME ) )
375        .collect( Collectors.toList() );
376      if ( !perInstanceParameters.isEmpty() )
377      {
378        if ( "PROVIDE".equals( injectMode ) )
379        {
380          throw new ArezProcessorException( "@ArezComponent target has specified at least one @PerInstance " +
381                                            "parameter on the constructor but has set inject parameter to PROVIDE. " +
382                                            "The component cannot be provided to other components if the invoker " +
383                                            "must supply per-instance parameters so either change the inject " +
384                                            "parameter to CONSUME or remove the @PerInstance parameter.",
385                                            ctor );
386        }
387        generatesFactoryToInject = true;
388      }
389    }
390
391    final List<ExecutableElement> methods =
392      ProcessorUtil.getMethods( typeElement, processingEnv.getElementUtils(), processingEnv.getTypeUtils() );
393    final boolean generateToString = methods.stream().
394      noneMatch( m -> m.getSimpleName().toString().equals( "toString" ) &&
395                      m.getParameters().size() == 0 &&
396                      !( m.getEnclosingElement().getSimpleName().toString().equals( "Object" ) &&
397                         "java.lang".equals( processingEnv.getElementUtils().
398                           getPackageOf( m.getEnclosingElement() ).getQualifiedName().toString() ) ) );
399
400    final ComponentDescriptor descriptor =
401      new ComponentDescriptor( processingEnv.getSourceVersion(),
402                               processingEnv.getElementUtils(),
403                               processingEnv.getTypeUtils(),
404                               type,
405                               nameIncludesId,
406                               allowEmpty,
407                               generated,
408                               observableFlag,
409                               disposeTrackableFlag,
410                               disposeOnDeactivate,
411                               injectMode,
412                               dagger,
413                               generatesFactoryToInject,
414                               nonConstructorInjections,
415                               requireEquals,
416                               requireVerify,
417                               scopeAnnotation,
418                               deferSchedule,
419                               generateToString,
420                               packageElement,
421                               typeElement );
422
423    descriptor.analyzeCandidateMethods( methods, processingEnv.getTypeUtils() );
424    descriptor.validate();
425
426    for ( final ObservableDescriptor observable : descriptor.getObservables() )
427    {
428      if ( observable.expectSetter() )
429      {
430        final TypeMirror returnType = observable.getGetterType().getReturnType();
431        final TypeMirror parameterType = observable.getSetterType().getParameterTypes().get( 0 );
432        if ( !processingEnv.getTypeUtils().isSameType( parameterType, returnType ) &&
433             !parameterType.toString().equals( returnType.toString() ) )
434        {
435          throw new ArezProcessorException( "@Observable property defines a setter and getter with different types." +
436                                            " Getter type: " + returnType + " Setter type: " + parameterType + ".",
437                                            observable.getGetter() );
438        }
439      }
440    }
441
442    final AnnotationMirror repository =
443      ProcessorUtil.findAnnotationByType( typeElement, Constants.REPOSITORY_ANNOTATION_CLASSNAME );
444    if ( null != repository )
445    {
446      final List<TypeElement> extensions =
447        ProcessorUtil.getTypeMirrorsAnnotationParameter( processingEnv.getElementUtils(),
448                                                         typeElement,
449                                                         Constants.REPOSITORY_ANNOTATION_CLASSNAME,
450                                                         "extensions" ).stream().
451          map( typeMirror -> (TypeElement) processingEnv.getTypeUtils().asElement( typeMirror ) ).
452          collect( Collectors.toList() );
453      final String name = getAnnotationParameter( repository, "name" );
454      final String repositoryInjectConfig = getRepositoryInjectMode( repository );
455      final String repositoryDaggerConfig = getRepositoryDaggerConfig( repository );
456      descriptor.configureRepository( name, extensions, repositoryInjectConfig, repositoryDaggerConfig );
457    }
458    if ( !observableFlag && descriptor.hasRepository() )
459    {
460      throw new ArezProcessorException( "@ArezComponent target has specified observable = DISABLE and " +
461                                        "but is also annotated with the @Repository annotation which requires " +
462                                        "that the observable != DISABLE.", typeElement );
463    }
464    if ( descriptor.hasRepository() &&
465         null != ProcessorUtil.findAnnotationByType( typeElement, Constants.SINGLETON_ANNOTATION_CLASSNAME ) )
466    {
467      throw new ArezProcessorException( "@ArezComponent target is annotated with both the " +
468                                        "@arez.annotations.Repository annotation and the " +
469                                        "javax.inject.Singleton annotation which is an invalid " +
470                                        "combination.", typeElement );
471    }
472    if ( !descriptor.isDisposeTrackable() && descriptor.hasRepository() )
473    {
474      throw new ArezProcessorException( "@ArezComponent target has specified the disposeTrackable = DISABLE " +
475                                        "annotation parameter but is also annotated with @Repository that " +
476                                        "requires disposeTrackable = ENABLE.", typeElement );
477    }
478
479    final boolean idRequired = isIdRequired( descriptor, arezComponent );
480    descriptor.setIdRequired( idRequired );
481    if ( !idRequired )
482    {
483      if ( descriptor.hasRepository() )
484      {
485        throw new ArezProcessorException( "@ArezComponent target has specified the idRequired = DISABLE " +
486                                          "annotation parameter but is also annotated with @Repository that " +
487                                          "requires idRequired = ENABLE.", typeElement );
488      }
489      if ( descriptor.hasComponentIdMethod() )
490      {
491        throw new ArezProcessorException( "@ArezComponent target has specified the idRequired = DISABLE " +
492                                          "annotation parameter but also has annotated a method with @ComponentId " +
493                                          "that requires idRequired = ENABLE.", typeElement );
494      }
495      if ( descriptor.hasComponentIdRefMethod() )
496      {
497        throw new ArezProcessorException( "@ArezComponent target has specified the idRequired = DISABLE " +
498                                          "annotation parameter but also has annotated a method with @ComponentIdRef " +
499                                          "that requires idRequired = ENABLE.", typeElement );
500      }
501    }
502
503    return descriptor;
504  }

506  private boolean isScopeAnnotation( @Nonnull final AnnotationMirror a )
507  {
508    final Element element = processingEnv.getTypeUtils().asElement( a.getAnnotationType() );
509    return null != ProcessorUtil.findAnnotationByType( element, Constants.SCOPE_ANNOTATION_CLASSNAME );
510  }

512  private boolean isComponentObservableRequired( @Nonnull final AnnotationMirror arezComponent,
513                                                 @Nonnull final TypeElement typeElement,
514                                                 final boolean disposeOnDeactivate )
515  {
516    final VariableElement variableElement = getAnnotationParameter( arezComponent, "observable" );
517    switch ( variableElement.getSimpleName().toString() )
518    {
519      case "ENABLE":
520        return true;
521      case "DISABLE":
522        return false;
523      default:
524        return disposeOnDeactivate ||
525               null != ProcessorUtil.findAnnotationByType( typeElement, Constants.REPOSITORY_ANNOTATION_CLASSNAME );
526    }
527  }

529  @Nonnull
530  private String getInjectMode( @Nonnull final AnnotationMirror arezComponent,
531                                @Nonnull final TypeElement typeElement,
532                                @Nullable final AnnotationMirror scopeAnnotation,
533                                final String daggerMode,
534                                final boolean fieldInjections,
535                                final boolean methodInjections )
536  {
537    final VariableElement injectParameter = getAnnotationParameter( arezComponent, "inject" );
538    final String mode = injectParameter.getSimpleName().toString();
539    if ( "AUTODETECT".equals( mode ) )
540    {
541      final boolean shouldInject =
542        daggerMode.equals( "ENABLE" ) || null != scopeAnnotation || fieldInjections || methodInjections;
543      return shouldInject ? "PROVIDE" : "NONE";
544    }
545    else if ( "NONE".equals( mode ) )
546    {
547      if ( daggerMode.equals( "ENABLE" ) )
548      {
549        throw new ArezProcessorException( "@ArezComponent target has a dagger parameter that resolved to ENABLE " +
550                                          "but the inject parameter is set to NONE and this is not a valid " +
551                                          "combination of parameters.", typeElement );
552      }
553      if ( fieldInjections )
554      {
555        throw new ArezProcessorException( "@ArezComponent target has fields annotated with the javax.inject.Inject " +
556                                          "annotation but the inject parameter is set to NONE and this is not a " +
557                                          "valid scenario. Remove the @Inject annotation(s) or change the inject " +
558                                          "parameter to a value other than NONE.", typeElement );
559      }
560      if ( methodInjections )
561      {
562        throw new ArezProcessorException( "@ArezComponent target has methods annotated with the javax.inject.Inject " +
563                                          "annotation but the inject parameter is set to NONE and this is not a " +
564                                          "valid scenario. Remove the @Inject annotation(s) or change the inject " +
565                                          "parameter to a value other than NONE.", typeElement );
566      }
567      if ( null != scopeAnnotation )
568      {
569        throw new ArezProcessorException( "@ArezComponent target is annotated with scope annotation " +
570                                          scopeAnnotation + " but the inject parameter is set to NONE and this " +
571                                          "is not a valid scenario. Remove the scope annotation or change the " +
572                                          "inject parameter to a value other than NONE.", typeElement );
573      }
574      return mode;
575    }
576    else
577    {
578      return mode;
579    }
580  }

582  private boolean isVerifyRequired( @Nonnull final AnnotationMirror arezComponent,
583                                    @Nonnull final TypeElement typeElement )
584  {
585    final VariableElement daggerParameter = getAnnotationParameter( arezComponent, "verify" );
586    switch ( daggerParameter.getSimpleName().toString() )
587    {
588      case "ENABLE":
589        return true;
590      case "DISABLE":
591        return false;
592      default:
593        return ProcessorUtil.getMethods( typeElement, processingEnv.getElementUtils(), processingEnv.getTypeUtils() ).
594          stream().anyMatch( this::hasReferenceAnnotations );
595    }
596  }

598  private boolean hasReferenceAnnotations( @Nonnull final Element method )
599  {
600    return null != ProcessorUtil.findAnnotationByType( method, Constants.REFERENCE_ANNOTATION_CLASSNAME ) ||
601           null != ProcessorUtil.findAnnotationByType( method, Constants.REFERENCE_ID_ANNOTATION_CLASSNAME ) ||
602           null != ProcessorUtil.findAnnotationByType( method, Constants.INVERSE_ANNOTATION_CLASSNAME );
603  }

605  private boolean isEqualsRequired( @Nonnull final AnnotationMirror arezComponent,
606                                    @Nonnull final TypeElement typeElement )
607  {
608    final VariableElement injectParameter = getAnnotationParameter( arezComponent, "requireEquals" );
609    switch ( injectParameter.getSimpleName().toString() )
610    {
611      case "ENABLE":
612        return true;
613      case "DISABLE":
614        return false;
615      default:
616        return null != ProcessorUtil.findAnnotationByType( typeElement, Constants.REPOSITORY_ANNOTATION_CLASSNAME );
617    }
618  }

620  private boolean isIdRequired( @Nonnull final ComponentDescriptor descriptor,
621                                @Nonnull final AnnotationMirror arezComponent )
622  {
623    final VariableElement injectParameter = getAnnotationParameter( arezComponent, "requireId" );
624    switch ( injectParameter.getSimpleName().toString() )
625    {
626      case "ENABLE":
627        return true;
628      case "DISABLE":
629        return false;
630      default:
631        return descriptor.hasRepository() || descriptor.hasComponentIdMethod() || descriptor.hasComponentIdRefMethod();
632    }
633  }

638    final VariableElement injectParameter = getAnnotationParameter( repository, "inject" );
