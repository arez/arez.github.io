001package arez;
002
003import arez.spy.Spy;
004import java.util.Objects;
005import javax.annotation.Nonnull;
006import javax.annotation.Nullable;
007import static arez.Guards.*;
008
009/**
010 * A node within an Arez dependency graph.
011 * The node is a named element within a specific Arez system that forms part of the
012 * dependency graph.
013 *
014 * <p>The Node class can be extended by classes outside the Arez core package. Typically this is
015 * done when a collection of primitive types (i.e. Observables, Observers, ComputableValues etc) are
016 * aggregated to form a single abstraction within the reactive system.</p>
017 */
018public abstract class Node
019  implements Disposable
020{
021  /**
022   * Reference to the context to which this node belongs.
023   */
024  @Nullable
025  private final ArezContext _context;
026  /**
027   * A human consumable name for node. It should be non-null if {@link Arez#areNamesEnabled()} returns
028   * true and <tt>null</tt> otherwise.
029   */
030  @Nullable
031  private final String _name;
032
033  Node( @Nullable final ArezContext context, @Nullable final String name )
034  {
035    if ( Arez.shouldCheckApiInvariants() )
036    {
037      apiInvariant( () -> Arez.areZonesEnabled() || null == context,
038                    () -> "Arez-0180: Node passed a context but Arez.areZonesEnabled() is false" );
039      apiInvariant( () -> Arez.areNamesEnabled() || null == name,
040                    () -> "Arez-0052: Node passed a name '" + name + "' but Arez.areNamesEnabled() is false" );
041    }
042    _context = Arez.areZonesEnabled() ? Objects.requireNonNull( context ) : null;
043    _name = Arez.areNamesEnabled() ? Objects.requireNonNull( name ) : null;
044  }
045
046  /**
047   * Return the name of the node.
048   * This method should NOT be invoked unless {@link Arez#areNamesEnabled()} returns <code>true</code>.
049   *
050   * @return the name of the node.
051   */
052  @Nonnull
053  public final String getName()
054  {
055    if ( Arez.shouldCheckApiInvariants() )
056    {
057      apiInvariant( Arez::areNamesEnabled,
058                    () -> "Arez-0053: Node.getName() invoked when Arez.areNamesEnabled() is false" );
059    }
060    assert null != _name;
061    return _name;
062  }
063
064  /**
065   * Return the context that the node is associated with.
066   *
067   * @return the associated ArezContext.
068   */
069  @Nonnull
070  public final ArezContext getContext()
071  {
072    return Arez.areZonesEnabled() ? Objects.requireNonNull( _context ) : Arez.context();
073  }
074
075  /**
076   * {@inheritDoc}
077   */
078  @Nonnull
079  @Override
080  public final String toString()
081  {
082    if ( Arez.areNamesEnabled() )
083    {
084      return getName();
085    }
086    else
087    {
088      return super.toString();
089    }
090  }
091
092  /**
093   * Return true if spy events will be propagated.
094   * This means spies are enabled and there is at least one spy event handler present.
095   *
096   * @return true if spy events will be propagated, false otherwise.
097   */
098  final boolean willPropagateSpyEvents()
099  {
100    return Arez.areSpiesEnabled() && getSpy().willPropagateSpyEvents();
101  }
102
103  /**
104   * Return the spy associated with context.
105   * This method should not be invoked unless {@link Arez#areSpiesEnabled()} returns true.
106   *
107   * @return the spy associated with context.
108   */
109  @Nonnull
110  final Spy getSpy()
111  {
112    return getContext().getSpy();
113  }
114
115  /**
116   * Report a spy event.
117   *
118   * @param event the event that occurred.
119   */
120  final void reportSpyEvent( @Nonnull final Object event )
121  {
122    getSpy().reportSpyEvent( event );
123  }
124}
