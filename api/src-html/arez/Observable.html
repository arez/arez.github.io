017/**
018 * The observable represents state that can be observed within the system.
019 */
020public final class Observable<T>
021  extends Node
022{
023  /**
024   * The value of _workState when the Observable is should longer be used.
025   */
026  static final int DISPOSED = -2;
027  /**
028   * The value that _workState is set to to optimize the detection of duplicate,
029   * existing and new dependencies during tracking completion.
030   */
031  static final int IN_CURRENT_TRACKING = -1;
032  /**
033   * The value that _workState is when the observer has been added as new dependency
034   * to derivation.
035   */
036  static final int NOT_IN_CURRENT_TRACKING = 0;
037
038  private final ArrayList<Observer> _observers = new ArrayList<>();
039  /**
040   * True if deactivation has been requested.
041   * Used to avoid adding duplicates to pending deactivation list.
042   */
043  private boolean _pendingDeactivation;
044  /**
045   * The workState variable contains some data used during processing of observable
046   * at various stages.
047   *
048   * Within the scope of a tracking transaction, it is set to the id of the tracking
049   * observer if the observable was observed. This enables an optimization that skips
050   * adding this observer to the same observer multiple times. This optimization sometimes
051   * ignored as nested transactions that observe the same observer will reset this value.
052   *
053   * When completing a tracking transaction the value may be set to {@link #IN_CURRENT_TRACKING}
054   * or {@link #NOT_IN_CURRENT_TRACKING} but should be set to {@link #NOT_IN_CURRENT_TRACKING} after
055   * {@link Transaction#completeTracking()} method is completed..
056   */
057  private int _workState;
058  /**
059   * The state of the observer that is least stale.
060   * This cached value is used to avoid redundant propagations.
061   */
062  @Nonnull
063  private ObserverState _leastStaleObserverState = ObserverState.UP_TO_DATE;
064  /**
065   * The observer that created this observable if any.
066   */
067  @Nullable
068  private final Observer _owner;
069  /**
070   * The component that this observable is contained within.
071   * This should only be set if {@link Arez#areNativeComponentsEnabled()} is true but may also be null if
072   * the observable is a "top-level" observable.
073   */
074  @Nullable
075  private final Component _component;
076  /**
077   * The accessor method to retrieve the value.
078   * This should only be set if {@link Arez#arePropertyIntrospectorsEnabled()} is true but may also be elided if the
079   * value should not be accessed even by DevTools.
080   */
081  @Nullable
082  private final PropertyAccessor<T> _accessor;
083  /**
084   * The mutator method to change the value.
085   * This should only be set if {@link Arez#arePropertyIntrospectorsEnabled()} is true but may also be elided if the
086   * value should not be mutated even by DevTools.
087   */
088  @Nullable
089  private final PropertyMutator<T> _mutator;

091  Observable( @Nullable final ArezContext context,
092              @Nullable final Component component,
093              @Nullable final String name,
094              @Nullable final Observer owner,
095              @Nullable final PropertyAccessor<T> accessor,
096              @Nullable final PropertyMutator<T> mutator )
097  {
098    super( context, name );
099    _component = Arez.areNativeComponentsEnabled() ? component : null;
100    _owner = owner;
101    _accessor = accessor;
102    _mutator = mutator;
103    if ( Arez.shouldCheckInvariants() )
104    {
105      invariant( () -> Arez.areNativeComponentsEnabled() || null == component,
106                 () -> "Arez-0054: Observable named '" + getName() + "' has component specified but " +
107                       "Arez.areNativeComponentsEnabled() is false." );
108    }
109    if ( Arez.shouldCheckApiInvariants() )
110    {
111      apiInvariant( () -> Arez.arePropertyIntrospectorsEnabled() || null == accessor,
112                    () -> "Arez-0055: Observable named '" + getName() + "' has accessor specified but " +
113                          "Arez.arePropertyIntrospectorsEnabled() is false." );
114      apiInvariant( () -> Arez.arePropertyIntrospectorsEnabled() || null == mutator,
115                    () -> "Arez-0056: Observable named '" + getName() + "' has mutator specified but " +
116                          "Arez.arePropertyIntrospectorsEnabled() is false." );
117    }
118    if ( null != _owner )
119    {
120      // This invariant can not be checked if Arez.shouldEnforceTransactionType() is false as
121      // the variable has yet to be assigned and no transaction mode set. Thus just skip the
122      // check in this scenario.
123      if ( Arez.shouldCheckInvariants() )
124      {
125        invariant( () -> !Arez.shouldEnforceTransactionType() || _owner.isDerivation(),
126                   () -> "Arez-0057: Observable named '" + getName() + "' has owner specified " +
127                         "but owner is not a derivation." );
128      }
129      assert !Arez.areNamesEnabled() || _owner.getName().equals( name );
130    }
131    if ( !hasOwner() )
132    {
133      if ( null != _component )
134      {
135        _component.addObservable( this );
136      }
137      else if ( Arez.areRegistriesEnabled() )
138      {
139        getContext().registerObservable( this );
140      }
141    }
142  }
143
144  /**
145   * {@inheritDoc}
146   */
147  @Override
148  public void dispose()
149  {
150    if ( !isDisposed() )
151    {
152      getContext().dispose( Arez.areNamesEnabled() ? getName() : null, this::performDispose );
153      // All dependencies should have been released by the time it comes to deactivate phase.
154      // The Observable has been marked as changed, forcing all observers to re-evaluate and
155      // ultimately this will result in their removal of this Observable as a dependency as
156      // it is an error to invoke reportObserved(). Once all dependencies are removed then
157      // this Observable will be deactivated if it is a ComputedValue. Thus no need to call
158      // queueForDeactivation() here.
159      if ( hasOwner() )
160      {
161        /*
162         * Dispose the owner first so that it is removed as a dependency and thus will not have a reaction
163         * scheduled.
164         */
165        getOwner().dispose();
166      }
167      else
168      {
169        if ( willPropagateSpyEvents() )
170        {
171          reportSpyEvent( new ObservableDisposedEvent( new ObservableInfoImpl( getSpy(), this ) ) );
172        }
173        if ( null != _component )
174        {
175          _component.removeObservable( this );
176        }
177        else if ( Arez.areRegistriesEnabled() )
178        {
179          getContext().deregisterObservable( this );
180        }
181      }
182    }
183  }
184
185  private void performDispose()
186  {
187    getContext().getTransaction().reportDispose( this );
188    reportChanged();
189    _workState = DISPOSED;
190  }

277  /**
278   * Return true if this observable can deactivate when it is no longer observable and activate when it is observable again.
279   */

282    return hasOwner() && !getOwner().getComputedValue().isKeepAlive();
283  }

301  /**
302   * Deactivate the observable.
303   * This means that the observable no longer has any listeners and can release resources associated
304   * with generating values. (i.e. remove observers on any observables that are used to compute the
305   * value of this observable).
306   */
307  void deactivate()
308  {
309    if ( Arez.shouldCheckInvariants() )
310    {
311      invariant( () -> getContext().isTransactionActive(),
312                 () -> "Arez-0060: Attempt to invoke deactivate on observable named '" + getName() +
313                       "' when there is no active transaction." );
314      invariant( this::canDeactivate,
315                 () -> "Arez-0061: Invoked deactivate on observable named '" + getName() + "' but observable " +
316                       "can not be deactivated. Either owner is null or the associated ComputedValue " +
317                       "has keepAlive enabled." );
318    }
319    assert null != _owner;
320    if ( _owner.isActive() )
321    {
322      /*
323       * It is possible for the owner to already be deactivated if dispose() is explicitly
324       * called within the transaction.
325       */
326      _owner.setState( ObserverState.INACTIVE );
327      if ( willPropagateSpyEvents() )
328      {
329        reportSpyEvent( new ComputedValueDeactivatedEvent( new ComputedValueInfoImpl( getSpy(),
330                                                                                      _owner.getComputedValue() ) ) );
331      }
332    }
333  }
334
335  /**
336   * Activate the observable.
337   * The reverse of {@link #deactivate()}.
338   */
339  void activate()
340  {
341    if ( Arez.shouldCheckInvariants() )
342    {
343      invariant( () -> getContext().isTransactionActive(),
344                 () -> "Arez-0062: Attempt to invoke activate on observable named '" + getName() + "' when there " +
345                       "is no active transaction." );
346      invariant( () -> null != _owner,
347                 () -> "Arez-0063: Invoked activate on observable named '" + getName() + "' when owner is null." );
348      assert null != _owner;
349      invariant( _owner::isInactive,
350                 () -> "Arez-0064: Invoked activate on observable named '" + getName() + "' when " +
351                       "observable is already active." );
352    }
353    assert null != _owner;
354    _owner.setState( ObserverState.UP_TO_DATE );
355    if ( willPropagateSpyEvents() )
356    {
357      reportSpyEvent( new ComputedValueActivatedEvent( new ComputedValueInfoImpl( getSpy(),
358                                                                                  _owner.getComputedValue() ) ) );
359    }
360  }

378  void addObserver( @Nonnull final Observer observer )
379  {
380    if ( Arez.shouldCheckInvariants() )
381    {
382      invariant( () -> getContext().isTransactionActive(),
383                 () -> "Arez-0065: Attempt to invoke addObserver on observable named '" + getName() +
384                       "' when there is no active transaction." );
385      invariantObserversLinked();
386      invariant( () -> !hasObserver( observer ),
387                 () -> "Arez-0066: Attempting to add observer named '" + observer.getName() + "' to " +
388                       "observable named '" + getName() + "' when observer is already observing observable." );
389      invariant( () -> !isDisposed(),
390                 () -> "Arez-0067: Attempting to add observer named '" + observer.getName() + "' to " +
391                       "observable named '" + getName() + "' when observable is disposed." );
392      invariant( observer::isLive,
393                 () -> "Arez-0068: Attempting to add observer named '" + observer.getName() + "' to observable " +
394                       "named '" + getName() + "' when observer is disposed." );
395      invariant( () -> !hasOwner() || !observer.isHighPriority() || getOwner().isHighPriority(),
396                 () -> "Arez-0183: Attempting to add observer named '" + observer.getName() + "' to observable " +
397                       "named '" + getName() + "' where the observer is scheduled at a high priority but the " +
398                       "observables owner is scheduled at a normal priority." );
399    }
400    getObservers().add( observer );
401
402    final ObserverState state =
403      ObserverState.INACTIVE == observer.getState() ? ObserverState.UP_TO_DATE : observer.getState();
404    if ( _leastStaleObserverState.ordinal() > state.ordinal() )
405    {
406      _leastStaleObserverState = state;
407    }
408  }

410  void removeObserver( @Nonnull final Observer observer )
411  {
412    if ( Arez.shouldCheckInvariants() )
413    {
414      invariant( () -> getContext().isTransactionActive(),
415                 () -> "Arez-0069: Attempt to invoke removeObserver on observable named '" + getName() + "' " +
416                       "when there is no active transaction." );
417      invariantObserversLinked();
418      invariant( () -> hasObserver( observer ),
419                 () -> "Arez-0070: Attempting to remove observer named '" + observer.getName() + "' from observable " +
420                       "named '" + getName() + "' when observer is already observing observable." );
421    }
422    final ArrayList<Observer> observers = getObservers();
423    observers.remove( observer );
424    if ( observers.isEmpty() && canDeactivate() )
425    {
426      queueForDeactivation();
427    }
428    if ( Arez.shouldCheckInvariants() )
429    {
430      invariantObserversLinked();
431    }
432  }

434  void queueForDeactivation()
435  {
436    if ( Arez.shouldCheckInvariants() )
437    {
438      invariant( () -> getContext().isTransactionActive(),
439                 () -> "Arez-0071: Attempt to invoke queueForDeactivation on observable named '" + getName() +
440                       "' when there is no active transaction." );
441      invariant( this::canDeactivate,
442                 () -> "Arez-0072: Attempted to invoke queueForDeactivation() on observable named '" + getName() +
443                       "' but observable is not able to be deactivated." );
444      invariant( () -> !hasObservers(),
445                 () -> "Arez-0073: Attempted to invoke queueForDeactivation() on observable named '" + getName() +
446                       "' but observable has observers." );
447    }
448    if ( !isPendingDeactivation() )
449    {
450      getContext().getTransaction().queueForDeactivation( this );
451    }
452  }

522  /**
523   * Return the value from observable if introspectors are enabled and an accessor has been supplied.
524   */
525  @Nullable
526  private Object getObservableValue()
527  {
528    if ( Arez.arePropertyIntrospectorsEnabled() && null != getAccessor() )
529    {
530      try
531      {
532        return getAccessor().get();
533      }
534      catch ( final Throwable ignored )
535      {
536      }
537    }

541  void invariantOwner()
542  {
543    if ( Arez.shouldCheckInvariants() && null != _owner )
544    {
545      invariant( () -> Objects.equals( _owner.getDerivedValue(), this ),
546                 () -> "Arez-0076: Observable named '" + getName() + "' has owner specified but owner does " +
547                       "not link to observable as derived value." );
548    }
549  }
550
551  void invariantObserversLinked()
552  {
553    if ( Arez.shouldCheckInvariants() )
554    {
555      getObservers().forEach( observer ->
556                                invariant( () -> observer.getDependencies().contains( this ),
557                                           () -> "Arez-0077: Observable named '" + getName() + "' has observer " +
558                                                 "named '" + observer.getName() + "' which does not contain " +
559                                                 "observable as dependency." ) );
560    }
561  }
562
563  void invariantLeastStaleObserverState()
564  {
565    if ( Arez.shouldCheckInvariants() )
566    {
567      final ObserverState leastStaleObserverState =
568        getObservers().stream().
569          map( Observer::getState ).map( s -> s == ObserverState.INACTIVE ? ObserverState.UP_TO_DATE : s ).
570          min( Comparator.comparing( Enum::ordinal ) ).orElse( ObserverState.UP_TO_DATE );
571      invariant( () -> leastStaleObserverState.ordinal() >= _leastStaleObserverState.ordinal(),
572                 () -> "Arez-0078: Calculated leastStaleObserverState on observable named '" + getName() +
573                       "' is '" + leastStaleObserverState.name() + "' which is unexpectedly less " +
574                       "than cached value '" + _leastStaleObserverState.name() + "'." );
575    }
576  }
