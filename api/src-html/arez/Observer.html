020/**
021 * A node within Arez that is notified of changes in 0 or more Observables.
022 */
023public final class Observer
024  extends Node
025{
026  /**
027   * The component that this observer is contained within.
028   * This should only be set if {@link Arez#areNativeComponentsEnabled()} is true but may also be null if
029   * the observer is a "top-level" observer.
030   */
031  @OmitSymbol( unless = "arez.enable_native_components" )
032  @Nullable
033  private final Component _component;
034  /**
035   * The reference to the ComputableValue if this observer is a derivation.
036   */
037  @Nullable
038  private final ComputableValue<?> _computableValue;
039  /**
040   * The observables that this observer receives notifications from.
041   * These are the dependencies within the dependency graph and will
042   * typically correspond to the observables that were accessed in last
043   * transaction that this observer was tracking.
044   *
045   * This list should contain no duplicates.
046   */
047  @Nonnull
048  private List<ObservableValue<?>> _dependencies = new ArrayList<>();
049  /**
050   * Observe function to invoke if any.
051   * This may be null if external executor is responsible for executing the observe function via
052   * methods such as {@link ArezContext#observe(Observer, Function, Object...)}. If this is null then
053   * {@link #_onDepsChange} must not be null.
054   */
055  @Nullable
056  private final Procedure _observe;
057  /**
058   * Callback invoked when dependencies are updated.
059   * This may be null when the observer re-executes the observe function when dependencies change
060   * but in that case {@link #_observe} must not be null.
061   */
062  @Nullable
063  private final Procedure _onDepsChange;
064  /**
065   * Cached info object associated with element.
066   * This should be null if {@link Arez#areSpiesEnabled()} is false.
067   */
068  @OmitSymbol( unless = "arez.enable_spies" )
069  @Nullable
070  private ObserverInfo _info;
071  /**
072   * A bitfield that contains config time and runtime flags/state.
073   * See the values in {@link Flags} that are covered by the masks
074   * {@link Flags#RUNTIME_FLAGS_MASK} and {@link Flags#CONFIG_FLAGS_MASK}
075   * for acceptable values.
076   */
077  private int _flags;
078  @Nonnull
079  private final Task _task;
080
081  Observer( @Nonnull final ComputableValue<?> computableValue, final int flags )
082  {
083    this( Arez.areZonesEnabled() ? computableValue.getContext() : null,
084          null,
085          Arez.areNamesEnabled() ? computableValue.getName() : null,
086          computableValue,
087          computableValue::compute,
088          null,
089          flags |
090          ( Flags.KEEPALIVE == Flags.getScheduleType( flags ) ? 0 : Flags.DEACTIVATE_ON_UNOBSERVE ) |
091          Task.Flags.runType( flags, Flags.KEEPALIVE == Flags.getScheduleType( flags ) ?
092                                     Task.Flags.RUN_NOW :
093                                     Task.Flags.RUN_LATER ) |
094          ( Arez.shouldEnforceTransactionType() ? Flags.READ_ONLY : 0 ) |
095          Flags.NESTED_ACTIONS_DISALLOWED |
096          Flags.dependencyType( flags ) );
097  }

099  Observer( @Nullable final ArezContext context,
100            @Nullable final Component component,
101            @Nullable final String name,
102            @Nullable final Procedure observe,
103            @Nullable final Procedure onDepsChange,
104            final int flags )
105  {
106    this( context,
107          component,
108          name,
109          null,
110          observe,
111          onDepsChange,
112          flags |
113          ( null == observe ? Flags.APPLICATION_EXECUTOR : Flags.KEEPALIVE ) |
114          Task.Flags.runType( flags, null == observe ? Task.Flags.RUN_LATER : Task.Flags.RUN_NOW ) |
115          Flags.nestedActionRule( flags ) |
116          Flags.dependencyType( flags ) |
117          Transaction.Flags.transactionMode( flags ) );
118  }

120  private Observer( @Nullable final ArezContext context,
121                    @Nullable final Component component,
122                    @Nullable final String name,
123                    @Nullable final ComputableValue<?> computableValue,
124                    @Nullable final Procedure observe,
125                    @Nullable final Procedure onDepsChange,
126                    final int flags )
127  {
128    super( context, name );
129    _task = new Task( context,
130                      name,
131                      this::invokeReaction,
132                      ( flags & Task.Flags.OBSERVER_TASK_FLAGS_MASK ) |
133                      Task.Flags.NO_REGISTER_TASK |
134                      Task.Flags.NO_WRAP_TASK );
135    _flags = ( flags & ~Task.Flags.OBSERVER_TASK_FLAGS_MASK ) | Flags.STATE_INACTIVE;
136    if ( Arez.shouldCheckInvariants() )
137    {
138      if ( Arez.shouldEnforceTransactionType() )
139      {
140        invariant( () -> Transaction.Flags.isTransactionModeValid( flags ),
141                   () -> "Arez-0079: Observer named '" + getName() + "' incorrectly specified both READ_ONLY " +
142                         "and READ_WRITE transaction mode flags." );
143      }
144      else
145      {
146        invariant( () -> !Transaction.Flags.isTransactionModeSpecified( flags ),
147                   () -> "Arez-0082: Observer named '" + getName() + "' specified transaction mode '" +
148                         Transaction.Flags.getTransactionModeName( flags ) + "' when " +
149                         "Arez.enforceTransactionType() is false." );
150      }
151      invariant( () -> Task.Flags.isPriorityValid( _task.getFlags() ),
152                 () -> "Arez-0080: Observer named '" + getName() + "' has invalid priority " +
153                       Task.Flags.getPriorityIndex( _task.getFlags() ) + "." );
154      invariant( () -> Task.Flags.isRunTypeValid( _task.getFlags() ),
155                 () -> "Arez-0081: Observer named '" + getName() + "' incorrectly specified both " +
156                       "RUN_NOW and RUN_LATER flags." );
157      invariant( () -> 0 == ( flags & Observer.Flags.RUN_NOW ) || null != observe,
158                 () -> "Arez-0206: Observer named '" + getName() + "' incorrectly specified " +
159                       "RUN_NOW flag but the observe function is null." );
160      invariant( () -> Arez.areNativeComponentsEnabled() || null == component,
161                 () -> "Arez-0083: Observer named '" + getName() + "' has component specified but " +
162                       "Arez.areNativeComponentsEnabled() is false." );
163      invariant( () -> Task.Flags.getPriority( flags ) != Task.Flags.PRIORITY_LOWEST ||
164                       0 == ( flags & Flags.OBSERVE_LOWER_PRIORITY_DEPENDENCIES ),
165                 () -> "Arez-0184: Observer named '" + getName() + "' has LOWEST priority but has passed " +
166                       "OBSERVE_LOWER_PRIORITY_DEPENDENCIES option which should not be present as the observer " +
167                       "has no lower priority." );
168      invariant( () -> null != observe || null != onDepsChange,
169                 () -> "Arez-0204: Observer named '" + getName() + "' has not supplied a value for either the " +
170                       "observe parameter or the onDepsChange parameter." );
171      // Next lines are impossible situations to create from tests. Add asserts to verify this.
172      assert Flags.KEEPALIVE != Flags.getScheduleType( flags ) || null != observe;
173      assert Flags.APPLICATION_EXECUTOR != Flags.getScheduleType( flags ) || null == observe;
174      assert !( Observer.Flags.RUN_NOW == ( flags & Observer.Flags.RUN_NOW ) &&
175                Flags.KEEPALIVE != Flags.getScheduleType( flags ) &&
176                null != computableValue );
177      invariant( () -> Flags.isNestedActionsModeValid( flags ),
178                 () -> "Arez-0209: Observer named '" + getName() + "' incorrectly specified both the " +
179                       "NESTED_ACTIONS_ALLOWED flag and the NESTED_ACTIONS_DISALLOWED flag." );
180      invariant( () -> Flags.isScheduleTypeValid( flags ),
181                 () -> "Arez-0210: Observer named '" + getName() + "' incorrectly specified multiple " +
182                       "schedule type flags (KEEPALIVE, DEACTIVATE_ON_UNOBSERVE, APPLICATION_EXECUTOR)." );
183      invariant( () -> ( ~( Flags.RUNTIME_FLAGS_MASK | Flags.CONFIG_FLAGS_MASK ) & flags ) == 0,
184                 () -> "Arez-0207: Observer named '" + getName() + "' specified illegal flags: " +
185                       ( ~( Flags.RUNTIME_FLAGS_MASK | Flags.CONFIG_FLAGS_MASK ) & flags ) );
186    }
187    assert null == computableValue || !Arez.areNamesEnabled() || computableValue.getName().equals( name );
188    _component = Arez.areNativeComponentsEnabled() ? component : null;
189    _computableValue = computableValue;
190    _observe = observe;
191    _onDepsChange = onDepsChange;
192
193    executeObserveNextIfPresent();
194
195    if ( null == _computableValue )
196    {
197      if ( null != _component )
198      {
199        _component.addObserver( this );
200      }
201      else if ( Arez.areRegistriesEnabled() )
202      {
203        getContext().registerObserver( this );
204      }
205    }
206    if ( null == _computableValue )
207    {
208      if ( willPropagateSpyEvents() )
209      {
210        getSpy().reportSpyEvent( new ObserverCreateEvent( asInfo() ) );
211      }
212      if ( null != _observe )
213      {
214        initialSchedule();
215      }
216    }
217  }

222    _task.triggerSchedulerInitiallyUnlessRunLater();
223  }

277  /**
278   * Make the Observer INACTIVE and release any resources associated with observer.
279   * The applications should NOT interact with the Observer after it has been disposed.
280   */
281  @Override
282  public void dispose()
283  {
284    if ( isNotDisposedOrDisposing() )
285    {
286      getContext().safeAction( Arez.areNamesEnabled() ? getName() + ".dispose" : null,
287                               this::performDispose,
288                               ActionFlags.NO_VERIFY_ACTION_REQUIRED );
289      if ( !isComputableValue() )
290      {
291        if ( willPropagateSpyEvents() )
292        {
293          reportSpyEvent( new ObserverDisposeEvent( asInfo() ) );
294        }
295        if ( null != _component )
296        {
297          _component.removeObserver( this );
298        }
299        else if ( Arez.areRegistriesEnabled() )
300        {
301          getContext().deregisterObserver( this );
302        }
303      }
304      if ( null != _computableValue )
305      {
306        _computableValue.dispose();
307      }
308      _task.dispose();
309      markAsDisposed();
310    }
311  }

313  private void performDispose()
314  {
315    getContext().getTransaction().reportDispose( this );
316    markDependenciesLeastStaleObserverAsUpToDate();
317    setState( Flags.STATE_DISPOSING );
318  }

399  /**
400   * This method should be invoked if the observer has non-arez dependencies and one of
401   * these dependencies has been updated. This will mark the observer as stale and reschedule
402   * the reaction if necessary. The method must be invoked from within a read-write transaction.
403   * the reaction if necessary. The method must be invoked from within a read-write transaction.
404   */
405  public void reportStale()
406  {
407    if ( Arez.shouldCheckApiInvariants() )
408    {
409      apiInvariant( this::areExternalDependenciesAllowed,
410                    () -> "Arez-0199: Observer.reportStale() invoked on observer named '" + getName() +
411                          "' but the observer has not specified AREZ_OR_EXTERNAL_DEPENDENCIES flag." );
412      apiInvariant( () -> getContext().isTransactionActive(),
413                    () -> "Arez-0200: Observer.reportStale() invoked on observer named '" + getName() +
414                          "' when there is no active transaction." );
415      apiInvariant( () -> getContext().getTransaction().isMutation(),
416                    () -> "Arez-0201: Observer.reportStale() invoked on observer named '" + getName() +
417                          "' when the active transaction '" + getContext().getTransaction().getName() +
418                          "' is READ_ONLY rather than READ_WRITE." );
419      getContext().getTransaction().markTransactionAsUsed();
420    }
421    setState( Flags.STATE_STALE );
422  }

436  /**
437   * Set the state of the observer.
438   * Call the hook actions for relevant state change.
439   *
440   * @param state    the new state of the observer.
441   * @param schedule true if a state transition can cause observer to reschedule, false otherwise.
442   */
443  void setState( final int state, final boolean schedule )
444  {
445    if ( Arez.shouldCheckInvariants() )
446    {
447      invariant( () -> getContext().isTransactionActive(),
448                 () -> "Arez-0086: Attempt to invoke setState on observer named '" + getName() + "' when there is " +
449                       "no active transaction." );
450      invariantState();
451    }
452    final int originalState = getState();
453    if ( state != originalState )
454    {
455      _flags = Flags.setState( _flags, state );
456      if ( Arez.shouldCheckInvariants() && Flags.STATE_DISPOSED == originalState )
457      {
458        fail( () -> "Arez-0087: Attempted to activate disposed observer named '" + getName() + "'." );
459      }
460      else if ( null == _computableValue && Flags.STATE_STALE == state )
461      {
462        if ( schedule )
463        {
464          scheduleReaction();
465        }
466      }
467      else if ( null != _computableValue &&
468                Flags.STATE_UP_TO_DATE == originalState &&
469                ( Flags.STATE_STALE == state || Flags.STATE_POSSIBLY_STALE == state ) )
470      {
471        _computableValue.getObservableValue().reportPossiblyChanged();
472        runHook( _computableValue.getOnStale(), ObserverError.ON_STALE_ERROR );
473        if ( schedule )
474        {
475          scheduleReaction();
476        }
477      }
478      else if ( Flags.STATE_INACTIVE == state ||
479                ( Flags.STATE_INACTIVE != originalState && Flags.STATE_DISPOSING == state ) )
480      {
481        if ( isComputableValue() )
482        {
483          final ComputableValue<?> computableValue = getComputableValue();
484          runHook( computableValue.getOnDeactivate(), ObserverError.ON_DEACTIVATE_ERROR );
485          computableValue.completeDeactivate();
486        }
487        clearDependencies();
488      }
489      else if ( Flags.STATE_INACTIVE == originalState &&
490                ( ( Flags.STATE_DISPOSED | Flags.STATE_DISPOSING ) & state ) == 0 )
491      {
492        if ( isComputableValue() )
493        {
494          runHook( getComputableValue().getOnActivate(), ObserverError.ON_ACTIVATE_ERROR );
495        }
496      }
497      if ( Arez.shouldCheckInvariants() )
498      {
499        invariantState();
500      }
501    }
502  }

504  /**
505   * Run the supplied hook if non null.
506   *
507   * @param hook the hook to run.
508   */
509  void runHook( @Nullable final Procedure hook, @Nonnull final ObserverError error )
510  {
511    if ( null != hook )
512    {
513      try
514      {
515        hook.call();
516      }
517      catch ( final Throwable t )
518      {
519        getContext().reportObserverError( this, error, t );
520      }
521    }
522  }

524  /**
525   * Remove all dependencies, removing this observer from all dependencies in the process.
526   */
527  void clearDependencies()
528  {
529    getDependencies().forEach( dependency -> {
530      dependency.removeObserver( this );
531      if ( !dependency.hasObservers() )
532      {
533        dependency.setLeastStaleObserverState( Flags.STATE_UP_TO_DATE );
534      }
535    } );
536    getDependencies().clear();
537  }

551  /**
552   * Schedule this observer if it does not already have a reaction pending.
553   * The observer will not actually react if it is not already marked as STALE.
554   */
555  public void schedule()
556  {
557    if ( Arez.shouldCheckApiInvariants() )
558    {
559      apiInvariant( this::supportsManualSchedule,
560                    () -> "Arez-0202: Observer.schedule() invoked on observer named '" + getName() +
561                          "' but supportsManualSchedule() returns false." );
562    }
563    if ( Arez.shouldEnforceTransactionType() && getContext().isTransactionActive() && Arez.shouldCheckInvariants() )
564    {
565      getContext().getTransaction().markTransactionAsUsed();
566    }
567    executeObserveNextIfPresent();
568    scheduleReaction();
569    getContext().triggerScheduler();
570  }

572  /**
573   * Schedule this observer if it does not already have a reaction pending.
574   */
575  void scheduleReaction()
576  {
577    if ( isNotDisposed() )
578    {
579      if ( Arez.shouldCheckInvariants() )
580      {
581        invariant( this::isActive,
582                   () -> "Arez-0088: Observer named '" + getName() + "' is not active but an attempt has been made " +
583                         "to schedule observer." );
584      }
585      if ( !getTask().isQueued() )
586      {
587        getContext().scheduleReaction( this );
588      }
589    }
590  }

592  /**
593   * Run the reaction in a transaction with the name and mode defined
594   * by the observer. If the reaction throws an exception, the exception is reported
595   * to the context global ObserverErrorHandlers
596   */
597  void invokeReaction()
598  {
599    if ( isNotDisposed() )
600    {
601      final long start;
602      if ( willPropagateSpyEvents() )
603      {
604        start = System.currentTimeMillis();
605        if ( isComputableValue() )
606        {
607          reportSpyEvent( new ComputeStartEvent( getComputableValue().asInfo() ) );
608        }
609        else
610        {
611          reportSpyEvent( new ObserveStartEvent( asInfo() ) );
612        }
613      }
614      else
615      {
616        start = 0;
617      }
618      Throwable error = null;
619      try
620      {
621        // ComputableValues may have calculated their values and thus be up to date so no need to recalculate.
622        if ( Flags.STATE_UP_TO_DATE != getState() )
623        {
624          if ( shouldExecuteObserveNext() )
625          {
626            executeOnDepsChangeNextIfPresent();
627            runObserveFunction();
628          }
629          else
630          {
631            assert null != _onDepsChange;
632            _onDepsChange.call();
633          }
634        }
635        else if ( shouldExecuteObserveNext() )
636        {
637          /*
638           * The observer should invoke onDepsChange next if the following conditions hold.
639           *  - a manual schedule() invocation
640           *  - the observer is not stale, and
641           *  - there is an onDepsChange hook present
642           *
643           *  This block ensures this is the case.
644           */
645          executeOnDepsChangeNextIfPresent();
646        }
647      }
648      catch ( final Throwable t )
649      {
650        error = t;
651        getContext().reportObserverError( this, ObserverError.REACTION_ERROR, t );
652      }
653      if ( willPropagateSpyEvents() )
654      {
655        final long duration = System.currentTimeMillis() - start;
656        if ( isComputableValue() )
657        {
658          final ComputableValue<?> computableValue = getComputableValue();
659          reportSpyEvent( new ComputeCompleteEvent( computableValue.asInfo(),
660                                                    noReportResults() ? null : computableValue.getValue(),
661                                                    computableValue.getError(),
662                                                    (int) duration ) );
663        }
664        else
665        {
666          reportSpyEvent( new ObserveCompleteEvent( asInfo(), error, (int) duration ) );
667        }
668      }
669    }
670  }

672  private void runObserveFunction()
673    throws Throwable
674  {
675    assert null != _observe;
676    final Procedure action;
677    if ( Arez.shouldCheckInvariants() && areArezDependenciesRequired() )
678    {
679      action = () -> {
680        _observe.call();
681        final Transaction current = Transaction.current();
682
683        final List<ObservableValue<?>> observableValues = current.getObservableValues();
684        invariant( () -> Objects.requireNonNull( current.getTracker() ).isDisposing() ||
685                         ( null != observableValues && !observableValues.isEmpty() ),
686                   () -> "Arez-0172: Observer named '" + getName() + "' that does not use an external executor " +
687                         "completed observe function but is not observing any properties. As a result the observer " +
688                         "will never be rescheduled." );
689      };
690    }
691    else
692    {
693      action = _observe;
694    }
695    getContext().rawObserve( this, action, null );
696  }

698  /**
699   * Utility to mark all dependencies least stale observer as up to date.
700   * Used when the Observer is determined to be up todate.
701   */
702  void markDependenciesLeastStaleObserverAsUpToDate()
703  {
704    for ( final ObservableValue<?> dependency : getDependencies() )
705    {
706      dependency.setLeastStaleObserverState( Flags.STATE_UP_TO_DATE );
707    }

710  /**
711   * Determine if any dependency of the Observer has actually changed.
712   * If the state is POSSIBLY_STALE then recalculate any ComputableValue dependencies.
713   * If any ComputableValue dependencies actually changed then the STALE state will
714   * be propagated.
715   *
716   * <p>By iterating over the dependencies in the same order that they were reported and
717   * stopping on the first change, all the recalculations are only called for ComputableValues
718   * that will be tracked by derivation. That is because we assume that if the first N
719   * dependencies of the derivation doesn't change then the derivation should run the same way
720   * up until accessing N-th dependency.</p>
721   *
722   * @return true if the Observer should be recomputed.
723   */
724  boolean shouldCompute()
725  {
726    final int state = getState();
727    switch ( state )
728    {
729      case Flags.STATE_UP_TO_DATE:
730        return false;
731      case Flags.STATE_INACTIVE:
732      case Flags.STATE_STALE:
733        return true;
734      case Flags.STATE_POSSIBLY_STALE:
735      {
736        for ( final ObservableValue<?> observableValue : getDependencies() )
737        {
738          if ( observableValue.isComputableValue() )
739          {
740            final Observer owner = observableValue.getObserver();
741            final ComputableValue<?> computableValue = owner.getComputableValue();
742            try
743            {
744              computableValue.get();
745            }
746            catch ( final Throwable ignored )
747            {
748            }
749            // Call to get() will update this state if ComputableValue changed
750            if ( Flags.STATE_STALE == getState() )
751            {
752              return true;
753            }
754          }
755        }
756      }
757      break;
758      default:
759        if ( Arez.shouldCheckInvariants() )
760        {
761          fail( () -> "Arez-0205: Observer.shouldCompute() invoked on observer named '" + getName() +
762                      "' but observer is in state " + Flags.getStateName( getState() ) );
763        }
764    }
765    /*
766     * This results in POSSIBLY_STALE returning to UP_TO_DATE
767     */
768    markDependenciesLeastStaleObserverAsUpToDate();
769    return false;
770  }

785   * This should be the only mechanism via which the dependencies are updated.
786   *
787   * @param dependencies the new set of dependencies.
788   */
789  void replaceDependencies( @Nonnull final List<ObservableValue<?>> dependencies )
790  {
791    if ( Arez.shouldCheckInvariants() )

796    if ( Arez.shouldCheckInvariants() )

823   * Should be optimized away if invariant checking is disabled.
824   *

827  void invariantDependenciesBackLink( @Nonnull final String context )

857  /**
858   * Ensure that state field and other fields of the Observer are consistent.
859   */
860  void invariantState()
861  {
862    if ( Arez.shouldCheckInvariants() )
863    {
864      if ( isInactive() && !isDisposing() )
865      {
866        invariant( () -> getDependencies().isEmpty(),
867                   () -> "Arez-0092: Observer named '" + getName() + "' is inactive but still has dependencies: " +
868                         getDependencies().stream().map( Node::getName ).collect( Collectors.toList() ) + "." );
869      }
870      if ( null != _computableValue && _computableValue.isNotDisposed() )
871      {
872        final ObservableValue<?> observable = _computableValue.getObservableValue();
873        invariant( () -> Objects.equals( observable.isComputableValue() ? observable.getObserver() : null, this ),
874                   () -> "Arez-0093: Observer named '" + getName() + "' is associated with an ObservableValue that " +
875                         "does not link back to observer." );
876      }
877    }
878  }

880  void invariantComputableValueObserverState()
881  {
882    if ( Arez.shouldCheckInvariants() )
883    {
884      if ( isComputableValue() && isActive() && isNotDisposed() )
885      {
886        invariant( () -> !getComputableValue().getObservableValue().getObservers().isEmpty() ||
887                         Objects.equals( getContext().getTransaction().getTracker(), this ),
888                   () -> "Arez-0094: Observer named '" + getName() + "' is a ComputableValue and active but the " +
889                         "associated ObservableValue has no observers." );
890      }
891    }
892  }

980  public static final class Flags
981  {
982    /**
983     * The flag can be passed to actions or observers to force the action to not report result to spy infrastructure.
984     */
985    public static final int NO_REPORT_RESULT = 1 << 12;
986    /**
987     * Highest priority.
988     * This priority should be used when the observer will dispose or release other reactive elements
989     * (and thus remove elements from being scheduled).
990     * <p>Only one of the PRIORITY_* flags should be applied to observer.</p>
991     *
992     * @see arez.annotations.Priority#HIGHEST
993     * @see arez.spy.Priority#HIGHEST
994     * @see Task.Flags#PRIORITY_HIGHEST
995     */
996    public static final int PRIORITY_HIGHEST = 0b001 << 15;
997    /**
998     * High priority.
999     * To reduce the chance that downstream elements will react multiple times within a single
1000     * reaction round, this priority should be used when the observer may trigger many downstream
1001     * reactions.
1002     * <p>Only one of the PRIORITY_* flags should be applied to observer.</p>
1003     *
1004     * @see arez.annotations.Priority#HIGH
1005     * @see arez.spy.Priority#HIGH
1006     * @see Task.Flags#PRIORITY_HIGH
1007     */
1008    public static final int PRIORITY_HIGH = 0b010 << 15;
1009    /**
1010     * Normal priority if no other priority otherwise specified.
1011     * <p>Only one of the PRIORITY_* flags should be applied to observer.</p>
1012     *
1013     * @see arez.annotations.Priority#NORMAL
1014     * @see arez.spy.Priority#NORMAL
1015     * @see Task.Flags#PRIORITY_NORMAL
1016     */
1017    public static final int PRIORITY_NORMAL = 0b011 << 15;
1018    /**
1019     * Low priority.
1020     * Usually used to schedule observers that reflect state onto non-reactive
1021     * application components. i.e. Observers that are used to build html views,
1022     * perform network operations etc. These reactions are often at low priority
1023     * to avoid recalculation of dependencies (i.e. {@link ComputableValue}s) triggering
1024     * this reaction multiple times within a single reaction round.
1025     * <p>Only one of the PRIORITY_* flags should be applied to observer.</p>
1026     *
1027     * @see arez.annotations.Priority#LOW
1028     * @see arez.spy.Priority#LOW
1029     * @see Task.Flags#PRIORITY_LOW
1030     */
1031    public static final int PRIORITY_LOW = 0b100 << 15;
1032    /**
1033     * Lowest priority. Use this priority if the observer is a {@link ComputableValue} that
1034     * may be unobserved when a {@link #PRIORITY_LOW} observer reacts. This is used to avoid
1035     * recomputing state that is likely to either be unobserved or recomputed as part of
1036     * another observers reaction.
1037     * <p>Only one of the PRIORITY_* flags should be applied to observer.</p>
1038     *
1039     * @see arez.annotations.Priority#LOWEST
1040     * @see arez.spy.Priority#LOWEST
1041     * @see Task.Flags#PRIORITY_LOWEST

1043    public static final int PRIORITY_LOWEST = 0b101 << 15;
1044    /**

1046     */
1047    public static final int PRIORITY_MASK = 0b111 << 15;
1048    /**
1049     * The observer can only read arez state.
1050     */
1051    public static final int READ_ONLY = 1 << 24;
1052    /**
1053     * The observer can read or write arez state.
1054     */
1055    public static final int READ_WRITE = 1 << 23;

1059     * is no {@link Observer#_observe} function supplied. This should not be
1060     * specified if {@link #RUN_LATER} is specified.
1061     */

1064    /**
1065     * The scheduler will not be triggered when the observer is created. The observer either
1066     * has no {@link Observer#_observe} function or is responsible for ensuring that
1067     * {@link ArezContext#triggerScheduler()} is invoked at a later time. This should not be
1068     * specified if {@link #RUN_NOW} is specified.
1069     */
1070    public static final int RUN_LATER = 1 << 21;
1071    /**
1072     * Flag indicating that the Observer is allowed to observe {@link ComputableValue} instances with a lower priority.
1073     */
1074    public static final int OBSERVE_LOWER_PRIORITY_DEPENDENCIES = 1 << 30;
1075    /**
1076     * Indicates that the an action can be created from within the Observers observed function.
1077     */
1078    public static final int NESTED_ACTIONS_ALLOWED = 1 << 29;
1079    /**
1080     * Indicates that the an action must not be created from within the Observers observed function.
1081     */
1082    public static final int NESTED_ACTIONS_DISALLOWED = 1 << 28;

1084     * Flag set set if the application code can not invoke the {@link Observer#reportStale()} method.

1088    public static final int AREZ_DEPENDENCIES = 1 << 27;

1105     * observers that supply a observed function.

1119     */
