107  Observer( @Nullable final ArezContext context,
108            @Nullable final Component component,
109            @Nullable final String name,
110            @Nullable final ComputedValue<?> computedValue,
111            @Nullable final TransactionMode mode,
112            @Nonnull final Reaction reaction,
113            @Nonnull final Priority priority,
114            final boolean canTrackExplicitly )
115  {
116    super( context, name );
117    if ( Arez.shouldCheckInvariants() )
118    {
119      if ( Arez.shouldEnforceTransactionType() )
120      {
121        if ( TransactionMode.READ_WRITE_OWNED == mode )
122        {
123          invariant( () -> null != computedValue,
124                     () -> "Arez-0079: Attempted to construct an observer named '" + getName() + "' with " +
125                           "READ_WRITE_OWNED transaction mode but no ComputedValue." );
126          assert null != computedValue;
127          invariant( () -> !canTrackExplicitly,
128                     () -> "Arez-0080: Attempted to construct an ComputedValue '" + getName() + "' that could " +
129                           "track explicitly." );
130        }
131        else if ( TransactionMode.DISPOSE == mode )
132        {
133          fail( () -> "Arez-0173: Attempted to construct an observer named '" + getName() + "' with DISPOSE " +
134                      "transaction mode. Observers must not specify this mode." );
135        }
136        else if ( null != computedValue )
137        {
138          fail( () -> "Arez-0081: Attempted to construct an observer named '" + getName() + "' with " + mode +
139                      " transaction mode and a ComputedValue." );
140        }
141      }
142      else
143      {
144        invariant( () -> null == mode,
145                   () -> "Arez-0082: Observer named '" + getName() + "' specified mode '" + mode + "' when " +
146                         "Arez.enforceTransactionType() is false." );
147        assert null == mode;
148      }
149      invariant( () -> Arez.areNativeComponentsEnabled() || null == component,
150                 () -> "Arez-0083: Observer named '" + getName() + "' has component specified but " +
151                       "Arez.areNativeComponentsEnabled() is false." );
152    }
153    assert null == computedValue || !Arez.areNamesEnabled() || computedValue.getName().equals( name );
154    _component = Arez.areNativeComponentsEnabled() ? component : null;
155    _computedValue = computedValue;
156    _mode = Arez.shouldEnforceTransactionType() ? Objects.requireNonNull( mode ) : null;
157    _reaction = Objects.requireNonNull( reaction );
158    _priority = Objects.requireNonNull( priority );
159    _canTrackExplicitly = canTrackExplicitly;
160    if ( null != _computedValue )
161    {
162      _derivedValue =
163        new Observable<>( context,
164                          null,
165                          name,
166                          this,
167                          Arez.arePropertyIntrospectorsEnabled() ? _computedValue::getValue : null,
168                          null );
169    }
170    else
171    {
172      _derivedValue = null;
173      if ( null != _component )
174      {
175        _component.addObserver( this );
176      }
177      else if ( Arez.areRegistriesEnabled() )
178      {
179        getContext().registerObserver( this );
180      }
181    }
182  }

211  boolean isDerivation()
212  {
213    /*
214     * We do not use "null != _derivedValue" as it is called from constructor of observable
215     * prior to assigning it to _derivedValue. However it is only called if Arez.shouldEnforceTransactionType()
216     * so we can use "null != _derivedValue" when not enabled.
217     */
218    return Arez.shouldEnforceTransactionType() ? TransactionMode.READ_WRITE_OWNED == getMode() : null != _derivedValue;
219  }

221  /**
222   * Make the Observer INACTIVE and release any resources associated with observer.
223   * The applications should NOT interact with the Observer after it has been disposed.
224   */
225  @Override
226  public void dispose()
227  {
228    if ( !_disposed && !_disposing )
229    {
230      _disposing = true;
231      runHook( getOnDispose(), ObserverError.ON_DISPOSE_ERROR );
232      getContext().dispose( Arez.areNamesEnabled() ? getName() : null, this::performDispose );
233      if ( !isDerivation() )
234      {
235        if ( willPropagateSpyEvents() )
236        {
237          reportSpyEvent( new ObserverDisposedEvent( new ObserverInfoImpl( getContext().getSpy(), this ) ) );
238        }
239        if ( null != _component )
240        {
241          _component.removeObserver( this );
242        }
243        else if ( Arez.areRegistriesEnabled() )
244        {
245          getContext().deregisterObserver( this );
246        }
247      }
248      if ( null != _computedValue )
249      {
250        _computedValue.dispose();
251      }
252      if ( null != _derivedValue )
253      {
254        _derivedValue.dispose();
255      }
256      _disposing = false;
257    }
258  }
259
260  private void performDispose()
261  {
262    getContext().getTransaction().reportDispose( this );
263    markDependenciesLeastStaleObserverAsUpToDate();
264    clearDependencies();
265    setDisposed( true );
266    setState( ObserverState.INACTIVE );
267  }

375  /**
376   * Set the state of the observer.
377   * Call the hook actions for relevant state change.
378   *
379   * @param state    the new state of the observer.
380   * @param schedule true if a state transition can cause observer to reschedule, false otherwise.
381   */
382  void setState( @Nonnull final ObserverState state, final boolean schedule )
383  {
384    if ( Arez.shouldCheckInvariants() )
385    {
386      invariant( () -> getContext().isTransactionActive(),
387                 () -> "Arez-0086: Attempt to invoke setState on observer named '" + getName() + "' when there is " +
388                       "no active transaction." );
389      invariantState();
390    }
391    if ( !state.equals( _state ) )
392    {
393      final ObserverState originalState = _state;
394      _state = state;
395      if ( null == _derivedValue && ObserverState.STALE == state )
396      {
397        runHook( getOnStale(), ObserverError.ON_STALE_ERROR );
398        if ( schedule )
399        {
400          schedule();
401        }
402      }
403      else if ( null != _derivedValue &&
404                ObserverState.UP_TO_DATE == originalState &&
405                ( ObserverState.STALE == state || ObserverState.POSSIBLY_STALE == state ) )
406      {
407        // Have to check _derivedValue here as isDerivation() will be true
408        // during construction, prior to _derivedValue being set.
409        _derivedValue.reportPossiblyChanged();
410        runHook( getOnStale(), ObserverError.ON_STALE_ERROR );
411        if ( schedule )
412        {
413          schedule();
414        }
415      }
416      else if ( ObserverState.INACTIVE == _state )
417      {
418        runHook( getOnDeactivate(), ObserverError.ON_DEACTIVATE_ERROR );
419        if ( isDerivation() )
420        {
421          getComputedValue().setValue( null );
422        }
423        clearDependencies();
424      }
425      else if ( ObserverState.INACTIVE == originalState )
426      {
427        if ( Arez.shouldCheckInvariants() )
428        {
429          invariant( this::isLive,
430                     () -> "Arez-0087: Attempted to activate disposed observer named '" + getName() + "'." );
431        }
432        runHook( getOnActivate(), ObserverError.ON_ACTIVATE_ERROR );
433      }
434      if ( Arez.shouldCheckInvariants() )
435      {
436        invariantState();
437      }
438    }
439  }

548  void clearDependencies()
549  {
550    getDependencies().forEach( dependency -> {
551      dependency.removeObserver( this );
552      if ( !dependency.hasObservers() )

586  /**
587   * Schedule this observer if it does not already have a reaction pending.
588   */
589  void schedule()
590  {
591    if ( isLive() )
592    {
593      if ( Arez.shouldCheckInvariants() )
594      {
595        invariant( this::isActive,
596                   () -> "Arez-0088: Observer named '" + getName() + "' is not active but an attempt has been made " +
597                         "to schedule observer." );
598      }
599      if ( !isScheduled() )
600      {
601        getContext().scheduleReaction( this );
602      }
603    }
604  }

606  /**
607   * Run the reaction in a transaction with the name and mode defined
608   * by the observer. If the reaction throws an exception, the exception is reported
609   * to the context global ObserverErrorHandlers
610   */
611  void invokeReaction()
612  {
613    if ( isLive() )
614    {
615      final long start;
616      if ( willPropagateSpyEvents() )
617      {
618        start = System.currentTimeMillis();
619        if ( isDerivation() )
620        {
621          reportSpyEvent( new ComputeStartedEvent( new ComputedValueInfoImpl( getSpy(), getComputedValue() ) ) );
622        }
623        else
624        {
625          reportSpyEvent( new ReactionStartedEvent( new ObserverInfoImpl( getSpy(), this ) ) );
626        }
627      }
628      else
629      {
630        start = 0;
631      }
632      try
633      {
634        // ComputedValues may have calculated their values and thus be up to date so no need to recalculate.
635        if ( ObserverState.UP_TO_DATE != getState() )
636        {
637          getReaction().react( this );
638        }
639      }
640      catch ( final Throwable t )
641      {
642        getContext().reportObserverError( this, ObserverError.REACTION_ERROR, t );
643      }
644      if ( willPropagateSpyEvents() )
645      {
646        final long duration = System.currentTimeMillis() - start;
647        if ( isDerivation() )
648        {
649          reportSpyEvent( new ComputeCompletedEvent( new ComputedValueInfoImpl( getSpy(), getComputedValue() ),
650                                                     duration ) );
651        }
652        else
653        {
654          reportSpyEvent( new ReactionCompletedEvent( new ObserverInfoImpl( getSpy(), this ), duration ) );
655        }
656      }
657    }
658  }

660  /**
661   * Utility to mark all dependencies least stale observer as up to date.
662   * Used when the Observer is determined to be up todate.
663   */
664  void markDependenciesLeastStaleObserverAsUpToDate()
665  {
666    for ( final Observable dependency : getDependencies() )
667    {
668      dependency.setLeastStaleObserverState( ObserverState.UP_TO_DATE );
669    }
670  }

672  /**
673   * Determine if any dependency of the Observer has actually changed.
674   * If the state is POSSIBLY_STALE then recalculate any ComputedValue dependencies.
675   * If any ComputedValue dependencies actually changed then the STALE state will
676   * be propagated.
677   *
678   * <p>By iterating over the dependencies in the same order that they were reported and
679   * stopping on the first change, all the recalculations are only called for ComputedValues
680   * that will be tracked by derivation. That is because we assume that if the first N
681   * dependencies of the derivation doesn't change then the derivation should run the same way
682   * up until accessing N-th dependency.</p>
683   *
684   * @return true if the Observer should be recomputed.
685   */
686  boolean shouldCompute()
687  {
688    switch ( getState() )
689    {
690      case UP_TO_DATE:
691        return false;
692      case INACTIVE:
693      case STALE:
694        return true;
695      case POSSIBLY_STALE:
696      {
697        for ( final Observable observable : getDependencies() )
698        {
699          if ( observable.hasOwner() )
700          {
701            final Observer owner = observable.getOwner();
702            final ComputedValue computedValue = owner.getComputedValue();
703            try
704            {
705              computedValue.get();
706            }
707            catch ( final Throwable ignored )
708            {
709            }
710            // Call to get() will update this state if ComputedValue changed
711            if ( ObserverState.STALE == getState() )
712            {
713              return true;
714            }
715          }
716        }
717      }
718    }
719    /*
720     * This results in POSSIBLY_STALE returning to UP_TO_DATE
721     */
722    markDependenciesLeastStaleObserverAsUpToDate();
723    return false;
724  }

737  /**
738   * Replace the current set of dependencies with supplied dependencies.
739   * This should be the only mechanism via which the dependencies are updated.
740   *
741   * @param dependencies the new set of dependencies.
742   */
743  void replaceDependencies( @Nonnull final ArrayList<Observable<?>> dependencies )
744  {
745    if ( Arez.shouldCheckInvariants() )
746    {
747      invariantDependenciesUnique( "Pre replaceDependencies" );
748    }
749    _dependencies = Objects.requireNonNull( dependencies );
750    if ( Arez.shouldCheckInvariants() )
751    {
752      invariantDependenciesUnique( "Post replaceDependencies" );
753      invariantDependenciesBackLink( "Post replaceDependencies" );
754      invariantDependenciesNotDisposed();
755    }
756  }

758  /**
759   * Ensure the dependencies list contain no duplicates.
760   * Should be optimized away if invariant checking is disabled.
761   *
762   * @param context some useful debugging context used in invariant checks.
763   */
764  void invariantDependenciesUnique( @Nonnull final String context )
765  {
766    if ( Arez.shouldCheckInvariants() )
767    {
768      invariant( () -> getDependencies().size() == new HashSet<>( getDependencies() ).size(),
769                 () -> "Arez-0089: " + context + ": The set of dependencies in observer named '" +
770                       getName() + "' is not unique. Current list: '" +
771                       getDependencies().stream().map( Node::getName ).collect( Collectors.toList() ) + "'." );
772    }
773  }

775  /**
776   * Ensure all dependencies contain this observer in the list of observers.
777   * Should be optimized away if invariant checking is disabled.
778   *
779   * @param context some useful debugging context used in invariant checks.
780   */
781  void invariantDependenciesBackLink( @Nonnull final String context )
782  {
783    if ( Arez.shouldCheckInvariants() )
784    {
785      getDependencies().forEach( observable ->
786                                   invariant( () -> observable.getObservers().contains( this ),

788                                                    "' has dependency observable named '" + observable.getName() +
789                                                    "' which does not contain the observer in the list of " +
790                                                    "observers." ) );
791      invariantDerivationState();
792    }
793  }

811  /**
812   * Ensure that state field and other fields of the Observer are consistent.
813   */
814  void invariantState()
815  {
816    if ( Arez.shouldCheckInvariants() )
817    {
818      if ( isInactive() )
819      {
820        invariant( () -> getDependencies().isEmpty(),
821                   () -> "Arez-0092: Observer named '" + getName() + "' is inactive but still has dependencies: " +
822                         getDependencies().stream().map( Node::getName ).collect( Collectors.toList() ) + "." );
823      }
824      if ( isDerivation() && isLive() )
825      {
826        invariant( () -> Objects.equals( getDerivedValue().hasOwner() ? getDerivedValue().getOwner() : null, this ),
827                   () -> "Arez-0093: Observer named '" + getName() + "' has a derived value that does not " +
828                         "link back to observer." );
829      }
830    }
831  }
832
833  void invariantDerivationState()
834  {
835    if ( Arez.shouldCheckInvariants() )
836    {
837      if ( isDerivation() && isActive() && !isDisposed() )
838      {
839        invariant( () -> !getDerivedValue().getObservers().isEmpty() ||
840                         Objects.equals( getContext().getTransaction().getTracker(), this ),
841                   () -> "Arez-0094: Observer named '" + getName() + "' is a derivation and active but the " +
842                         "derived value has no observers." );
843      }
844    }
845  }

847  /**
848   * Return the ComputedValue for Observer.
849   * This should not be called if observer is not a derivation and will generate an invariant failure
850   * if invariants are enabled.
851   *
852   * @return the associated ComputedValue.
853   */
854  @Nonnull
855  ComputedValue<?> getComputedValue()
856  {
857    if ( Arez.shouldCheckInvariants() )
858    {
859      invariant( this::isDerivation,
860                 () -> "Arez-0095: Attempted to invoke getComputedValue on observer named '" + getName() + "' when " +
861                       "is not a computed observer." );
862    }
863    assert null != _computedValue;
864    return _computedValue;
865  }
