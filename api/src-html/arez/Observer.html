019/**
020 * A node within Arez that is notified of changes in 0 or more Observables.
021 */
022public final class Observer
023  extends Node
024{
025  /**
026   * The component that this observer is contained within.
027   * This should only be set if {@link Arez#areNativeComponentsEnabled()} is true but may also be null if
028   * the observer is a "top-level" observer.
029   */
030  @Nullable
031  private final Component _component;
032  /**
033   * The reference to the ComputedValue if this observer is a derivation.
034   */
035  @Nullable
036  private final ComputedValue<?> _computedValue;
037  /**
038   * The observables that this observer receives notifications from.
039   * These are the dependencies within the dependency graph and will
040   * typically correspond to the observables that were accessed in last
041   * transaction that this observer was tracking.
042   *
043   * This list should contain no duplicates.
044   */
045  @Nonnull
046  private ArrayList<ObservableValue<?>> _dependencies = new ArrayList<>();
047  /**
048   * Observed function to invoke if any.
049   * This may be null if external executor is responsible for executing the observed function via
050   * methods such as {@link ArezContext#observe(Observer, Function, Object...)}. If this is null then
051   * {@link #_onDepsChanged} must not be null.
052   */
053  @Nullable
054  private final Procedure _observed;
055  /**
056   * Callback invoked when dependencies are updated.
057   * This may be null when the observer re-executes the observed function when dependencies change
058   * but in that case {@link #_observed} must not be null.
059   */
060  @Nullable
061  private final Procedure _onDepsChanged;
062  /**
063   * Cached info object associated with element.
064   * This should be null if {@link Arez#areSpiesEnabled()} is false;
065   */
066  @Nullable
067  private ObserverInfo _info;
068  /**
069   * A bitfield that contains config time and runtime flags/state.
070   * See the values in {@link Flags} that are covered by the masks
071   * {@link Flags#RUNTIME_FLAGS_MASK} and {@link Flags#CONFIG_FLAGS_MASK}
072   * for acceptable values.
073   */
074  private int _flags;

076  Observer( @Nonnull final ComputedValue<?> computedValue, final int flags )
077  {
078    this( Arez.areZonesEnabled() ? computedValue.getContext() : null,
079          null,
080          Arez.areNamesEnabled() ? computedValue.getName() : null,
081          computedValue,
082          computedValue::compute,
083          null,
084          flags |
085          ( Flags.KEEPALIVE == Flags.getScheduleType( flags ) ? 0 : Flags.DEACTIVATE_ON_UNOBSERVE ) |
086          Flags.runType( flags, Flags.KEEPALIVE == Flags.getScheduleType( flags ) ? Flags.RUN_NOW : Flags.RUN_LATER ) |
087          Flags.environmentFlag( flags ) |
088          ( Arez.shouldEnforceTransactionType() ? Flags.READ_ONLY : 0 ) |
089          Flags.NESTED_ACTIONS_DISALLOWED |
090          Flags.dependencyType( flags ) |
091          Flags.priority( flags ) );
092  }

094  Observer( @Nullable final ArezContext context,
095            @Nullable final Component component,
096            @Nullable final String name,
097            @Nullable final Procedure observed,
098            @Nullable final Procedure onDepsChanged,
099            final int flags )
100  {
101    this( context,
102          component,
103          name,
104          null,
105          observed,
106          onDepsChanged,
107          flags |
108          ( null == observed ? Flags.APPLICATION_EXECUTOR : Flags.KEEPALIVE ) |
109          ( null != observed ? Flags.ENVIRONMENT_REQUIRED : Flags.environmentFlag( flags ) ) |
110          Flags.runType( flags, null == observed ? Flags.RUN_LATER : Flags.RUN_NOW ) |
111          Flags.priority( flags ) |
112          Flags.nestedActionRule( flags ) |
113          Flags.dependencyType( flags ) |
114          Flags.transactionMode( flags ) );
115  }

117  private Observer( @Nullable final ArezContext context,
118                    @Nullable final Component component,
119                    @Nullable final String name,
120                    @Nullable final ComputedValue<?> computedValue,
121                    @Nullable final Procedure observed,
122                    @Nullable final Procedure onDepsChanged,
123                    final int flags )
124  {
125    super( context, name );
126    _flags = flags | Flags.STATE_INACTIVE;
127    if ( Arez.shouldCheckInvariants() )
128    {
129      if ( Arez.shouldEnforceTransactionType() )
130      {
131        invariant( () -> Flags.isTransactionModeValid( flags ),
132                   () -> "Arez-0079: Observer named '" + getName() + "' incorrectly specified both READ_ONLY " +
133                         "and READ_WRITE transaction mode flags." );
134      }
135      else
136      {
137        invariant( () -> !Flags.isTransactionModeSpecified( flags ),
138                   () -> "Arez-0082: Observer named '" + getName() + "' specified transaction mode '" +
139                         Flags.getTransactionModeName( flags ) + "' when Arez.enforceTransactionType() is false." );
140      }
141      invariant( () -> Flags.isPriorityValid( flags ),
142                 () -> "Arez-0080: Observer named '" + getName() + "' has invalid priority " +
143                       Flags.getPriorityIndex( flags ) + "." );
144      invariant( () -> Flags.isRunTypeValid( flags ),
145                 () -> "Arez-0081: Observer named '" + getName() + "' incorrectly specified both " +
146                       "RUN_NOW and RUN_LATER flags." );
147      invariant( () -> Flags.isEnvironmentValid( flags ),
148                 () -> "Arez-0121: Observer named '" + getName() + "' incorrectly specified both " +
149                       "the ENVIRONMENT_REQUIRED flag and the ENVIRONMENT_NOT_REQUIRED flag." );
150      invariant( () -> 0 == ( flags & Flags.RUN_NOW ) || null != observed,
151                 () -> "Arez-0206: Observer named '" + getName() + "' incorrectly specified " +
152                       "RUN_NOW flag but the observed function is null." );
153      invariant( () -> Arez.areNativeComponentsEnabled() || null == component,
154                 () -> "Arez-0083: Observer named '" + getName() + "' has component specified but " +
155                       "Arez.areNativeComponentsEnabled() is false." );
156      invariant( () -> Flags.getPriority( flags ) != Flags.PRIORITY_LOWEST ||
157                       0 == ( flags & Flags.OBSERVE_LOWER_PRIORITY_DEPENDENCIES ),
158                 () -> "Arez-0184: Observer named '" + getName() + "' has LOWEST priority but has passed " +
159                       "OBSERVE_LOWER_PRIORITY_DEPENDENCIES option which should not be present as the observer " +
160                       "has no lower priority." );
161      invariant( () -> null != observed || null != onDepsChanged,
162                 () -> "Arez-0204: Observer named '" + getName() + "' has not supplied a value for either the " +
163                       "observed parameter or the onDepsChanged parameter." );
164      // Next lines are impossible situations to create from tests. Add asserts to verify this.
165      assert Flags.KEEPALIVE != Flags.getScheduleType( flags ) || null != observed;
166      assert Flags.APPLICATION_EXECUTOR != Flags.getScheduleType( flags ) || null == observed;
167      invariant( () -> !( Flags.RUN_NOW == ( flags & Flags.RUN_NOW ) &&
168                          Flags.KEEPALIVE != Flags.getScheduleType( flags ) &&
169                          null != computedValue ),
170                 () -> "Arez-0208: ComputedValue named '" + getName() + "' incorrectly specified " +
171                       "RUN_NOW flag but not the KEEPALIVE flag." );
172      invariant( () -> Flags.isNestedActionsModeValid( flags ),
173                 () -> "Arez-0209: Observer named '" + getName() + "' incorrectly specified both the " +
174                       "NESTED_ACTIONS_ALLOWED flag and the NESTED_ACTIONS_DISALLOWED flag." );
175      invariant( () -> Flags.isScheduleTypeValid( flags ),
176                 () -> "Arez-0210: Observer named '" + getName() + "' incorrectly specified multiple " +
177                       "schedule type flags (KEEPALIVE, DEACTIVATE_ON_UNOBSERVE, APPLICATION_EXECUTOR)." );
178      invariant( () -> ( ~( Flags.RUNTIME_FLAGS_MASK | Flags.CONFIG_FLAGS_MASK ) & flags ) == 0,
179                 () -> "Arez-0207: Observer named '" + getName() + "' specified illegal flags: " +
180                       ( ~( Flags.RUNTIME_FLAGS_MASK | Flags.CONFIG_FLAGS_MASK ) & flags ) );
181    }
182    assert null == computedValue || !Arez.areNamesEnabled() || computedValue.getName().equals( name );
183    _component = Arez.areNativeComponentsEnabled() ? component : null;
184    _computedValue = computedValue;
185    _observed = observed;
186    _onDepsChanged = onDepsChanged;
187
188    executeObservedNextIfPresent();
189
190    if ( null == _computedValue )
191    {
192      if ( null != _component )
193      {
194        _component.addObserver( this );
195      }
196      else if ( Arez.areRegistriesEnabled() )
197      {
198        getContext().registerObserver( this );
199      }
200    }
201    if ( null == _computedValue )
202    {
203      if ( willPropagateSpyEvents() )
204      {
205        getSpy().reportSpyEvent( new ObserverCreatedEvent( asInfo() ) );
206      }
207      if ( null != _observed )
208      {
209        initialSchedule();
210      }
211    }
212  }

214  void initialSchedule()
215  {
216    final ArezContext context = getContext();
217    context.scheduleReaction( this );
218    if ( 0 != ( _flags & Flags.RUN_NOW ) )
219    {
220      context.triggerScheduler();
221    }

292  /**
293   * Make the Observer INACTIVE and release any resources associated with observer.
294   * The applications should NOT interact with the Observer after it has been disposed.
295   */
296  @Override
297  public void dispose()
298  {
299    if ( !isDisposedOrDisposing() )
300    {
301      getContext().safeAction( Arez.areNamesEnabled() ? getName() + ".dispose" : null,
302                               this::performDispose,
303                               Flags.NO_VERIFY_ACTION_REQUIRED );
304      if ( !isComputedValue() )
305      {
306        if ( willPropagateSpyEvents() )
307        {
308          reportSpyEvent( new ObserverDisposedEvent( asInfo() ) );
309        }
310        if ( null != _component )
311        {
312          _component.removeObserver( this );
313        }
314        else if ( Arez.areRegistriesEnabled() )
315        {
316          getContext().deregisterObserver( this );
317        }
318      }
319      if ( null != _computedValue )
320      {
321        _computedValue.dispose();
322      }
323      markAsDisposed();
324    }
325  }

416  /**
417   * This method should be invoked if the observer has non-arez dependencies and one of
418   * these dependencies has been updated. This will mark the observer as stale and reschedule
419   * the reaction if necessary. The method must be invoked from within a read-write transaction.
420   * the reaction if necessary. The method must be invoked from within a read-write transaction.

422  public void reportStale()
423  {
424    if ( Arez.shouldCheckApiInvariants() )
425    {
426      apiInvariant( this::areExternalDependenciesAllowed,
427                    () -> "Arez-0199: Observer.reportStale() invoked on observer named '" + getName() +
428                          "' but the observer has not specified AREZ_OR_EXTERNAL_DEPENDENCIES flag." );
429      apiInvariant( () -> getContext().isTransactionActive(),
430                    () -> "Arez-0200: Observer.reportStale() invoked on observer named '" + getName() +
431                          "' when there is no active transaction." );
432      apiInvariant( () -> getContext().getTransaction().isMutation(),
433                    () -> "Arez-0201: Observer.reportStale() invoked on observer named '" + getName() +
434                          "' when the active transaction '" + getContext().getTransaction().getName() +
435                          "' is READ_ONLY rather than READ_WRITE." );
436      getContext().getTransaction().markTransactionAsUsed();
437    }
438    setState( Flags.STATE_STALE );
439  }

453  /**
454   * Set the state of the observer.
455   * Call the hook actions for relevant state change.
456   *
457   * @param state    the new state of the observer.
458   * @param schedule true if a state transition can cause observer to reschedule, false otherwise.
459   */
460  void setState( final int state, final boolean schedule )
461  {
462    if ( Arez.shouldCheckInvariants() )
463    {
464      invariant( () -> getContext().isTransactionActive(),
465                 () -> "Arez-0086: Attempt to invoke setState on observer named '" + getName() + "' when there is " +
466                       "no active transaction." );
467      invariantState();
468    }
469    final int originalState = getState();
470    if ( state != originalState )
471    {
472      _flags = Flags.setState( _flags, state );
473      if ( Arez.shouldCheckInvariants() && Flags.STATE_DISPOSED == originalState )
474      {
475        fail( () -> "Arez-0087: Attempted to activate disposed observer named '" + getName() + "'." );
476      }
477      else if ( null == _computedValue && Flags.STATE_STALE == state )
478      {
479        if ( schedule )
480        {
481          scheduleReaction();
482        }
483      }
484      else if ( null != _computedValue &&
485                Flags.STATE_UP_TO_DATE == originalState &&
486                ( Flags.STATE_STALE == state || Flags.STATE_POSSIBLY_STALE == state ) )
487      {
488        _computedValue.getObservableValue().reportPossiblyChanged();
489        runHook( _computedValue.getOnStale(), ObserverError.ON_STALE_ERROR );
490        if ( schedule )
491        {
492          scheduleReaction();
493        }
494      }
495      else if ( Flags.STATE_INACTIVE == state ||
496                ( Flags.STATE_INACTIVE != originalState && Flags.STATE_DISPOSING == state ) )
497      {
498        if ( isComputedValue() )
499        {
500          final ComputedValue<?> computedValue = getComputedValue();
501          runHook( computedValue.getOnDeactivate(), ObserverError.ON_DEACTIVATE_ERROR );
502          computedValue.setValue( null );
503        }
504        clearDependencies();
505      }
506      else if ( Flags.STATE_INACTIVE == originalState )
507      {
508        if ( isComputedValue() )
509        {
510          runHook( getComputedValue().getOnActivate(), ObserverError.ON_ACTIVATE_ERROR );
511        }
512      }
513      if ( Arez.shouldCheckInvariants() )
514      {
515        invariantState();
516      }
517    }
518  }

522   *
523   * @param hook the hook to run.
524   */

529      try

540  /**
541   * Remove all dependencies, removing this observer from all dependencies in the process.
542   */
543  void clearDependencies()
544  {
545    getDependencies().forEach( dependency -> {
546      dependency.removeObserver( this );
547      if ( !dependency.hasObservers() )
548      {
549        dependency.setLeastStaleObserverState( Flags.STATE_UP_TO_DATE );
550      }
551    } );
552    getDependencies().clear();
553  }

581  /**
582   * Schedule this observer if it does not already have a reaction pending.
583   * The observer will not actually react if it is not already marked as STALE.
584   */
585  public void schedule()
586  {
587    if ( Arez.shouldCheckApiInvariants() )
588    {
589      apiInvariant( this::supportsManualSchedule,
590                    () -> "Arez-0202: Observer.schedule() invoked on observer named '" + getName() +
591                          "' but supportsManualSchedule() returns false." );
592    }
593    if ( Arez.shouldEnforceTransactionType() && getContext().isTransactionActive() && Arez.shouldCheckInvariants() )
594    {
595      getContext().getTransaction().markTransactionAsUsed();
596    }
597    executeObservedNextIfPresent();
598    scheduleReaction();
599    getContext().triggerScheduler();
600  }

602  /**
603   * Schedule this observer if it does not already have a reaction pending.
604   */
605  void scheduleReaction()
606  {
607    if ( isNotDisposed() )
608    {
609      if ( Arez.shouldCheckInvariants() )
610      {
611        invariant( this::isActive,
612                   () -> "Arez-0088: Observer named '" + getName() + "' is not active but an attempt has been made " +
613                         "to schedule observer." );
614      }
615      if ( !isScheduled() )
616      {
617        getContext().scheduleReaction( this );
618      }
619    }
620  }

622  /**
623   * Run the reaction in a transaction with the name and mode defined
624   * by the observer. If the reaction throws an exception, the exception is reported
625   * to the context global ObserverErrorHandlers
626   */
627  void invokeReaction()
628  {
629    if ( isNotDisposed() )
630    {
631      final long start;
632      if ( willPropagateSpyEvents() )
633      {
634        start = System.currentTimeMillis();
635        if ( isComputedValue() )
636        {
637          reportSpyEvent( new ComputeStartedEvent( getComputedValue().asInfo() ) );
638        }
639        else
640        {
641          reportSpyEvent( new ObserveStartedEvent( asInfo() ) );
642        }
643      }
644      else
645      {
646        start = 0;
647      }
648      try
649      {
650        // ComputedValues may have calculated their values and thus be up to date so no need to recalculate.
651        if ( Flags.STATE_UP_TO_DATE != getState() )
652        {
653          if ( shouldExecuteObservedNext() )
654          {
655            executeOnDepsChangedNextIfPresent();
656            runObservedFunction();
657          }
658          else
659          {
660            assert null != _onDepsChanged;
661            _onDepsChanged.call();
662          }
663        }
664        else if ( shouldExecuteObservedNext() )
665        {
666          /*
667           * The observer should invoke onDepsChanged next if the following conditions hold.
668           *  - a manual schedule() invocation
669           *  - the observer is not stale, and
670           *  - there is an onDepsChanged hook present
671           *
672           *  This block ensures this is the case.
673           */
674          executeOnDepsChangedNextIfPresent();
675        }
676      }
677      catch ( final Throwable t )
678      {
679        getContext().reportObserverError( this, ObserverError.REACTION_ERROR, t );
680      }
681      if ( willPropagateSpyEvents() )
682      {
683        final long duration = System.currentTimeMillis() - start;
684        if ( isComputedValue() )
685        {
686          final ComputedValue<?> computedValue = getComputedValue();
687          reportSpyEvent( new ComputeCompletedEvent( computedValue.asInfo(),
688                                                     computedValue.getValue(),
689                                                     computedValue.getError(),
690                                                     (int) duration ) );
691        }
692        else
693        {
694          reportSpyEvent( new ObserveCompletedEvent( asInfo(), (int) duration ) );
695        }
696      }
697    }
698  }

700  private void runObservedFunction()
701    throws Throwable
702  {
703    assert null != _observed;
704    final Procedure action;
705    if ( Arez.shouldCheckInvariants() && areArezDependenciesRequired() )
706    {
707      action = () -> {
708        _observed.call();
709        final Transaction current = Transaction.current();
710
711        final ArrayList<ObservableValue<?>> observableValues = current.getObservableValues();
712        invariant( () -> Objects.requireNonNull( current.getTracker() ).isDisposing() ||
713                         ( null != observableValues && !observableValues.isEmpty() ),
714                   () -> "Arez-0172: Observer named '" + getName() + "' that does not use an external executor " +
715                         "completed observed function but is not observing any properties. As a result the observer " +
716                         "will never be rescheduled." );
717      };
718    }
719    else
720    {
721      action = _observed;
722    }
723    getContext().rawObserve( this, action, null );
724  }

726  /**
727   * Utility to mark all dependencies least stale observer as up to date.
728   * Used when the Observer is determined to be up todate.
729   */
730  void markDependenciesLeastStaleObserverAsUpToDate()
731  {
732    for ( final ObservableValue dependency : getDependencies() )
733    {
734      dependency.setLeastStaleObserverState( Flags.STATE_UP_TO_DATE );
735    }
736  }

738  /**
739   * Determine if any dependency of the Observer has actually changed.
740   * If the state is POSSIBLY_STALE then recalculate any ComputedValue dependencies.
741   * If any ComputedValue dependencies actually changed then the STALE state will
742   * be propagated.
743   *
744   * <p>By iterating over the dependencies in the same order that they were reported and
745   * stopping on the first change, all the recalculations are only called for ComputedValues
746   * that will be tracked by derivation. That is because we assume that if the first N
747   * dependencies of the derivation doesn't change then the derivation should run the same way
748   * up until accessing N-th dependency.</p>
749   *
750   * @return true if the Observer should be recomputed.
751   */
752  boolean shouldCompute()
753  {
754    final int state = getState();
755    switch ( state )
756    {
757      case Flags.STATE_UP_TO_DATE:
758        return false;
759      case Flags.STATE_INACTIVE:
760      case Flags.STATE_STALE:
761        return true;
762      case Flags.STATE_POSSIBLY_STALE:
763      {
764        for ( final ObservableValue observableValue : getDependencies() )
765        {
766          if ( observableValue.isComputedValue() )
767          {
768            final Observer owner = observableValue.getObserver();
769            final ComputedValue computedValue = owner.getComputedValue();
770            try
771            {
772              computedValue.get();
773            }
774            catch ( final Throwable ignored )
775            {
776            }
777            // Call to get() will update this state if ComputedValue changed
778            if ( Flags.STATE_STALE == getState() )
779            {
780              return true;
781            }
782          }
783        }
784      }
785      break;
786      default:
787        if ( Arez.shouldCheckInvariants() )
788        {
789          fail( () -> "Arez-0205: Observer.shouldCompute() invoked on observer named '" + getName() +
790                      "' but observer is in state " + Flags.getStateName( getState() ) );
791        }
792    }
793    /*
794     * This results in POSSIBLY_STALE returning to UP_TO_DATE
795     */
796    markDependenciesLeastStaleObserverAsUpToDate();
797    return false;
798  }

811  /**
812   * Replace the current set of dependencies with supplied dependencies.
813   * This should be the only mechanism via which the dependencies are updated.
814   *
815   * @param dependencies the new set of dependencies.
816   */
817  void replaceDependencies( @Nonnull final ArrayList<ObservableValue<?>> dependencies )
818  {
819    if ( Arez.shouldCheckInvariants() )
820    {
821      invariantDependenciesUnique( "Pre replaceDependencies" );
822    }
823    _dependencies = Objects.requireNonNull( dependencies );
824    if ( Arez.shouldCheckInvariants() )
825    {
826      invariantDependenciesUnique( "Post replaceDependencies" );
827      invariantDependenciesBackLink( "Post replaceDependencies" );
828      invariantDependenciesNotDisposed();
829    }
830  }

833   * Ensure the dependencies list contain no duplicates.
834   * Should be optimized away if invariant checking is disabled.

844                       getName() + "' is not unique. Current list: '" +
845                       getDependencies().stream().map( Node::getName ).collect( Collectors.toList() ) + "'." );

849  /**
850   * Ensure all dependencies contain this observer in the list of observers.
851   * Should be optimized away if invariant checking is disabled.
852   *
853   * @param context some useful debugging context used in invariant checks.
854   */
855  void invariantDependenciesBackLink( @Nonnull final String context )
856  {
857    if ( Arez.shouldCheckInvariants() )
858    {
859      getDependencies().forEach( observable ->
860                                   invariant( () -> observable.getObservers().contains( this ),
861                                              () -> "Arez-0090: " + context + ": Observer named '" + getName() +
862                                                    "' has ObservableValue dependency named '" + observable.getName() +
863                                                    "' which does not contain the observer in the list of " +
864                                                    "observers." ) );
865      invariantComputedValueObserverState();
866    }

870   * Ensure all dependencies are not disposed.
871   */
872  void invariantDependenciesNotDisposed()
873  {
874    if ( Arez.shouldCheckInvariants() )
875    {
876      getDependencies().forEach( observable ->
877                                   invariant( observable::isNotDisposed,
878                                              () -> "Arez-0091: Observer named '" + getName() + "' has " +
879                                                    "ObservableValue dependency named '" + observable.getName() +
880                                                    "' which is disposed." ) );
881      invariantComputedValueObserverState();
882    }

885  /**
886   * Ensure that state field and other fields of the Observer are consistent.
887   */
888  void invariantState()
889  {
890    if ( Arez.shouldCheckInvariants() )
891    {
892      if ( isInactive() && !isDisposing() )
893      {
894        invariant( () -> getDependencies().isEmpty(),
895                   () -> "Arez-0092: Observer named '" + getName() + "' is inactive but still has dependencies: " +
896                         getDependencies().stream().map( Node::getName ).collect( Collectors.toList() ) + "." );
897      }
898      if ( null != _computedValue && _computedValue.isNotDisposed() )
899      {
900        final ObservableValue<?> observable = _computedValue.getObservableValue();
901        invariant( () -> Objects.equals( observable.isComputedValue() ? observable.getObserver() : null, this ),
902                   () -> "Arez-0093: Observer named '" + getName() + "' is associated with an ObservableValue that " +
903                         "does not link back to observer." );
904      }
905    }
906  }
907
908  void invariantComputedValueObserverState()
909  {
910    if ( Arez.shouldCheckInvariants() )
911    {
912      if ( isComputedValue() && isActive() && isNotDisposed() )
913      {
914        invariant( () -> !getComputedValue().getObservableValue().getObservers().isEmpty() ||
915                         Objects.equals( getContext().getTransaction().getTracker(), this ),
916                   () -> "Arez-0094: Observer named '" + getName() + "' is a ComputedValue and active but the " +
917                         "associated ObservableValue has no observers." );
918      }
919    }
920  }
