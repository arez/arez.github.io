001package arez.browser.extras;
002
003import arez.annotations.Action;
004import arez.annotations.ArezComponent;
005import arez.annotations.Observable;
006import elemental2.promise.IThenable;
007import elemental2.promise.Promise;
008import java.util.Objects;
009import javax.annotation.Nonnull;
010import jsinterop.base.Js;
011import org.realityforge.anodoc.Unsupported;
012import static org.realityforge.braincheck.Guards.*;
013
014/**
015 * An observable model that wraps a Promise and exposes observable state that track
016 * the state of the promise. The observable exposes the state of the promise as well
017 * as the value that it resolves to or the error it was rejected with as observable
018 * properties.
019 *
020 * <h1>A very simple example</h1>
021 * <pre>{@code
022 * import com.google.gwt.core.client.EntryPoint;
023 * import elemental2.dom.DomGlobal;
024 * import elemental2.dom.Response;
025 * import elemental2.promise.Promise;
026 * import arez.Arez;
027 * import arez.browser.extras.ObservablePromise;
028 *
029 * public class Example
030 *   implements EntryPoint
031 * {
032 *   public void onModuleLoad()
033 *   {
034 *     final Promise<Response> promise = DomGlobal.fetch( "https://example.com/" );
035 *     final ObservablePromise<Response, Object> observablePromise = ObservablePromise.create( promise );
036 *     Arez.context().autorun( () -> {
037 *       final String message = "Promise Status: " + observablePromise.getState();
038 *       DomGlobal.console.log( message );
039 *     } );
040 *   }
041 * }
042 * }</pre>
043 *
044 * @param <T> the type of the value that the promise will resolve to.
045 * @param <E> the type of the error if the promise is rejected.
046 */
047@Unsupported( "This is still considered experimental and will likely evolve over time" )
048@ArezComponent
049public abstract class ObservablePromise<T, E>
050{
051  /**
052   * The state of the promise.
053   */
054  public enum State
055  {
056    PENDING, FULFILLED, REJECTED
057  }
058
059  /**
060   * The underlying promise.
061   * This is not converted to a local variable to make it easy to debug scenarios from within the
062   * browsers DevTools.
063   */
064  @SuppressWarnings( "FieldCanBeLocal" )
065  private final Promise<T> _promise;
066  /**
067   * The state of the promise. Starts as {@link State#PENDING} and then transitions to either
068   * {@link State#FULFILLED} or {@link State#REJECTED}.
069   */
070  @Nonnull
071  private State _state;
072  /**
073   * The value that the promise resolved to. This is not valid unless the state is {@link State#FULFILLED}.
074   */
075  private T _value;
076  /**
077   * The error that the promise was rejected with. This is not valid unless the state is {@link State#REJECTED}.
078   */
079  private E _error;
080
081  /**
082   * Create the observable model that wraps specified promise.
083   *
084   * @param <T>     the type of the value that the promise will resolve to.
085   * @param <E>     the type of the error if the promise is rejected.
086   * @param promise the promise to wrap.
087   * @return the ObservablePromise
088   */
089  public static <T, E> ObservablePromise<T, E> create( @Nonnull final Promise<T> promise )
090  {
091    return new Arez_ObservablePromise<>( promise );
092  }
093
094  ObservablePromise( @Nonnull final Promise<T> promise )
095  {
096    _state = State.PENDING;
097    _promise = Objects.requireNonNull( promise );
098    _promise.then( this::onFulfilled ).catch_( this::onRejected );
099  }
100
101  /**
102   * Return the promise state.
103   *
104   * @return the promise state.
105   */
106  @Observable
107  @Nonnull
108  public State getState()
109  {
110    return _state;
111  }
112
113  void setState( @Nonnull final State state )
114  {
115    _state = Objects.requireNonNull( state );
116  }
117
118  /**
119   * Return the value that the promise was resolved to.
120   * This should NOT be called if the state is not {@link State#FULFILLED} and will result in an invariant
121   * failure if invariants are enabled.
122   *
123   * @return the value that the promise was resolved to.
124   */
125  @Observable
126  public T getValue()
127  {
128    apiInvariant( () -> _state == State.FULFILLED,
129                  () -> "ObservablePromise.getValue() called when the promise is not in fulfilled state. " +
130                        "State: " + _state + ", Promise: " + _promise );
131    return _value;
132  }
133
134  void setValue( final T value )
135  {
136    invariant( () -> _state == State.FULFILLED,
137               () -> "ObservablePromise.setValue() called when promise is in incorrect state. " +
138                     "State: " + _state + ", Promise: " + _promise );
139    _value = value;
140  }
141
142  /**
143   * Return the error that the promise was rejected with.
144   * This should NOT be called if the state is not {@link State#REJECTED} and will result in an invariant
145   * failure if invariants are enabled.
146   *
147   * @return the error that the promise was rejected with.
148   */
149  @Observable
150  public E getError()
151  {
152    apiInvariant( () -> _state == State.REJECTED,
153                  () -> "ObservablePromise.getError() called when the promise is not in rejected state. " +
154                        "State: " + _state + ", Promise: " + _promise );
155    return _error;
156  }
157
158  void setError( final E error )
159  {
160    invariant( () -> _state == State.REJECTED,
161               () -> "ObservablePromise.setError() called when promise is in incorrect state. " +
162                     "State: " + _state + ", Promise: " + _promise );
163    _error = error;
164  }
165
166  @Action
167  @Nonnull
168  IThenable<T> onFulfilled( final T value )
169  {
170    setState( State.FULFILLED );
171    setValue( value );
172    return Promise.resolve( value );
173  }
174
175  @Action
176  IThenable<Object> onRejected( @Nonnull final Object error )
177  {
178    setState( State.REJECTED );
179    setError( Js.uncheckedCast( error ) );
180    return Promise.reject( error );
181  }
182}
