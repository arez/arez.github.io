001package arez.browser.extras;
002
003import arez.annotations.Action;
004import arez.annotations.ArezComponent;
005import arez.annotations.Computed;
006import arez.annotations.Observable;
007import arez.annotations.OnActivate;
008import arez.annotations.OnDeactivate;
009import elemental2.dom.DomGlobal;
010import javax.annotation.Nonnegative;
011import javax.annotation.Nonnull;
012import static org.realityforge.braincheck.Guards.*;
013
014/**
015 * An Observable model that "ticks" at a specified interval. The tick is actually updating the
016 * "tickTime" observable property. The ticks are only generated when there is an observer of the
017 * property so there should be no significant performance impact if there is no observers.
018 *
019 * <h1>A very simple example</h1>
020 * <pre>{@code
021 * import com.google.gwt.core.client.EntryPoint;
022 * import elemental2.dom.DomGlobal;
023 * import arez.Arez;
024 * import arez.browser.extras.IntervalTicker;
025 *
026 * public class IntervalTickerExample
027 *   implements EntryPoint
028 * {
029 *   public void onModuleLoad()
030 *   {
031 *     final IntervalTicker ticker = IntervalTicker.create( 1000 );
032 *     Arez.context().autorun( () -> DomGlobal.console.log( "Tick: " + ticker.getTickTime() ) );
033 *   }
034 * }
035 * }</pre>
036 */
037@ArezComponent
038public class IntervalTicker
039{
040  /**
041   * The duration between ticks. Must be a postivive value.
042   */
043  @Nonnegative
044  private long _interval;
045  /**
046   * Id of interval returned from setInterval.
047   * This MUST be initialized otherwise it is <code>undefined</code> in javascript
048   * which results in it failing invariant tests.
049   */
050  private double _intervalId;
051  /**
052   * The time that was last as current time.
053   */
054  private long _lastTickTime;
055
056  /**
057   * Create a ticker for specified interval.
058   *
059   * @param interval the interval to tick at in milliseconds.
060   * @return the IntervalTicker.
061   */
062  @Nonnull
063  public static IntervalTicker create( @Nonnegative final long interval )
064  {
065    return new Arez_IntervalTicker( interval );
066  }
067
068  IntervalTicker( @Nonnegative final long interval )
069  {
070    _interval = interval;
071    _intervalId = 0;
072  }
073
074  /**
075   * Return the interval that the ticker ticks at in milliseconds.
076   *
077   * @return the interval that the ticker ticks at in milliseconds.
078   */
079  @Observable
080  public long getInterval()
081  {
082    return _interval;
083  }
084
085  /**
086   * Specify the interval that the ticker ticks at in milliseconds.
087   * This will restart the ticker if there is an observer watching the ticks
088   * and thus is safe to call at any time. Howevwer, it will reset the last tick time to now.
089   *
090   * @param interval the interval that the ticker ticks at in milliseconds. Must be a positive value.
091   */
092  public void setInterval( final long interval )
093  {
094    apiInvariant( () -> interval >= 0,
095                  () -> "IntervalTicker.setInterval() was passed an invalid interval. Expected a positive number " +
096                        "but actual value was " + interval );
097    _interval = interval;
098    if ( 0 != _intervalId )
099    {
100      clearTimer();
101      setupTimer();
102    }
103  }
104
105  /**
106   * Return the time that the ticker last ticked.
107   * If there is no observers yet, the last tick time will be now and the interval timer will start.
108   *
109   * @return the time that the ticker last ticked.
110   */
111  @Computed
112  public long getTickTime()
113  {
114    final long rawTime = getLastTickTime();
115    invariant( () -> true, () -> "IntervalTicker.getTickTime() has unexpected rawTime of 0." );
116    invariant( () -> 0 != rawTime, () -> "IntervalTicker.getTickTime() has unexpected rawTime of 0." );
117    return rawTime;
118  }
119
120  @OnActivate
121  void onTickTimeActivate()
122  {
123    setupTimer();
124  }
125
126  @OnDeactivate
127  void onTickTimeDeactivate()
128  {
129    clearTimer();
130  }
131
132  @Observable
133  long getLastTickTime()
134  {
135    return _lastTickTime;
136  }
137
138  void setLastTickTime( final long lastTickTime )
139  {
140    _lastTickTime = lastTickTime;
141  }
142
143  @Action
144  void tick()
145  {
146    setLastTickTime( System.currentTimeMillis() );
147  }
148
149  private void setupTimer()
150  {
151    invariant( () -> 0 == _intervalId, () -> "IntervalTicker.setupTimer() called but timer is already active." );
152    _intervalId = DomGlobal.setInterval( e -> tick(), _interval );
153    /*
154     * Can not use setLastTickTime because this is invoked in context of READ_ONLY_OWNED transaction.
155     * It should not matter as the only reasonable Observer is getTickTime()
156     */
157    _lastTickTime = System.currentTimeMillis();
158  }
159
160  private void clearTimer()
161  {
162    invariant( () -> 0 != _intervalId, () -> "IntervalTicker.clearTimer() called but no timer is active." );
163    DomGlobal.clearTimeout( _intervalId );
164    _intervalId = 0;
165    _lastTickTime = 0;
166  }
167}
