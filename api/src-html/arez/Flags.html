001package arez;
002
003import javax.annotation.Nonnull;
004
005/**
006 * Flags that can be passed when creating observers that determine the configuration of the observer.
007 * The class also contains constants and methods used to extract runtime state from Observers.
008 */
009public final class Flags
010{
011  /**
012   * Flag indicating that the Observer is allowed to observe {@link ComputedValue} instances with a lower priority.
013   */
014  public static final int OBSERVE_LOWER_PRIORITY_DEPENDENCIES = 1 << 30;
015  /**
016   * Indicates that the an action can be created from within the Observers tracked function.
017   */
018  public static final int NESTED_ACTIONS_ALLOWED = 1 << 29;
019  /**
020   * Indicates that the an action must not be created from within the Observers tracked function.
021   */
022  public static final int NESTED_ACTIONS_DISALLOWED = 1 << 28;
023  /**
024   * Mask to extract "NESTED_ACTIONS" option so can derive default value if required.
025   */
026  private static final int NESTED_ACTIONS_MASK = NESTED_ACTIONS_ALLOWED | NESTED_ACTIONS_DISALLOWED;
027  /**
028   * Flag set if the application code can invoke {@link Observer#reportStale()} or {@link ComputedValue#reportPossiblyChanged()} to indicate non-arez dependency has changed.
029   */
030  public static final int NON_AREZ_DEPENDENCIES = 1 << 27;
031  /**
032   * Flag set set if the application code can not invoke {@link Observer#reportStale()} or {@link ComputedValue#reportPossiblyChanged()} to indicate dependency has changed.
033   */
034  public static final int AREZ_DEPENDENCIES_ONLY = 1 << 26;
035  /**
036   * Mask used to extract dependency type bits.
037   */
038  private static final int DEPENDENCIES_TYPE_MASK = NON_AREZ_DEPENDENCIES | AREZ_DEPENDENCIES_ONLY;
039  /**
040   * The observer can only read arez state.
041   */
042  public static final int READ_ONLY = 1 << 25;
043  /**
044   * The observer can read or write arez state.
045   */
046  public static final int READ_WRITE = 1 << 24;
047  /**
048   * Mask used to extract transaction mode bits.
049   */
050  private static final int TRANSACTION_MASK = READ_ONLY | READ_WRITE;
051  /**
052   * The scheduler will be triggered when the observer is created to immediately invoke the
053   * {@link Observer#_tracked} function. This configuration should not be specified if there
054   * is no {@link Observer#_tracked} function supplied. This should not be
055   * specified if {@link #RUN_LATER} is specified.
056   */
057  public static final int RUN_NOW = 1 << 23;
058  /**
059   * The scheduler will not be triggered when the observer is created. The observer either
060   * has no {@link Observer#_tracked} function or is responsible for ensuring that
061   * {@link ArezContext#triggerScheduler()} is invoked at a later time. This should not be
062   * specified if {@link #RUN_NOW} is specified.
063   */
064  public static final int RUN_LATER = 1 << 22;
065  /**
066   * Mask used to extract run type bits.
067   */
068  private static final int RUN_TYPE_MASK = RUN_NOW | RUN_LATER;
069  /**
070   * The runtime will keep the observer reacting to dependencies until disposed. This should not be
071   * specified if {@link #DEACTIVATE_ON_UNOBSERVE} is specified. This is the default value for observers
072   * that supply a tracked function.
073   */
074  public static final int KEEPALIVE = 1 << 21;
075  /**
076   * The flag is valid on observers associated with computed values and will deactivate the observer if the
077   * computed value has no observers. This should not be specified if {@link #KEEPALIVE} is specified.
078   */
079  public static final int DEACTIVATE_ON_UNOBSERVE = 1 << 20;
080  /**
081   * The flag is valid on observers associated with computed values and will deactivate the observer if the
082   * computed value has no observers. This should not be specified if {@link #KEEPALIVE} is specified.
083   */
084  public static final int SCHEDULED_EXTERNALLY = 1 << 19;
085  /**
086   * Mask used to extract react type bits.
087   */
088  private static final int SCHEDULE_TYPE_MASK = KEEPALIVE | DEACTIVATE_ON_UNOBSERVE | SCHEDULED_EXTERNALLY;
089  /**
090   * Highest priority.
091   * This priority should be used when the observer will dispose or release other reactive elements
092   * (and thus remove elements from being scheduled).
093   * <p>Only one of the PRIORITY_* flags should be applied to observer.</p>
094   */
095  public static final int PRIORITY_HIGHEST = 0b001 << 16;
096  /**
097   * High priority.
098   * To reduce the chance that downstream elements will react multiple times within a single
099   * reaction round, this priority should be used when the observer may trigger many downstream
100   * reactions.
101   * <p>Only one of the PRIORITY_* flags should be applied to observer.</p>
102   */
103  public static final int PRIORITY_HIGH = 0b010 << 16;
104  /**
105   * Normal priority if no other priority otherwise specified.
106   * <p>Only one of the PRIORITY_* flags should be applied to observer.</p>
107   */
108  public static final int PRIORITY_NORMAL = 0b011 << 16;
109  /**
110   * Low priority.
111   * Usually used to schedule observers that reflect state onto non-reactive
112   * application components. i.e. Observers that are used to build html views,
113   * perform network operations etc. These reactions are often at low priority
114   * to avoid recalculation of dependencies (i.e. {@link ComputedValue}s) triggering
115   * this reaction multiple times within a single reaction round.
116   * <p>Only one of the PRIORITY_* flags should be applied to observer.</p>
117   */
118  public static final int PRIORITY_LOW = 0b100 << 16;
119  /**
120   * Lowest priority. Use this priority if the observer is a {@link ComputedValue} that
121   * may be unobserved when a {@link #PRIORITY_LOW} observer reacts. This is used to avoid
122   * recomputing state that is likely to either be unobserved or recomputed as part of
123   * another observers reaction.
124   * <p>Only one of the PRIORITY_* flags should be applied to observer.</p>
125   */
126  public static final int PRIORITY_LOWEST = 0b101 << 16;
127  /**
128   * Mask used to extract priority bits.
129   */
130  private static final int PRIORITY_MASK = 0b111 << 16;
131  /**
132   * Shift used to extract priority after applying mask.
133   */
134  private static final int PRIORITY_SHIFT = 16;
135  /**
136   * Mask that identifies the bits associated with static configuration.
137   */
138  static final int CONFIG_FLAGS_MASK =
139    OBSERVE_LOWER_PRIORITY_DEPENDENCIES |
140    NESTED_ACTIONS_MASK |
141    DEPENDENCIES_TYPE_MASK |
142    TRANSACTION_MASK |
143    RUN_TYPE_MASK |
144    SCHEDULE_TYPE_MASK |
145    PRIORITY_MASK;
146  /**
147   * Flag indicating whether next scheduled invocation of {@link Observer} should invoke {@link Observer#_tracked} or {@link Observer#_onDepsUpdated}.
148   */
149  static final int EXECUTE_TRACKED_NEXT = 1 << 10;
150  /**
151   * The observer has been scheduled.
152   */
153  static final int SCHEDULED = 1 << 9;
154  /**
155   * Mask used to extract state bits.
156   * State is the lowest bits as it is the most frequently accessed numeric fields
157   * and placing values at lower part of integer avoids a shift.
158   */
159  private static final int STATE_MASK = 0b111;
160  /**
161   * The observer has been disposed.
162   */
163  static final int STATE_DISPOSED = 0b001;
164  /**
165   * The observer is in the process of being disposed.
166   */
167  static final int STATE_DISPOSING = 0b010;
168  /**
169   * The observer is not active and is not holding any data about it's dependencies.
170   * Typically mean this tracker observer has not been run or if it is a ComputedValue that
171   * there is no observer observing the associated ObservableValue.
172   */
173  static final int STATE_INACTIVE = 0b011;
174  /**
175   * No change since last time observer was notified.
176   */
177  static final int STATE_UP_TO_DATE = 0b100;
178  /**
179   * A transitive dependency has changed but it has not been determined if a shallow
180   * dependency has changed. The observer will need to check if shallow dependencies
181   * have changed. Only Derived observables will propagate POSSIBLY_STALE state.
182   */
183  static final int STATE_POSSIBLY_STALE = 0b101;
184  /**
185   * A dependency has changed so the observer will need to recompute.
186   */
187  static final int STATE_STALE = 0b110;
188  /**
189   * Mask that identifies the bits associated with runtime configuration.
190   */
191  static final int RUNTIME_FLAGS_MASK = EXECUTE_TRACKED_NEXT | SCHEDULED | STATE_MASK;
192
193  /**
194   * Return true if flags contains priority.
195   *
196   * @param flags the flags.
197   * @return true if flags contains priority.
198   */
199  static boolean isPrioritySpecified( final int flags )
200  {
201    return 0 != ( flags & PRIORITY_MASK );
202  }
203
204  /**
205   * Return true if flags contains valid priority.
206   *
207   * @param flags the flags.
208   * @return true if flags contains priority.
209   */
210  static boolean isPriorityValid( final int flags )
211  {
212    final int priorityIndex = getPriorityIndex( flags );
213    return priorityIndex <= 4 && priorityIndex >= 0;
214  }
215
216  /**
217   * Extract and return the priority value ranging from the highest priority 0 and lowest priority 4.
218   * This method assumes that flags has valid priority and will not attempt to re-check.
219   *
220   * @param flags the flags.
221   * @return the priority.
222   */
223  static int getPriority( final int flags )
224  {
225    return flags & PRIORITY_MASK;
226  }
227
228  /**
229   * Extract and return the priority value ranging from the highest priority 0 and lowest priority 4.
230   * This method assumes that flags has valid priority and will not attempt to re-check.
231   *
232   * @param flags the flags.
233   * @return the priority.
234   */
235  static int getPriorityIndex( final int flags )
236  {
237    return ( getPriority( flags ) >> PRIORITY_SHIFT ) - 1;
238  }
239
240  static int priority( final int flags )
241  {
242    return defaultFlagUnlessSpecified( flags, PRIORITY_MASK, PRIORITY_NORMAL );
243  }
244
245  static int nestedActionRule( final int flags )
246  {
247    return Arez.shouldCheckInvariants() ?
248           defaultFlagUnlessSpecified( flags, NESTED_ACTIONS_MASK, NESTED_ACTIONS_DISALLOWED ) :
249           0;
250  }
251
252  /**
253   * Return true if flags contains a valid nested action mode.
254   *
255   * @param flags the flags.
256   * @return true if flags contains valid nested action mode.
257   */
258  static boolean isNestedActionsModeValid( final int flags )
259  {
260    return NESTED_ACTIONS_ALLOWED == ( flags & NESTED_ACTIONS_ALLOWED ) ^
261           NESTED_ACTIONS_DISALLOWED == ( flags & NESTED_ACTIONS_DISALLOWED );
262  }
263
264  /**
265   * Return true if flags contains transaction mode.
266   *
267   * @param flags the flags.
268   * @return true if flags contains transaction mode.
269   */
270  static boolean isTransactionModeSpecified( final int flags )
271  {
272    return 0 != ( flags & TRANSACTION_MASK );
273  }
274
275  static int transactionMode( final int flags )
276  {
277    return Arez.shouldEnforceTransactionType() ?
278           defaultFlagUnlessSpecified( flags, TRANSACTION_MASK, READ_ONLY ) :
279           0;
280  }
281
282  /**
283   * Return true if flags contains a valid transaction mode.
284   *
285   * @param flags the flags.
286   * @return true if flags contains transaction mode.
287   */
288  static boolean isTransactionModeValid( final int flags )
289  {
290    return 0 != ( flags & READ_ONLY ) ^ 0 != ( flags & READ_WRITE );
291  }
292
293  /**
294   * Return true if flags contains a valid react type.
295   *
296   * @param flags the flags.
297   * @return true if flags contains react type.
298   */
299  static boolean isRunTypeValid( final int flags )
300  {
301    return RUN_NOW == ( flags & RUN_NOW ) ^ RUN_LATER == ( flags & RUN_LATER );
302  }
303
304  /**
305   * Return the default run type flag if run type not specified.
306   *
307   * @param flags       the flags.
308   * @param defaultFlag the default flag to apply
309   * @return the default run type if run type unspecified else 0.
310   */
311  static int runType( final int flags, final int defaultFlag )
312  {
313    return defaultFlagUnlessSpecified( flags, RUN_TYPE_MASK, defaultFlag );
314  }
315
316  /**
317   * Return name of transaction mode.
318   *
319   * @param flags the flags.
320   * @return true if flags contains transaction mode.
321   */
322  @Nonnull
323  static String getTransactionModeName( final int flags )
324  {
325    assert Arez.shouldCheckInvariants() || Arez.shouldCheckApiInvariants();
326    if ( 0 != ( flags & READ_ONLY ) )
327    {
328      return "READ_ONLY";
329    }
330    else if ( 0 != ( flags & READ_WRITE ) )
331    {
332      return "READ_WRITE";
333    }
334    else
335    {
336      return "UNKNOWN(" + flags + ")";
337    }
338  }
339
340  /**
341   * Extract and return the schedule type.
342   *
343   * @param flags the flags.
344   * @return the schedule type.
345   */
346  static int getScheduleType( final int flags )
347  {
348    return flags & SCHEDULE_TYPE_MASK;
349  }
350
351  /**
352   * Return true if flags contains a valid ScheduleType.
353   *
354   * @param flags the flags.
355   * @return true if flags contains a valid ScheduleType.
356   */
357  static boolean isScheduleTypeValid( final int flags )
358  {
359    return KEEPALIVE == ( flags & KEEPALIVE ) ^
360           DEACTIVATE_ON_UNOBSERVE == ( flags & DEACTIVATE_ON_UNOBSERVE ) ^
361           SCHEDULED_EXTERNALLY == ( flags & SCHEDULED_EXTERNALLY );
362  }
363
364  /**
365   * Extract and return the observer's state.
366   *
367   * @param flags the flags.
368   * @return the state.
369   */
370  static int getState( final int flags )
371  {
372    return flags & STATE_MASK;
373  }
374
375  /**
376   * Return the new value of flags when supplied with specified state.
377   *
378   * @param flags the flags.
379   * @param state the new state.
380   * @return the new flags.
381   */
382  static int setState( final int flags, final int state )
383  {
384    return ( flags & ~STATE_MASK ) | state;
385  }
386
387  /**
388   * Return true if the state is UP_TO_DATE, POSSIBLY_STALE or STALE.
389   * The inverse of {@link #isNotActive(int)}
390   *
391   * @param flags the flags to check.
392   * @return true if the state is UP_TO_DATE, POSSIBLY_STALE or STALE.
393   */
394  static boolean isActive( final int flags )
395  {
396    return getState( flags ) > STATE_INACTIVE;
397  }
398
399  /**
400   * Return true if the state is INACTIVE, DISPOSING or DISPOSED.
401   * The inverse of {@link #isActive(int)}
402   *
403   * @param flags the flags to check.
404   * @return true if the state is INACTIVE, DISPOSING or DISPOSED.
405   */
406  static boolean isNotActive( final int flags )
407  {
408    return !isActive( flags );
409  }
410
411  /**
412   * Return the least stale observer state. if the state is not active
413   * then the {@link #STATE_UP_TO_DATE} will be returned.
414   *
415   * @param flags the flags to check.
416   * @return the least stale observer state.
417   */
418  static int getLeastStaleObserverState( final int flags )
419  {
420    final int state = getState( flags );
421    return state > STATE_INACTIVE ? state : STATE_UP_TO_DATE;
422  }
423
424  /**
425   * Return the state as a string.
426   *
427   * @param state the state value. One of the STATE_* constants
428   * @return the string describing state.
429   */
430  @Nonnull
431  static String getStateName( final int state )
432  {
433    assert Arez.shouldCheckInvariants() || Arez.shouldCheckApiInvariants();
434    switch ( state )
435    {
436      case STATE_DISPOSED:
437        return "DISPOSED";
438      case STATE_DISPOSING:
439        return "DISPOSING";
440      case STATE_INACTIVE:
441        return "INACTIVE";
442      case STATE_POSSIBLY_STALE:
443        return "POSSIBLY_STALE";
444      case STATE_STALE:
445        return "STALE";
446      case STATE_UP_TO_DATE:
447        return "UP_TO_DATE";
448      default:
449        return "UNKNOWN(" + state + ")";
450    }
451  }
452
453  private Flags()
454  {
455  }
456
457  /**
458   * Return the default flag unless a value in mask is specified.
459   *
460   * @param flags       the flags.
461   * @param mask        the mask.
462   * @param defaultFlag the default flag to apply
463   * @return the default flag unless flag has value.
464   */
465  private static int defaultFlagUnlessSpecified( final int flags, final int mask, final int defaultFlag )
466  {
467    return 0 != ( flags & mask ) ? 0 : defaultFlag;
468  }
