001package arez.component;
002
003import arez.Arez;
004import arez.Locator;
005import java.util.HashMap;
006import java.util.Map;
007import java.util.function.Function;
008import javax.annotation.Nonnull;
009import javax.annotation.Nullable;
010import static arez.Guards.*;
011
012/**
013 * An Locator implementation where you can register a function-per type to be resolved.
014 */
015public final class TypeBasedLocator
016  implements Locator
017{
018  /**
019   * Factory methods for looking entities up by type.
020   */
021  private final Map<Class<?>, Function<Object, ?>> _findByIdFunctions = new HashMap<>();
022
023  /**
024   * Register a function that will find entities of specified type by id.
025   * This must not be invoked if another function has already been registered for type.
026   *
027   * @param <T>              the type of the entity.
028   * @param type             the type of the entity.
029   * @param findByIdFunction the function that looks up the entity by id.
030   */
031  public final <T> void registerLookup( @Nonnull final Class<T> type,
032                                        @Nonnull final Function<Object, T> findByIdFunction )
033  {
034    if ( Arez.shouldCheckApiInvariants() )
035    {
036      apiInvariant( () -> !_findByIdFunctions.containsKey( type ),
037                    () -> "Arez-0188: Attempting to register lookup function for type " + type +
038                          " when a function for type already exists." );
039    }
040    _findByIdFunctions.put( type, findByIdFunction );
041  }
042
043  @Nullable
044  @Override
045  @SuppressWarnings( "unchecked" )
046  public final <T> T findById( @Nonnull final Class<T> type, @Nonnull final Object id )
047  {
048    final Function<Object, ?> function = _findByIdFunctions.get( type );
049    if ( null != function )
050    {
051      return (T) function.apply( id );
052    }
053    else
054    {
055      return null;
056    }
057  }
058}
