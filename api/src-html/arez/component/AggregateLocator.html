001package arez.component;
002
003import arez.Arez;
004import java.util.ArrayList;
005import javax.annotation.Nonnull;
006import javax.annotation.Nullable;
007import static org.realityforge.braincheck.Guards.*;
008
009/**
010 * An Locator implementation that delegates queries to a list of other Locator implementations.
011 * This implementation will search through the list of registered locators whenever the query method
012 * {@link #findById(Class, Object)} is invoked.
013 */
014public class AggregateLocator
015  implements Locator
016{
017  /**
018   * A list of Locator instances.
019   */
020  private final ArrayList<Locator> _locators = new ArrayList<>();
021
022  /**
023   * Create the locator registering the supplied locators.
024   *
025   * @param locators the ordered list of locators to register.
026   */
027  public AggregateLocator( @Nonnull final Locator... locators )
028  {
029    for ( final Locator locator : locators )
030    {
031      registerLocator( locator );
032    }
033  }
034
035  /**
036   * Register an entity locator to be searched.
037   * The Locator must not already be registered.
038   *
039   * @param locator the Locator to register.
040   */
041  protected final void registerLocator( @Nonnull final Locator locator )
042  {
043    if ( Arez.shouldCheckApiInvariants() )
044    {
045      apiInvariant( () -> !_locators.contains( locator ),
046                    () -> "Arez-0189: Attempting to register locator " + locator + " when the " +
047                          "Locator is already present." );
048    }
049    _locators.add( locator );
050  }
051
052  /**
053   * {@inheritDoc}
054   */
055  @Nullable
056  @Override
057  public final <T> T findById( @Nonnull final Class<T> type, @Nonnull final Object id )
058  {
059    for ( final Locator locator : _locators )
060    {
061      final T entity = locator.findById( type, id );
062      if ( null != entity )
063      {
064        return entity;
065      }
066    }
067    return null;
068  }
069}
