001package arez.component;
002
003import arez.Arez;
004import arez.ArezContext;
005import arez.Component;
006import arez.Disposable;
007import arez.Observable;
008import arez.Observer;
009import arez.annotations.Action;
010import arez.annotations.ComponentNameRef;
011import arez.annotations.ComponentRef;
012import arez.annotations.ContextRef;
013import arez.annotations.ObservableRef;
014import arez.annotations.PreDispose;
015import java.util.Comparator;
016import java.util.HashMap;
017import java.util.List;
018import java.util.function.Predicate;
019import java.util.stream.Stream;
020import javax.annotation.Nonnull;
021import javax.annotation.Nullable;
022import org.realityforge.braincheck.Guards;
023
024/**
025 * Abstract base class for repositories that contain Arez components.
026 * This class is used by the annotation processor as a base class from which to derive the actual
027 * repositories for each type.
028 *
029 * <p>When multiple results are returned as a list, they are passed through {@link RepositoryUtil#asList(Stream)} or
030 * {@link RepositoryUtil#toResults(List)} and this will convert the result set to an unmodifiable list if
031 * {@link Arez#areRepositoryResultsModifiable()} returns true. Typically this means that in
032 * development mode these will be made immutable but that the lists will be passed through as-is
033 * in production mode for maximum performance.</p>
034 */
035public abstract class AbstractRepository<K, T, R extends AbstractRepository<K, T, R>>
036{
037  /**
038   * A map of all the entities ArezId to entity.
039   */
040  private final HashMap<K, RepositoryEntry<T>> _entities = new HashMap<>();
041
042  /**
043   * Return the component associated with component if native components enabled.
044   *
045   * @return the component associated with component if native components enabled.
046   */
047  @ComponentRef
048  @Nonnull
049  protected abstract Component component();
050
051  /**
052   * Register specified entity in list of entities managed by the repository.
053   * The expectation that this is invoked after the entity has been created but before it is returned
054   * to the repository user.
055   *
056   * @param entity the entity to register.
057   */
058  protected final void registerEntity( @Nonnull final T entity )
059  {
060    getEntitiesObservable().preReportChanged();
061    final RepositoryEntry<T> entry = new RepositoryEntry<>( entity );
062    final K arezId = Identifiable.getArezId( entity );
063    _entities.put( arezId, entry );
064    final Observer monitor =
065      getContext().when( Arez.areNativeComponentsEnabled() ? component() : null,
066                         Arez.areNamesEnabled() ? getRepositoryName() + ".Watcher." + arezId : null,
067                         true,
068                         () -> Disposable.isDisposed( entity ),
069                         () -> destroy( entity ),
070                         true,
071                         true );
072    entry.setMonitor( monitor );
073    getEntitiesObservable().reportChanged();
074  }
075
076  /**
077   * Dispose all the entities associated with the repository.
078   */
079  @PreDispose
080  protected void preDispose()
081  {
082    getEntitiesObservable().preReportChanged();
083    _entities.values().forEach( e -> Disposable.dispose( e ) );
084    _entities.clear();
085    getEntitiesObservable().reportChanged();
086  }
087
088  /**
089   * Return true if the specified entity is contained in the repository.
090   *
091   * @param entity the entity.
092   * @return true if the specified entity is contained in the repository, false otherwise.
093   */
094  public boolean contains( @Nonnull final T entity )
095  {
096    getEntitiesObservable().reportObserved();
097    return !Disposable.isDisposed( entity ) && _entities.containsKey( Identifiable.<K>getArezId( entity ) );
098  }
099
100  /**
101   * Destroy the supplied entity.
102   * The entity must have been created in this repository and must not have already been destroyed.
103   * The entity will be removed from the map and will be disposed.
104   *
105   * @param entity the entity to destroy.
106   */
107  @Action
108  public void destroy( @Nonnull final T entity )
109  {
110    final RepositoryEntry<T> entry = _entities.remove( Identifiable.<K>getArezId( entity ) );
111    if ( null != entry )
112    {
113      getEntitiesObservable().preReportChanged();
114      Disposable.dispose( entry );
115      getEntitiesObservable().reportChanged();
116    }
117    else
118    {
119      Guards.fail( () -> "Called destroy() passing an entity that was not in the repository. Entity: " + entity );
120    }
121  }
122
123  /**
124   * Return all the entities.
125   *
126   * @return all the entities.
127   */
128  @Nonnull
129  public final List<T> findAll()
130  {
131    return RepositoryUtil.asList( entities() );
132  }
133
134  /**
135   * Return all entities sorted by supplied comparator.
136   *
137   * @param sorter the comparator used to sort entities.
138   * @return the entity list result.
139   */
140  @Nonnull
141  public final List<T> findAll( @Nonnull final Comparator<T> sorter )
142  {
143    return RepositoryUtil.asList( entities().sorted( sorter ) );
144  }
145
146  /**
147   * Return all entities that match query.
148   *
149   * @param query the predicate used to select entities.
150   * @return the entity list result.
151   */
152  @Nonnull
153  public final List<T> findAllByQuery( @Nonnull final Predicate<T> query )
154  {
155    return RepositoryUtil.asList( entities().filter( query ) );
156  }
157
158  /**
159   * Return all entities that match query sorted by supplied comparator.
160   *
161   * @param query  the predicate used to select entities.
162   * @param sorter the comparator used to sort entities.
163   * @return the entity list result.
164   */
165  @Nonnull
166  public final List<T> findAllByQuery( @Nonnull final Predicate<T> query, @Nonnull final Comparator<T> sorter )
167  {
168    return RepositoryUtil.asList( entities().filter( query ).sorted( sorter ) );
169  }
170
171  /**
172   * Return the entity that matches query or null if unable to locate matching entity.
173   *
174   * @param query the predicate used to select entity.
175   * @return the entity or null if unable to locate matching entity.
176   */
177  @Nullable
178  public final T findByQuery( @Nonnull final Predicate<T> query )
179  {
180    return entities().filter( query ).findFirst().orElse( null );
181  }
182
183  /**
184   * Return the entity that matches query else throw an exception.
185   *
186   * @param query the predicate used to select entity.
187   * @return the entity.
188   * @throws NoResultException if unable to locate matching entity.
189   */
190  @Nonnull
191  public final T getByQuery( @Nonnull final Predicate<T> query )
192    throws NoResultException
193  {
194    final T entity = findByQuery( query );
195    if ( null == entity )
196    {
197      throw new NoResultException();
198    }
199    return entity;
200  }
201
202  /**
203   * Return the entity with the specified ArezId or null if no such entity.
204   *
205   * @param arezId the id of entity.
206   * @return the entity or null if no such entity.
207   */
208  @Nullable
209  public T findByArezId( @Nonnull final K arezId )
210  {
211    final RepositoryEntry<T> entry = _entities.get( arezId );
212    if ( null != entry && !Disposable.isDisposed( entry ) )
213    {
214      final T entity = entry.getEntity();
215      ComponentObservable.observe( entity );
216      return entity;
217    }
218    else
219    {
220      getEntitiesObservable().reportObserved();
221      return null;
222    }
223  }
224
225  /**
226   * Return the entity with the specified ArezId else throw an exception.
227   *
228   * @param arezId the id of entity.
229   * @return the entity.
230   * @throws NoSuchEntityException if unable to locate entity with specified ArezId.
231   */
232  @Nonnull
233  public final T getByArezId( @Nonnull final K arezId )
234    throws NoSuchEntityException
235  {
236    final T entity = findByArezId( arezId );
237    if ( null == entity )
238    {
239      throw new NoSuchEntityException( arezId );
240    }
241    return entity;
242  }
243
244  /**
245   * Return the repository instance cast to typed subtype.
246   *
247   * @return the repository instance.
248   */
249  @SuppressWarnings( "unchecked" )
250  @Nonnull
251  public final R self()
252  {
253    return (R) this;
254  }
255
256  /**
257   * Return the context associated with the repository.
258   *
259   * @return the context associated with the repository.
260   */
261  @ContextRef
262  @Nonnull
263  protected abstract ArezContext getContext();
264
265  /**
266   * Return the name associated with the repository.
267   *
268   * @return the name associated with the repository.
269   */
270  @ComponentNameRef
271  @Nonnull
272  protected abstract String getRepositoryName();
273
274  /**
275   * Return the observable associated with entities.
276   * This template method is implemented by the Arez annotation processor and is used internally
277   * to repository. It should not be invoked by extensions.
278   *
279   * @return the Arez observable associated with entities observable property.
280   */
281  @ObservableRef
282  @Nonnull
283  protected abstract Observable getEntitiesObservable();
284
285  /**
286   * Return a stream of all entities in the repository.
287   * This method is typically used by repository extensions.
288   *
289   * @return the underlying entities.
290   */
291  @arez.annotations.Observable( expectSetter = false )
292  @Nonnull
293  public Stream<T> entities()
294  {
295    return _entities.values().stream().filter( this::notDisposed ).map( RepositoryEntry::getEntity );
296  }
297
298  /**
299   * Return true if entry is not disposed.
300   *
301   * @return true if entry is not disposed.
302   */
303  private boolean notDisposed( @Nonnull final RepositoryEntry<T> entry )
304  {
305    return !Disposable.isDisposed( entry );
306  }
307}
