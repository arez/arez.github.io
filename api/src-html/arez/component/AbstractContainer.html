001package arez.component;
002
003import arez.Arez;
004import arez.ArezContext;
005import arez.Component;
006import arez.Disposable;
007import arez.Observable;
008import arez.Observer;
009import arez.annotations.ComponentNameRef;
010import arez.annotations.ComponentRef;
011import arez.annotations.ContextRef;
012import arez.annotations.ObservableRef;
013import arez.annotations.PreDispose;
014import java.util.HashMap;
015import java.util.List;
016import java.util.stream.Stream;
017import javax.annotation.Nonnull;
018import javax.annotation.Nullable;
019import org.realityforge.anodoc.TestOnly;
020import org.realityforge.braincheck.Guards;
021
022/**
023 * Abstract base class for observable Arez components that contain other components.
024 *
025 * <p>When multiple results are returned as a list, they are passed through {@link CollectionsUtil#asList(Stream)} or
026 * {@link CollectionsUtil#wrap(List)} and this will convert the result set to an unmodifiable variant if
027 * {@link Arez#areCollectionsPropertiesUnmodifiable()} returns true. Typically this means that in
028 * development mode these will be made immutable but that the lists will be passed through as-is
029 * in production mode for maximum performance.</p>
030 */
031public abstract class AbstractContainer<K, T>
032{
033  /**
034   * A map of all the entities ArezId to entity.
035   */
036  private final HashMap<K, EntityReference<T>> _entities = new HashMap<>();
037
038  /**
039   * Attach specified entity to the set of entities managed by the container.
040   * This should not be invoked if the entity is already attached to the repository.
041   *
042   * @param entity the entity to register.
043   */
044  protected void attach( @Nonnull final T entity )
045  {
046    getEntitiesObservable().preReportChanged();
047    final EntityReference<T> entry = new EntityReference<>( entity );
048    final K arezId = Identifiable.getArezId( entity );
049    _entities.put( arezId, entry );
050    final Observer monitor =
051      getContext().when( Arez.areNativeComponentsEnabled() ? component() : null,
052                         Arez.areNamesEnabled() ? getContainerName() + ".EntityWatcher." + arezId : null,
053                         true,
054                         () -> ComponentObservable.notObserved( entity ),
055                         () -> detach( entity ),
056                         true,
057                         true );
058    entry.setMonitor( monitor );
059    getEntitiesObservable().reportChanged();
060  }
061
062  /**
063   * Dispose all the entities associated with the container.
064   */
065  @PreDispose
066  protected void preDispose()
067  {
068    _entities.values().forEach( e -> Disposable.dispose( e ) );
069    _entities.clear();
070  }
071
072  /**
073   * Return true if the specified entity is contained in the container.
074   *
075   * @param entity the entity.
076   * @return true if the specified entity is contained in the container, false otherwise.
077   */
078  protected boolean contains( @Nonnull final T entity )
079  {
080    getEntitiesObservable().reportObserved();
081    return Disposable.isNotDisposed( entity ) && _entities.containsKey( Identifiable.<K>getArezId( entity ) );
082  }
083
084  /**
085   * Detach the entity from the container and dispose the entity.
086   * The entity must be attached to the container.
087   *
088   * @param entity the entity to destroy.
089   */
090  protected void destroy( @Nonnull final T entity )
091  {
092    detach( entity, true );
093  }
094
095  /**
096   * Detach entity from container without disposing entity.
097   * The entity must be attached to the container.
098   *
099   * @param entity the entity to detach.
100   */
101  protected void detach( @Nonnull final T entity )
102  {
103    detach( entity, false );
104  }
105
106  /**
107   * Detach entity from container without disposing entity.
108   * The entity must be attached to the container.
109   *
110   * @param entity the entity to detach.
111   */
112  private void detach( @Nonnull final T entity, final boolean disposeEntity )
113  {
114    // This method has been extracted to try and avoid GWT inlining into invoker
115    final EntityReference<T> entry = _entities.remove( Identifiable.<K>getArezId( entity ) );
116    if ( null != entry )
117    {
118      getEntitiesObservable().preReportChanged();
119      if ( disposeEntity )
120      {
121        Disposable.dispose( entry );
122      }
123      else
124      {
125        final Observer monitor = entry.getMonitor();
126        if ( null != monitor )
127        {
128          Disposable.dispose( monitor );
129        }
130      }
131      getEntitiesObservable().reportChanged();
132    }
133    else
134    {
135      Guards.fail( () -> "Arez-0157: Called detach() passing an entity that was not attached to the container. Entity: " +
136                         entity );
137    }
138  }
139
140  /**
141   * Return the entity with the specified ArezId or null if no such entity.
142   *
143   * @param arezId the id of entity.
144   * @return the entity or null if no such entity.
145   */
146  @Nullable
147  protected T findByArezId( @Nonnull final K arezId )
148  {
149    final EntityReference<T> entry = _entities.get( arezId );
150    if ( null != entry && Disposable.isNotDisposed( entry ) )
151    {
152      final T entity = entry.getEntity();
153      ComponentObservable.observe( entity );
154      return entity;
155    }
156    else
157    {
158      getEntitiesObservable().reportObserved();
159      return null;
160    }
161  }
162
163  /**
164   * Return the entity with the specified ArezId else throw an exception.
165   *
166   * @param arezId the id of entity.
167   * @return the entity.
168   * @throws NoSuchEntityException if unable to locate entity with specified ArezId.
169   */
170  @Nonnull
171  protected T getByArezId( @Nonnull final K arezId )
172    throws NoSuchEntityException
173  {
174    final T entity = findByArezId( arezId );
175    if ( null == entity )
176    {
177      throw new NoSuchEntityException( arezId );
178    }
179    return entity;
180  }
181
182  /**
183   * Return the component associated with the container if native components enabled.
184   *
185   * @return the component associated with the container if native components enabled.
186   */
187  @ComponentRef
188  @Nonnull
189  protected abstract Component component();
190
191  /**
192   * Return the context associated with the container.
193   *
194   * @return the context associated with the container.
195   */
196  @ContextRef
197  @Nonnull
198  protected abstract ArezContext getContext();
199
200  /**
201   * Return the name associated with the container.
202   *
203   * @return the name associated with the container.
204   */
205  @ComponentNameRef
206  @Nonnull
207  protected abstract String getContainerName();
208
209  /**
210   * Return the observable associated with entities.
211   * This template method is implemented by the Arez annotation processor and is used internally
212   * to container. It should not be invoked by extensions.
213   *
214   * @return the Arez observable associated with entities observable property.
215   */
216  @ObservableRef
217  @Nonnull
218  protected abstract Observable getEntitiesObservable();
219
220  /**
221   * Return a stream of all entities in the container.
222   *
223   * @return the underlying entities.
224   */
225  @arez.annotations.Observable( expectSetter = false )
226  @Nonnull
227  public Stream<T> entities()
228  {
229    return _entities.values().stream().filter( Disposable::isNotDisposed ).map( EntityReference::getEntity );
230  }
231
232  @TestOnly
233  final HashMap<K, EntityReference<T>> getEntities()
234  {
235    return _entities;
236  }
237}
