001package arez.component;
002
003import arez.Arez;
004import arez.Disposable;
005import arez.ObservableValue;
006import arez.annotations.ObservableRef;
007import arez.annotations.PreDispose;
008import java.util.HashMap;
009import java.util.List;
010import java.util.stream.Stream;
011import javax.annotation.Nonnull;
012import javax.annotation.Nullable;
013import org.realityforge.braincheck.Guards;
014import static org.realityforge.braincheck.Guards.*;
015
016/**
017 * Abstract base class for observable Arez components that contain other components.
018 *
019 * <p>When multiple results are returned as a list, they are passed through {@link CollectionsUtil#asList(Stream)} or
020 * {@link CollectionsUtil#wrap(List)} and this will convert the result set to an unmodifiable variant if
021 * {@link Arez#areCollectionsPropertiesUnmodifiable()} returns true. Typically this means that in
022 * development mode these will be made immutable but that the lists will be passed through as-is
023 * in production mode for maximum performance.</p>
024 */
025public abstract class AbstractContainer<K, T>
026{
027  /**
028   * A map of all the entities ArezId to entity.
029   */
030  private final HashMap<K, T> _entities = new HashMap<>();
031
032  /**
033   * Attach specified entity to the set of entities managed by the container.
034   * This should not be invoked if the entity is already attached to the repository.
035   *
036   * @param entity the entity to register.
037   */
038  @SuppressWarnings( "SuspiciousMethodCalls" )
039  protected void attach( @Nonnull final T entity )
040  {
041    if ( Arez.shouldCheckApiInvariants() )
042    {
043      apiInvariant( () -> Disposable.isNotDisposed( entity ),
044                    () -> "Arez-0168: Called attach() passing an entity that is disposed. Entity: " + entity );
045      apiInvariant( () -> !_entities.containsKey( Identifiable.getArezId( entity ) ),
046                    () -> "Arez-0136: Called attach() passing an entity that is already attached " +
047                          "to the container. Entity: " + entity );
048    }
049    getEntitiesObservable().preReportChanged();
050    attachEntity( entity );
051    _entities.put( Identifiable.getArezId( entity ), entity );
052    getEntitiesObservable().reportChanged();
053  }
054
055  /**
056   * Return true if disposing the container should result in disposing contained entities or just detaching the entities.
057   *
058   * @return true to dispose entities on container dispose, false to detach entities on container disposed.
059   */
060  protected abstract boolean shouldDisposeEntryOnDispose();
061
062  /**
063   * Dispose or detach all the entities associated with the container.
064   */
065  @PreDispose
066  protected void preDispose()
067  {
068    _entities.values().forEach( entry -> detachEntity( entry, shouldDisposeEntryOnDispose() ) );
069    _entities.clear();
070  }
071
072  /**
073   * Return true if the specified entity is contained in the container.
074   *
075   * @param entity the entity.
076   * @return true if the specified entity is contained in the container, false otherwise.
077   */
078  protected boolean contains( @Nonnull final T entity )
079  {
080    getEntitiesObservable().reportObserved();
081    return _entities.containsKey( Identifiable.<K>getArezId( entity ) );
082  }
083
084  /**
085   * Detach the entity from the container and dispose the entity.
086   * The entity must be attached to the container.
087   *
088   * @param entity the entity to destroy.
089   */
090  protected void destroy( @Nonnull final T entity )
091  {
092    detach( entity, true );
093  }
094
095  /**
096   * Detach entity from container without disposing entity.
097   * The entity must be attached to the container.
098   *
099   * @param entity the entity to detach.
100   */
101  protected void detach( @Nonnull final T entity )
102  {
103    detach( entity, false );
104  }
105
106  /**
107   * Detach entity from container without disposing entity.
108   * The entity must be attached to the container.
109   *
110   * @param entity the entity to detach.
111   */
112  private void detach( @Nonnull final T entity, final boolean disposeEntity )
113  {
114    // This method has been extracted to try and avoid GWT inlining into invoker
115    final T removed = _entities.remove( Identifiable.<K>getArezId( entity ) );
116    if ( null != removed )
117    {
118      getEntitiesObservable().preReportChanged();
119      detachEntity( entity, disposeEntity );
120      getEntitiesObservable().reportChanged();
121    }
122    else
123    {
124      Guards.fail( () -> "Arez-0157: Called detach() passing an entity that was not attached to the container. Entity: " +
125                         entity );
126    }
127  }
128
129  /**
130   * Return the entity with the specified ArezId or null if no such entity.
131   *
132   * @param arezId the id of entity.
133   * @return the entity or null if no such entity.
134   */
135  @Nullable
136  protected T findByArezId( @Nonnull final K arezId )
137  {
138    final T entity = _entities.get( arezId );
139    if ( null != entity )
140    {
141      ComponentObservable.observe( entity );
142      return entity;
143    }
144    getEntitiesObservable().reportObserved();
145    return null;
146  }
147
148  /**
149   * Return the entity with the specified ArezId else throw an exception.
150   *
151   * @param arezId the id of entity.
152   * @return the entity.
153   * @throws NoSuchEntityException if unable to locate entity with specified ArezId.
154   */
155  @Nonnull
156  protected T getByArezId( @Nonnull final K arezId )
157    throws NoSuchEntityException
158  {
159    final T entity = findByArezId( arezId );
160    if ( null == entity )
161    {
162      throw new NoSuchEntityException( arezId );
163    }
164    return entity;
165  }
166
167  /**
168   * Return the observable associated with entities.
169   * This template method is implemented by the Arez annotation processor and is used internally
170   * to container. It should not be invoked by extensions.
171   *
172   * @return the Arez observable associated with entities observable property.
173   */
174  @ObservableRef
175  @Nonnull
176  protected abstract ObservableValue getEntitiesObservable();
177
178  /**
179   * Return a stream of all entities in the container.
180   *
181   * @return the underlying entities.
182   */
183  @arez.annotations.Observable( expectSetter = false )
184  @Nonnull
185  public Stream<T> entities()
186  {
187    return _entities.values().stream();
188  }
189
190  private void attachEntity( @Nonnull final T entity )
191  {
192    DisposeTrackable
193      .asDisposeTrackable( entity )
194      .getNotifier()
195      .addOnDisposeListener( this, () -> {
196        getEntitiesObservable().preReportChanged();
197        detach( entity, false );
198        getEntitiesObservable().reportChanged();
199      } );
200  }
201
202  private void detachEntity( @Nonnull final T entity, final boolean disposeOnDetach )
203  {
204    detachEntity( entity );
205    if ( disposeOnDetach )
206    {
207      Disposable.dispose( entity );
208    }
209  }
210
211  private void detachEntity( @Nonnull final T entity )
212  {
213    DisposeTrackable
214      .asDisposeTrackable( entity )
215      .getNotifier()
216      .removeOnDisposeListener( this );
217  }
218
219  final HashMap<K, T> entityMap()
220  {
221    return _entities;
222  }
223}
