001package arez.component;
002
003import arez.Arez;
004import arez.Disposable;
005import arez.Observable;
006import arez.annotations.ObservableRef;
007import arez.annotations.PreDispose;
008import java.util.HashMap;
009import java.util.List;
010import java.util.stream.Stream;
011import javax.annotation.Nonnull;
012import javax.annotation.Nullable;
013import org.realityforge.anodoc.TestOnly;
014import org.realityforge.braincheck.Guards;
015import static org.realityforge.braincheck.Guards.*;
016
017/**
018 * Abstract base class for observable Arez components that contain other components.
019 *
020 * <p>When multiple results are returned as a list, they are passed through {@link CollectionsUtil#asList(Stream)} or
021 * {@link CollectionsUtil#wrap(List)} and this will convert the result set to an unmodifiable variant if
022 * {@link Arez#areCollectionsPropertiesUnmodifiable()} returns true. Typically this means that in
023 * development mode these will be made immutable but that the lists will be passed through as-is
024 * in production mode for maximum performance.</p>
025 */
026public abstract class AbstractContainer<K, T>
027  extends AbstractEntryContainer<T>
028{
029  /**
030   * A map of all the entities ArezId to entity.
031   */
032  private final HashMap<K, EntityEntry<T>> _entities = new HashMap<>();
033
034  /**
035   * Attach specified entity to the set of entities managed by the container.
036   * This should not be invoked if the entity is already attached to the repository.
037   *
038   * @param entity the entity to register.
039   */
040  @SuppressWarnings( "SuspiciousMethodCalls" )
041  protected void attach( @Nonnull final T entity )
042  {
043    if ( Arez.shouldCheckApiInvariants() )
044    {
045      apiInvariant( () -> !_entities.containsKey( Identifiable.getArezId( entity ) ),
046                    () -> "Arez-0136: Called attach() passing an entity that ia already attached " +
047                          "to the container. Entity: " + entity );
048    }
049    getEntitiesObservable().preReportChanged();
050    final EntityEntry<T> entry = createEntityEntry( entity, reference -> detach( reference.getEntity() ) );
051    _entities.put( Identifiable.getArezId( entity ), entry );
052    getEntitiesObservable().reportChanged();
053  }
054
055  /**
056   * Return true if disposing the container should result in disposing contained entities or just detaching the entities.
057   *
058   * @return true to dispose entities on container dispose, false to detach entities on container disposed.
059   */
060  protected abstract boolean shouldDisposeEntryOnDispose();
061
062  /**
063   * Dispose or detach all the entities associated with the container.
064   */
065  @PreDispose
066  protected void preDispose()
067  {
068    _entities.values().forEach( entry -> detachEntry( entry, shouldDisposeEntryOnDispose() ) );
069    _entities.clear();
070  }
071
072  /**
073   * Return true if the specified entity is contained in the container.
074   *
075   * @param entity the entity.
076   * @return true if the specified entity is contained in the container, false otherwise.
077   */
078  protected boolean contains( @Nonnull final T entity )
079  {
080    getEntitiesObservable().reportObserved();
081    return Disposable.isNotDisposed( entity ) && _entities.containsKey( Identifiable.<K>getArezId( entity ) );
082  }
083
084  /**
085   * Detach the entity from the container and dispose the entity.
086   * The entity must be attached to the container.
087   *
088   * @param entity the entity to destroy.
089   */
090  protected void destroy( @Nonnull final T entity )
091  {
092    detach( entity, true );
093  }
094
095  /**
096   * Detach entity from container without disposing entity.
097   * The entity must be attached to the container.
098   *
099   * @param entity the entity to detach.
100   */
101  protected void detach( @Nonnull final T entity )
102  {
103    detach( entity, false );
104  }
105
106  /**
107   * Detach entity from container without disposing entity.
108   * The entity must be attached to the container.
109   *
110   * @param entity the entity to detach.
111   */
112  private void detach( @Nonnull final T entity, final boolean disposeEntity )
113  {
114    // This method has been extracted to try and avoid GWT inlining into invoker
115    final EntityEntry<T> entry = _entities.remove( Identifiable.<K>getArezId( entity ) );
116    if ( null != entry )
117    {
118      getEntitiesObservable().preReportChanged();
119      detachEntry( entry, disposeEntity );
120      getEntitiesObservable().reportChanged();
121    }
122    else
123    {
124      Guards.fail( () -> "Arez-0157: Called detach() passing an entity that was not attached to the container. Entity: " +
125                         entity );
126    }
127  }
128
129  /**
130   * Return the entity with the specified ArezId or null if no such entity.
131   *
132   * @param arezId the id of entity.
133   * @return the entity or null if no such entity.
134   */
135  @Nullable
136  protected T findByArezId( @Nonnull final K arezId )
137  {
138    final EntityEntry<T> entry = _entities.get( arezId );
139    if ( null != entry && Disposable.isNotDisposed( entry ) )
140    {
141      final T entity = entry.getEntity();
142      ComponentObservable.observe( entity );
143      return entity;
144    }
145    else
146    {
147      getEntitiesObservable().reportObserved();
148      return null;
149    }
150  }
151
152  /**
153   * Return the entity with the specified ArezId else throw an exception.
154   *
155   * @param arezId the id of entity.
156   * @return the entity.
157   * @throws NoSuchEntityException if unable to locate entity with specified ArezId.
158   */
159  @Nonnull
160  protected T getByArezId( @Nonnull final K arezId )
161    throws NoSuchEntityException
162  {
163    final T entity = findByArezId( arezId );
164    if ( null == entity )
165    {
166      throw new NoSuchEntityException( arezId );
167    }
168    return entity;
169  }
170
171  /**
172   * Return the observable associated with entities.
173   * This template method is implemented by the Arez annotation processor and is used internally
174   * to container. It should not be invoked by extensions.
175   *
176   * @return the Arez observable associated with entities observable property.
177   */
178  @ObservableRef
179  @Nonnull
180  protected abstract Observable getEntitiesObservable();
181
182  /**
183   * Return a stream of all entities in the container.
184   *
185   * @return the underlying entities.
186   */
187  @arez.annotations.Observable( expectSetter = false )
188  @Nonnull
189  public Stream<T> entities()
190  {
191    return _entities.values().stream().filter( Disposable::isNotDisposed ).map( EntityEntry::getEntity );
192  }
193
194  @TestOnly
195  final HashMap<K, EntityEntry<T>> getEntities()
196  {
197    return _entities;
198  }
199}
