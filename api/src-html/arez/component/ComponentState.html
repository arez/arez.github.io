001package arez.component;
002
003import javax.annotation.Nonnull;
004
005/**
006 * This class is used in generated code to manage the state of the component.
007 * This is centralized here so that it is easier for GWT to optimize it either by sharing the methods
008 * implementation or inlining as appropriate.
009 */
010public final class ComponentState
011{
012  /**
013   * The component has been created, but not yet initialized.
014   */
015  final static byte COMPONENT_CREATED = 0;
016  /**
017   * The components constructor has been called, the {@link arez.ArezContext} field initialized (if necessary),
018   * and the synthetic id has been generated (if required).
019   */
020  public final static byte COMPONENT_INITIALIZED = 1;
021  /**
022   * The reactive elements have been created (i.e. the {@link arez.Observable}, {@link arez.Observer},
023   * {@link arez.ComputedValue} etc.). The {@link arez.annotations.PostConstruct} has NOT been invoked nor
024   * has the {@link arez.Component} been instantiated. This means the component is ready to be interacted with
025   * in a {@link arez.annotations.PostConstruct} method but has not been fully constructed.
026   */
027  public final static byte COMPONENT_CONSTRUCTED = 2;
028  /**
029   * The {@link arez.annotations.PostConstruct} method has been invoked and
030   * the {@link arez.Component} has been instantiated. The scheduler has not been triggered.
031   */
032  public final static byte COMPONENT_COMPLETE = 3;
033  /**
034   * The scheduler has been triggered and any {@link arez.annotations.Autorun} methods have been invoked
035   * or scheduled.
036   */
037  public final static byte COMPONENT_READY = 4;
038  /**
039   * The component is disposing.
040   */
041  public final static byte COMPONENT_DISPOSING = -2;
042  /**
043   * The component has been disposed.
044   */
045  public final static byte COMPONENT_DISPOSED = -1;
046
047  /**
048   * Return true if the component has been initialized.
049   *
050   * @param state the component state.
051   * @return true if the component has been initialized.
052   */
053  public static boolean hasBeenInitialized( final byte state )
054  {
055    return COMPONENT_CREATED != state;
056  }
057
058  /**
059   * Return true if the component has been constructed.
060   *
061   * @param state the component state.
062   * @return true if the component has been constructed.
063   */
064  public static boolean hasBeenConstructed( final byte state )
065  {
066    return hasBeenInitialized( state ) && COMPONENT_INITIALIZED != state;
067  }
068
069  /**
070   * Return true if the component has been completed.
071   *
072   * @param state the component state.
073   * @return true if the component has been completed.
074   */
075  public static boolean hasBeenCompleted( final byte state )
076  {
077    return hasBeenConstructed( state ) && COMPONENT_CONSTRUCTED != state;
078  }
079
080  /**
081   * Return true if the component is active and can be interacted with.
082   * This means that the component has been constructed and has not started to be disposed.
083   *
084   * @param state the component state.
085   * @return true if the component is active.
086   */
087  public static boolean isActive( final byte state )
088  {
089    return COMPONENT_CONSTRUCTED == state || COMPONENT_COMPLETE == state || COMPONENT_READY == state;
090  }
091
092  /**
093   * Return true if the component is disposing or disposed.
094   *
095   * @param state the component state.
096   * @return true if the component is disposing or disposed.
097   */
098  public static boolean isDisposingOrDisposed( final byte state )
099  {
100    return state < 0;
101  }
102
103  /**
104   * Describe specified state. This is usually used to provide error messages.
105   *
106   * @param state the component state.
107   * @return a string description of the state.
108   */
109  @Nonnull
110  public static String describe( final byte state )
111  {
112    switch ( state )
113    {
114      case COMPONENT_CREATED:
115        return "created";
116      case COMPONENT_INITIALIZED:
117        return "initialized";
118      case COMPONENT_CONSTRUCTED:
119        return "constructed";
120      case COMPONENT_COMPLETE:
121        return "complete";
122      case COMPONENT_READY:
123        return "ready";
124      case COMPONENT_DISPOSING:
125        return "disposing";
126      case COMPONENT_DISPOSED:
127        return "disposed";
128      default:
129        throw new IllegalStateException( "Unexpected state passed into ComponentState.describe: " + state );
130    }
131  }
132
133  private ComponentState()
134  {
135  }
136}
