001package arez.component;
002
003import arez.Arez;
004import arez.ArezContext;
005import arez.Component;
006import arez.ComputedValue;
007import arez.Disposable;
008import arez.Priority;
009import arez.Procedure;
010import arez.SafeFunction;
011import java.util.HashMap;
012import java.util.Map;
013import java.util.Objects;
014import java.util.Stack;
015import javax.annotation.Nonnull;
016import javax.annotation.Nullable;
017import static org.realityforge.braincheck.Guards.*;
018
019/**
020 * The class responsible for caching
021 */
022public final class MemoizeCache<T>
023  implements Disposable
024{
025  /**
026   * Functional interface for calculating memoizable value.
027   *
028   * @param <T> The type of the returned value.
029   */
030  @FunctionalInterface
031  public interface Function<T>
032  {
033    /**
034     * Return calculated memoizable value.
035     *
036     * @param args the functions arguments.
037     * @return the value generated by function.
038     */
039    T call( @Nonnull final Object... args );
040  }
041
042  /**
043   * Reference to the system to which this node belongs.
044   */
045  @Nullable
046  private final ArezContext _context;
047  /**
048   * A human consumable prefix for computed values. It should be non-null if {@link Arez#areNamesEnabled()} returns
049   * true and <tt>null</tt> otherwise.
050   */
051  @Nullable
052  private final String _name;
053  /**
054   * The component that this memoize cache is contained within.
055   * This should only be set if {@link Arez#areNativeComponentsEnabled()} is true but can be null even if this is true.
056   */
057  @Nullable
058  private final Component _component;
059  /**
060   * The function memoized.
061   */
062  @Nonnull
063  private final Function<T> _function;
064  /**
065   * The cache of all the ComputedValues created for each unique combination of parameters.
066   */
067  private final Map<Object, Object> _cache = new HashMap<>();
068  /**
069   * The number of arguments passed to memoized function.
070   */
071  private final int _argCount;
072  /**
073   * The priority of the created computeds.
074   */
075  @Nonnull
076  private final Priority _priority;
077  /**
078   * Parameter passed to created computeds .
079   */
080  private final boolean _observeLowerPriorityDependencies;
081  /**
082   * The index of the next ComputedValue created.
083   * This is only used when creating unique names for ComputedValues.
084   */
085  private int _nextIndex;
086  /**
087   * Flag indicating that the cache is currently being disposed.
088   */
089  private boolean _disposed;
090
091  /**
092   * Create the Memoize method cache.
093   *
094   * @param context                          the context in which to create ComputedValue instances.
095   * @param component                        the associated native component if any. This should only be set if {@link Arez#areNativeComponentsEnabled()} returns true.
096   * @param name                             a human consumable prefix for computed values.
097   * @param function                         the memoized function.
098   * @param argCount                         the number of arguments expected to be passed to memoized function.
099   * @param priority                         the priority that used to create the ComputedValue instances.
100   * @param observeLowerPriorityDependencies the observeLowerPriorityDependencies flag that used when creating ComputedValue instances.
101   */
102  public MemoizeCache( @Nullable final ArezContext context,
103                       @Nullable final Component component,
104                       @Nullable final String name,
105                       @Nonnull final Function<T> function,
106                       final int argCount,
107                       @Nonnull final Priority priority,
108                       final boolean observeLowerPriorityDependencies )
109  {
110    if ( Arez.shouldCheckInvariants() )
111    {
112      apiInvariant( () -> Arez.areZonesEnabled() || null == context,
113                    () -> "Arez-174: MemoizeCache passed a context but Arez.areZonesEnabled() is false" );
114      apiInvariant( () -> Arez.areNamesEnabled() || null == name,
115                    () -> "Arez-0159: MemoizeCache passed a name '" + name + "' but Arez.areNamesEnabled() is false" );
116      apiInvariant( () -> argCount > 0,
117                    () -> "Arez-0160: MemoizeCache constructed with invalid argCount: " + argCount +
118                          ". Expected positive value." );
119    }
120    _context = Arez.areZonesEnabled() ? Objects.requireNonNull( context ) : null;
121    _component = Arez.areNativeComponentsEnabled() ? component : null;
122    _name = Arez.areNamesEnabled() ? Objects.requireNonNull( name ) : null;
123    _function = Objects.requireNonNull( function );
124    _argCount = argCount;
125    _priority = Objects.requireNonNull( priority );
126    _observeLowerPriorityDependencies = observeLowerPriorityDependencies;
127  }
128
129  /**
130   * Return the result of the memoized function, calculating if necessary.
131   *
132   * @param args the arguments passed to the memoized function.
133   * @return the result of the memoized function.
134   */
135  public T get( @Nonnull final Object... args )
136  {
137    if ( Arez.shouldCheckApiInvariants() )
138    {
139      apiInvariant( this::isNotDisposed,
140                    () -> "Arez-0161: MemoizeCache named '" + _name + "' had get() invoked when disposed." );
141    }
142    return getComputedValue( args ).get();
143  }
144
145  /**
146   * {@inheritDoc}
147   */
148  @Override
149  public boolean isDisposed()
150  {
151    return _disposed;
152  }
153
154  @Override
155  public void dispose()
156  {
157    if ( !_disposed )
158    {
159      _disposed = true;
160      getContext().safeAction( Arez.areNamesEnabled() ? _name : null, true, false, () -> {
161        disposeMap( _cache, _argCount );
162        _cache.clear();
163      } );
164    }
165  }
166
167  @Nonnull
168  final ArezContext getContext()
169  {
170    return Arez.areZonesEnabled() ? Objects.requireNonNull( _context ) : Arez.context();
171  }
172
173  /**
174   * Traverse to leaf map elements and dispose all contained ComputedValue instances.
175   */
176  @SuppressWarnings( "unchecked" )
177  private void disposeMap( @Nonnull final Map<Object, Object> map, final int depth )
178  {
179    if ( 1 == depth )
180    {
181      for ( final Map.Entry<Object, Object> entry : map.entrySet() )
182      {
183        final ComputedValue<?> computedValue = (ComputedValue<?>) entry.getValue();
184        computedValue.dispose();
185      }
186    }
187    else
188    {
189      for ( final Map.Entry<Object, Object> entry : map.entrySet() )
190      {
191        disposeMap( (Map<Object, Object>) entry.getValue(), depth - 1 );
192      }
193    }
194  }
195
196  /**
197   * Retrieve the computed value for specified parameters, creating it if necessary.
198   *
199   * @param args the arguments passed to the memoized function.
200   */
201  @SuppressWarnings( "unchecked" )
202  ComputedValue<T> getComputedValue( @Nonnull final Object... args )
203  {
204    if ( Arez.shouldCheckApiInvariants() )
205    {
206      apiInvariant( () -> args.length == _argCount,
207                    () -> "Arez-0162: MemoizeCache.getComputedValue called with " + args.length +
208                          " arguments but expected " + _argCount + " arguments." );
209    }
210    Map<Object, Object> map = _cache;
211    final int size = args.length - 1;
212    for ( int i = 0; i < size; i++ )
213    {
214      map = (Map<Object, Object>) map.computeIfAbsent( args[ i ], v -> new HashMap<>() );
215    }
216    ComputedValue<T> computedValue =
217      (ComputedValue<T>) map.computeIfAbsent( args[ size ], v -> createComputedValue( args ) );
218    if ( Disposable.isDisposed( computedValue ) )
219    {
220      computedValue = createComputedValue( args );
221      map.put( args[ size ], computedValue );
222    }
223    return computedValue;
224  }
225
226  /**
227   * Create computed value for specified parameters.
228   *
229   * @param args the arguments passed to the memoized function.
230   */
231  private ComputedValue<T> createComputedValue( @Nonnull final Object... args )
232  {
233    final Component component = Arez.areNativeComponentsEnabled() ? _component : null;
234    final String name = Arez.areNamesEnabled() ? _name + "." + _nextIndex++ : null;
235    final Procedure onDeactivate = () -> disposeComputedValue( args );
236    final SafeFunction<T> function = () -> _function.call( args );
237    return getContext().computed( component,
238                                  name,
239                                  function,
240                                  null,
241                                  onDeactivate,
242                                  null,
243                                  null,
244                                  _priority,
245                                  false,
246                                  true,
247                                  _observeLowerPriorityDependencies );
248  }
249
250  /**
251   * Method invoked to dispose memoized value.
252   * This is called from deactivate hook so there should always by a cached value present
253   * and thus we never check for missing elements in chain.
254   *
255   * @param args the arguments originally passed to the memoized function.
256   */
257  @SuppressWarnings( "unchecked" )
258  void disposeComputedValue( @Nonnull final Object... args )
259  {
260    if ( Arez.shouldCheckInvariants() )
261    {
262      invariant( () -> args.length == _argCount,
263                 () -> "Arez-0163: MemoizeCache.disposeComputedValue called with " + args.length +
264                       " argument but expected " + _argCount + " arguments." );
265    }
266    if ( _disposed )
267    {
268      return;
269    }
270    final Stack<Map> stack = new Stack<>();
271    stack.push( _cache );
272    final int size = args.length - 1;
273    for ( int i = 0; i < size; i++ )
274    {
275      stack.push( (Map) stack.peek().get( args[ i ] ) );
276    }
277    getContext().scheduleDispose( ( (ComputedValue<T>) stack.peek().remove( args[ size ] ) ) );
278    while ( stack.size() > 1 )
279    {
280      final Map map = stack.pop();
281      if ( map.isEmpty() )
282      {
283        stack.peek().remove( args[ stack.size() - 1 ] );
284      }
285      else
286      {
287        return;
288      }
289    }
290  }
291
292  Map<Object, Object> getCache()
293  {
294    return _cache;
295  }
296
297  int getNextIndex()
298  {
299    return _nextIndex;
300  }
301
302  @Nonnull
303  Priority getPriority()
304  {
305    return _priority;
306  }
307
308  boolean canObserveLowerPriorityDependencies()
309  {
310    return _observeLowerPriorityDependencies;
311  }
