001package arez.component;
002
003import arez.Arez;
004import arez.ArezContext;
005import arez.Component;
006import arez.ComputedValue;
007import arez.Disposable;
008import arez.EqualityComparator;
009import arez.Procedure;
010import arez.SafeFunction;
011import java.util.HashMap;
012import java.util.Map;
013import java.util.Objects;
014import java.util.Stack;
015import javax.annotation.Nonnegative;
016import javax.annotation.Nonnull;
017import javax.annotation.Nullable;
018import org.realityforge.anodoc.TestOnly;
019import static org.realityforge.braincheck.Guards.*;

021/**
022 * The class responsible for caching
023 */
024public final class MemoizeCache<T>
025  implements Disposable
026{
027  /**
028   * Functional interface for calculating memoizable value.
029   *
030   * @param <T> The type of the returned value.
031   */
032  @FunctionalInterface
033  public interface Function<T>
034  {
035    /**
036     * Return calculated memoizable value.
037     *
038     * @param args the functions arguments.
039     * @return the value generated by function.
040     */
041    T call( @Nonnull final Object... args );
042  }
043
044  /**
045   * Reference to the system to which this node belongs.
046   */
047  @Nonnull
048  private final ArezContext _context;
049  /**
050   * A human consumable prefix for computed values. It should be non-null if {@link Arez#areNamesEnabled()} returns
051   * true and <tt>null</tt> otherwise.
052   */
053  @Nullable
054  private final String _name;
055  /**
056   * The component that this memoize cache is contained within.
057   * This should only be set if {@link Arez#areNativeComponentsEnabled()} is true but can be null even if this is true.
058   */
059  @Nullable
060  private final Component _component;
061  /**
062   * The function memoized.
063   */
064  @Nonnull
065  private final Function<T> _function;
066  /**
067   * The cache of all the ComputedValues created for each unique combination of parameters.
068   */
069  private final Map<Object, Object> _cache = new HashMap<>();
070  /**
071   * The number of arguments passed to memoized function.
072   */
073  private final int _argCount;
074  /**
075   * The index of the next ComputedValue created.
076   * This is only used when creating unique names for ComputedValues.
077   */
078  private int _nextIndex;
079  /**
080   * Flag indicating that the cache is currently being disposed.
081   */
082  private boolean _disposed;
083
084  /**
085   * Create the Memoize method cache.
086   *
087   * @param context   the context in which to create ComputedValue instances.
088   * @param component the associated native component if any. This should only be set if {@link Arez#areNativeComponentsEnabled()} returns true.
089   * @param name      a human consumable prefix for computed values.
090   * @param function  the memoized function.
091   * @param argCount  the number of arguments expected to be passed to memoized function.
092   */
093  public MemoizeCache( @Nonnull final ArezContext context,
094                       @Nullable final Component component,
095                       @Nullable final String name,
096                       @Nonnull final Function<T> function,
097                       @Nonnegative final int argCount )
098  {
099    if ( Arez.shouldCheckApiInvariants() )
100    {
101      apiInvariant( () -> Arez.areNamesEnabled() || null == name,
102                    () -> "Arez-0159: MemoizeCache passed a name '" + name + "' but Arez.areNamesEnabled() is false" );
103      apiInvariant( () -> argCount > 0,
104                    () -> "Arez-0160: MemoizeCache constructed with invalid argCount: " + argCount +
105                          ". Expected positive value." );
106    }
107    _context = Objects.requireNonNull( context );
108    _component = Arez.areNativeComponentsEnabled() ? component : null;
109    _name = Arez.areNamesEnabled() ? Objects.requireNonNull( name ) : null;
110    _function = Objects.requireNonNull( function );
111    _argCount = argCount;
112  }
113
114  /**
115   * Return the result of the memoized function, calculating if necessary.
116   *
117   * @param args the arguments passed to the memoized function.
118   * @return the result of the memoized function.
119   */
120  public T get( @Nonnull final Object... args )
121  {
122    if ( Arez.shouldCheckApiInvariants() )
123    {
124      apiInvariant( () -> !isDisposed(),
125                    () -> "Arez-0161: MemoizeCache named '" + _name + "' had get() invoked when disposed." );
126    }
127    return getComputedValue( args ).get();
128  }
129
130  /**
131   * {@inheritDoc}
132   */
133  @Override
134  public boolean isDisposed()
135  {
136    return _disposed;
137  }
138
139  @Override
140  public void dispose()
141  {
142    if ( !_disposed )
143    {
144      _disposed = true;
145      disposeMap( _cache, _argCount );
146      _cache.clear();
147    }
148  }
149
150  /**
151   * Traverse to leaf map elements and dispose all contained ComputedValue instances.
152   */
153  @SuppressWarnings( "unchecked" )
154  private void disposeMap( @Nonnull final Map<Object, Object> map, final int depth )
155  {
156    if ( 1 == depth )
157    {
158      for ( final Map.Entry<Object, Object> entry : map.entrySet() )
159      {
160        final ComputedValue<?> computedValue = (ComputedValue<?>) entry.getValue();
161        computedValue.dispose();
162      }
163    }
164    else
165    {
166      for ( final Map.Entry<Object, Object> entry : map.entrySet() )
167      {
168        disposeMap( (Map<Object, Object>) entry.getValue(), depth - 1 );
169      }
170    }
171  }
172
173  /**
174   * Retrieve the computed value for specified parameters, creating it if necessary.
175   *
176   * @param args the arguments passed to the memoized function.
177   */
178  @SuppressWarnings( "unchecked" )
179  ComputedValue<T> getComputedValue( @Nonnull final Object... args )
180  {
181    if ( Arez.shouldCheckApiInvariants() )
182    {
183      apiInvariant( () -> args.length == _argCount,
184                    () -> "Arez-0162: MemoizeCache.getComputedValue called with " + args.length +
185                          " arguments but expected " + _argCount + " arguments." );
186    }
187    Map<Object, Object> map = _cache;
188    final int size = args.length - 1;
189    for ( int i = 0; i < size; i++ )
190    {
191      map = (Map<Object, Object>) map.computeIfAbsent( args[ i ], v -> new HashMap<>() );
192    }
193    ComputedValue<T> computedValue =
194      (ComputedValue<T>) map.computeIfAbsent( args[ size ], v -> createComputedValue( args ) );
195    if ( Disposable.isDisposed( computedValue ) )
196    {
197      computedValue = createComputedValue( args );
198      map.put( args[ size ], computedValue );
199    }
200    return computedValue;
201  }
202
203  /**
204   * Create computed value for specified parameters.
205   *
206   * @param args the arguments passed to the memoized function.
207   */
208  private ComputedValue<T> createComputedValue( @Nonnull final Object... args )
209  {
210    final Component component = Arez.areNativeComponentsEnabled() ? _component : null;
211    final String name = Arez.areNamesEnabled() ? _name + "." + _nextIndex++ : null;
212    final Procedure onDeactivate = () -> disposeComputedValue( args );
213    final SafeFunction<T> function = () -> _function.call( args );
214    return _context.createComputedValue( component,
215                                         name,
216                                         function,
217                                         EqualityComparator.defaultComparator(),
218                                         null,
219                                         onDeactivate,
220                                         null,
221                                         null );
222  }
223
224  /**
225   * Method invoked to dispose memoized value.
226   * This is called from deactivate hook so there should always by a cached value present
227   * and thus we never check for missing elements in chain.
228   *
229   * @param args the arguments originally passed to the memoized function.
230   */
231  @SuppressWarnings( "unchecked" )
232  void disposeComputedValue( @Nonnull final Object... args )
233  {
234    if ( Arez.shouldCheckInvariants() )
235    {
236      invariant( () -> args.length == _argCount,
237                 () -> "Arez-0163: MemoizeCache.disposeComputedValue called with " + args.length +
238                       " argument but expected " + _argCount + " arguments." );
239    }
240    if ( _disposed )
241    {
242      return;
243    }
244    final Stack<Map> stack = new Stack<>();
245    stack.push( _cache );
246    final int size = args.length - 1;
247    for ( int i = 0; i < size; i++ )
248    {
249      stack.push( (Map) stack.peek().get( args[ i ] ) );
250    }
251    ( (ComputedValue<T>) stack.peek().remove( args[ size ] ) ).dispose();
252    while ( stack.size() > 1 )
253    {
254      final Map map = stack.pop();
255      if ( map.isEmpty() )
256      {
257        stack.peek().remove( args[ stack.size() - 1 ] );
258      }
259      else
260      {
261        return;
262      }
263    }
264  }
265
266  @TestOnly
267  Map<Object, Object> getCache()
268  {
269    return _cache;
270  }
271
272  @TestOnly
273  int getNextIndex()
274  {
275    return _nextIndex;
276  }
