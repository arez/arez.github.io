001package arez.component;
002
003import arez.Arez;
004import java.util.Collections;
005import java.util.List;
006import java.util.stream.Collectors;
007import java.util.stream.Stream;
008import javax.annotation.Nonnull;
009
010/**
011 * Utility methods used when returning results from repositories.
012 */
013public final class RepositoryUtil
014{
015  private RepositoryUtil()
016  {
017  }
018
019  /**
020   * Convert specified list to results to return from query.
021   * The result list will be wrapped in an unmodifiable list if {@link Arez#areRepositoryResultsModifiable()}
022   * returns true, otherwise it will just return the supplied list.
023   * This method should be called by repository extensions when returning list results
024   * when not using {@link #asList(Stream)}.
025   *
026   * @param <T>  the type of elements in list.
027   * @param list the input list.
028   * @return the output list
029   */
030  @Nonnull
031  public static <T> List<T> toResults( @Nonnull final List<T> list )
032  {
033    return Arez.areRepositoryResultsModifiable() ? list : Collections.unmodifiableList( list );
034  }
035
036  /**
037   * Convert specified stream to a list, wrapping as an immutable list if required.
038   * This method should be called by repository extensions when returning list results.
039   *
040   * @param <T>    the type of elements in stream.
041   * @param stream the input stream.
042   * @return the output list
043   */
044  @Nonnull
045  public static <T> List<T> asList( @Nonnull final Stream<T> stream )
046  {
047    return toResults( stream.collect( Collectors.toList() ) );
048  }
049}
