001package arez.component;
002
003import arez.Arez;
004import arez.ArezContext;
005import arez.Component;
006import arez.ComputedValue;
007import arez.Disposable;
008import arez.Procedure;
009import arez.SafeFunction;
010import java.util.HashMap;
011import java.util.Map;
012import java.util.Objects;
013import java.util.Stack;
014import javax.annotation.Nonnegative;
015import javax.annotation.Nonnull;
016import javax.annotation.Nullable;
017import org.realityforge.anodoc.TestOnly;
018import static org.realityforge.braincheck.Guards.*;
019
020/**
021 * The class responsible for caching
022 */
023public final class MemoizeCache<T>
024  implements Disposable
025{
026  /**
027   * Functional interface for calculating memoizable value.
028   *
029   * @param <T> The type of the returned value.
030   */
031  @FunctionalInterface
032  public interface Function<T>
033  {
034    /**
035     * Return calculated memoizable value.
036     *
037     * @param args the functions arguments.
038     * @return the value generated by function.
039     */
040    T call( @Nonnull final Object... args );
041  }
042
043  /**
044   * Reference to the system to which this node belongs.
045   */
046  @Nonnull
047  private final ArezContext _context;
048  /**
049   * A human consumable prefix for computed values. It should be non-null if {@link Arez#areNamesEnabled()} returns
050   * true and <tt>null</tt> otherwise.
051   */
052  @Nullable
053  private final String _name;
054  /**
055   * The component that this memoize cache is contained within.
056   * This should only be set if {@link Arez#areNativeComponentsEnabled()} is true but can be null even if this is true.
057   */
058  @Nullable
059  private final Component _component;
060  /**
061   * The function memoized.
062   */
063  @Nonnull
064  private final Function<T> _function;
065  /**
066   * The cache of all the ComputedValues created for each unique combination of parameters.
067   */
068  private final Map<Object, Object> _cache = new HashMap<>();
069  /**
070   * The number of arguments passed to memoized function.
071   */
072  private final int _argCount;
073  /**
074   * The index of the next ComputedValue created.
075   * This is only used when creating unique names for ComputedValues.
076   */
077  private int _nextIndex;
078  /**
079   * Flag indicating that the cache is currently being disposed.
080   */
081  private boolean _disposed;
082
083  /**
084   * Create the Memoize method cache.
085   *
086   * @param context   the context in which to create ComputedValue instances.
087   * @param component the associated native component if any. This should only be set if {@link Arez#areNativeComponentsEnabled()} returns true.
088   * @param name      a human consumable prefix for computed values.
089   * @param function  the memoized function.
090   * @param argCount  the number of arguments expected to be passed to memoized function.
091   */
092  public MemoizeCache( @Nonnull final ArezContext context,
093                       @Nullable final Component component,
094                       @Nullable final String name,
095                       @Nonnull final Function<T> function,
096                       @Nonnegative final int argCount )
097  {
098    if ( Arez.shouldCheckApiInvariants() )
099    {
100      apiInvariant( () -> Arez.areNamesEnabled() || null == name,
101                    () -> "Arez-0159: MemoizeCache passed a name '" + name + "' but Arez.areNamesEnabled() is false" );
102      apiInvariant( () -> argCount > 0,
103                    () -> "Arez-0160: MemoizeCache constructed with invalid argCount: " + argCount +
104                          ". Expected positive value." );
105    }
106    _context = Objects.requireNonNull( context );
107    _component = Arez.areNativeComponentsEnabled() ? component : null;
108    _name = Arez.areNamesEnabled() ? Objects.requireNonNull( name ) : null;
109    _function = Objects.requireNonNull( function );
110    _argCount = argCount;
111  }
112
113  /**
114   * Return the result of the memoized function, calculating if necessary.
115   *
116   * @param args the arguments passed to the memoized function.
117   * @return the result of the memoized function.
118   */
119  public T get( @Nonnull final Object... args )
120  {
121    if ( Arez.shouldCheckApiInvariants() )
122    {
123      apiInvariant( () -> !isDisposed(),
124                    () -> "Arez-0161: MemoizeCache named '" + _name + "' had get() invoked when disposed." );
125    }
126    return getComputedValue( args ).get();
127  }
128
129  /**
130   * {@inheritDoc}
131   */
132  @Override
133  public boolean isDisposed()
134  {
135    return _disposed;
136  }
137
138  @Override
139  public void dispose()
140  {
141    if ( !_disposed )
142    {
143      _disposed = true;
144      disposeMap( _cache, _argCount );
145      _cache.clear();
146    }
147  }
148
149  /**
150   * Traverse to leaf map elements and dispose all contained ComputedValue instances.
151   */
152  @SuppressWarnings( "unchecked" )
153  private void disposeMap( @Nonnull final Map<Object, Object> map, final int depth )
154  {
155    if ( 1 == depth )
156    {
157      for ( final Map.Entry<Object, Object> entry : map.entrySet() )
158      {
159        final ComputedValue<?> computedValue = (ComputedValue<?>) entry.getValue();
160        computedValue.dispose();
161      }
162    }
163    else
164    {
165      for ( final Map.Entry<Object, Object> entry : map.entrySet() )
166      {
167        disposeMap( (Map<Object, Object>) entry.getValue(), depth - 1 );
168      }
169    }
170  }
171
172  /**
173   * Retrieve the computed value for specified parameters, creating it if necessary.
174   *
175   * @param args the arguments passed to the memoized function.
176   */
177  @SuppressWarnings( "unchecked" )
178  ComputedValue<T> getComputedValue( @Nonnull final Object... args )
179  {
180    if ( Arez.shouldCheckApiInvariants() )
181    {
182      apiInvariant( () -> args.length == _argCount,
183                    () -> "Arez-0162: MemoizeCache.getComputedValue called with " + args.length +
184                          " arguments but expected " + _argCount + " arguments." );
185    }
186    Map<Object, Object> map = _cache;
187    final int size = args.length - 1;
188    for ( int i = 0; i < size; i++ )
189    {
190      map = (Map<Object, Object>) map.computeIfAbsent( args[ i ], v -> new HashMap<>() );
191    }
192    ComputedValue<T> computedValue =
193      (ComputedValue<T>) map.computeIfAbsent( args[ size ], v -> createComputedValue( args ) );
194    if ( Disposable.isDisposed( computedValue ) )
195    {
196      computedValue = createComputedValue( args );
197      map.put( args[ size ], computedValue );
198    }
199    return computedValue;
200  }
201
202  /**
203   * Create computed value for specified parameters.
204   *
205   * @param args the arguments passed to the memoized function.
206   */
207  private ComputedValue<T> createComputedValue( @Nonnull final Object... args )
208  {
209    final Component component = Arez.areNativeComponentsEnabled() ? _component : null;
210    final String name = Arez.areNamesEnabled() ? _name + "." + _nextIndex++ : null;
211    final Procedure onDeactivate = () -> disposeComputedValue( args );
212    final SafeFunction<T> function = () -> _function.call( args );
213    return _context.createComputedValue( component, name, function, Objects::equals, null, onDeactivate, null, null );
214  }
215
216  /**
217   * Method invoked to dispose memoized value.
218   * This is called from deactivate hook so there should always by a cached value present
219   * and thus we never check for missing elements in chain.
220   *
221   * @param args the arguments originally passed to the memoized function.
222   */
223  @SuppressWarnings( "unchecked" )
224  void disposeComputedValue( @Nonnull final Object... args )
225  {
226    if ( Arez.shouldCheckInvariants() )
227    {
228      invariant( () -> args.length == _argCount,
229                 () -> "Arez-0163: MemoizeCache.disposeComputedValue called with " + args.length +
230                       " argument but expected " + _argCount + " arguments." );
231    }
232    if ( _disposed )
233    {
234      return;
235    }
236    final Stack<Map> stack = new Stack<>();
237    stack.push( _cache );
238    final int size = args.length - 1;
239    for ( int i = 0; i < size; i++ )
240    {
241      stack.push( (Map) stack.peek().get( args[ i ] ) );
242    }
243    ( (ComputedValue<T>) stack.peek().remove( args[ size ] ) ).dispose();
244    while ( stack.size() > 1 )
245    {
246      final Map map = stack.pop();
247      if ( map.isEmpty() )
248      {
249        stack.peek().remove( args[ stack.size() - 1 ] );
250      }
251      else
252      {
253        return;
254      }
255    }
256  }
257
258  @TestOnly
259  Map<Object, Object> getCache()
260  {
261    return _cache;
262  }
263
264  @TestOnly
265  int getNextIndex()
266  {
267    return _nextIndex;
268  }
