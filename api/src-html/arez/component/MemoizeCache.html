004import arez.ArezContext;

006import arez.ComputableValue;
007import arez.Disposable;
008import arez.Flags;
009import arez.Procedure;

020/**
021 * The class responsible for caching
022 */
023public final class MemoizeCache<T>
024  implements Disposable
025{
026  /**
027   * Functional interface for calculating memoizable value.
028   *
029   * @param <T> The type of the returned value.
030   */
031  @FunctionalInterface
032  public interface Function<T>
033  {
034    /**
035     * Return calculated memoizable value.
036     *
037     * @param args the functions arguments.
038     * @return the value generated by function.
039     */
040    T call( @Nonnull final Object... args );
041  }
042
043  /**
044   * Reference to the system to which this node belongs.
045   */
046  @Nullable
047  private final ArezContext _context;
048  /**
049   * A human consumable prefix for computable values. It should be non-null if {@link Arez#areNamesEnabled()} returns
050   * true and <tt>null</tt> otherwise.
051   */
052  @Nullable
053  private final String _name;
054  /**
055   * The component that this memoize cache is contained within.
056   * This should only be set if {@link Arez#areNativeComponentsEnabled()} is true but can be null even if this is true.
057   */
058  @Nullable
059  private final Component _component;
060  /**
061   * The function memoized.
062   */
063  @Nonnull
064  private final Function<T> _function;
065  /**
066   * The cache of all the ComputableValue created for each unique combination of parameters.
067   */
068  private final Map<Object, Object> _cache = new HashMap<>();
069  /**
070   * The number of arguments passed to memoized function.
071   */
072  private final int _argCount;
073  /**
074   * The flags passed to the created ComputableValues.
075   */
076  private final int _flags;
077  /**
078   * The index of the next ComputableValue created.
079   * This is only used when creating unique names for ComputableValues.
080   */
081  private int _nextIndex;
082  /**
083   * Flag indicating that the cache is currently being disposed.
084   */
085  private boolean _disposed;
086
087  /**
088   * Create the Memoize method cache.
089   *
090   * @param context   the context in which to create ComputableValue instances.
091   * @param component the associated native component if any. This should only be set if {@link Arez#areNativeComponentsEnabled()} returns true.
092   * @param name      a human consumable prefix for computable values.
093   * @param function  the memoized function.
094   * @param argCount  the number of arguments expected to be passed to memoized function.
095   */
096  public MemoizeCache( @Nullable final ArezContext context,
097                       @Nullable final Component component,
098                       @Nullable final String name,
099                       @Nonnull final Function<T> function,
100                       final int argCount )
101  {
102    this( context, component, name, function, argCount, 0 );
103  }
104
105  /**
106   * Create the Memoize method cache.
107   *
108   * @param context   the context in which to create ComputableValue instances.
109   * @param component the associated native component if any. This should only be set if {@link Arez#areNativeComponentsEnabled()} returns true.
110   * @param name      a human consumable prefix for computable values.
111   * @param function  the memoized function.
112   * @param argCount  the number of arguments expected to be passed to memoized function.
113   * @param flags     the flags that are used when creating ComputableValue instances. The only flags supported are the PRIORITY_* flags and {@link Flags#OBSERVE_LOWER_PRIORITY_DEPENDENCIES}.
114   */
115  public MemoizeCache( @Nullable final ArezContext context,
116                       @Nullable final Component component,
117                       @Nullable final String name,
118                       @Nonnull final Function<T> function,
119                       final int argCount,
120                       final int flags )
121  {
122    if ( Arez.shouldCheckApiInvariants() )
123    {
124      apiInvariant( () -> Arez.areZonesEnabled() || null == context,
125                    () -> "Arez-174: MemoizeCache passed a context but Arez.areZonesEnabled() is false" );
126      apiInvariant( () -> Arez.areNamesEnabled() || null == name,
127                    () -> "Arez-0159: MemoizeCache passed a name '" + name + "' but Arez.areNamesEnabled() is false" );
128      apiInvariant( () -> argCount > 0,
129                    () -> "Arez-0160: MemoizeCache constructed with invalid argCount: " + argCount +
130                          ". Expected positive value." );
131      final int mask = Flags.PRIORITY_HIGHEST |
132                       Flags.PRIORITY_HIGH |
133                       Flags.PRIORITY_NORMAL |
134                       Flags.PRIORITY_LOW |
135                       Flags.PRIORITY_LOWEST |
136                       Flags.ENVIRONMENT_REQUIRED |
137                       Flags.ENVIRONMENT_NOT_REQUIRED |
138                       Flags.AREZ_DEPENDENCIES |
139                       Flags.AREZ_OR_NO_DEPENDENCIES |
140                       Flags.AREZ_OR_EXTERNAL_DEPENDENCIES |
141                       Flags.OBSERVE_LOWER_PRIORITY_DEPENDENCIES;
142      apiInvariant( () -> ( ~mask & flags ) == 0,
143                    () -> "Arez-0211: MemoizeCache passed unsupported flags. Unsupported bits: " + ( ~mask & flags ) );
144    }
145    _context = Arez.areZonesEnabled() ? Objects.requireNonNull( context ) : null;
146    _component = Arez.areNativeComponentsEnabled() ? component : null;
147    _name = Arez.areNamesEnabled() ? Objects.requireNonNull( name ) : null;
148    _function = Objects.requireNonNull( function );
149    _argCount = argCount;
150    _flags = flags;
151  }
152
153  /**
154   * Return the result of the memoized function, calculating if necessary.
155   *
156   * @param args the arguments passed to the memoized function.
157   * @return the result of the memoized function.
158   */
159  public T get( @Nonnull final Object... args )
160  {
161    if ( Arez.shouldCheckApiInvariants() )
162    {
163      apiInvariant( this::isNotDisposed,
164                    () -> "Arez-0161: MemoizeCache named '" + _name + "' had get() invoked when disposed." );
165    }
166    return getComputableValue( args ).get();
167  }
168
169  /**
170   * {@inheritDoc}
171   */
172  @Override
173  public boolean isDisposed()
174  {
175    return _disposed;
176  }
177
178  @Override
179  public void dispose()
180  {
181    if ( !_disposed )
182    {
183      _disposed = true;
184      getContext().safeAction( Arez.areNamesEnabled() ? _name : null, () -> {
185        disposeMap( _cache, _argCount );
186        _cache.clear();
187      }, Flags.NO_VERIFY_ACTION_REQUIRED );
188    }
189  }
190
191  @Nonnull
192  final ArezContext getContext()
193  {
194    return Arez.areZonesEnabled() ? Objects.requireNonNull( _context ) : Arez.context();
195  }
196
197  /**
198   * Traverse to leaf map elements and dispose all contained ComputableValue instances.
199   */
200  @SuppressWarnings( "unchecked" )
201  private void disposeMap( @Nonnull final Map<Object, Object> map, final int depth )
202  {
203    if ( 1 == depth )
204    {
205      for ( final Map.Entry<Object, Object> entry : map.entrySet() )
206      {
207        final ComputableValue<?> computableValue = (ComputableValue<?>) entry.getValue();
208        computableValue.dispose();
209      }
210    }
211    else
212    {
213      for ( final Map.Entry<Object, Object> entry : map.entrySet() )
214      {
215        disposeMap( (Map<Object, Object>) entry.getValue(), depth - 1 );
216      }
217    }
218  }
219
220  /**
221   * Retrieve the computable value for specified parameters, creating it if necessary.
222   *
223   * @param args the arguments passed to the memoized function.
224   */
225  @SuppressWarnings( "unchecked" )
226  ComputableValue<T> getComputableValue( @Nonnull final Object... args )
227  {
228    if ( Arez.shouldCheckApiInvariants() )
229    {
230      apiInvariant( () -> args.length == _argCount,
231                    () -> "Arez-0162: MemoizeCache.getComputableValue called with " + args.length +
232                          " arguments but expected " + _argCount + " arguments." );
233    }
234    Map<Object, Object> map = _cache;
235    final int size = args.length - 1;
236    for ( int i = 0; i < size; i++ )
237    {
238      map = (Map<Object, Object>) map.computeIfAbsent( args[ i ], v -> new HashMap<>() );
239    }
240    ComputableValue<T> computableValue =
241      (ComputableValue<T>) map.computeIfAbsent( args[ size ], v -> createComputableValue( args ) );
242    if ( Disposable.isDisposed( computableValue ) )
243    {
244      computableValue = createComputableValue( args );
245      map.put( args[ size ], computableValue );
246    }
247    return computableValue;
248  }
249
250  /**
251   * Create computable value for specified parameters.
252   *
253   * @param args the arguments passed to the memoized function.
254   */
255  private ComputableValue<T> createComputableValue( @Nonnull final Object... args )
256  {
257    final Component component = Arez.areNativeComponentsEnabled() ? _component : null;
258    final String name = Arez.areNamesEnabled() ? _name + "." + _nextIndex++ : null;
259    final Procedure onDeactivate = () -> disposeComputableValue( args );
260    final SafeFunction<T> function = () -> _function.call( args );
261    return getContext().computable( component, name, function, null, onDeactivate, null, _flags );
262  }
263
264  /**
265   * Method invoked to dispose memoized value.
266   * This is called from deactivate hook so there should always by a cached value present
267   * and thus we never check for missing elements in chain.
268   *
269   * @param args the arguments originally passed to the memoized function.
270   */
271  @SuppressWarnings( "unchecked" )
272  void disposeComputableValue( @Nonnull final Object... args )
273  {
274    if ( Arez.shouldCheckInvariants() )
275    {
276      invariant( () -> args.length == _argCount,
277                 () -> "Arez-0163: MemoizeCache.disposeComputableValue called with " + args.length +
278                       " argument(s) but expected " + _argCount + " argument(s)." );
279    }
280    if ( _disposed )
281    {
282      return;
283    }
284    final Stack<Map> stack = new Stack<>();
285    stack.push( _cache );
286    final int size = args.length - 1;
287    for ( int i = 0; i < size; i++ )
288    {
289      stack.push( (Map) stack.peek().get( args[ i ] ) );
290    }
291    final ComputableValue<T> computableValue = (ComputableValue<T>) stack.peek().remove( args[ size ] );
292    if ( Arez.shouldCheckInvariants() )
293    {
294      invariant( () -> null != computableValue,
295                 () -> "Arez-0193: MemoizeCache.disposeComputableValue called with args " + Arrays.asList( args ) +
296                       " but unable to locate corresponding ComputableValue." );
297    }
298    assert null != computableValue;
299    getContext().scheduleDispose( computableValue );
300    while ( stack.size() > 1 )
301    {
302      final Map map = stack.pop();
303      if ( map.isEmpty() )
304      {
305        stack.peek().remove( args[ stack.size() - 1 ] );
306      }
307      else
308      {
309        return;
310      }
311    }
312  }
313
314  Map<Object, Object> getCache()
315  {
316    return _cache;
317  }
318
319  int getNextIndex()
320  {
321    return _nextIndex;
322  }
323
324  int getFlags()
325  {
326    return _flags;
327  }
