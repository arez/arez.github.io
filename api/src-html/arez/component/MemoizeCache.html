001package arez.component;
002
003import arez.Arez;
004import arez.ArezContext;
005import arez.Component;
006import arez.ComputedValue;
007import arez.Disposable;
008import arez.Procedure;
009import arez.SafeFunction;
010import java.util.HashMap;
011import java.util.Map;
012import java.util.Objects;
013import java.util.Stack;
014import javax.annotation.Nonnull;
015import javax.annotation.Nullable;
016import static org.realityforge.braincheck.Guards.*;
017
018/**
019 * The class responsible for caching
020 */
021public final class MemoizeCache<T>
022  implements Disposable
023{
024  /**
025   * Functional interface for calculating memoizable value.
026   *
027   * @param <T> The type of the returned value.
028   */
029  @FunctionalInterface
030  public interface Function<T>
031  {
032    /**
033     * Return calculated memoizable value.
034     *
035     * @param args the functions arguments.
036     * @return the value generated by function.
037     */
038    T call( @Nonnull final Object... args );
039  }
040
041  /**
042   * Reference to the system to which this node belongs.
043   */
044  @Nullable
045  private final ArezContext _context;
046  /**
047   * A human consumable prefix for computed values. It should be non-null if {@link Arez#areNamesEnabled()} returns
048   * true and <tt>null</tt> otherwise.
049   */
050  @Nullable
051  private final String _name;
052  /**
053   * The component that this memoize cache is contained within.
054   * This should only be set if {@link Arez#areNativeComponentsEnabled()} is true but can be null even if this is true.
055   */
056  @Nullable
057  private final Component _component;
058  /**
059   * The function memoized.
060   */
061  @Nonnull
062  private final Function<T> _function;
063  /**
064   * The cache of all the ComputedValues created for each unique combination of parameters.
065   */
066  private final Map<Object, Object> _cache = new HashMap<>();
067  /**
068   * The number of arguments passed to memoized function.
069   */
070  private final int _argCount;
071  /**
072   * The index of the next ComputedValue created.
073   * This is only used when creating unique names for ComputedValues.
074   */
075  private int _nextIndex;
076  /**
077   * Flag indicating that the cache is currently being disposed.
078   */
079  private boolean _disposed;
080
081  /**
082   * Create the Memoize method cache.
083   *
084   * @param context   the context in which to create ComputedValue instances.
085   * @param component the associated native component if any. This should only be set if {@link Arez#areNativeComponentsEnabled()} returns true.
086   * @param name      a human consumable prefix for computed values.
087   * @param function  the memoized function.
088   * @param argCount  the number of arguments expected to be passed to memoized function.
089   */
090  public MemoizeCache( @Nullable final ArezContext context,
091                       @Nullable final Component component,
092                       @Nullable final String name,
093                       @Nonnull final Function<T> function,
094                       final int argCount )
095  {
096    if ( Arez.shouldCheckInvariants() )
097    {
098      apiInvariant( () -> Arez.areZonesEnabled() || null == context,
099                    () -> "Arez-174: MemoizeCache passed a context but Arez.areZonesEnabled() is false" );
100      apiInvariant( () -> Arez.areNamesEnabled() || null == name,
101                    () -> "Arez-0159: MemoizeCache passed a name '" + name + "' but Arez.areNamesEnabled() is false" );
102      apiInvariant( () -> argCount > 0,
103                    () -> "Arez-0160: MemoizeCache constructed with invalid argCount: " + argCount +
104                          ". Expected positive value." );
105    }
106    _context = Arez.areZonesEnabled() ? Objects.requireNonNull( context ) : null;
107    _component = Arez.areNativeComponentsEnabled() ? component : null;
108    _name = Arez.areNamesEnabled() ? Objects.requireNonNull( name ) : null;
109    _function = Objects.requireNonNull( function );
110    _argCount = argCount;
111  }
112
113  /**
114   * Return the result of the memoized function, calculating if necessary.
115   *
116   * @param args the arguments passed to the memoized function.
117   * @return the result of the memoized function.
118   */
119  public T get( @Nonnull final Object... args )
120  {
121    if ( Arez.shouldCheckApiInvariants() )
122    {
123      apiInvariant( this::isNotDisposed,
124                    () -> "Arez-0161: MemoizeCache named '" + _name + "' had get() invoked when disposed." );
125    }
126    return getComputedValue( args ).get();
127  }
128
129  /**
130   * {@inheritDoc}
131   */
132  @Override
133  public boolean isDisposed()
134  {
135    return _disposed;
136  }
137
138  @Override
139  public void dispose()
140  {
141    if ( !_disposed )
142    {
143      _disposed = true;
144      getContext().safeAction( Arez.areNamesEnabled() ? _name : null, () -> {
145        disposeMap( _cache, _argCount );
146        _cache.clear();
147      } );
148    }
149  }
150
151  @Nonnull
152  final ArezContext getContext()
153  {
154    return Arez.areZonesEnabled() ? Objects.requireNonNull( _context ) : Arez.context();
155  }
156
157  /**
158   * Traverse to leaf map elements and dispose all contained ComputedValue instances.
159   */
160  @SuppressWarnings( "unchecked" )
161  private void disposeMap( @Nonnull final Map<Object, Object> map, final int depth )
162  {
163    if ( 1 == depth )
164    {
165      for ( final Map.Entry<Object, Object> entry : map.entrySet() )
166      {
167        final ComputedValue<?> computedValue = (ComputedValue<?>) entry.getValue();
168        computedValue.dispose();
169      }
170    }
171    else
172    {
173      for ( final Map.Entry<Object, Object> entry : map.entrySet() )
174      {
175        disposeMap( (Map<Object, Object>) entry.getValue(), depth - 1 );
176      }
177    }
178  }
179
180  /**
181   * Retrieve the computed value for specified parameters, creating it if necessary.
182   *
183   * @param args the arguments passed to the memoized function.
184   */
185  @SuppressWarnings( "unchecked" )
186  ComputedValue<T> getComputedValue( @Nonnull final Object... args )
187  {
188    if ( Arez.shouldCheckApiInvariants() )
189    {
190      apiInvariant( () -> args.length == _argCount,
191                    () -> "Arez-0162: MemoizeCache.getComputedValue called with " + args.length +
192                          " arguments but expected " + _argCount + " arguments." );
193    }
194    Map<Object, Object> map = _cache;
195    final int size = args.length - 1;
196    for ( int i = 0; i < size; i++ )
197    {
198      map = (Map<Object, Object>) map.computeIfAbsent( args[ i ], v -> new HashMap<>() );
199    }
200    ComputedValue<T> computedValue =
201      (ComputedValue<T>) map.computeIfAbsent( args[ size ], v -> createComputedValue( args ) );
202    if ( Disposable.isDisposed( computedValue ) )
203    {
204      computedValue = createComputedValue( args );
205      map.put( args[ size ], computedValue );
206    }
207    return computedValue;
208  }
209
210  /**
211   * Create computed value for specified parameters.
212   *
213   * @param args the arguments passed to the memoized function.
214   */
215  private ComputedValue<T> createComputedValue( @Nonnull final Object... args )
216  {
217    final Component component = Arez.areNativeComponentsEnabled() ? _component : null;
218    final String name = Arez.areNamesEnabled() ? _name + "." + _nextIndex++ : null;
219    final Procedure onDeactivate = () -> disposeComputedValue( args );
220    final SafeFunction<T> function = () -> _function.call( args );
221    return getContext().createComputedValue( component,
222                                             name,
223                                             function,
224                                             null,
225                                             onDeactivate,
226                                             null,
227                                             null );
228  }
229
230  /**
231   * Method invoked to dispose memoized value.
232   * This is called from deactivate hook so there should always by a cached value present
233   * and thus we never check for missing elements in chain.
234   *
235   * @param args the arguments originally passed to the memoized function.
236   */
237  @SuppressWarnings( "unchecked" )
238  void disposeComputedValue( @Nonnull final Object... args )
239  {
240    if ( Arez.shouldCheckInvariants() )
241    {
242      invariant( () -> args.length == _argCount,
243                 () -> "Arez-0163: MemoizeCache.disposeComputedValue called with " + args.length +
244                       " argument but expected " + _argCount + " arguments." );
245    }
246    if ( _disposed )
247    {
248      return;
249    }
250    final Stack<Map> stack = new Stack<>();
251    stack.push( _cache );
252    final int size = args.length - 1;
253    for ( int i = 0; i < size; i++ )
254    {
255      stack.push( (Map) stack.peek().get( args[ i ] ) );
256    }
257    getContext().scheduleDispose( ( (ComputedValue<T>) stack.peek().remove( args[ size ] ) ) );
258    while ( stack.size() > 1 )
259    {
260      final Map map = stack.pop();
261      if ( map.isEmpty() )
262      {
263        stack.peek().remove( args[ stack.size() - 1 ] );
264      }
265      else
266      {
267        return;
268      }
269    }
270  }
271
272  Map<Object, Object> getCache()
273  {
274    return _cache;
275  }
276
277  int getNextIndex()
278  {
279    return _nextIndex;
280  }
