027/**
028 * The "kernel" of the components generated by the annotation processor.
029 * This class exists so that code common across multiple components is not present in every
030 * generated class but is instead in a single location. This results in smaller, faster code.
031 */
032@OmitClinit
033public final class ComponentKernel
034  implements Disposable, ComponentObservable
035{
036  /**
037   * The component has been created, but not yet initialized.
038   */
039  private final static byte COMPONENT_CREATED = 0;
040  /**
041   * The components constructor has been called, the {@link ArezContext} field initialized (if necessary),
042   * and the synthetic id has been generated (if required).
043   */
044  private final static byte COMPONENT_INITIALIZED = 1;
045  /**
046   * The reactive elements have been created (i.e. the {@link ObservableValue}, {@link arez.Observer},
047   * {@link ComputableValue} etc.). The {@link arez.annotations.PostConstruct} has NOT been invoked nor
048   * has the {@link Component} been instantiated. This means the component is ready to be interacted with
049   * in a {@link arez.annotations.PostConstruct} method but has not been fully constructed.
050   */
051  private final static byte COMPONENT_CONSTRUCTED = 2;
052  /**
053   * The {@link arez.annotations.PostConstruct} method has been invoked and
054   * the {@link Component} has been instantiated. Observers have been scheduled but the scheduler
055   * has not been triggered.
056   */
057  private final static byte COMPONENT_COMPLETE = 3;
058  /**
059   * The scheduler has been triggered and any {@link Observe} methods have been invoked if runtime managed.
060   */
061  private final static byte COMPONENT_READY = 4;
062  /**
063   * The component is disposing.
064   */
065  private final static byte COMPONENT_DISPOSING = -2;
066  /**
067   * The component has been disposed.
068   */
069  private final static byte COMPONENT_DISPOSED = -1;
070  /**
071   * Reference to the context to which this component belongs.
072   */
073  @OmitSymbol( unless = "arez.enable_zones" )
074  @Nullable
075  private final ArezContext _context;
076  /**
077   * A human consumable name for component. It should be non-null if {@link Arez#areNamesEnabled()} returns
078   * <code>true</code> and <tt>null</tt> otherwise.
079   */
080  @Nullable
081  @OmitSymbol( unless = "arez.enable_names" )
082  private final String _name;
083  /**
084   * The runtime managed synthetic id for component. This will be 0 if the component has supplied a custom
085   * id via a method annotated with {@link arez.annotations.ComponentId} or the annotation processor has
086   * determined that no id is required. The id must be supplied with a non-zero value if:
087   *
088   * <ul>
089   * <li>the component type is annotated with the {@link arez.annotations.Repository} annotation.</li>
090   * <li>the component declared it requires an id (i.e. {@link ArezComponent#requireId()} is <code>true</code>) but
091   * no method annotated with {@link arez.annotations.ComponentId} is present on the components type.</li>
092   * <li>The runtime requires an id as part of debugging infrastructure. (i.e. @link Arez#areNamesEnabled(), {@link Arez#areRegistriesEnabled()}
093   * or {@link Arez#areNativeComponentsEnabled()} returns <code>true</code>.</li>
094   * </ul>
095   */
096  private final int _id;
097  /**
098   * The initialization state of the component. Possible values are defined by the constants in the
099   * this class however this field is only used for determining whether a component
100   * is disposed when invariant checking is disabled so states other than {@link #COMPONENT_DISPOSING} are not set
101   * when invariant checking is disabled.
102   */
103  private byte _state;
104  /**
105   * The native component associated with the component. This should be non-null if {@link Arez#areNativeComponentsEnabled()}
106   * returns <code>true</code> and <tt>null</tt> otherwise.
107   */
108  @OmitSymbol( unless = "arez.enable_native_components" )
109  @Nullable
110  private final Component _component;
111  /**
112   * This callback is invoked before the component is disposed.
113   */
114  @Nullable
115  private final SafeProcedure _preDisposeCallback;
116  /**
117   * This callback is invoked to dispose the reactive elements of the component.
118   */
119  @Nullable
120  private final SafeProcedure _disposeCallback;
121  /**
122   * This callback is invoked after the component is disposed.
123   */
124  @Nullable
125  private final SafeProcedure _postDisposeCallback;
126  /**
127   * The mechanisms to notify downstream elements that the component has been disposed. This should be non-null
128   * if the {@link ArezComponent#disposeNotifier()} is enabled, and <code>null</code> otherwise.
129   */
130  @Nullable
131  private final Map<Object, SafeProcedure> _onDisposeListeners;
132  /**
133   * Mechanism for implementing {@link ComponentObservable} on the component.
134   */
135  @Nullable
136  private final ObservableValue<Boolean> _componentObservable;
137  /**
138   * Mechanism for implementing {@link ArezComponent#disposeOnDeactivate()} on the component.
139   */
140  @Nullable
141  private final ComputableValue<Boolean> _disposeOnDeactivate;
142  /**
143   * Guard to ensure we never try to schedule a dispose multiple times, otherwise the underlying task
144   * system will detect multiple tasks with the same name and object.
145   */
146  private boolean _disposeScheduled;
147
148  public ComponentKernel( @Nullable final ArezContext context,
149                          @Nullable final String name,
150                          final int id,
151                          @Nullable final Component component,
152                          @Nullable final SafeProcedure preDisposeCallback,
153                          @Nullable final SafeProcedure disposeCallback,
154                          @Nullable final SafeProcedure postDisposeCallback,
155                          final boolean notifyOnDispose,
156                          final boolean isComponentObservable,
157                          final boolean disposeOnDeactivate )
158  {
159    if ( Arez.shouldCheckApiInvariants() )
160    {
161      apiInvariant( () -> Arez.areZonesEnabled() || null == context,
162                    () -> "Arez-0100: ComponentKernel passed a context but Arez.areZonesEnabled() is false" );
163      apiInvariant( () -> Arez.areNamesEnabled() || null == name,
164                    () -> "Arez-0156: ComponentKernel passed a name '" + name +
165                          "' but Arez.areNamesEnabled() returns false." );
166      apiInvariant( () -> !Arez.areNativeComponentsEnabled() ||
167                          null == component ||
168                          0 == id ||
169                          ( (Integer) id ).equals( component.getId() ),
170                    () -> "Arez-0222: ComponentKernel named '" + name +
171                          "' passed an id " + id + " and a component but the component had a different id (" +
172                          Objects.requireNonNull( component ).getId() + ")" );
173    }
174
175    if ( Arez.shouldCheckApiInvariants() )
176    {
177      _state = COMPONENT_INITIALIZED;
178    }
179    _name = Arez.areNamesEnabled() ? name : null;
180    _context = Arez.areZonesEnabled() ? context : null;
181    _component = Arez.areNativeComponentsEnabled() ? Objects.requireNonNull( component ) : null;
182    _id = id;
183    _onDisposeListeners = notifyOnDispose ? new HashMap<>() : null;
184    _preDisposeCallback = Arez.areNativeComponentsEnabled() ? null : preDisposeCallback;
185    _disposeCallback = Arez.areNativeComponentsEnabled() ? null : disposeCallback;
186    _postDisposeCallback = Arez.areNativeComponentsEnabled() ? null : postDisposeCallback;
187    _componentObservable = isComponentObservable ? createComponentObservable() : null;
188    _disposeOnDeactivate = disposeOnDeactivate ? createDisposeOnDeactivate() : null;
189  }
190
191  @Nonnull
192  private ComputableValue<Boolean> createDisposeOnDeactivate()
193  {
194    return getContext().computable( Arez.areNativeComponentsEnabled() ? getComponent() : null,
195                                    Arez.areNamesEnabled() ? getName() + ".disposeOnDeactivate" : null,
196                                    this::observe0,
197                                    null,
198                                    this::scheduleDispose,
199                                    null,
200                                    ComputableValue.Flags.PRIORITY_HIGHEST );
201  }
202
203  private void scheduleDispose()
204  {
205    /*
206     * Guard against a scenario where due to interleaving of scheduled tasks a component is disposed due,
207     * to deactivation and then is re-observed and deactivated again prior to the dispose task running.
208     * This scenario was thought to be practically impossible but several applications did the impossible.
209     *
210     * There is still a bug or at least an ambiguity where a disposeOnDeactivate component deactivates, schedules
211     * dispose and then activates before the dispose task runs. Should the dispose be aborted or should it go ahead?
212     * Currently the Arez API does not expose a flag indicating whether computableValues are observed and not possible
213     * to implement the first strategy even though it may seem to be the right one.
214     */
215    if ( !_disposeScheduled )
216    {
217      _disposeScheduled = true;
218      getContext().task( Arez.areNamesEnabled() ? getName() + ".disposeOnDeactivate.task" : null,
219                         this::dispose,
220                         Task.Flags.PRIORITY_HIGHEST | Task.Flags.DISPOSE_ON_COMPLETE | Task.Flags.NO_WRAP_TASK );
221    }
222  }
223
224  @Nonnull
225  private ObservableValue<Boolean> createComponentObservable()
226  {
227    return getContext().observable( Arez.areNativeComponentsEnabled() ? getComponent() : null,
228                                    Arez.areNamesEnabled() ? getName() + ".isDisposed" : null,
229                                    Arez.arePropertyIntrospectorsEnabled() ? () -> _state > 0 : null );
230  }
231
232  @Override
233  public boolean observe()
234  {
235    if ( Arez.shouldCheckApiInvariants() )
236    {
237      apiInvariant( () -> null != _disposeOnDeactivate || null != _componentObservable,
238                    () -> "Arez-0221: ComponentKernel.observe() invoked on component named '" + getName() +
239                          "' but observing is not enabled for component." );
240    }
241    if ( null != _disposeOnDeactivate )
242    {
243      return isNotDisposed() ? _disposeOnDeactivate.get() : false;
244    }
245    else
246    {
247      return observe0();
248    }
249  }
250
251  /**
252   * Internal observe method that may be directly used or used from computable if disposeOnDeactivate is true.
253   */
254  private boolean observe0()
255  {
256    assert null != _componentObservable;
257    final boolean isNotDisposed = isNotDisposed();
258    if ( isNotDisposed )
259    {
260      _componentObservable.reportObserved();
261    }
262    return isNotDisposed;
263  }
264
265  @Override
266  public void dispose()
267  {
268    if ( isNotDisposed() )
269    {
270      // Note that his state transition occurs outside the guard as it is required to compute isDisposed() state
271      _state = COMPONENT_DISPOSING;
272      if ( Arez.areNativeComponentsEnabled() )
273      {
274        assert null != _component;
275        _component.dispose();
276      }
277      else
278      {
279        getContext().safeAction( Arez.areNamesEnabled() ? getName() + ".dispose" : null,
280                                 this::performDispose,
281                                 ActionFlags.NO_VERIFY_ACTION_REQUIRED );
282      }
283      if ( Arez.shouldCheckApiInvariants() )
284      {
285        _state = COMPONENT_DISPOSED;
286      }
287    }
288  }
289
290  private void performDispose()
291  {
292    invokeCallbackIfNecessary( _preDisposeCallback );
293    releaseResources();
294    invokeCallbackIfNecessary( _disposeCallback );
295    invokeCallbackIfNecessary( _postDisposeCallback );
296  }
297
298  private void invokeCallbackIfNecessary( @Nullable final SafeProcedure callback )
299  {
300    if ( null != callback )
301    {
302      callback.call();
303    }
304  }
305
306  @Override
307  public boolean isDisposed()
308  {
309    return _state < 0;
310  }
311
312  private void releaseResources()
313  {
314    if ( null != _onDisposeListeners )
315    {
316      notifyOnDisposeListeners();
317    }
318    // If native components are enabled, these elements are registered with native component
319    // and will thus be disposed as part
320    if ( !Arez.areNativeComponentsEnabled() )
321    {
322      Disposable.dispose( _componentObservable );
323      Disposable.dispose( _disposeOnDeactivate );
324    }
325  }
326
327  /**
328   * Notify an OnDispose listeners that have been added to the component.
329   * This method MUST only be called if the component has enabled onDisposeNotify feature.
330   */
331  public void notifyOnDisposeListeners()
332  {
333    assert null != _onDisposeListeners;
334    for ( final Map.Entry<Object, SafeProcedure> entry : new ArrayList<>( _onDisposeListeners.entrySet() ) )
335    {
336      final Object key = entry.getKey();
337      /*
338       * There is scenarios where there is multiple elements being simultaneously disposed and
339       * the @CascadeDispose has not triggered so a disposed object is in this list waiting to
340       * be called back. If the callback is triggered and the @CascadeDispose is on an observable
341       * property then the framework will attempt to null field and generate invariant failures
342       * or runtime errors unless we skip the callback and just remove the listener.
343       */
344      if ( !Disposable.isDisposed( key ) )
345      {
346        entry.getValue().call();
347      }
348    }
349  }

447  @Nonnull
448  private String describeState( final int state )
449  {
450    assert Arez.shouldCheckInvariants() || Arez.shouldCheckApiInvariants();
451    switch ( state )
452    {
453      case ComponentKernel.COMPONENT_CREATED:
454        return "created";
455      case ComponentKernel.COMPONENT_INITIALIZED:
456        return "initialized";
457      case ComponentKernel.COMPONENT_CONSTRUCTED:
458        return "constructed";
459      case ComponentKernel.COMPONENT_COMPLETE:
460        return "complete";
461      case ComponentKernel.COMPONENT_READY:
462        return "ready";
463      case ComponentKernel.COMPONENT_DISPOSING:
464        return "disposing";
465      default:
466        assert ComponentKernel.COMPONENT_DISPOSED == state;
467        return "disposed";
468    }
469  }

471  /**
472   * Transition component state from {@link ComponentKernel#COMPONENT_INITIALIZED} to {@link ComponentKernel#COMPONENT_CONSTRUCTED}.
473   */
474  public void componentConstructed()
475  {
476    if ( Arez.shouldCheckApiInvariants() )
477    {
478      apiInvariant( () -> COMPONENT_INITIALIZED == _state,
479                    () -> "Arez-0219: Bad state transition from " + describeState( _state ) +
480                          " to " + describeState( COMPONENT_CONSTRUCTED ) +
481                          " on component named '" + getName() + "'." );
482      _state = COMPONENT_CONSTRUCTED;
483    }
484  }
485
486  /**
487   * Transition component state from {@link ComponentKernel#COMPONENT_INITIALIZED} to
488   * {@link ComponentKernel#COMPONENT_CONSTRUCTED} and then to {@link ComponentKernel#COMPONENT_READY}.
489   * This should only be called if there is active elements that are part of the component that need to be scheduled,
490   * otherwise the component can transition directly to ready.
491   */
492  public void componentComplete()
493  {
494    completeNativeComponent();
495    if ( Arez.shouldCheckApiInvariants() )
496    {
497      apiInvariant( () -> COMPONENT_CONSTRUCTED == _state,
498                    () -> "Arez-0220: Bad state transition from " + describeState( _state ) +
499                          " to " + describeState( COMPONENT_COMPLETE ) +
500                          " on component named '" + getName() + "'." );
501      _state = COMPONENT_COMPLETE;
502    }
503    // Trigger scheduler so active parts of components can react
504    getContext().triggerScheduler();
505    makeComponentReady();
506  }
507
508  /**
509   * Transition component state from {@link ComponentKernel#COMPONENT_CONSTRUCTED} to {@link ComponentKernel#COMPONENT_READY}.
510   * This should be invoked rather than {@link #componentComplete()} if there is no active elements of the component that
511   * need to be scheduled.
512   */
513  public void componentReady()
514  {
515    completeNativeComponent();
516    makeComponentReady();
517  }

519  /**
520   * Mark the native component if present as complete.
521   */
522  private void completeNativeComponent()
523  {
524    if ( Arez.areNativeComponentsEnabled() )
525    {
526      assert null != _component;
527      _component.complete();
528    }
529  }

531  private void makeComponentReady()
532  {
533    if ( Arez.shouldCheckApiInvariants() )
534    {
535      apiInvariant( () -> COMPONENT_CONSTRUCTED == _state || COMPONENT_COMPLETE == _state,
536                    () -> "Arez-0218: Bad state transition from " + describeState( _state ) +
537                          " to " + describeState( COMPONENT_READY ) +
538                          " on component named '" + getName() + "'." );
539      _state = COMPONENT_READY;
540    }
541  }

651  /**
652   * Add the listener to notify list under key.
653   * This method MUST NOT be invoked after {@link #dispose()} has been invoked.
654   * This method should not be invoked if another listener has been added with the same key without
655   * being removed.
656   *
657   * <p>If the key implements {@link Disposable} and {@link Disposable#isDisposed()} returns <code>true</code>
658   * when invoking the calback then the callback will be skipped. This rare situation only occurs when there is
659   * circular dependency in the object model usually involving {@link CascadeDispose}.</p>
660   *
661   * @param key    the key to uniquely identify listener.
662   * @param action the listener callback.
663   */
664  public void addOnDisposeListener( @Nonnull final Object key, @Nonnull final SafeProcedure action )
665  {
666    assert null != _onDisposeListeners;
667    if ( Arez.shouldCheckApiInvariants() )
668    {
669      invariant( this::isNotDisposed,
670                 () -> "Arez-0170: Attempting to add OnDispose listener but ComponentKernel has been disposed." );
671      invariant( () -> !_onDisposeListeners.containsKey( key ),
672                 () -> "Arez-0166: Attempting to add OnDispose listener with key '" + key +
673                       "' but a listener with that key already exists." );
674    }
675    _onDisposeListeners.put( key, action );
676  }

678  /**
679   * Remove the listener with specified key from the notify list.
680   * This method should only be invoked when a listener has been added for specific key using
681   * {@link #addOnDisposeListener(Object, SafeProcedure)} and has not been removed by another
682   * call to this method.
683   *
684   * @param key the key under which the listener was previously added.
685   */
686  public void removeOnDisposeListener( @Nonnull final Object key )
687  {
688    assert null != _onDisposeListeners;
689    // This method can be called when the notifier is disposed to avoid the caller (i.e. per-component
690    // generated code) from checking the disposed state of the notifier before invoking this method.
691    // This is necessary in a few rare circumstances but requiring the caller to check before invocation
692    // increases the generated code size.
693    final SafeProcedure removed = _onDisposeListeners.remove( key );
694    if ( Arez.shouldCheckApiInvariants() )
695    {
696      invariant( () -> null != removed,
697                 () -> "Arez-0167: Attempting to remove OnDispose listener with key '" + key +
698                       "' but no such listener exists." );
699    }
700  }
701
702  boolean hasOnDisposeListeners()
703  {
704    return null != _onDisposeListeners;
705  }
706
707  @Nonnull
708  Map<Object, SafeProcedure> getOnDisposeListeners()
709  {
710    assert null != _onDisposeListeners;
711    return _onDisposeListeners;
712  }
