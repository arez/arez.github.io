001package arez.component;
002
003import arez.Arez;
004import arez.ArezContext;
005import arez.Component;
006import arez.Disposable;
007import arez.Observer;
008import arez.Priority;
009import arez.annotations.ComponentNameRef;
010import arez.annotations.ComponentRef;
011import arez.annotations.ContextRef;
012import java.util.function.Consumer;
013import javax.annotation.Nonnull;
014
015/**
016 * Abstract base class for classes that are expected to clear references to components when the
017 * component is disposed.
018 */
019public abstract class AbstractEntryContainer<T>
020{
021  /**
022   * Create an EntityReference for specified entity.
023   * This class also sets up monitoring observer so that if the entity is disposed then the
024   * onDisposeAction is invoked. The action is expected to detach the reference and may potentially
025   * dispose the entity and/or remove the entry from a container.
026   *
027   * @param entity          the entity.
028   * @param onDisposeAction the action to take when the entity is disposed.
029   * @return the reference.
030   */
031  @Nonnull
032  final EntityEntry<T> createEntityEntry( @Nonnull final T entity,
033                                          @Nonnull final Consumer<EntityEntry<T>> onDisposeAction )
034  {
035    final EntityEntry<T> entry = new EntityEntry<>( entity );
036    final String name =
037      Arez.areNamesEnabled() ? getName() + ".EntityWatcher." + Identifiable.getArezId( entity ) : null;
038    final Observer monitor =
039      getContext().when( Arez.areNativeComponentsEnabled() ? component() : null,
040                         name,
041                         true,
042                         () -> ComponentObservable.notObserved( entity ),
043                         () -> onDisposeAction.accept( entry ),
044                         Priority.HIGH,
045                         true );
046    entry.setMonitor( monitor );
047    return entry;
048  }
049
050  /**
051   * Dispose the entry and the associated monitor and if shouldDisposeEntity is true, dispose the entity.
052   *
053   * @param entry               the entry to detach.
054   * @param shouldDisposeEntity true to also dispose the entity, false otherwise.
055   */
056  final void detachEntry( @Nonnull final EntityEntry<T> entry, final boolean shouldDisposeEntity )
057  {
058    final Observer monitor = entry.getMonitor();
059    if ( null != monitor )
060    {
061      Disposable.dispose( monitor );
062    }
063    if ( shouldDisposeEntity )
064    {
065      Disposable.dispose( entry.getEntity() );
066    }
067  }
068
069  /**
070   * Return the native component associated with the current component.
071   *
072   * @return the native component associated with the current component.
073   */
074  @ComponentRef
075  @Nonnull
076  protected abstract Component component();
077
078  /**
079   * Return the context associated with the current component.
080   *
081   * @return the context associated with the current component.
082   */
083  @ContextRef
084  @Nonnull
085  protected abstract ArezContext getContext();
086
087  /**
088   * Return the component name associated with the current component.
089   *
090   * @return the component name associated with the current component.
091   */
092  @ComponentNameRef
093  @Nonnull
094  protected abstract String getName();
095}
