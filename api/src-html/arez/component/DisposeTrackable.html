001package arez.component;
002
003import arez.Arez;
004import javax.annotation.Nonnull;
005import javax.annotation.Nullable;
006import static org.realityforge.braincheck.Guards.*;
007
008/**
009 * Interface implemented by a component if it supports notifying listeners
010 * when the component is disposed. This notification occurs using a call-back
011 * and occurs within the dispose transaction (after {@link arez.annotations.PreDispose}
012 * is invoked if present) using a callback. Contrast this with the strategy used by
013 * {@link ComponentObservable} which uses standard Arez observables to track when
014 * a component is disposed.
015 */
016public interface DisposeTrackable
017{
018  /**
019   * Return the notifier associated with the component.
020   *
021   * @return the notifier associated with the component.
022   */
023  @Nonnull
024  DisposeNotifier getNotifier();
025
026  /**
027   * Return the notifier associated with the supplied object if object implements DisposeTrackable.
028   *
029   * @param object the object.
030   * @return the notifier associated with the supplied object if object implements DisposeTrackable.
031   */
032  @Nullable
033  static DisposeNotifier getNotifier( @Nonnull final Object object )
034  {
035    return object instanceof DisposeTrackable ? ( (DisposeTrackable) object ).getNotifier() : null;
036  }
037
038  /**
039   * Cast the specified object to an instance of DisposeTrackable.
040   * Invariant checks will verify that the cast is valid before proceeding.
041   *
042   * @param object the object.
043   * @return the object cast to DisposeTrackable.
044   */
045  @Nonnull
046  static DisposeTrackable asDisposeTrackable( @Nonnull final Object object )
047  {
048    if ( Arez.shouldCheckApiInvariants() )
049    {
050      apiInvariant( () -> object instanceof DisposeTrackable,
051                    () -> "Arez-0178: Object passed to asDisposeTrackable does not implement " +
052                          "DisposeTrackable. Object: " + object );
053    }
054    return (DisposeTrackable) object;
055  }
056}
