001package arez.component;
002
003import javax.annotation.Nonnull;
004import javax.annotation.Nullable;
005
006/**
007 * The interface used to look up components by type and id.
008 * This is primarily used by components that represent entities that relate to other entities.
009 */
010public interface Locator
011{
012  /**
013   * Return the entity with the specified type and the specified id, throwing an exception if not present.
014   *
015   * @param <T>  the entity type.
016   * @param type the type of the entity.
017   * @param id   the id of the entity.
018   * @return the entity.
019   * @throws NoSuchEntityException if unable to locate entity.
020   */
021  @Nonnull
022  default <T> T getById( @Nonnull final Class<T> type, @Nonnull final Object id )
023    throws NoSuchEntityException
024  {
025    final T entity = findById( type, id );
026    if ( null == entity )
027    {
028      throw new NoSuchEntityException( id );
029    }
030    return entity;
031  }
032
033  /**
034   * Lookup the entity with the specified type and the specified id, returning null if not present.
035   *
036   * @param <T>  the entity type.
037   * @param type the type of the entity.
038   * @param id   the id of the entity.
039   * @return the entity or null if no such entity.
040   */
041  @Nullable
042  <T> T findById( @Nonnull Class<T> type, @Nonnull Object id );
043}
