001package arez.dom;
002
003import akasha.AddEventListenerOptions;
004import akasha.EventListener;
005import akasha.WindowGlobal;
006import akasha.TimerHandler;
007import arez.annotations.Action;
008import arez.annotations.ArezComponent;
009import arez.annotations.Feature;
010import arez.annotations.Memoize;
011import arez.annotations.Observable;
012import arez.annotations.OnActivate;
013import arez.annotations.OnDeactivate;
014import arez.annotations.PostConstruct;
015import arez.annotations.PreDispose;
016import java.util.Arrays;
017import java.util.HashSet;
018import java.util.Set;
019import javax.annotation.Nonnull;
020
021/**
022 * An Arez browser component that tracks when the user is idle. A user is considered idle if they have not
023 * interacted with the browser for a specified amount of time. The component declares state that tracks when
024 * the user is "idle". A user is considered idle if they have not interacted with the browser
025 * for a specified amount of time.
026 *
027 * <p>Application code can observe the idle state via accessing {@link #isIdle()}.
028 * Typically this is done in a tracking transaction such as those defined by autorun.</p>
029 *
030 * <p>The "amount of time" is defined by the Observable value "timeout" accessible via
031 * {@link #getTimeout()} and mutable via {@link #setTimeout(long)}.</p>
032 *
033 * <p>The "not interacted with the browser" is detected by listening for interaction
034 * events on the browser. The list of events that the model listens for is controlled via
035 * {@link #getEvents()} and {@link #setEvents(Set)}. It should be noted that if
036 * there is no observer observing the idle state then the model will remove listeners
037 * so as not to have any significant performance impact.</p>
038 *
039 * <h1>A very simple example</h1>
040 * <pre>{@code
041 * import com.google.gwt.core.client.EntryPoint;
042 * import akasha.Console;
043 * import arez.Arez;
044 * import arez.dom.IdleStatus;
045 *
046 * public class IdleStatusExample
047 *   implements EntryPoint
048 * {
049 *   public void onModuleLoad()
050 *   {
051 *     final IdleStatus idleStatus = IdleStatus.create();
052 *     Arez.context().autorun( () -> {
053 *       final String message = "Interaction Status: " + ( idleStatus.isIdle() ? "Idle" : "Active" );
054 *       Console.log( message );
055 *     } );
056 *   }
057 * }
058 * }</pre>
059 */
060@ArezComponent( requireId = Feature.DISABLE )
061public abstract class IdleStatus
062{
063  private static final long DEFAULT_TIMEOUT = 2000L;
064  @Nonnull
065  private final TimerHandler _timeoutCallback = this::onTimeout;
066  @Nonnull
067  private final EventListener _listener = e -> resetLastActivityTime();
068  @Nonnull
069  private Set<String> _events =
070    new HashSet<>( Arrays.asList( "keydown", "touchstart", "scroll", "mousemove", "mouseup", "mousedown", "wheel" ) );
071  /**
072   * True if an Observer is watching idle state.
073   */
074  private boolean _active;
075  /**
076   * The id of timeout scheduled action, 0 if none set.
077   */
078  private int _timeoutId;
079
080  /**
081   * Create an instance of this model.
082   *
083   * @return an instance of IdleStatus.
084   */
085  @Nonnull
086  public static IdleStatus create()
087  {
088    return create( DEFAULT_TIMEOUT );
089  }
090
091  /**
092   * Create an instance of this model.
093   *
094   * @param timeout the duration to after activity before becoming idle.
095   * @return an instance of IdleStatus.
096   */
097  @Nonnull
098  public static IdleStatus create( final long timeout )
099  {
100    return new Arez_IdleStatus( timeout );
101  }
102
103  IdleStatus()
104  {
105  }
106
107  @PostConstruct
108  void postConstruct()
109  {
110    resetLastActivityTime();
111  }
112
113  @PreDispose
114  void preDispose()
115  {
116    cancelTimeout();
117  }
118
119  /**
120   * Return true if the user is idle.
121   *
122   * @return true if the user is idle, false otherwise.
123   */
124  @Memoize
125  public boolean isIdle()
126  {
127    if ( isRawIdle() )
128    {
129      return true;
130    }
131    else
132    {
133      final int timeToWait = getTimeToWait();
134      if ( timeToWait > 0 )
135      {
136        if ( 0 == _timeoutId )
137        {
138          scheduleTimeout( timeToWait );
139        }
140        return false;
141      }
142      else
143      {
144        return true;
145      }
146    }
147  }
148
149  @OnActivate
150  void onIdleActivate()
151  {
152    _active = true;
153    _events.forEach( e -> WindowGlobal.addEventListener( e, _listener, AddEventListenerOptions.create().passive( true ) ) );
154  }
155
156  @OnDeactivate
157  void onIdleDeactivate()
158  {
159    _active = false;
160    _events.forEach( e -> WindowGlobal.removeEventListener( e, _listener ) );
161  }
162
163  /**
164   * Short cut observable field checked after idle state is confirmed.
165   */
166  @Observable
167  abstract void setRawIdle( boolean rawIdle );
168
169  abstract boolean isRawIdle();
170
171  /**
172   * Return the duration for which no events should be received for the idle condition to be triggered.
173   *
174   * @return the timeout.
175   */
176  @Observable( initializer = Feature.ENABLE )
177  public abstract long getTimeout();
178
179  /**
180   * Set the timeout.
181   *
182   * @param timeout the timeout.
183   */
184  public abstract void setTimeout( long timeout );
185
186  /**
187   * Return the set of events to listen to.
188   *
189   * @return the set of events.
190   */
191  @Nonnull
192  @Observable
193  public Set<String> getEvents()
194  {
195    return _events;
196  }
197
198  /**
199   * Specify the set of events to listen to.
200   * If the model is already active, the listeners will be updated to reflect the new events.
201   *
202   * @param events the set of events.
203   */
204  public void setEvents( @Nonnull final Set<String> events )
205  {
206    final Set<String> oldEvents = _events;
207    _events = new HashSet<>( events );
208    updateListeners( oldEvents );
209  }
210
211  /**
212   * Synchronize listeners against the dom based on new events.
213   */
214  private void updateListeners( @Nonnull final Set<String> oldEvents )
215  {
216    if ( _active )
217    {
218      //Remove any old events
219      oldEvents.stream().
220        filter( e -> !_events.contains( e ) ).
221        forEach( e -> WindowGlobal.removeEventListener( e, _listener ) );
222      // Add any new events
223      _events.stream().
224        filter( e -> !oldEvents.contains( e ) ).
225        forEach( e -> WindowGlobal.addEventListener( e, _listener ) );
226    }
227  }
228
229  /**
230   * Return the time at which the last monitored event was received.
231   *
232   * @return the time at which the last event was received.
233   */
234  @Observable
235  public abstract long getLastActivityAt();
236
237  abstract void setLastActivityAt( long lastActivityAt );
238
239  private int getTimeToWait()
240  {
241    return (int) ( getLastActivityAt() + getTimeout() - System.currentTimeMillis() );
242  }
243
244  private void cancelTimeout()
245  {
246    WindowGlobal.clearTimeout( _timeoutId );
247    _timeoutId = 0;
248  }
249
250  private void scheduleTimeout( final int timeToWait )
251  {
252    _timeoutId = WindowGlobal.setTimeout( _timeoutCallback, timeToWait );
253  }
254
255  @Action
256  void onTimeout()
257  {
258    _timeoutId = 0;
259    final int timeToWait = getTimeToWait();
260    if ( timeToWait > 0 )
261    {
262      scheduleTimeout( timeToWait );
263    }
264    else
265    {
266      setRawIdle( true );
267    }
268  }
269
270  @Action
271  void resetLastActivityTime()
272  {
273    setRawIdle( false );
274    setLastActivityAt( System.currentTimeMillis() );
275  }
276}
