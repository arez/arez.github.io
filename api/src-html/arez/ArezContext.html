209   *
210   * @param type        the component type.
211   * @param id          the component id.
212   * @param name        the name of the component. Should be null if {@link Arez#areNamesEnabled()} returns false.
213   * @param preDispose  the hook action called just before the Component is disposed. The hook method is called from within the dispose transaction.
214   * @param postDispose the hook action called just after the Component is disposed. The hook method is called from within the dispose transaction.
215   * @return true if component is defined in context.
216   */
217  @Nonnull
218  public Component component( @Nonnull final String type,
219                              @Nonnull final Object id,
220                              @Nullable final String name,
221                              @Nullable final SafeProcedure preDispose,
222                              @Nullable final SafeProcedure postDispose )
223  {
224    if ( Arez.shouldCheckApiInvariants() )
225    {
226      apiInvariant( Arez::areNativeComponentsEnabled,
227                    () -> "Arez-0008: ArezContext.component() invoked when Arez.areNativeComponentsEnabled() returns false." );
228    }
229    final HashMap<Object, Component> map = getComponentByTypeMap( type );
230    if ( Arez.shouldCheckApiInvariants() )
231    {
232      apiInvariant( () -> !map.containsKey( id ),
233                    () -> "Arez-0009: ArezContext.component() invoked for type '" + type + "' and id '" +
234                          id + "' but a component already exists for specified type+id." );
235    }
236    final Component component =
237      new Component( Arez.areZonesEnabled() ? this : null, type, id, name, preDispose, postDispose );
238    map.put( id, component );
239    if ( willPropagateSpyEvents() )
240    {
241      getSpy().reportSpyEvent( new ComponentCreateStartedEvent( getSpy().asComponentInfo( component ) ) );
242    }
243    return component;
244  }

246  /**
247   * Invoked by the component during it's dispose to release resources associated with the component.
248   *
249   * @param component the component.
250   */
251  void deregisterComponent( @Nonnull final Component component )
252  {
253    if ( Arez.shouldCheckInvariants() )
254    {
255      invariant( Arez::areNativeComponentsEnabled,
256                 () -> "Arez-0006: ArezContext.deregisterComponent() invoked when Arez.areNativeComponentsEnabled() returns false." );
257    }
258    final String type = component.getType();
259    final HashMap<Object, Component> map = getComponentByTypeMap( type );
260    final Component removed = map.remove( component.getId() );
261    if ( Arez.shouldCheckInvariants() )
262    {
263      invariant( () -> component == removed,
264                 () -> "Arez-0007: ArezContext.deregisterComponent() invoked for '" + component + "' but was " +
265                       "unable to remove specified component from registry. Actual component removed: " + removed );
266    }
267    if ( map.isEmpty() )
268    {
269      assert _components != null;
270      _components.remove( type );
271    }
272  }

1140  /**
1141   * Method invoked to trigger the scheduler to run any pending reactions. The scheduler will only be
1142   * triggered if there is no transaction active. This method is typically used after one or more Observers
1143   * have been created outside a transaction with the runImmediately flag set to false and the caller wants
1144   * to force the observers to react. Otherwise the Observers will not be schedule until the next top-level
1145   * transaction completes. Pending reactions are run in the environment specified by {@link #setEnvironment(Environment)}
1146   * if any has been specified.
1147   */
1148  public void triggerScheduler()
1149  {
1150    if ( isSchedulerEnabled() && !isSchedulerPaused() )
1151    {
1152      // Each reaction creates a top level transaction that attempts to run call
1153      // this method when it completes. Rather than allow this if it is detected
1154      // that we are running reactions already then just abort and assume the top
1155      // most invocation of runPendingTasks will handle scheduling
1156      if ( !_schedulerActive )
1157      {
1158        _schedulerActive = true;
1159        try
1160        {
1161          if ( Arez.areEnvironmentsEnabled() && null != _environment )
1162          {
1163            // The environment wrapper can perform actions that trigger the need for the Arez
1164            // scheduler to re-run so we keep checking until there is no more work to be done.
1165            // This is typically used when the environment reacts to changes that Arez triggered
1166            // (i.e. via @Track callbacks) which in turn reschedules Arez changes. This happens
1167            // in frameworks like react4j which have only scheduler that responds to changes and
1168            // feeds back into Arez.
1169            do
1170            {
1171              safeRunInEnvironment( safeProcedureToFunction( _scheduler::runPendingTasks ) );
1172            }
1173            while ( _scheduler.hasTasksToSchedule() );
1174          }
1175          else
1176          {
1177            _scheduler.runPendingTasks();
1178          }
1179        }
1180        finally
1181        {
1182          _schedulerActive = false;
1183        }
1184      }
1185    }
1186  }

1787                             @Nonnull final SafeFunction<T> executable,
1788                             final int flags,
1789                             @Nullable final Observer observer,
1790                             @Nullable final Object[] parameters,
1791                             final boolean expectResult )
1792  {
1793    final String name = generateName( "Action", specifiedName );
1794
1795    verifyActionFlags( name, flags );
1796
1797    final boolean observed = null != observer;
1798    Throwable t = null;
1799    boolean completed = false;
1800    long startedAt = 0L;
1801    T result;
1802    try
1803    {
1804      if ( willPropagateSpyEvents() )
1805      {
1806        startedAt = System.currentTimeMillis();
1807        reportActionStarted( name, parameters, observed );
1808      }

1980                                         final int flags,
1981                                         final Transaction transaction )
1982  {
1983    if ( Arez.shouldCheckInvariants() )
1984    {
1985      if ( null == observer )
1986      {
1987        verifyActionRequired( transaction, flags );
1988      }
1989      else if ( Flags.AREZ_DEPENDENCIES == ( flags & Flags.AREZ_DEPENDENCIES ) )

1995                         ( null != observableValues && !observableValues.isEmpty() ),
1996                   () -> "Arez-0118: Observer named '" + name + "' completed observed function (executed by " +
1997                         "application) but is not observing any properties." );
1998      }
1999    }
2000  }
