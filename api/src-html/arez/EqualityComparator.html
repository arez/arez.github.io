001package arez;
002
003import java.util.Objects;
004import javax.annotation.Nullable;
005
006/**
007 * Functional interface for detecting if two values are equal.
008 *
009 * @param <T> The type of the values checked.
010 */
011@FunctionalInterface
012public interface EqualityComparator<T>
013{
014  /**
015   * The default comparator.
016   */
017  EqualityComparator<?> DEFAULT_COMPARATOR = Objects::equals;
018
019  /**
020   * Return a default equality comparator.
021   *
022   * @param <T> the type to compare.
023   * @return the comparator.
024   */
025  @SuppressWarnings( "unchecked" )
026  static <T> EqualityComparator<T> defaultComparator()
027  {
028    return (EqualityComparator<T>) DEFAULT_COMPARATOR;
029  }
030
031  /**
032   * Return true if values are considered to be equal, false otherwise.
033   *
034   * @param value1 the first value.
035   * @param value2 the second value.
036   * @return true if values are considered to be equal, false otherwise.
037   */
038  boolean equals( @Nullable T value1, @Nullable T value2 );
039}
