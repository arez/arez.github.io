003import arez.spy.ComponentCreateCompleteEvent;
004import arez.spy.ComponentDisposeCompleteEvent;
005import arez.spy.ComponentDisposeStartEvent;
006import arez.spy.ComponentInfo;

013/**
014 * The component is an abstraction representation of a reactive component within Arez.
015 * Each component is made up of one or more of the core Arez reactive elements: {@link ObservableValue}s,
016 * {@link Observer}s or {@link ComputableValue}s.
017 */
018public final class Component
019  implements Disposable
020{
021  /**
022   * Reference to the system to which this node belongs.
023   */
024  @Nullable
025  private final ArezContext _context;
026  /**
027   * A opaque string describing the type of the component.
028   * It corresponds to @ArezComponent.name parameter if this component was built using the annotation processor.
029   */
030  @Nonnull
031  private final String _type;
032  /**
033   * The if of the component. This should be null for singletons and non-null for non-singletons.
034   */
035  @Nonnull
036  private final Object _id;
037  /**
038   * A human consumable name for node. It should be non-null if {@link Arez#areNamesEnabled()} returns
039   * true and <tt>null</tt> otherwise.
040   */
041  @Nullable
042  private final String _name;
043  private final ArrayList<ObservableValue<?>> _observableValues = new ArrayList<>();
044  private final ArrayList<Observer> _observers = new ArrayList<>();
045  private final ArrayList<ComputableValue<?>> _computableValues = new ArrayList<>();
046  /**
047   * Hook action called just before the Component is disposed.
048   * Occurs inside the dispose transaction.
049   */
050  @Nullable
051  private final SafeProcedure _preDispose;
052  /**
053   * Hook action called just after the Component is disposed.
054   */
055  @Nullable
056  private final SafeProcedure _postDispose;
057  private boolean _complete;
058  private boolean _disposed;
059  /**
060   * Cached info object associated with element.
061   * This should be null if {@link Arez#areSpiesEnabled()} is false;
062   */
063  @Nullable
064  private ComponentInfo _info;
065
066  Component( @Nullable final ArezContext context,
067             @Nonnull final String type,
068             @Nonnull final Object id,
069             @Nullable final String name,
070             @Nullable final SafeProcedure preDispose,
071             @Nullable final SafeProcedure postDispose )
072  {
073    if ( Arez.shouldCheckApiInvariants() )
074    {
075      apiInvariant( () -> Arez.areNamesEnabled() || null == name,
076                    () -> "Arez-0037: Component passed a name '" + name + "' but Arez.areNamesEnabled() is false" );
077      invariant( () -> Arez.areZonesEnabled() || null == context,
078                 () -> "Arez-0175: Component passed a context but Arez.areZonesEnabled() is false" );
079    }
080    _context = Arez.areZonesEnabled() ? Objects.requireNonNull( context ) : null;
081    _type = Objects.requireNonNull( type );
082    _id = Objects.requireNonNull( id );
083    _name = Arez.areNamesEnabled() ? Objects.requireNonNull( name ) : null;
084    _preDispose = preDispose;
085    _postDispose = postDispose;
086  }
087
088  /**
089   * Return the component type.
090   * This is an opaque string specified by the user.
091   *
092   * @return the component type.
093   */
094  @Nonnull
095  public String getType()
096  {
097    return _type;
098  }
099
100  /**
101   * Return the unique id of the component.
102   * This will return null for singletons.
103   *
104   * @return the unique id of the component.
105   */
106  @Nonnull
107  public Object getId()
108  {
109    return _id;
110  }
111
112  /**
113   * Return the unique name of the component.
114   *
115   * @return the name of the component.
116   */
117  @Nonnull
118  public String getName()
119  {
120    if ( Arez.shouldCheckApiInvariants() )
121    {
122      apiInvariant( Arez::areNamesEnabled,
123                    () -> "Arez-0038: Component.getName() invoked when Arez.areNamesEnabled() is false" );
124    }
125    assert null != _name;
126    return _name;
127  }
128
129  @Nonnull
130  ArezContext getContext()
131  {
132    return Arez.areZonesEnabled() ? Objects.requireNonNull( _context ) : Arez.context();
133  }
134
135  /**
136   * {@inheritDoc}
137   */
138  @Override
139  public void dispose()
140  {
141    if ( !_disposed )
142    {
143      _disposed = true;
144      if ( Arez.areSpiesEnabled() && getContext().getSpy().willPropagateSpyEvents() )
145      {
146        final ComponentInfo info = getContext().getSpy().asComponentInfo( this );
147        getContext().getSpy().reportSpyEvent( new ComponentDisposeStartEvent( info ) );
148      }
149      getContext().safeAction( Arez.areNamesEnabled() ? getName() + ".dispose" : null, () -> {
150        if ( null != _preDispose )
151        {
152          _preDispose.call();
153        }
154        getContext().deregisterComponent( this );
155        /*
156         * Create a new list and perform dispose on each list to avoid concurrent mutation exceptions.
157         * This can probably be significantly optimized when translated to javascript. However native
158         * components are not typically used in production mode so no effort has been made to optimize
159         * the next steps.
160         */
161        new ArrayList<>( _observers ).forEach( o -> Disposable.dispose( o ) );
162        new ArrayList<>( _computableValues ).forEach( v -> Disposable.dispose( v ) );
163        new ArrayList<>( _observableValues ).forEach( o -> Disposable.dispose( o ) );
164        if ( null != _postDispose )
165        {
166          _postDispose.call();
167        }
168      }, Flags.NO_VERIFY_ACTION_REQUIRED );
169      if ( Arez.areSpiesEnabled() && getContext().getSpy().willPropagateSpyEvents() )
170      {
171        final ComponentInfo info = getContext().getSpy().asComponentInfo( this );
172        getContext().getSpy().reportSpyEvent( new ComponentDisposeCompleteEvent( info ) );
173      }
174    }
175  }
176
177  /**
178   * {@inheritDoc}
179   */
180  @Override
181  public boolean isDisposed()
182  {
183    return _disposed;
184  }
185
186  /**
187   * {@inheritDoc}
188   */
189  @Nonnull
190  @Override
191  public final String toString()
192  {
193    if ( Arez.areNamesEnabled() )
194    {
195      return getName();
196    }
197    else
198    {
199      return super.toString();
200    }
201  }
202
203  /**
204   * Return true if the creation of this component is complete.
205   *
206   * @return true if the creation of this component is complete, false otherwise.
207   */
208  public boolean isComplete()
209  {
210    return _complete;
211  }
212
213  /**
214   * The toolkit user should call this method when the component is complete.
215   * After this method has been invoked the user should not attempt to define any more {@link ObservableValue}s,
216   * {@link Observer}s or {@link ComputableValue}s on the component.
217   */
218  public void complete()
219  {
220    if ( !_complete )
221    {
222      _complete = true;
223      if ( Arez.areSpiesEnabled() && getContext().getSpy().willPropagateSpyEvents() )
224      {
225        final ComponentInfo component = getContext().getSpy().asComponentInfo( this );
226        getContext().getSpy().reportSpyEvent( new ComponentCreateCompleteEvent( component ) );
227      }
228    }
229  }
230
231  /**
232   * Return the info associated with this class.
233   *
234   * @return the info associated with this class.
235   */
236  @SuppressWarnings( "ConstantConditions" )
237  @Nonnull
238  ComponentInfo asInfo()
239  {
240    if ( Arez.shouldCheckInvariants() )
241    {
242      invariant( Arez::areSpiesEnabled,
243                 () -> "Arez-0194: Component.asInfo() invoked but Arez.areSpiesEnabled() returned false." );
244    }
245    if ( Arez.areSpiesEnabled() && null == _info )
246    {
247      _info = new ComponentInfoImpl( this );
248    }
249    return Arez.areSpiesEnabled() ? _info : null;
250  }
251
252  /**
253   * Return the observers associated with the component.
254   *
255   * @return the observers associated with the component.
256   */
257  @Nonnull
258  ArrayList<Observer> getObservers()
259  {
260    return _observers;
261  }
262
263  /**
264   * Add observer to component.
265   * Observer should not be part of observer.
266   *
267   * @param observer the observer.
268   */
269  void addObserver( @Nonnull final Observer observer )
270  {
271    if ( Arez.shouldCheckApiInvariants() )
272    {
273      apiInvariant( () -> !_observers.contains( observer ),
274                    () -> "Arez-0040: Component.addObserver invoked on component '" + getName() + "' specifying " +
275                          "observer named '" + observer.getName() + "' when observer already exists for component." );
276    }
277    _observers.add( observer );
278  }
279
280  /**
281   * Remove observer from the component.
282   * Observer should be part of component.
283   *
284   * @param observer the observer.
285   */
286  void removeObserver( @Nonnull final Observer observer )
287  {
288    if ( Arez.shouldCheckApiInvariants() )
289    {
290      apiInvariant( () -> _observers.contains( observer ),
291                    () -> "Arez-0041: Component.removeObserver invoked on component '" + getName() + "' specifying " +
292                          "observer named '" + observer.getName() + "' when observer does not exist for component." );
293    }
294    _observers.remove( observer );
295  }

297  /**
298   * Return the observables associated with the component.
299   *
300   * @return the observables associated with the component.
301   */
302  @Nonnull
303  ArrayList<ObservableValue<?>> getObservableValues()
304  {
305    return _observableValues;
306  }
307
308  /**
309   * Add observableValue to component.
310   * ObservableValue should not be part of component.
311   *
312   * @param observableValue the observableValue.
313   */
314  void addObservableValue( @Nonnull final ObservableValue observableValue )
315  {
316    if ( Arez.shouldCheckApiInvariants() )
317    {
318      apiInvariant( () -> !_complete,
319                    () -> "Arez-0042: Component.addObservableValue invoked on component '" +
320                          getName() +
321                          "' " +
322                          "specifying ObservableValue named '" +
323                          observableValue.getName() +
324                          "' when component.complete() " +
325                          "has already been called." );
326      apiInvariant( () -> !_observableValues.contains( observableValue ),
327                    () -> "Arez-0043: Component.addObservableValue invoked on component '" +
328                          getName() +
329                          "' " +
330                          "specifying ObservableValue named '" +
331                          observableValue.getName() +
332                          "' when ObservableValue already " +
333                          "exists for component." );
334    }
335    _observableValues.add( observableValue );
336  }
337
338  /**
339   * Remove observableValue from the component.
340   * ObservableValue should be part of component.
341   *
342   * @param observableValue the observableValue.
343   */
344  void removeObservableValue( @Nonnull final ObservableValue observableValue )
345  {
346    if ( Arez.shouldCheckApiInvariants() )
347    {
348      apiInvariant( () -> _observableValues.contains( observableValue ),
349                    () -> "Arez-0044: Component.removeObservableValue invoked on component '" +
350                          getName() +
351                          "' " +
352                          "specifying ObservableValue named '" +
353                          observableValue.getName() +
354                          "' when ObservableValue does not " +
355                          "exist for component." );
356    }
357    _observableValues.remove( observableValue );
358  }
