001package arez.spy;
002
003import java.util.Map;
004import java.util.Objects;
005import javax.annotation.Nonnull;
006import javax.annotation.Nullable;
007
008/**
009 * Notification when Action completes.
010 */
011public final class ActionCompletedEvent
012  implements SerializableEvent
013{
014  public static final String TYPE_NAME = EventUtil.getName( ActionCompletedEvent.class );
015
016  @Nonnull
017  private final String _name;
018  /**
019   * Is the action a "tracking" action.
020   */
021  private final boolean _tracked;
022  @Nonnull
023  private final Object[] _parameters;
024  /**
025   * True if the action returns a result during normal completion.
026   */
027  private final boolean _returnsResult;
028  @Nullable
029  private final Object _result;
030  @Nullable
031  private final Throwable _throwable;
032  private final long _duration;
033
034  public ActionCompletedEvent( @Nonnull final String name,
035                               final boolean tracked,
036                               @Nonnull final Object[] parameters,
037                               final boolean returnsResult,
038                               @Nullable final Object result,
039                               @Nullable final Throwable throwable,
040                               final long duration )
041  {
042    assert duration >= 0;
043    assert null == throwable || null == result;
044    _name = Objects.requireNonNull( name );
045    _tracked = tracked;
046    _parameters = Objects.requireNonNull( parameters );
047    _returnsResult = returnsResult;
048    _result = result;
049    _throwable = throwable;
050    _duration = duration;
051  }
052
053  @Nonnull
054  public String getName()
055  {
056    return _name;
057  }
058
059  public boolean isTracked()
060  {
061    return _tracked;
062  }
063
064  @Nonnull
065  public Object[] getParameters()
066  {
067    return _parameters;
068  }
069
070  public boolean returnsResult()
071  {
072    return _returnsResult;
073  }
074
075  @Nullable
076  public Object getResult()
077  {
078    return _result;
079  }
080
081  @Nullable
082  public Throwable getThrowable()
083  {
084    return _throwable;
085  }
086
087  public long getDuration()
088  {
089    return _duration;
090  }
091
092  /**
093   * {@inheritDoc}
094   */
095  @Override
096  public void toMap( @Nonnull final Map<String, Object> map )
097  {
098    map.put( "type", TYPE_NAME );
099    map.put( "action", getName() );
100    map.put( "tracked", isTracked() );
101    map.put( "duration", getDuration() );
102    final Throwable throwable = getThrowable();
103    map.put( "normalCompletion", null == throwable );
104    final String message =
105      null == throwable ? null : null == throwable.getMessage() ? throwable.toString() : throwable.getMessage();
106    map.put( "errorMessage", message );
107    map.put( "parameters", getParameters() );
108    map.put( "returnsResult", returnsResult() );
109    map.put( "result", getResult() );
110  }
111}
