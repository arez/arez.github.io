001package arez.spy;
002
003import arez.Arez;
004import arez.Priority;
005import java.util.List;
006import javax.annotation.Nonnull;
007import javax.annotation.Nullable;
008
009/**
010 * A representation of a ComputedValue instance exposed to spy framework.
011 */
012public interface ComputedValueInfo
013  extends ElementInfo
014{
015  /**
016   * Return true if the ComputedValue is "computing".
017   * This implies that the current transaction or one of the parent transactions is calculating the
018   * ComputedValue at the moment.
019   *
020   * @return true if there is a transaction active.
021   * @see arez.Spy#isComputing(arez.ComputedValue)
022   */
023  boolean isComputing();
024
025  /**
026   * Return the priority of the ComputedValue.
027   *
028   * @return the priority of the ComputedValue.
029   */
030  @Nonnull
031  Priority getPriority();
032
033  /**
034   * Return true if the ComputedValue is active.
035   * A ComputedValue is active if there is one or more Observers and the value will be calculated.
036   *
037   * @return true if the ComputedValue is active.
038   * @see arez.Spy#isActive(arez.ComputedValue)
039   */
040  boolean isActive();
041
042  /**
043   * Return the list of observers for ComputedValue.
044   * The list is an immutable copy of the observers of the {@link arez.ComputedValue}.
045   *
046   * @return the list of observers for ComputedValue.
047   * @see arez.Spy#getObservers(arez.ComputedValue)
048   */
049  @Nonnull
050  List<ObserverInfo> getObservers();
051
052  /**
053   * Return the list of dependencies of the ComputedValue.
054   * The list is an immutable copy of the dependencies of the {@link arez.ComputedValue}.
055   * If the {@link arez.ComputedValue} is currently being computed (i.e. {@link #isComputing()}
056   * returns true) then the dependencies are provisional and may be added to as transaction
057   * completes.
058   *
059   * @return the list of dependencies for ComputedValue.
060   * @see arez.Spy#getDependencies(arez.ComputedValue)
061   */
062  @Nonnull
063  List<ObservableValueInfo> getDependencies();
064
065  /**
066   * Return the component for the ComputedValue.
067   * This method should not be invoked if {@link Arez#areNativeComponentsEnabled()} returns false.
068   *
069   * @return the component that contains ComputedValue if any.
070   * @see arez.Spy#getComponent(arez.ComputedValue)
071   */
072  @Nullable
073  ComponentInfo getComponent();
074
075  /**
076   * Return the value of the ComputedValue.
077   * This method should only be invoked if {@link Arez#arePropertyIntrospectorsEnabled()} returns true.
078   *
079   * @return the value of the ComputedValue.
080   * @throws Throwable if the property accessor throws an exception.
081   * @see arez.Spy#getValue(arez.ComputedValue)
082   */
083  @Nullable
084  Object getValue()
085    throws Throwable;
086}
