001package arez.annotations;
002
003/**
004 * Enumeration that describes the types of dependencies supported by an {@link arez.Observer} or a {@link arez.ComputedValue}.
005 */
006public enum DepType
007{
008  /**
009   * The reactive element MUST have at least one dependency on an {@link arez.ObservableValue}.
010   * This means that the <code>computed</code> or <code>observed</code> method must invoke
011   * {@link arez.ObservableValue#reportObserved()} at least once within the scope of the method.
012   */
013  AREZ,
014  /**
015   * The reactive element may have dependencies on zero or more {@link arez.ObservableValue} instances.
016   * Using this dependency type also allows for the scenario where the <code>computed</code> or
017   * <code>observed</code> does not invoke {@link arez.ObservableValue#reportObserved()} within the scope
018   * of the method. The <code>computed</code> or <code>observed</code> will not be re-scheduled by the
019   * runtime when there are no dependencies. (This is sometimes acceptable, particularly during when a
020   * collection of components are being disposed over multiple scheduling rounds).
021   */
022  AREZ_OR_NONE,
023  /**
024   * The reactive element may have zero or more dependencies on {@link arez.ObservableValue} instances and/or
025   * may have dependencies on non-arez elements. The application can invoke {@link arez.Observer#reportStale()}
026   * or {@link arez.ComputedValue#reportPossiblyChanged()} to indicate non-arez dependency has changed.
027   */
028  AREZ_OR_EXTERNAL
029}
