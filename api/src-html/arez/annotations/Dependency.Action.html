001package arez.annotations;
002
003import java.lang.annotation.Documented;
004import java.lang.annotation.ElementType;
005import java.lang.annotation.Target;
006import javax.annotation.Nonnull;
007
008/**
009 * Methods marked with this annotation are dependencies of the component.
010 * THe component observes the dependency and if the dependency is disposed takes
011 * an action to cascade the dispose or null the property referencing dependency.
012 *
013 * <p>The method that is annotated with @Dependency must comply with the additional constraints:</p>
014 * <ul>
015 * <li>Must not be annotated with any other arez annotation except {@link Observable} or {@link Computed}</li>
016 * <li>Must have 0 parameters</li>
017 * <li>Must return a value</li>
018 * <li>Must not be private</li>
019 * <li>Must not be static</li>
020 * <li>Must not be abstract</li>
021 * <li>Must not throw exceptions</li>
022 * </ul>
023 */
024@Documented
025@Target( ElementType.METHOD )
026public @interface Dependency
027{
028  /**
029   * The action to take when dependency is disposed.
030   */
031  enum Action
032  {
033    /**
034     * Remove the component.
035     */
036    CASCADE,
037    /**
038     * Set the {@link Observable} property that holds dependency to null.
039     */
040    SET_NULL
041  }
042
043  /**
044   * Return the action to take when the dependency is disposed.
045   * A {@link Action#CASCADE} value indicates that the component should be deleted, while a
046   * {@link Action#SET_NULL} value indicates that the observable field should be set to null.
047   * The {@link Action#SET_NULL} is only valid on {@link Observable} properties that have an
048   * associated setter and are not annotated with {@link javax.annotation.Nonnull}.
049   *
050   * @return the action to take when the dependency is disposed.
051   */
052  @Nonnull
053  Action action() default Action.CASCADE;
054}
