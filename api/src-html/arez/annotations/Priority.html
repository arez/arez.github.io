001package arez.annotations;
002
003/**
004 * Enum to control scheduling priority of observers/reactions.
005 * Observers with higher priorities will react first. If observers have equal priorities then observers
006 * scheduled first will react first. Observers must not depend upon ComputedValue instances with
007 * a lower priority otherwise priority is ignored.
008 *
009 * <p>A user should be very careful when specifying a {@link #HIGH} priority as it is possible that
010 * the the reaction will be scheduled part way through the process of disposing and/or unlinking one-or-more
011 * components. Dispose reactions will often be scheduled with a higher priority but reactions unlinking disposed
012 * arez components from remaining arez components. In many cases this may mean invoking
013 * <code>Disposable.isDisposed(component)</code> before accessing arez components.</p>
014 */
015public enum Priority
016{
017  /**
018   * Highest user controllable priority.
019   * This priority should be used when the reaction will dispose other reactive elements (and thus they
020   * need not be scheduled) or if the reaction will trigger many downstream reactions.
021   */
022  HIGH,
023  /**
024   * Normal priority if not otherwise specified.
025   */
026  NORMAL,
027  /**
028   * Lowest priority.
029   */
030  LOW
031}
