001package arez.annotations;
002
003import arez.ComputedValue;
004import arez.Priority;
005import java.lang.annotation.Documented;
006import java.lang.annotation.ElementType;
007import java.lang.annotation.Target;
008import javax.annotation.Nonnull;
009
010/**
011 * Methods marked with this annotation are ComputedValues within Arez.
012 *
013 * <p>The return value should be derived from other Observables within the Arez system
014 * and the value returned by the method should not change unless the state of the other
015 * {@link Observable}s change. The method is wrapped in a READ_ONLY transaction and
016 * thus can not modify other state in the system.</p>
017 *
018 * <p>The method that is annotated with @Computed must comply with the additional constraints:</p>
019 * <ul>
020 * <li>Must not be annotated with any other arez annotation</li>
021 * <li>Must have 0 parameters</li>
022 * <li>Must return a value</li>
023 * <li>Must not be private</li>
024 * <li>Must not be static</li>
025 * <li>Must not be final</li>
026 * <li>Must not be abstract</li>
027 * <li>Must not throw exceptions</li>
028 * </ul>
029 */
030@Documented
031@Target( ElementType.METHOD )
032public @interface Computed
033{
034  /**
035   * Return the name of the ComputedValue relative to the component.
036   * The value must conform to the requirements of a java identifier.
037   * The name must also be unique across {@link Observable}s,
038   * {@link Computed}s and {@link Action}s within the scope of the
039   * {@link ArezComponent} annotated element.
040   *
041   * @return the name of the ComputedValue relative to the component.
042   */
043  @Nonnull
044  String name() default "<default>";
045
046  /**
047   * A flag indicating whether the computed should be "kept alive". A computed that is kept alive
048   * is activated on creation and never deactivates. This is useful if the computed property is only
049   * accessed from within actions but should be kept up to date and not recomputed on each access.
050   *
051   * @return true to keep computed alive.
052   */
053  boolean keepAlive() default false;
054
055  /**
056   * The priority of the underlying ComputedValue observer
057   *
058   * @return the priority of the ComputedValue observer.
059   */
060  Priority priority() default Priority.NORMAL;
061
062  /**
063   * Flag controlling whether the underlying observer can observe ComputedValue instances with lower priorities.
064   * The default value of false will result in an invariant failure (in development mode) if a lower priority
065   * dependency is observed by the observer. This is to prevent priority inversion when scheduling a higher
066   * priority observer is dependent upon a lower priority computed value. If the value is true then the no
067   * invariant failure is triggered and the component relies on the component author to handle possible priority
068   * inversion.
069   *
070   * @return false if observing lower priority dependencies should result in invariant failure in development mode.
071   */
072  boolean observeLowerPriorityDependencies() default false;
073
074  /**
075   * Flag indicating whether the value of the computed is solely dependent on arez elements.
076   * If set to true then Arez will verify that the function annotated by this accesses arez elements
077   * (i.e. instances of {@link arez.ObservableValue} or instances of {@link arez.ComputedValue}). If set
078   * to false then the component must define a {@link ComputedValueRef} method and should invoke
079   * {@link ComputedValue#reportPossiblyChanged()} when the non-arez dependencies are changed.
080   *
081   * @return true if the computed is solely dependent on arez elements.
082   */
083  boolean arezOnlyDependencies() default true;
084}
