001package arez.annotations;
002
003import java.lang.annotation.Documented;
004import java.lang.annotation.ElementType;
005import java.lang.annotation.Target;
006
007/**
008 * Annotate the method that will be overridden to provide the "name" of the Arez component.
009 * This is useful when the user wants to manually create Arez elements (i.e. {@link arez.ObservableValue} instances,
010 * {@link arez.Observer} instances or {@link arez.ComputableValue} instances etc) and wants to use the same naming
011 * convention as the generated Arez subclass.
012 *
013 * <p>This annotation should appear at most once on a component. The
014 * annotation should be on a method that accepts no parameters and returns
015 * a String.</p>
016 *
017 * <p>The method that is annotated with the annotation must comply with the additional constraints:</p>
018 * <ul>
019 * <li>Must not be annotated with any other arez annotation</li>
020 * <li>Must have 0 parameters</li>
021 * <li>Must return a String</li>
022 * <li>Must not be private</li>
023 * <li>Must not be static</li>
024 * <li>Must not be final</li>
025 * <li>Must be abstract</li>
026 * <li>Must not throw exceptions</li>
027 * <li>The method must be accessible to the component subclass which means it must not be package access unless it is in the same package as the arez component.</li>
028 * </ul>
029 *
030 * @see ComponentId
031 * @see ComponentTypeNameRef
032 */
033@Documented
034@Target( ElementType.METHOD )
035public @interface ComponentNameRef
036{
037}
