001package arez.annotations;
002
003import java.lang.annotation.Documented;
004import java.lang.annotation.ElementType;
005import java.lang.annotation.Target;
006import javax.annotation.Nonnull;
007
008/**
009 * Methods marked with this annotation are invoked in an Arez transaction.
010 *
011 * <p>The method that is annotated with @Action must comply with the additional constraints:</p>
012 * <ul>
013 * <li>Must not be annotated with any other arez annotation</li>
014 * <li>Must not be private</li>
015 * <li>Must not be static</li>
016 * <li>Must not be final</li>
017 * <li>Must not be abstract</li>
018 * </ul>
019 */
020@Documented
021@Target( ElementType.METHOD )
022public @interface Action
023{
024  /**
025   * Return the name of the Action relative to the component.
026   * The value must conform to the requirements of a java identifier.
027   * The name must also be unique across {@link Observable}s,
028   * {@link Computed}s and {@link Action}s within the scope of the
029   * {@link ArezComponent} annotated element.
030   *
031   * @return the name of the Action relative to the component.
032   */
033  @Nonnull
034  String name() default "<default>";
035
036  /**
037   * Does the action mutate state or not.
038   *
039   * @return true if method should be wrapped in READ_WRITE transaction, false if it should it should be wrapped in READ_ONLY transaction.
040   */
041  boolean mutation() default true;
042
043  /**
044   * Return true if the parameters should be reported to the core Arez runtime.
045   * It is useful to disable reporting for large, circular or just uninteresting parameters to the spy infrastructure.
046   *
047   * @return true to report the parameters, false otherwise.
048   */
049  boolean reportParameters() default true;
050
051  /**
052   * True if the action should always start a new transaction. A false value indicates that the action will
053   * use the invoking transaction if present, otherwise will create a new transaction to invoke action.
054   *
055   * @return true if the action will create a new transaction, false if it will use the existing transaction if present.
056   */
057  boolean requireNewTransaction() default false;
058
059  /**
060   * Flag indicating whether the code should verify that at least one read or write occurs within
061   * the scope of the action.
062   *
063   * @return true to verify action reads or writes observable data.
064   */
065  boolean verifyRequired() default true;
066}
