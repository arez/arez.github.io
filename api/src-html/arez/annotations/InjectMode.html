001package arez.annotations;
002
003/**
004 * Enum that controls how the component interacts with the injection framework if at all.
005 *
006 * <ul>
007 * <li>
008 * A component configured with {@link #PROVIDE} and either has no {@link PostConstruct} annotated method
009 * and no {@link Observe} annotated methods and no {@link Memoize} with keep alive set to <code>true</code>
010 * will have the <code>javax.inject.Inject</code> annotation added to the constructor and a Dagger2 module
011 * created named <code>[MyComponent]DaggerModule</code> which must be added to the desired component.
012 * </li>
013 * </ul>
014 */
015public enum InjectMode
016{
017  /**
018   * InjectMode is disabled. The component can not be created by the injection framework.
019   */
020  NONE,
021  /**
022   * The component must be created and injected by the injection framework. The component MUST have a single
023   * constructor otherwise the injection framework will not know how to create the component. The component
024   * may be injected into other components managed by the injection framework.
025   */
026  PROVIDE,
027  /**
028   * Allow the annotation processor to determine whether injection should be enabled.
029   * If the class has an annotation that is itself annotated with the <code>javax.inject.Scope</code>
030   * annotation then the component will be assumed to be {@link #PROVIDE} otherwise the value {@link #NONE}
031   * is assumed.
032   */
033  AUTODETECT
034}
