001package arez.annotations;
002
003import java.lang.annotation.Documented;
004import java.lang.annotation.ElementType;
005import java.lang.annotation.Target;
006import javax.annotation.Nonnull;
007
008/**
009 * Methods marked with this annotation are <a href="https://en.wikipedia.org/wiki/Memoization">memoized</a> while there is an observer.
010 *
011 * <p>The return value should be derived from the parameters and any other Observables or {@link arez.ComputableValue}s
012 * within the Arez system. The value returned by the method should not change unless the parameters or the
013 * state of the other {@link Observable}s change. The method is wrapped in a READ_ONLY transaction and
014 * thus can not modify other state in the system.</p>
015 *
016 * <p>This is implemented by creating a separate ComputableValue instance for each unique combination of
017 * parameters. When the ComputableValue is deactivated, a hook triggers that removes the ComputableValue
018 * from the local cache.</p>
019 *
020 * <p>The method that is annotated with @Memoize must comply with the additional constraints:</p>
021 * <ul>
022 * <li>Must not be annotated with any other arez annotation</li>
023 * <li>Must have 1 or more parameters</li>
024 * <li>Must return a value</li>
025 * <li>Must not be private</li>
026 * <li>Must not be static</li>
027 * <li>Must not be final</li>
028 * <li>Must not be abstract</li>
029 * <li>Must not throw exceptions</li>
030 * </ul>
031 */
032@Documented
033@Target( ElementType.METHOD )
034public @interface Memoize
035{
036  /**
037   * Return the root name of the Memoized value relative to the component. This
038   * will be used in combination with a sequence when naming the synthesized
039   * ComputableValue instances. The value must conform to the requirements of a
040   * java identifier. The name must also be unique across {@link Observable}s,
041   * {@link Computed}s and {@link Action}s within the scope of the
042   * {@link ArezComponent} annotated element.
043   *
044   * @return the root name of the Memoized value relative to the component.
045   */
046  @Nonnull
047  String name() default "<default>";
048
049  /**
050   * The priority of the underlying ComputableValue observer
051   *
052   * @return the priority of the ComputableValue observer.
053   */
054  Priority priority() default Priority.NORMAL;
055
056  /**
057   * Flag controlling whether the underlying observer can observe ComputableValue instances with lower priorities.
058   * The default value of false will result in an invariant failure (in development mode) if a lower priority
059   * dependency is observed by the observer. This is to prevent priority inversion when scheduling a higher
060   * priority observer is dependent upon a lower priority computable value. If the value is true then the no
061   * invariant failure is triggered and the component relies on the component author to handle possible priority
062   * inversion.
063   *
064   * @return false if observing lower priority dependencies should result in invariant failure in development mode.
065   */
066  boolean observeLowerPriorityDependencies() default false;
067
068  /**
069   * True if invocations of the underlying computable must be executed in an {@link arez.Environment}.
070   *
071   * @return true if invocations of the underlying computable must be executed in an {@link arez.Environment}, false otherwise.
072   */
073  boolean requireEnvironment() default false;
074
075  /**
076   * Enum indicating whether the value of the underlying computable is derived from arez elements and/or
077   * external dependencies. The current implementation does not support setting this to {@link DepType#AREZ_OR_EXTERNAL}.
078   *
079   * @return the types of dependencies allowed on the underlying computable.
080   * @see Computed#depType()
081   */
082  @Nonnull
083  DepType depType() default DepType.AREZ;
084}
