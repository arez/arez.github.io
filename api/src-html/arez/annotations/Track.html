001package arez.annotations;
002
003import java.lang.annotation.Documented;
004import java.lang.annotation.ElementType;
005import java.lang.annotation.Target;
006import javax.annotation.Nonnull;
007
008/**
009 * Methods marked with this annotation are run in tracking transactions.
010 * Any Observables or ComputedValues accessed within the scope of the method will be added
011 * as a dependency of the tracker. If any of these dependencies are changed then class will
012 * be notified by way of a "OnDepsChanged" method. The method can be explicitly marked with the
013 * {@link OnDepsChanged} annotation or the framework will attempt to use the method named according
014 * to naming conventions. i.e. If the @Track annotated method is named "render" then the @OnDepsChanged
015 * annotated method will have default to being named "onRenderDepsUpdate".
016 *
017 * <p>The method that is annotated with @Track must comply with the additional constraints:</p>
018 * <ul>
019 * <li>Must not be annotated with any other arez annotation</li>
020 * <li>Must not be private</li>
021 * <li>Must not be static</li>
022 * <li>Must not be final</li>
023 * <li>Must not be abstract</li>
024 * </ul>
025 */
026@Documented
027@Target( ElementType.METHOD )
028public @interface Track
029{
030  /**
031   * Return the name of the Tracked relative to the component.
032   * The value must conform to the requirements of a java identifier.
033   * The name must also be unique across {@link Observable}s,
034   * {@link Computed}s and {@link Action}s within the scope of the
035   * {@link ArezComponent} annotated element.
036   *
037   * @return the name of the Tracked relative to the component.
038   */
039  @Nonnull
040  String name() default "<default>";
041
042  /**
043   * Does the tracker mutate state or not.
044   * The default value us false thus making the transaction mode read-only. The intention is that tracker observers
045   * should primarily be reflecting Arez state to external systems. (i.e. views, network layers etc.)
046   *
047   * @return true if the tracker should run in a READ_WRITE transaction, false if it should run in a READ_ONLY transaction.
048   */
049  boolean mutation() default false;
050
051  /**
052   * Return true if the parameters should be reported to the core Arez runtime.
053   * It is useful to disable reporting for large, circular or just uninteresting parameters to the spy infrastructure.
054   *
055   * @return true to report the parameters, false otherwise.
056   */
057  boolean reportParameters() default true;
058
059  /**
060   * The priority of the underlying track observer
061   *
062   * @return the priority of the track observer.
063   */
064  Priority priority() default Priority.NORMAL;
065
066  /**
067   * Flag controlling whether the underlying observer can observe ComputedValue instances with lower priorities.
068   * The default value of false will result in an invariant failure (in development mode) if a lower priority
069   * dependency is observed by the observer. This is to prevent priority inversion when scheduling a higher
070   * priority observer is dependent upon a lower priority computed value. If the value is true then the no
071   * invariant failure is triggered and the component relies on the component author to handle possible priority
072   * inversion.
073   *
074   * @return false if observing lower priority dependencies should result in invariant failure in development mode.
075   */
076  boolean observeLowerPriorityDependencies() default false;
077}
