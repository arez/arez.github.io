001package arez.annotations;
002
003import java.lang.annotation.Documented;
004import java.lang.annotation.ElementType;
005import java.lang.annotation.Target;
006import javax.annotation.Nonnull;
007import org.realityforge.anodoc.Unsupported;
008
009/**
010 * Annotation that indicates that a repository must be generated for component.
011 * A repository is the default mechanism for managing instances of classes
012 * annotated with {@link ArezComponent}. The repository provides mechanisms to
013 * create a component, lookup a component by id or using a query and dispose
014 * the instance. These activities can be done by the user through other means
015 * but the repositories provide an easy solution.
016 *
017 * <p>This annotation can only be added to classes that have been annotated with the
018 * {@link ArezComponent} annotation.</p>
019 *
020 * <p>Annotating a class with this annotation will result in a class named "[MyComponent]Repository"</p>
021 *
022 * <p>The way to add custom queries or \@Computed queries is to define an interface that
023 * defines an abstract <code>self()</code> that returns the underlying repository. Using this
024 * combined with default methods, you can define as many new queries and mutations as is desired.
025 * The extension class then needs to be registered by setting the appropriate parameter on this
026 * annotation.</p>
027 *
028 * <p>An example of what an extension may look like for a <code>Todo</code> component. See below:</p>
029 *
030 * <pre>{@code
031 * public interface MyTodoRepositoryExtension
032 * {
033 *   default Todo findByTitle( final String title )
034 *   {
035 *     return self().findByQuery( todo -> todo.getTitle().equals( title ) );
036 *   }
037 *
038 *   \@Computed
039 *   default List<Todo> findAllCompleted()
040 *   {
041 *     return self().findAllByQuery( Todo::isCompleted );
042 *   }
043 *
044 *   MyTodoRepository self();
045 * }
046 * }</pre>
047 */
048@Documented
049@Target( ElementType.TYPE )
050@Unsupported( "This may change in the future as it is incorporated with replicant and the wider ecosystem" )
051public @interface Repository
052{
053  /**
054   * Return the name of the repository.
055   * The value must conform to the requirements of a java identifier.
056   * The default value if not specified is the name of the associated ArezComponent suffixed with "Repository".
057   *
058   * @return the name of the repository.
059   */
060  @Nonnull
061  String name() default "<default>";
062
063  /**
064   * Return the list of extension interfaces that the repository will implement.
065   * The extension interfaces should not define any non-default methods besides
066   * self().
067   *
068   * @return the list of extension interfaces that the repository will implement.
069   */
070  Class[] extensions() default {};
071
072  /**
073   * Indicate whether a dagger module should be generated for repository.
074   * {@link Feature#ENABLE} will force the generation of the module, {@link Feature#DISABLE}
075   * will result in no dagger module and {@link Feature#AUTODETECT} will add a dagger
076   * module if the <code>dagger.Module</code> class is present on the classpath.
077   *
078   * @return enum controlling whether a dagger module should be generated for repository.
079   */
080  Feature dagger() default Feature.AUTODETECT;
081
082  /**
083   * Indicate whether an @Inject annotation should be added to constructor of the generated repository.
084   * {@link Feature#ENABLE} will force the addition of an @Inject annotation, {@link Feature#DISABLE}
085   * will result in no @Inject annotation and {@link Feature#AUTODETECT} will add an @Inject
086   * if any it is present on the classpath. Note that this is effectively {@link Feature#ENABLE} if
087   * dagger parameter is true.
088   *
089   * @return enum controlling present of Inject annotation on constructor of repository.
090   */
091  Feature inject() default Feature.AUTODETECT;
092}
