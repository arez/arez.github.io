001package arez.annotations;
002
003import java.lang.annotation.Documented;
004import java.lang.annotation.ElementType;
005import java.lang.annotation.Target;
006
007/**
008 * Annotation added to a field to indicate that the field should be disposed when the component is disposed.
009 * The field will be disposed after the {@link PreDispose} method.
010 *
011 * <p>The field that is annotated with the annotation must comply with the additional constraints:</p>
012 * <ul>
013 * <li>Must not be private</li>
014 * <li>Must not be static</li>
015 * <li>The type of the field must implement {@link arez.Disposable} or must be annotated by {@link ArezComponent}</li>
016 * <li>The field must be accessible to the component subclass which means it must be protected of package access if it is in the same package as the arez component.</li>
017 * </ul>
018 */
019@Documented
020@Target( { ElementType.FIELD } )
021public @interface CascadeDispose
022{
023}
