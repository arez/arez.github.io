001package arez.annotations;
002
003import java.lang.annotation.Documented;
004import java.lang.annotation.ElementType;
005import java.lang.annotation.Target;
006
007/**
008 * Annotation that can be applied to constructor parameters to indicate the must be passed in when
009 * creating the instance. This annotation is only useful if using an injection framework to construct
010 * the component and the the {@link ArezComponent#inject()} parameter is set to {@link InjectMode#CONSUME}.
011 *
012 * <p>If a parameter is annotated with this then the annotation processor will generate a static subclass
013 * named <code>Factory</code> in the enhanced component class that is responsible for creating the component
014 * instance that has a create method that accepts all parameters that are marked using this annotation.</p>
015 */
016@Documented
017@Target( ElementType.PARAMETER )
018public @interface PerInstance
019{
020}
