001package arez.annotations;
002
003import arez.Priority;
004import java.lang.annotation.Documented;
005import java.lang.annotation.ElementType;
006import java.lang.annotation.Target;
007import javax.annotation.Nonnull;
008
009/**
010 * Methods marked with this annotation are converted to autoruns.
011 * Autorun methods are invoked by the Arez runtime and should never be invoked directly by user code.
012 *
013 * <p>The method that is annotated with @Autorun must comply with the additional constraints:</p>
014 * <ul>
015 * <li>Must not be annotated with any other arez annotation</li>
016 * <li>Must not be private</li>
017 * <li>Must not be public</li>
018 * <li>Must not be static</li>
019 * <li>Must not be final</li>
020 * <li>Must not be abstract</li>
021 * </ul>
022 */
023@Documented
024@Target( ElementType.METHOD )
025public @interface Autorun
026{
027  /**
028   * Return the name of the Autorun relative to the component.
029   * The value must conform to the requirements of a java identifier.
030   * The name must also be unique across {@link Observable}s,
031   * {@link Computed}s and {@link Action}s within the scope of the
032   * {@link ArezComponent} annotated element.
033   *
034   * @return the name of the Autorun relative to the component.
035   */
036  @Nonnull
037  String name() default "<default>";
038
039  /**
040   * Does the autorun mutate state or not.
041   * The default value us false thus making the transaction mode read-only. The intention is that autorun observers
042   * should primarily be reflecting Arez state to external systems. (i.e. views, network layers etc.)
043   *
044   * @return true if the autorun should run in a READ_WRITE transaction, false if it should run in a READ_ONLY transaction.
045   */
046  boolean mutation() default false;
047
048  /**
049   * The priority of the underlying autorun observer
050   *
051   * @return the priority of the autorun observer.
052   */
053  Priority priority() default Priority.NORMAL;
054
055  /**
056   * Flag controlling whether the underlying observer can observe ComputedValue instances with lower priorities.
057   * The default value of false will result in an invariant failure (in development mode) if a lower priority
058   * dependency is observed by the observer. This is to prevent priority inversion when scheduling a higher
059   * priority observer is dependent upon a lower priority computed value. If the value is true then the no
060   * invariant failure is triggered and the component relies on the component author to handle possible priority
061   * inversion.
062   *
063   * @return false if observing lower priority dependencies should result in invariant failure in development mode.
064   */
065  boolean observeLowerPriorityDependencies() default false;
066
067  /**
068   * Flag controlling whether the observer can invoke actions.
069   * An action that specifies {@link Action#requireNewTransaction()} as true will start a new transaction
070   * and any observables accessed within the action will not be dependencies of the observer. Sometimes this
071   * behaviour is desired. Sometimes an action that specifies {@link Action#requireNewTransaction()} as false
072   * will be used instead and any observable accessed within the scope of the action will be a dependency of
073   * the observer and thus changes in the observable will reschedule the observer. Sometimes this
074   * behaviour is desired. Either way the developer must be conscious of these decisions and thus must explicitly
075   * set this flag to true to invoke any actions within the scope of the observers reaction.
076   *
077   * @return true if the observer can invoke actions, false otherwise.
078   */
079  boolean canNestActions() default false;
080}
