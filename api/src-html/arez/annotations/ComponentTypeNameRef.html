001package arez.annotations;
002
003import java.lang.annotation.Documented;
004import java.lang.annotation.ElementType;
005import java.lang.annotation.Target;
006
007/**
008 * Annotate the method that will be overridden to return the value of ArezComponent.name().
009 * This is only useful when a class potentially has multiple {@link ArezComponent} annotated subclasses.
010 * This method is used when manually setting up debug context and you want it to align with the concrete
011 * implementations. However often {@link ComponentNameRef} is a better solution. If not specified Arez will
012 * generate a private method if needed.
013 *
014 * <p>This annotation should appear at most once on a component. The annotation should be on a method that
015 * can be overridden, accepts no parameters and returns a String.</p>
016 *
017 * <p>The method that is annotated with this annotation must comply with the additional constraints:</p>
018 * <ul>
019 * <li>Must not be annotated with any other arez annotation</li>
020 * <li>Must have 0 parameters</li>
021 * <li>Must return a String</li>
022 * <li>Must be abstract</li>
023 * <li>Must not throw exceptions</li>
024 * <li>Must be accessible to the class annotated by the {@link ArezComponent} annotation.</li>
025 * </ul>
026 */
027@Documented
028@Target( ElementType.METHOD )
029public @interface ComponentTypeNameRef
030{
031}
