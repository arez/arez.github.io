001package arez.annotations;
002
003import arez.Arez;
004import java.lang.annotation.Documented;
005import java.lang.annotation.ElementType;
006import java.lang.annotation.Target;
007import javax.annotation.Nonnull;
008
009/**
010 * Annotation that marks classes to be processed by Arez annotation processor.
011 * Classes with this annotation can contain {@link Observable} properties,
012 * {@link Computed} properties, {@link Autorun} methods, {@link Track} methods
013 * and {@link Action} methods.
014 *
015 * <p>The annotation controls the way that contained actions and observables are
016 * named (if names are enabled in the system.</p>
017 * <ul>
018 * <li>The value returned by {@link #name()} indicates the type name for instances
019 * of this object. If not specified it will default to the SimpleName of the class.
020 * i.e. The class <tt>com.biz.models.MyModel</tt> will default to a name of
021 * "MyModel".</li>
022 * <li>The {@link #nameIncludesId()} method indicates whether the names in generated components
023 * should include the id. If you expect only one instance of this component, it can be simpler
024 * to elide the id.</li>
025 * </ul>
026 * <p>The name of any elements contained within the component follows the pattern
027 * "<tt>[ArezComponent.name].[ArezComponent.id].[Element.name]</tt>". If the value of {@link #nameIncludesId()}
028 * is false then the "<tt>[ArezComponent.id].</tt>" element of the name will be elided.</p>
029 *
030 * <p>The type that is annotated with <tt>@ArezComponent</tt> annotation must comply with the additional constraints:</p>
031 * <ul>
032 * <li>Must be a class, not an interface or enum</li>
033 * <li>Must be concrete, not abstract</li>
034 * <li>Must not be final</li>
035 * <li>Must not be a non-static nested class</li>
036 * <li>Must have at least one method annotated with {@link Action}, {@link Autorun}, {@link Track}, {@link Computed} or {@link Observable}</li>
037 * </ul>
038 *
039 * <p>The annotation processor that handles this annotation will analyze all super classes and super
040 * interfaces. This includes analysis of default methods on interfaces. So it is perfectly valid to
041 * add {@link Observable}, {@link Computed}, {@link Action}, {@link PreDispose} and {@link PostDispose}
042 * annotations to default methods on implemented interfaces.</p>
043 */
044@Documented
045@Target( ElementType.TYPE )
046public @interface ArezComponent
047{
048  /**
049   * Return the name of the type.
050   * The value must conform to the requirements of a java identifier.
051   *
052   * @return the name of the type.
053   */
054  @Nonnull
055  String name() default "<default>";
056
057  /**
058   * Return true if the name derived for component should include the id, false otherwise.
059   * If the user does not specify a value for this parameter but does specify the <tt>javax.inject.Singleton</tt>
060   * on the same type then the default value of this parameter is false, otherwise it is true.
061   *
062   * @return true to include the id in the component name, false otherwise.
063   */
064  boolean nameIncludesId() default true;
065
066  /**
067   * Return true if the component does not need to explicitly declare elements.
068   * Otherwise if no elements (i.e. {@link Observable}s, {@link Action}s, {@link Autorun}s etc) are defined
069   * on a component it will generate an error.
070   *
071   * @return true if the component does not need to explicitly declare elements, false otherwise.
072   */
073  boolean allowEmpty() default false;
074
075  /**
076   * Return true if the component type does not need to be abstract.
077   * Otherwise an error will be generated if this annotation is added to a concrete type.
078   *
079   * @return true if the component type does not need to be abstract.
080   */
081  boolean allowConcrete() default false;
082
083  /**
084   * Return the enum to control whether the component should support being "observed" by
085   * {@link arez.component.ComponentObservable#observe(Object)}.
086   * {@link Feature#ENABLE} will force the implementation of the ComponentObservable interface,
087   * {@link Feature#DISABLE} will result in not implementing the ComponentObservable interface and
088   * {@link Feature#AUTODETECT} will cause the component to implement the interface if the component
089   * is also annotated with the {@link Repository} annotation or if the {@link #disposeOnDeactivate()}
090   * parameter is true.
091   *
092   * @return enum to control whether the component should support being "observed".
093   */
094  Feature observable() default Feature.AUTODETECT;
095
096  /**
097   * Return enum to control whether the component should support implement {@link arez.component.DisposeTrackable}.
098   * This will result in the component invoking dispose listener callbacks during dispose operation
099   * within the scope of the disposing transaction.
100   *
101   * <p>{@link Feature#ENABLE} will force the implementation of the DisposeTrackable interface,
102   * {@link Feature#DISABLE} will result in not implementing the DisposeTrackable interface and
103   * {@link Feature#AUTODETECT} will cause the component to implement the interface if the component
104   * is not annotated with <tt>javax.inject.Singleton</tt>.</p>
105   *
106   * @return Return enum to control whether the component should support implement DisposeTrackable.
107   */
108  Feature disposeTrackable() default Feature.AUTODETECT;
109
110  /**
111   * Return true if the component should dispose itself once it is no longer "observed".
112   * By "observed" it means that the component will have {@link arez.component.ComponentObservable#observe(Object)}
113   * called on it. This parameter MUST be false if {@link #observable()} has the value {@link Feature#DISABLE}.
114   *
115   * @return true if the component should dispose itself once it is no longer "observed".
116   */
117  boolean disposeOnDeactivate() default false;
118
119  /**
120   * Indicate whether an @Inject annotation should be added to constructor of the generated class.
121   * {@link Feature#ENABLE} will force the addition of an @Inject annotation, {@link Feature#DISABLE}
122   * will result in no @Inject annotation and {@link Feature#AUTODETECT} will add an @Inject
123   * if any fields or methods in the component or any parent type has an @Inject annotation OR the class
124   * has an annotation that is itself annotated with the javax.inject.Scope annotation.
125   *
126   * @return enum controlling present of Inject annotation on constructor.
127   */
128  Feature inject() default Feature.AUTODETECT;
129
130  /**
131   * Indicate whether a dagger module should be generated for enhanced component class.
132   * {@link Feature#ENABLE} will force the generation of the module, {@link Feature#DISABLE}
133   * will result in no dagger module and {@link Feature#AUTODETECT} will add a dagger
134   * module if the <code>dagger.Module</code> class is present on the classpath AND the class
135   * has an annotation that is itself annotated with the javax.inject.Scope annotation.
136   *
137   * @return enum controlling whether a dagger module should be generated for repository.
138   */
139  Feature dagger() default Feature.AUTODETECT;
140
141  /**
142   * Return true if an the generated component should NOT schedule autorun actions at the end of the
143   * constructor. This is useful if the component creator will trigger schedules manually at a later time.
144   * This MUST be false if there is no autorun methods annotated on the component.
145   *
146   * @return Return true if an the generated component should NOT schedule autorun actions at the end of the constructor.
147   */
148  boolean deferSchedule() default false;
149
150  /**
151   * Indicate whether a component requires that the {@link Object#hashCode()} and {@link Object#equals(Object)}
152   * methods are implemented. These methods MUST be implemented if the {@link Repository} annotation is present
153   * but may be implemented in other scenarios. {@link Feature#ENABLE} will force the generation of the methods,
154   * {@link Feature#DISABLE} will not generate these methods and {@link Feature#AUTODETECT} will generated these
155   * methods if the {@link Repository} annotation is present.
156   *
157   * @return enum controlling whether the {@link Object#hashCode()} and {@link Object#equals(Object)} methods are implemented.
158   */
159  Feature requireEquals() default Feature.AUTODETECT;
160
161  /**
162   * Indicates whether the component should support access of the id via {@link arez.component.Identifiable#getArezId(Object)}.
163   * This must be supported if the {@link Repository} annotation is present but may be supported in other scenarios.
164   * This feature is present on all components if {@link Arez#areNativeComponentsEnabled()} is true,
165   * {@link Arez#areRegistriesEnabled()} is true or {@link Arez#areNamesEnabled()} is true and {@link #nameIncludesId()}
166   * is true. In other scenarios it can be disabled. This is usually done to decrease code-size.
167   * {@link Feature#ENABLE} will require support, {@link Feature#DISABLE} will remove support and {@link Feature#AUTODETECT}
168   * will add support if the {@link Repository} annotation is present or a method annotated with {@link ComponentId}
169   * is present. If the {@link Repository} annotation is present or a method annotated with {@link ComponentId}
170   * is present then it is an error to set the value to {@link Feature#DISABLE}.
171   *
172   * @return enum controlling whether a unique if of the component can be accessed via {@link arez.component.Identifiable#getArezId(Object)}.
173   */
174  Feature requireId() default Feature.AUTODETECT;
175
176  /**
177   * Indicates whether the component should implement the interface {@link arez.component.Verifiable}.
178   * This feature is ignored unless {@link Arez#isVerifyEnabled()} is true. The {@link Feature#AUTODETECT}
179   * value will enable this feature if the component has any {@link Reference} methods or any {@link Inverse} methods.
180   *
181   * @return enum that indicates whether the component should implement the interface {@link arez.component.Verifiable}.
182   */
183  Feature verify() default Feature.AUTODETECT;
184}
