001package arez.annotations;
002
003import arez.Arez;
004import arez.component.DisposeNotifier;
005import java.lang.annotation.Documented;
006import java.lang.annotation.ElementType;
007import java.lang.annotation.Target;
008import javax.annotation.Nonnull;
009
010/**
011 * Annotation that marks classes or interfaces to be processed by Arez annotation processor.
012 * Classes or interfaces with this annotation can contain {@link Observable} properties,
013 * {@link Memoize} properties, {@link Observe} methods and {@link Action} methods.
014 *
015 * <p>The annotation controls the way that contained actions and observables are
016 * named (if names are enabled in the system.</p>
017 * <ul>
018 * <li>The value returned by {@link #name()} indicates the type name for instances
019 * of this object. If not specified it will default to the SimpleName of the class.
020 * i.e. The class <tt>com.biz.models.MyModel</tt> will default to a name of
021 * "MyModel".</li>
022 * </ul>
023 * <p>The name of any elements contained within the component follows the pattern
024 * "<tt>[ArezComponent.name].[ArezComponent.id].[Element.name]</tt>".</p>
025 *
026 * <p>The type that is annotated with this annotation must comply with the additional constraints:</p>
027 * <ul>
028 * <li>Must be a class or an interface</li>
029 * <li>Must be abstract</li>
030 * <li>Must not be final</li>
031 * <li>Must not be a non-static nested class</li>
032 * <li>Must have at least one method annotated with {@link Action}, {@link Observe}, {@link Memoize} or {@link Observable}</li>
033 * </ul>
034 *
035 * <p>The annotation processor that handles this annotation will analyze all super classes and super
036 * interfaces. This includes analysis of default methods on interfaces. So it is perfectly valid to
037 * add annotations such as {@link Observable}, {@link Memoize}, {@link Action}, {@link PreDispose} and/or
038 * {@link PostDispose} to default methods on implemented interfaces.</p>
039 */
040@Documented
041@Target( ElementType.TYPE )
042public @interface ArezComponent
043{
044  /**
045   * Return the name of the type.
046   * The value must conform to the requirements of a java identifier.
047   *
048   * @return the name of the type.
049   */
050  @Nonnull
051  String name() default "<default>";
052
053  /**
054   * Return true if the component does not need to explicitly declare elements.
055   * Otherwise if no elements (i.e. {@link Observable}s, {@link Action}s, {@link Observe}s etc) are defined
056   * on a component it will generate an error.
057   *
058   * @return true if the component does not need to explicitly declare elements, false otherwise.
059   */
060  boolean allowEmpty() default false;
061
062  /**
063   * An enum controlling whether the component is treated like a service or an entity.
064   * A service is expected to be a long running component that exists based on the functionality of
065   * the application while an entity represents data within the application and may come and go based
066   * on changes in the application data. Within arez, the only practical effect is to change the default
067   * behaviour of other features. See the documentation for other parameters for further details.
068   *
069   * <p>If set as {@link Feature#AUTODETECT} or left as the default value, then Arez will assumes that if
070   * the component is injected using dependency injection, then the component is a service. Arez detects whether
071   * the type is annotated by the <a href="https://sting-ioc.github.io/">sting</a> annotations {@code sting.Named},
072   * {@code sting.Typed} and {@code sting.Eager} or the jsr330 annotation {@code javax.inject.Named} or any
073   * annotation that is annotated with {@code javax.inject.Scope} annotation. If such an annotation is found then
074   * the component defaults to being treated like a service.</p>
075   *
076   * @return an enum controlling whether the component is treated like a service or an entity.
077   */
078  Feature service() default Feature.AUTODETECT;
079
080  /**
081   * Return the enum to control whether the component should support being "observed" by
082   * {@link arez.component.ComponentObservable#observe(Object)}.
083   * {@link Feature#ENABLE} will force the implementation of the ComponentObservable interface,
084   * {@link Feature#DISABLE} will result in not implementing the ComponentObservable interface and
085   * {@link Feature#AUTODETECT} will cause the component to implement the interface if the component
086   * is also annotated with the {@link Repository} annotation or if the {@link #disposeOnDeactivate()}
087   * parameter is true.
088   *
089   * @return enum to control whether the component should support being "observed".
090   */
091  Feature observable() default Feature.AUTODETECT;
092
093  /**
094   * Return enum to control whether the component should support implement the {@link DisposeNotifier} interface.
095   * This will result in the component invoking dispose listener callbacks during dispose operation
096   * within the scope of the disposing transaction.
097   *
098   * <p>If the value of this parameter is {@link Feature#AUTODETECT} then the component will implement the
099   * interface {@link DisposeNotifier} interface unless the component is annotated with
100   * <tt>javax.inject.Singleton</tt>.</p>
101   *
102   * @return Return enum to control whether the component should implement DisposeNotifier.
103   */
104  Feature disposeNotifier() default Feature.AUTODETECT;
105
106  /**
107   * Return true if the component should dispose itself once it is no longer "observed".
108   * By "observed" it means that the component will have {@link arez.component.ComponentObservable#observe(Object)}
109   * called on it. This parameter MUST be false if {@link #observable()} has the value {@link Feature#DISABLE}.
110   *
111   * @return true if the component should dispose itself once it is no longer "observed".
112   */
113  boolean disposeOnDeactivate() default false;
114
115  /**
116   * Specify how the component is integrated into the injection framework.
117   *
118   * @return enum controlling the integration into the injection framework.
119   * @see InjectMode
120   */
121  InjectMode inject() default InjectMode.AUTODETECT;
122
123  /**
124   * Indicate whether dagger artifacts should be generated to support injection.
125   * {@link Feature#ENABLE} will force the generation of the artifacts, {@link Feature#DISABLE}
126   * will result in no dagger artifacts and {@link Feature#AUTODETECT} will add a dagger
127   * module if the <code>dagger.Module</code> class is present on the classpath and the {@link #service()}
128   * parameter resolves to {@link Feature#ENABLE}.
129   *
130   * @return an enum controlling whether a dagger module should be generated.
131   */
132  Feature dagger() default Feature.AUTODETECT;
133
134  /**
135   * Return true if an the generated component should NOT trigger scheduler at the end of the constructor.
136   * This is useful if the component creator will trigger schedules manually at a later time.
137   * This MUST be false if there is no observed methods annotated on the component.
138   *
139   * @return return true if an the generated component should NOT trigger scheduler at the end of the constructor.
140   */
141  boolean deferSchedule() default false;
142
143  /**
144   * Indicate whether a component requires that the {@link Object#hashCode()} and {@link Object#equals(Object)}
145   * methods are implemented. These methods MUST be implemented if the {@link Repository} annotation is present
146   * but may be implemented in other scenarios. {@link Feature#ENABLE} will force the generation of the methods,
147   * {@link Feature#DISABLE} will not generate these methods and {@link Feature#AUTODETECT} will generated these
148   * methods if the {@link Repository} annotation is present.
149   *
150   * @return enum controlling whether the {@link Object#hashCode()} and {@link Object#equals(Object)} methods are implemented.
151   */
152  Feature requireEquals() default Feature.AUTODETECT;
153
154  /**
155   * Indicates whether the component should support access of the id via {@link arez.component.Identifiable#getArezId(Object)}.
156   * This feature must be enabled and will be enabled when the value of the parameter is {@link Feature#AUTODETECT}
157   * in the following scenarios:
158   *
159   * <ul>
160   * <li>the {@link Repository} annotation is present on the same type.</li>
161   * <li>a method annotated with the {@link Inverse} annotation is present.</li>
162   * <li>a method annotated with the {@link ComponentId} annotation is present.</li>
163   * <li>a method annotated with the {@link ComponentIdRef} annotation is present.</li>
164   * </ul>
165   *
166   * <p>The feature is also enabled on every components if
167   * {@link Arez#areNativeComponentsEnabled()} is true, {@link Arez#areRegistriesEnabled()} is true or
168   * {@link Arez#areNamesEnabled()} is true.</p>
169   *
170   * @return enum controlling whether a unique if of the component can be accessed via {@link arez.component.Identifiable#getArezId(Object)}.
171   */
172  Feature requireId() default Feature.AUTODETECT;
173
174  /**
175   * Indicates whether the component should implement the interface {@link arez.component.Verifiable}.
176   * This feature is ignored unless {@link Arez#isVerifyEnabled()} is true. The {@link Feature#AUTODETECT}
177   * value will enable this feature if the component has any {@link Reference} methods or any {@link Inverse} methods.
178   *
179   * @return enum that indicates whether the component should implement the interface {@link arez.component.Verifiable}.
180   */
181  Feature verify() default Feature.AUTODETECT;
182
183  /**
184   * Indicate whether references to the component should also be annotated with {@link CascadeDispose} or {@link ComponentDependency}.
185   * This is used to ensure that when a component is disposed that any reference to the component from another
186   * component is rectified. The annotation processor will warn if the above rules are violated.
187   * {@link Feature#ENABLE} will tell the annotation to warn if references to component are invalid.
188   * {@link Feature#DISABLE} disables the warning. {@link Feature#AUTODETECT} will enable the warning if
189   * the {@link #disposeNotifier()} resolves to {@link Feature#ENABLE}.
190   *
191   * @return enum controlling whether a references to components should be explicitly managed.
192   */
193  Feature verifyReferencesToComponent() default Feature.AUTODETECT;
194
195  /**
196   * The default priority used by {@link Memoize} and {@link Observe} annotated methods.
197   * This parameter should only be specified if there are {@link Memoize} or {@link Observe} annotated
198   * methods present on the component.
199   *
200   * @return the default priority used by {@link Memoize} and {@link Observe} annotated methods.
201   */
202  Priority defaultPriority() default Priority.NORMAL;
203
204  /**
205   * The default value for the readOutsideTransaction parameter of {@link Memoize} and {@link Observable}
206   * annotated methods.
207   *
208   * @return The flag indicating whether the default is to allow reads outside a transaction or to require a transaction to read observables and memoized values.
209   */
210  Feature defaultReadOutsideTransaction() default Feature.AUTODETECT;
211
212  /**
213   * The default value for the writeOutsideTransaction parameter of {@link Observable} annotated methods.
214   *
215   * @return The flag indicating whether the default is to allow writes outside a transaction or to require a transaction to write observables values.
216   */
217  Feature defaultWriteOutsideTransaction() default Feature.AUTODETECT;
218}
