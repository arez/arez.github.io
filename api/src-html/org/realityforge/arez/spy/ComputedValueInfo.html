001package org.realityforge.arez.spy;
002
003import java.util.List;
004import javax.annotation.Nonnull;
005import javax.annotation.Nullable;
006import org.realityforge.arez.Arez;
007
008/**
009 * A representation of a ComputedValue instance exposed to spy framework.
010 */
011public interface ComputedValueInfo
012  extends ElementInfo
013{
014  /**
015   * Return true if the specified ComputedValue is "computing".
016   * This implies that the current transaction or one of the parent transactions is calculating the
017   * ComputedValue at the moment.
018   *
019   * @return true if there is a transaction active.
020   * @see org.realityforge.arez.Spy#isComputing(org.realityforge.arez.ComputedValue)
021   */
022  boolean isComputing();
023
024  /**
025   * Return true if the ComputedValue is active.
026   * A ComputedValue is active if there is one or more Observers and the value will be calculated.
027   *
028   * @return true if the ComputedValue is active.
029   * @see org.realityforge.arez.Spy#isActive(org.realityforge.arez.ComputedValue)
030   */
031  boolean isActive();
032
033  /**
034   * Return the list of observers for ComputedValue.
035   * The list is an immutable copy of the observers of the {@link org.realityforge.arez.ComputedValue}.
036   *
037   * @return the list of observers for ComputedValue.
038   * @see org.realityforge.arez.Spy#getObservers(org.realityforge.arez.ComputedValue)
039   */
040  @Nonnull
041  List<ObserverInfo> getObservers();
042
043  /**
044   * Return the list of dependencies of the ComputedValue.
045   * The list is an immutable copy of the dependencies of the {@link org.realityforge.arez.ComputedValue}.
046   * If the {@link org.realityforge.arez.ComputedValue} is currently being computed (i.e. {@link #isComputing()}
047   * returns true) then the dependencies are provisional and may be added to as transaction
048   * completes.
049   *
050   * @return the list of dependencies for ComputedValue.
051   * @see org.realityforge.arez.Spy#getDependencies(org.realityforge.arez.ComputedValue)
052   */
053  @Nonnull
054  List<ObservableInfo> getDependencies();
055
056  /**
057   * Return the component for the ComputedValue.
058   * This method should not be invoked if {@link Arez#areNativeComponentsEnabled()} returns false.
059   *
060   * @return the component that contains ComputedValue if any.
061   * @see org.realityforge.arez.Spy#getComponent(org.realityforge.arez.ComputedValue)
062   */
063  @Nullable
064  ComponentInfo getComponent();
065
066  /**
067   * Return the value of the ComputedValue.
068   * This method should only be invoked if {@link Arez#arePropertyIntrospectorsEnabled()} returns true.
069   *
070   * @return the value of the ComputedValue.
071   * @throws Throwable if the property accessor throws an exception.
072   * @see org.realityforge.arez.Spy#getValue(org.realityforge.arez.ComputedValue)
073   */
074  @Nullable
075  Object getValue()
076    throws Throwable;
077}
