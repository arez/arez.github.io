001package org.realityforge.arez.extras;
002
003import javax.annotation.Nonnull;
004import javax.annotation.Nullable;
005import org.realityforge.anodoc.Unsupported;
006import org.realityforge.arez.Arez;
007import org.realityforge.arez.Disposable;
008import org.realityforge.arez.SafeFunction;
009import org.realityforge.arez.SafeProcedure;
010
011/**
012 * Simplified interface for interacting with extras package.
013 */
014@Unsupported( "Expect this class to evolve over time" )
015public final class ArezExtras
016{
017  private ArezExtras()
018  {
019  }
020
021  /**
022   * Wait until a condition is true, then run effect. The effect is run in a read-only transaction.
023   * See {@link #when(String, boolean, SafeFunction, SafeProcedure)} for further details.
024   *
025   * @param condition The function that determines when the effect is run.
026   * @param effect    The procedure that is executed when the condition is true.
027   * @return the Node representing the reactive component. The user can dispose the node if it is no longer required.
028   */
029  public static Disposable when( @Nonnull final SafeFunction<Boolean> condition,
030                                 @Nonnull final SafeProcedure effect )
031  {
032    return when( false, condition, effect );
033  }
034
035  /**
036   * Wait until a condition is true, then run effect.
037   * See {@link #when(String, boolean, SafeFunction, SafeProcedure)} for further details.
038   *
039   * @param mutation  true if the effect can mutate state, false otherwise.
040   * @param condition The function that determines when the effect is run.
041   * @param effect    The procedure that is executed when the condition is true.
042   * @return the Node representing the reactive component. The user can dispose the node if it is no longer required.
043   */
044  public static Disposable when( final boolean mutation,
045                                 @Nonnull final SafeFunction<Boolean> condition,
046                                 @Nonnull final SafeProcedure effect )
047  {
048    return when( null, mutation, condition, effect );
049  }
050
051  /**
052   * Wait until a condition is true, then run effect.
053   * The condition function is run in a read-only, tracking transaction and will be re-evaluated
054   * any time any of the observed elements are updated. The effect procedure is run in either a
055   * read-only or read-write, non-tracking transaction.
056   *
057   * @param name      the debug name (if any) used when naming the underlying Arez resources.
058   * @param mutation  true if the effect can mutate state, false otherwise.
059   * @param condition The function that determines when the effect is run.
060   * @param effect    The procedure that is executed when the condition is true.
061   * @return the Node representing the reactive component. The user can dispose the node if it is no longer required.
062   */
063  public static Disposable when( @Nullable final String name,
064                                 final boolean mutation,
065                                 @Nonnull final SafeFunction<Boolean> condition,
066                                 @Nonnull final SafeProcedure effect )
067  {
068    return new Arez_Watcher( Arez.context().generateNodeName( "When", name ), mutation, condition, effect );
069  }
070}
