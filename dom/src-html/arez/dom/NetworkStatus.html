001package arez.dom;
002
003import arez.ComputableValue;
004import arez.annotations.Action;
005import arez.annotations.ArezComponent;
006import arez.annotations.ComputableValueRef;
007import arez.annotations.DepType;
008import arez.annotations.Feature;
009import arez.annotations.Memoize;
010import arez.annotations.Observable;
011import arez.annotations.OnActivate;
012import arez.annotations.OnDeactivate;
013import elemental2.dom.DomGlobal;
014import elemental2.dom.EventListener;
015import java.util.Date;
016import javax.annotation.Nonnull;
017
018/**
019 * An observable model that declares state that tracks when the user is "online".
020 * The online state is essentially a reflection of the browsers "navigator.onLine"
021 * value. If an observer is observing the model, the model listens for changes from
022 * the browser and updates the online state as appropriate. However if there is no
023 * observer for the state, the model will not listen to to the browser events so as
024 * not to have any significant performance impact.
025 *
026 * <h1>A very simple example</h1>
027 * <pre>{@code
028 * import com.google.gwt.core.client.EntryPoint;
029 * import elemental2.dom.DomGlobal;
030 * import arez.Arez;
031 * import arez.networkstatus.NetworkStatus;
032 *
033 * public class NetworkStatusExample
034 *   implements EntryPoint
035 * {
036 *   public void onModuleLoad()
037 *   {
038 *     final NetworkStatus networkStatus = NetworkStatus.create();
039 *     Arez.context().observer( () -> {
040 *       final String message = "Network Status: " + ( networkStatus.isOnLine() ? "Online" : "Offline" );
041 *       DomGlobal.console.log( message );
042 *       if ( !networkStatus.isOnLine() )
043 *       {
044 *         DomGlobal.console.log( "Offline since: " + networkStatus.getLastChangedAt() );
045 *       }
046 *     } );
047 *   }
048 * }
049 * }</pre>
050 */
051@ArezComponent( requireId = Feature.DISABLE )
052public abstract class NetworkStatus
053{
054  @Nonnull
055  private final EventListener _listener = e -> updateOnlineStatus( getOnLineComputableValue() );
056
057  /**
058   * Create an instance of NetworkStatus.
059   *
060   * @return the NetworkStatus instance.
061   */
062  @Nonnull
063  public static NetworkStatus create()
064  {
065    return new Arez_NetworkStatus( new Date() );
066  }
067
068  NetworkStatus()
069  {
070  }
071
072  /**
073   * Return true if the browser is online, false otherwise.
074   *
075   * @return true if the browser is online, false otherwise.
076   */
077  @Memoize( depType = DepType.AREZ_OR_EXTERNAL )
078  public boolean isOnLine()
079  {
080    return DomGlobal.navigator.onLine;
081  }
082
083  /**
084   * Return the last time at which online status changed.
085   * This will default to the time the component was created, otherwise
086   * the time at which the online status was changed.
087   *
088   * @return the last time at which online status changed.
089   */
090  @Observable
091  @Nonnull
092  public abstract Date getLastChangedAt();
093
094  abstract void setLastChangedAt( @Nonnull Date lastChangedAt );
095
096  @ComputableValueRef
097  abstract ComputableValue<?> getOnLineComputableValue();
098
099  @OnActivate
100  void onOnLineActivate()
101  {
102    DomGlobal.window.addEventListener( "online", _listener );
103    DomGlobal.window.addEventListener( "offline", _listener );
104  }
105
106  @OnDeactivate
107  void onOnLineDeactivate()
108  {
109    DomGlobal.window.removeEventListener( "online", _listener );
110    DomGlobal.window.removeEventListener( "offline", _listener );
111  }
112
113  @Action
114  void updateOnlineStatus( @Nonnull final ComputableValue<?> computableValue )
115  {
116    computableValue.reportPossiblyChanged();
117    setLastChangedAt( new Date() );
118  }
119}
