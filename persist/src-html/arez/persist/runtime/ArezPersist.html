001package arez.persist.runtime;
002
003import arez.SafeProcedure;
004import arez.persist.StoreTypes;
005import grim.annotations.OmitClinit;
006import grim.annotations.OmitSymbol;
007import javax.annotation.Nonnull;
008import javax.annotation.Nullable;
009
010/**
011 * Provide an interface to register and access stores and access the root scope as well as global configuration settings.
012 */
013@OmitClinit
014public final class ArezPersist
015{
016  private ArezPersist()
017  {
018  }
019
020  /**
021   * Return true if apiInvariants will be checked.
022   *
023   * @return true if apiInvariants will be checked.
024   */
025  @OmitSymbol
026  public static boolean shouldCheckApiInvariants()
027  {
028    return Config.shouldCheckApiInvariants();
029  }
030
031  /**
032   * Return true if the in-memory application store should be registered by the framework.
033   *
034   * @return true if the in-memory application store should be registered by the framework.
035   */
036  @OmitSymbol
037  public static boolean isApplicationStoreEnabled()
038  {
039    return Config.isApplicationStoreEnabled();
040  }
041
042  /**
043   * Return the root scope under which all other scopes are nested.
044   *
045   * @return the root scope under which all other scopes are nested.
046   */
047  @Nonnull
048  public static Scope getRootScope()
049  {
050    return Registry.getRootScope();
051  }
052
053  /**
054   * Register a store with specified name and storage service.
055   * It is an error to register multiple stores with the same name.
056   *
057   * <p>As part of the register operation, the store will attempt to restore state from the storage service.
058   * If an error occurs during the restore, then the error will be logged and registration will complete.</p>
059   *
060   * @param name    the name of the store.
061   * @param service the associated StorageService.
062   * @return the action to invoke to deregister service.
063   */
064  @Nonnull
065  public static SafeProcedure registerStore( @Nonnull final String name, @Nonnull final StorageService service )
066  {
067    return Registry.registerStore( name, service );
068  }
069
070  /**
071   * Return the store that is registered with the specified name.
072   * It is an error to invoke this method without registering a store under this name.
073   *
074   * @param name the name of the store.
075   * @return the store.
076   */
077  @Nonnull
078  public static Store getStore( @Nonnull final String name )
079  {
080    return Registry.getStore( name );
081  }
082
083  /**
084   * Find the scope with the specified name.
085   * The name can actually consist of name components separated by a "." character. Each name component is
086   * nested within the scope identified by the prior name component. i.e. The name "dashboard.finance.entry"
087   * will look for the scope named "entry" nested in a scope named "finance" nested in a scope named "dashboard".
088   *
089   * @param qualifiedName the qualified scope name.
090   * @return the scope if it exists.
091   */
092  @Nullable
093  public static Scope findScope( @Nonnull final String qualifiedName )
094  {
095    Scope scope = getRootScope();
096    if ( Scope.ROOT_SCOPE_NAME.equals( qualifiedName ) )
097    {
098      return scope;
099    }
100    int start = 0;
101    int end;
102    while ( -1 != ( end = qualifiedName.indexOf( '.', start ) ) )
103    {
104      scope = scope.findScope( qualifiedName.substring( start, end ) );
105      if ( null == scope )
106      {
107        return null;
108      }
109      else
110      {
111        start = end + 1;
112      }
113    }
114    return scope.findScope( qualifiedName.substring( start ) );
115  }
116
117  /**
118   * Find the scope with the specified name and if it does not exist then create it.
119   * The name can actually consist of name components separated by a "." character. Each name component is
120   * nested within the scope identified by the prior name component. i.e. The name "dashboard.finance.entry"
121   * will look for the scope named "entry" nested in a scope named "finance" nested in a scope named "dashboard".
122   *
123   * @param qualifiedName the qualified scope name.
124   * @return the scope.
125   */
126  @Nonnull
127  public static Scope findOrCreateScope( @Nonnull final String qualifiedName )
128  {
129    Scope scope = getRootScope();
130    if ( Scope.ROOT_SCOPE_NAME.equals( qualifiedName ) )
131    {
132      return scope;
133    }
134    int start = 0;
135    int end;
136    while ( -1 != ( end = qualifiedName.indexOf( '.', start ) ) )
137    {
138      scope = scope.findOrCreateScope( qualifiedName.substring( start, end ) );
139      start = end + 1;
140    }
141    return scope.findOrCreateScope( qualifiedName.substring( start ) );
142  }
143
144  /**
145   * Dispose the specified scope.
146   * A dispose operation first performs a {@link #releaseScope(Scope)} on the scope, then attempts to
147   * dispose all nested scopes and finally disposes the specified scope. A disposed scope should no longer be
148   * used to store state. It is an error to attempt to dispose the root scope.
149   *
150   * @param scope the scope to dispose.
151   */
152  public static void disposeScope( @Nonnull final Scope scope )
153  {
154    Registry.disposeScope( scope );
155  }
156
157  /**
158   * Release the specified scope.
159   * A release operation removes any state associated with the scope and any nested scope.
160   *
161   * @param scope the scope to release.
162   */
163  public static void releaseScope( @Nonnull final Scope scope )
164  {
165    Registry.releaseScope( scope );
166  }
167
168  /**
169   * Register a converter for a type.
170   * It is an error to register multiple converters with the same name.
171   *
172   * @param type      the application type.
173   * @param converter the converter.
174   * @return the action to invoke to deregister converter.
175   * @param <A> the type of the value.
176   */
177  public static <A> SafeProcedure registerConverter( @Nonnull final Class<A> type,
178                                                     @Nonnull final Converter<A, ?> converter )
179  {
180    return Registry.registerConverter( type, converter );
181  }
182
183  /**
184   * Return the converter registered for the specified application type or the identity converter if none are specified.
185   *
186   * @param type the application type.
187   * @return the converter if any.
188   * @param <A> the type of the value.
189   */
190  @Nonnull
191  public static <A> Converter<A, ?> getConverter( @Nonnull final Class<A> type )
192  {
193    return Registry.getConverter( type );
194  }
195
196  static void registerApplicationStoreIfEnabled()
197  {
198    if ( isApplicationStoreEnabled() )
199    {
200      registerStore( StoreTypes.APPLICATION, new NoopStorageService() );
201    }
202  }
203}
