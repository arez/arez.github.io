001package arez.persist.runtime;
002
003import arez.Arez;
004import arez.Disposable;
005import arez.annotations.ArezComponent;
006import arez.annotations.ComponentId;
007import arez.annotations.PreDispose;
008import java.util.Collection;
009import java.util.Collections;
010import java.util.HashMap;
011import java.util.Map;
012import java.util.Objects;
013import javax.annotation.Nonnull;
014import javax.annotation.Nullable;
015import static org.realityforge.braincheck.Guards.*;
016
017/**
018 * A scope is used to control the lifecycle of state storage.
019 * State containing within a scope can be removed from the store using a single call
020 * to {@link ArezPersist#releaseScope(Scope)}. Scopes can be nested within other scopes
021 * and releasing a scope, releases all child scopes. There is a single root scope in which
022 * all other scopes are nested and it can be retrieved using the {@link ArezPersist#getRootScope()}
023 * method.
024 *
025 * <p>A scope may be disposed. It is no longer valid to create nested scopes, store state or retrieve
026 * state with disposed scopes.</p>
027 */
028@ArezComponent( allowEmpty = true )
029public abstract class Scope
030{
031  /**
032   * The name of the root scope.
033   */
034  @Nonnull
035  public static final String ROOT_SCOPE_NAME = "<>";
036  /**
037   * The parent scope. Every scope but the root scope must contain a parent.
038   */
039  @Nullable
040  private final Scope _parent;
041  /**
042   * The name of the scope. Scopes have alphanumeric names and may also include '-' or '_' characters.
043   */
044  @Nonnull
045  private final String _name;
046  /**
047   * Nested scopes if any.
048   */
049  @Nonnull
050  private final Map<String, Scope> _nestedScopes = new HashMap<>();
051
052  @Nonnull
053  static Scope create( @Nullable final Scope parent, @Nonnull final String name )
054  {
055    return new Arez_Scope( parent, name );
056  }
057
058  Scope( @Nullable final Scope parent, @Nonnull final String name )
059  {
060    assert ( null == parent && ROOT_SCOPE_NAME.equals( name ) ) ||
061           ( null != parent && !ROOT_SCOPE_NAME.equals( name ) );
062    _parent = parent;
063    _name = Objects.requireNonNull( name );
064  }
065
066  /**
067   * Return the simple name of the scope.
068   * The simple name is the name that was used to create the scope.
069   *
070   * @return the simple name of the scope.
071   */
072  @Nonnull
073  public String getName()
074  {
075    return _name;
076  }
077
078  /**
079   * Return the qualified name of the scope.
080   * The qualified name includes the parent scopes name followed by a "." character unless
081   * the parent scope is the root scope.
082   *
083   * @return the qualified name of the scope.
084   */
085  @ComponentId
086  @Nonnull
087  public String getQualifiedName()
088  {
089    return null == _parent || null == _parent._parent ? _name : _parent.getQualifiedName() + "." + _name;
090  }
091
092  /**
093   * Find or create a scope directly nested under the current scope.
094   * This must not be invoked on a disposed scope.
095   *
096   * @param name the name of the nested scope.
097   * @return the scope.
098   */
099  @Nonnull
100  public Scope findOrCreateScope( @Nonnull final String name )
101  {
102    if ( ArezPersist.shouldCheckApiInvariants() )
103    {
104      apiInvariant( () -> Disposable.isNotDisposed( this ),
105                    () -> "findOrCreateScope() invoked on disposed scope named '" + _name + "'" );
106      apiInvariant( () -> isValidName( name ),
107                    () -> "findOrCreateScope() invoked with name '" + name +
108                          "' but the name has invalid characters. Names must contain alphanumeric " +
109                          "characters, '-' or '_'" );
110    }
111    final Scope existing = findScope( name );
112    if ( null != existing )
113    {
114      return existing;
115    }
116    else
117    {
118      final Scope scope = Scope.create( this, name );
119      _nestedScopes.put( name, scope );
120      return scope;
121    }
122  }
123
124  @Override
125  public String toString()
126  {
127    if ( Arez.areNamesEnabled() )
128    {
129      return getQualifiedName();
130    }
131    else
132    {
133      return super.toString();
134    }
135  }
136
137  @Nullable
138  Scope getParent()
139  {
140    return _parent;
141  }
142
143  @Nullable
144  Scope findScope( @Nonnull final String name )
145  {
146    return _nestedScopes.get( name );
147  }
148
149  @Nonnull
150  public Collection<Scope> getNestedScopes()
151  {
152    final Collection<Scope> scopes = _nestedScopes.values();
153    return ArezPersist.shouldCheckApiInvariants() ? Collections.unmodifiableCollection( scopes ) : scopes;
154  }
155
156  @PreDispose
157  void preDispose()
158  {
159    assert _nestedScopes.isEmpty();
160    if ( null != _parent )
161    {
162      _parent._nestedScopes.remove( _name );
163    }
164  }
165
166  /**
167   * Return true if the name is valid.
168   *
169   * @param name the name to check.
170   * @return true if the name is valid.
171   */
172  private boolean isValidName( @Nonnull final String name )
173  {
174    if ( 0 == name.length() )
175    {
176      return false;
177    }
178    else
179    {
180      final int length = name.length();
181      for ( int i = 0; i < length; i++ )
182      {
183        final char ch = name.charAt( i );
184        if ( !Character.isLetterOrDigit( ch ) && '_' != ch && '-' != ch )
185        {
186          return false;
187        }
188      }
189      return true;
190    }
191  }
192}
