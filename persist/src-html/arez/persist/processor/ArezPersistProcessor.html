034/**
035 * Annotation processor that analyzes Arez annotated source and generates models from the annotations.
036 */
037@SupportedAnnotationTypes( { Constants.PERSIST_TYPE_CLASSNAME,
038                             Constants.PERSIST_ID_CLASSNAME,
039                             Constants.PERSIST_CLASSNAME } )
040@SupportedSourceVersion( SourceVersion.RELEASE_8 )
041@SupportedOptions( { "arez.persist.defer.unresolved", "arez.persist.defer.errors", "arez.persist.debug" } )
042public final class ArezPersistProcessor
043  extends AbstractStandardProcessor
044{
045  @Nonnull
046  private static final Pattern GETTER_PATTERN = Pattern.compile( "^get([A-Z].*)$" );
047  @Nonnull
048  private static final Pattern ISSER_PATTERN = Pattern.compile( "^is([A-Z].*)$" );
049  /**
050   * Sentinel value indicating the default value.
051   */
052  @Nonnull
053  private static final String DEFAULT_SENTINEL = "<default>";
054  @Nonnull
055  private final DeferredElementSet _deferredTypes = new DeferredElementSet();

057  @Override
058  @Nonnull
059  protected String getIssueTrackerURL()
060  {
061    return "https://github.com/arez/arez-persist/issues";
062  }

065  @Override

068    return "arez.persist";

071  @Override
072  public boolean process( @Nonnull final Set<? extends TypeElement> annotations, @Nonnull final RoundEnvironment env )
073  {
074    debugAnnotationProcessingRootElements( env );
075    collectRootTypeNames( env );
076    // Validate persist first so we don't have to perform validation inside type processing
077    // The framework assumes that Arez is also running the annotation processor and validating
078    // the shape of the same methods as they are marked as @Observable so this annotation processor
079    // performs minimal validation. This can result in this annotation processor producing bad code
080    // and the arez annotation processor failing later in the round. The expectation is that we will
081    // need to get the arez annotation processor passing and once it does then the code generated by
082    // this annotation processor will start to work.
083    annotations.stream()
084      .filter( a -> a.getQualifiedName().toString().equals( Constants.PERSIST_CLASSNAME ) )
085      .findAny()
086      .ifPresent( a -> verifyPersistElements( env, env.getElementsAnnotatedWith( a ) ) );
087
088    annotations.stream()
089      .filter( a -> a.getQualifiedName().toString().equals( Constants.PERSIST_ID_CLASSNAME ) )
090      .findAny()
091      .ifPresent( a -> verifyPersistIdElements( env, env.getElementsAnnotatedWith( a ) ) );
092
093    processTypeElements( annotations, env, Constants.PERSIST_TYPE_CLASSNAME, _deferredTypes, this::process );
094
095    errorIfProcessingOverAndInvalidTypesDetected( env );
096    clearRootTypeNamesIfProcessingOver( env );
097    return true;
098  }
099
100  private void process( @Nonnull final TypeElement element )
101    throws Exception
102  {
103    if ( !AnnotationsUtil.hasAnnotationOfType( element, Constants.AREZ_COMPONENT_CLASSNAME ) )
104    {
105      throw new ProcessorException( MemberChecks.must( Constants.PERSIST_TYPE_CLASSNAME,
106                                                       "be present on a type annotated with the " +
107                                                       MemberChecks.toSimpleName( Constants.AREZ_COMPONENT_CLASSNAME ) +
108                                                       " annotation" ),
109                                    element );
110    }
111
112    final AnnotationMirror annotation =
113      AnnotationsUtil.getAnnotationByType( element, Constants.PERSIST_TYPE_CLASSNAME );
114    final String name = extractPersistTypeName( element, annotation );
115    final boolean persistOnDispose = AnnotationsUtil.getAnnotationValueValue( annotation, "persistOnDispose" );
116    final String defaultStore = extractDefaultStore( element, annotation );
117    final List<ExecutableElement> methods =
118      ElementsUtil.getMethods( element, processingEnv.getElementUtils(), processingEnv.getTypeUtils() );
119    ExecutableElement idMethod = null;
120    final Map<String, PropertyDescriptor> properties = new HashMap<>();
121    for ( final ExecutableElement method : methods )
122    {
123      final AnnotationMirror persistAnnotation =
124        AnnotationsUtil.findAnnotationByType( method, Constants.PERSIST_CLASSNAME );
125      final AnnotationMirror persistIdAnnotation =
126        AnnotationsUtil.findAnnotationByType( method, Constants.PERSIST_ID_CLASSNAME );
127      if ( null != persistAnnotation && null != persistIdAnnotation )
128      {
129        throw new ProcessorException( MemberChecks.mustNot( Constants.PERSIST_ID_CLASSNAME,
130                                                            "also be annotated with the " +
131                                                            MemberChecks.toSimpleName( Constants.PERSIST_CLASSNAME ) +
132                                                            " annotation" ),
133                                      element );
134      }
135      else if ( null != persistAnnotation )
136      {
137        processPersistAnnotation( element, method, persistAnnotation, methods, defaultStore, properties );
138      }
139      else if ( null != persistIdAnnotation )
140      {
141        // These validations have already been performed by an earlier step in this annotation processor
142        // if and only if the type declaring the @PersistId is also compiled in this compilation round.
143        // If the type that encloses @PersistId annotated method is part of a library or compiled earlier
144        // we re-run the checks just in case to avoid scenarios where weird behaviour slips through in code
145        // generation step.
146        validatePersisIdMethod( method );
147        MemberChecks.mustNotBePackageAccessInDifferentPackage( element,
148                                                               Constants.PERSIST_CLASSNAME,
149                                                               Constants.PERSIST_ID_CLASSNAME,
150                                                               method );
151        if ( null != idMethod )
152        {
153          throw new ProcessorException( MemberChecks.mustNot( Constants.PERSIST_ID_CLASSNAME,
154                                                              "be present multiple times within a " +
155                                                              MemberChecks.toSimpleName( Constants.PERSIST_TYPE_CLASSNAME ) +
156                                                              " annotated type but another method annotated with " +
157                                                              MemberChecks.toSimpleName( Constants.PERSIST_ID_CLASSNAME ) +
158                                                              " exists and is named " +
159                                                              idMethod.getSimpleName() ),
160                                        element );
161        }
162        idMethod = method;
163      }
164    }
165
166    if ( properties.isEmpty() )
167    {
168      throw new ProcessorException( MemberChecks.must( Constants.PERSIST_TYPE_CLASSNAME,
169                                                       "contain one or more " +
170                                                       MemberChecks.toSimpleName( Constants.OBSERVABLE_CLASSNAME ) +
171                                                       " properties annotated with " +
172                                                       MemberChecks.toSimpleName( Constants.PERSIST_CLASSNAME ) ),
173                                    element );
174    }
175
176    emitSidecar( new TypeDescriptor( name,
177                                     persistOnDispose,
178                                     element,
179                                     idMethod,
180                                     new ArrayList<>( properties.values() ) ) );
181  }

183  private void processPersistAnnotation( @Nonnull final TypeElement element,
184                                         @Nonnull final ExecutableElement method,
185                                         @Nonnull final AnnotationMirror persistAnnotation,
186                                         @Nonnull final List<ExecutableElement> methods,
187                                         @Nonnull final String defaultStore,
188                                         @Nonnull final Map<String, PropertyDescriptor> properties )
189  {
190    final TypeMirror returnType = method.getReturnType();
191    if ( TypeKind.VOID == returnType.getKind() )
192    {
193      throw new ProcessorException( MemberChecks.must( Constants.PERSIST_CLASSNAME,
194                                                       "be present on the accessor method of the " +
195                                                       MemberChecks.toSimpleName( Constants.OBSERVABLE_CLASSNAME ) +
196                                                       " property" ),
197                                    method );
198    }
199
200    final String persistName = extractPersistName( method, persistAnnotation );
201    final String persistStore = extractStore( element, method, persistAnnotation, defaultStore );
202
203    final String setterName = "set" + firstCharacterToUpperCase( persistName );
204    final ExecutableElement setter = methods
205      .stream()
206      .filter( m -> TypeKind.VOID == m.getReturnType().getKind() &&
207                    setterName.equals( m.getSimpleName().toString() ) &&
208                    1 == m.getParameters().size() &&
209                    processingEnv.getTypeUtils().isSameType( m.getParameters().get( 0 ).asType(), returnType ) )
210      .findAny()
211      .orElse( null );
212    if ( null == setter )
213    {
214      throw new ProcessorException( MemberChecks.must( Constants.PERSIST_CLASSNAME,
215                                                       "be paired with a setter named " + setterName ),
216                                    method );
217    }
218
219    final PropertyDescriptor existing = properties.get( persistName );
220    if ( null != existing )
221    {
222      throw new ProcessorException( MemberChecks.must( Constants.PERSIST_CLASSNAME,
223                                                       "has the same name '" + persistName +
224                                                       "' as another persistent property declared by the name. " +
225                                                       "The other property is accessed by the method named " +
226                                                       existing.getGetter().getSimpleName() ),
227                                    method );
228    }
229    else
230    {
231      properties.put( persistName, new PropertyDescriptor( persistName, persistStore, method, setter ) );
232    }
233  }

235  @Nonnull
236  private String extractPersistTypeName( @Nonnull final TypeElement element,
237                                         @Nonnull final AnnotationMirror annotation )
238  {
239    final String declaredValue = AnnotationsUtil.getAnnotationValueValue( annotation, "name" );
240    if ( DEFAULT_SENTINEL.equals( declaredValue ) )
241    {
242      return element.getSimpleName().toString();
243    }
244    else if ( SourceVersion.isIdentifier( declaredValue ) )
245    {
246      return declaredValue;
247    }
248    else
249    {
250      throw new ProcessorException( MemberChecks.mustNot( Constants.PERSIST_TYPE_CLASSNAME,
251                                                          "specify a name parameter that is not a valid java identifier" ),
252                                    element,
253                                    annotation );
254    }
255  }
256
257  @Nonnull
258  private String extractDefaultStore( @Nonnull final TypeElement element, @Nonnull final AnnotationMirror annotation )
259  {
260    final String defaultStore = AnnotationsUtil.getAnnotationValueValue( annotation, "defaultStore" );
261    if ( !SourceVersion.isIdentifier( defaultStore ) )
262    {
263      throw new ProcessorException( MemberChecks.mustNot( Constants.PERSIST_TYPE_CLASSNAME,
264                                                          "specify a defaultStore parameter that is not a valid java identifier" ),
265                                    element,
266                                    annotation );
267    }
268    return defaultStore;
269  }
270
271  @Nonnull
272  private String extractPersistName( @Nonnull final ExecutableElement method,
273                                     @Nonnull final AnnotationMirror annotation )
274  {
275    final String declaredValue = AnnotationsUtil.getAnnotationValueValue( annotation, "name" );
276    final String value = getPropertyAccessorName( method, declaredValue );
277    if ( SourceVersion.isIdentifier( value ) )
278    {
279      return value;
280    }
281    else
282    {
283      throw new ProcessorException( MemberChecks.mustNot( Constants.PERSIST_CLASSNAME,
284                                                          "specify a name parameter that is not a valid java identifier" ),
285                                    method,
286                                    annotation );
287    }
288  }
289
290  @Nonnull
291  private String extractStore( @Nonnull final TypeElement typeElement,
292                               @Nonnull final ExecutableElement element,
293                               @Nonnull final AnnotationMirror annotation,
294                               @Nonnull final String defaultStore )
295  {
296    final String store = AnnotationsUtil.getAnnotationValueValue( annotation, "store" );
297    if ( DEFAULT_SENTINEL.equals( store ) )
298    {
299      return defaultStore;
300    }
301    else if ( SourceVersion.isIdentifier( store ) )
302    {
303      if ( defaultStore.equals( store ) &&
304           typeElement == element.getEnclosingElement() &&
305           ElementsUtil.isWarningNotSuppressed( element, Constants.WARNING_UNNECESSARY_STORE ) )
306      {
307        final String message =
308          MemberChecks.shouldNot( Constants.PERSIST_CLASSNAME,
309                                  "specify the store parameter when it is the same as the defaultStore " +
310                                  "parameter in the specified by the " +
311                                  MemberChecks.toSimpleName( Constants.PERSIST_TYPE_CLASSNAME ) +
312                                  " annotation on the enclosing type. " +
313                                  MemberChecks.suppressedBy( Constants.WARNING_UNNECESSARY_STORE ) );
314        processingEnv.getMessager().printMessage( Diagnostic.Kind.WARNING, message, element );
315      }
316      return store;
317    }
318    else
319
320    {
321      throw new ProcessorException( MemberChecks.mustNot( Constants.PERSIST_CLASSNAME,
322                                                          "specify a store parameter that is not a valid java identifier" ),
323                                    element,
324                                    annotation );
325    }
326  }

328  private void verifyPersistElements( @Nonnull final RoundEnvironment env,
329                                      @Nonnull final Set<? extends Element> elements )
330  {
331    for ( final Element element : elements )
332    {
333      assert ElementKind.METHOD == element.getKind();
334      if ( !AnnotationsUtil.hasAnnotationOfType( element, Constants.OBSERVABLE_CLASSNAME ) )
335      {
336        reportError( env,
337                     MemberChecks.must( Constants.PERSIST_CLASSNAME,
338                                        "be also be annotated with the " + Constants.OBSERVABLE_CLASSNAME +
339                                        " annotation" ),
340                     element );
341      }
342    }
343  }
344
345  private void verifyPersistIdElements( @Nonnull final RoundEnvironment env,
346                                        @Nonnull final Set<? extends Element> elements )
347  {
348    for ( final Element element : elements )
349    {
350      performAction( env, e -> verifyPersistIdElement( env, e ), element );
351    }
352  }
353
354  private void verifyPersistIdElement( @Nonnull final RoundEnvironment env,  @Nonnull final Element element )
355  {
356    assert ElementKind.METHOD == element.getKind();
357    final Element type = element.getEnclosingElement();
358    if ( !AnnotationsUtil.hasAnnotationOfType( type, Constants.AREZ_COMPONENT_CLASSNAME ) &&
359         !AnnotationsUtil.hasAnnotationOfType( type, Constants.ACT_AS_COMPONENT_CLASSNAME ) )
360    {
361      reportError( env,
362                   MemberChecks.must( Constants.PERSIST_ID_CLASSNAME,
363                                      "be enclosed within a type annotated by either the " +
364                                      Constants.AREZ_COMPONENT_CLASSNAME + " annotation or the " +
365                                      Constants.ACT_AS_COMPONENT_CLASSNAME + " annotation" ),
366                   element );
367    }
368
369    validatePersisIdMethod( (ExecutableElement) element );
370  }
371
372  private void validatePersisIdMethod( @Nonnull final ExecutableElement method )
373  {
374    MemberChecks.mustNotBeAbstract( Constants.PERSIST_ID_CLASSNAME, method );
375    MemberChecks.mustNotThrowAnyExceptions( Constants.PERSIST_ID_CLASSNAME, method );
376    MemberChecks.mustNotHaveAnyTypeParameters( Constants.PERSIST_ID_CLASSNAME, method );
377    MemberChecks.mustNotHaveAnyParameters( Constants.PERSIST_ID_CLASSNAME, method );
378    MemberChecks.mustReturnAValue( Constants.PERSIST_ID_CLASSNAME, method );
379    MemberChecks.mustNotBeStatic( Constants.PERSIST_ID_CLASSNAME, method );
380    MemberChecks.mustNotBePrivate( Constants.PERSIST_ID_CLASSNAME, method );
381    MemberChecks.mustNotBeProtected( Constants.PERSIST_ID_CLASSNAME, method );
382  }

384  private void emitSidecar( @Nonnull final TypeDescriptor type )
385    throws IOException
386  {
387    final String packageName = GeneratorUtil.getQualifiedPackageName( type.getElement() );
388    emitTypeSpec( packageName, SidecarGenerator.buildType( processingEnv, type ) );
389  }

391  @SuppressWarnings( "SameParameterValue" )
392  @Nonnull
393  private String getPropertyAccessorName( @Nonnull final ExecutableElement method, @Nonnull final String specifiedName )
394    throws ProcessorException
395  {
396    String name = deriveName( method, GETTER_PATTERN, specifiedName );
397    if ( null != name )
398    {
399      return name;
400    }
401    if ( method.getReturnType().getKind() == TypeKind.BOOLEAN )
402    {
403      name = deriveName( method, ISSER_PATTERN, specifiedName );
404      if ( null != name )
405      {
406        return name;
407      }
408    }
409    return method.getSimpleName().toString();
410  }
411
412  @Nullable
413  private String deriveName( @Nonnull final ExecutableElement method,
414                             @Nonnull final Pattern pattern,
415                             @Nonnull final String name )
416    throws ProcessorException
417  {
418    if ( DEFAULT_SENTINEL.equals( name ) )
419    {
420      final String methodName = method.getSimpleName().toString();
421      final Matcher matcher = pattern.matcher( methodName );
422      if ( matcher.find() )
423      {
424        final String candidate = matcher.group( 1 );
425        return firstCharacterToLowerCase( candidate );
426      }
427      else
428      {
429        return null;
430      }
431    }
432    else
433    {
434      return name;
435    }
436  }
437
438  @Nonnull
439  private String firstCharacterToLowerCase( @Nonnull final String name )
440  {
441    return Character.toLowerCase( name.charAt( 0 ) ) + name.substring( 1 );
442  }
443
444  @Nonnull
445  private String firstCharacterToUpperCase( @Nonnull final String name )
446  {
447    return Character.toUpperCase( name.charAt( 0 ) ) + name.substring( 1 );
448  }
