034/**
035 * Annotation processor that analyzes Arez annotated source and generates models from the annotations.
036 */
037@SupportedAnnotationTypes( { Constants.PERSIST_TYPE_CLASSNAME,
038                             Constants.PERSIST_CLASSNAME } )
039@SupportedSourceVersion( SourceVersion.RELEASE_8 )
040@SupportedOptions( { "arez.persist.defer.unresolved", "arez.persist.defer.errors", "arez.persist.debug" } )
041public final class ArezPersistProcessor
042  extends AbstractStandardProcessor
043{
044  @Nonnull
045  private static final Pattern GETTER_PATTERN = Pattern.compile( "^get([A-Z].*)$" );
046  @Nonnull
047  private static final Pattern ISSER_PATTERN = Pattern.compile( "^is([A-Z].*)$" );
048  /**
049   * Sentinel value indicating the default value.
050   */
051  @Nonnull
052  private static final String DEFAULT_SENTINEL = "<default>";
053  @Nonnull
054  private final DeferredElementSet _deferredTypes = new DeferredElementSet();
055
056  @Override
057  @Nonnull
058  protected String getIssueTrackerURL()
059  {
060    return "https://github.com/arez/arez-persist/issues";
061  }
062
063  @Nonnull
064  @Override
065  protected String getOptionPrefix()
066  {
067    return "arez.persist";
068  }

070  @Override
071  public boolean process( @Nonnull final Set<? extends TypeElement> annotations, @Nonnull final RoundEnvironment env )
072  {
073    // Validate persist first so we don't have to perform validation inside type processing
074    // The framework assumes that Arez is also running the annotation processor and validating
075    // the shape of the same methods as they are marked as @Observable so this annotation processor
076    // performs minimal validation. This can result in this annotation processor producing bad code
077    // and the arez annotation processor failing later in the round. The expectation is that we will
078    // need to get the arez annotation processor passing and once it does then the code generated by
079    // this annotation processor will start to work.
080    annotations.stream()
081      .filter( a -> a.getQualifiedName().toString().equals( Constants.PERSIST_CLASSNAME ) )
082      .findAny()
083      .ifPresent( a -> verifyPersistElements( env, env.getElementsAnnotatedWith( a ) ) );
084
085    processTypeElements( annotations, env, Constants.PERSIST_TYPE_CLASSNAME, _deferredTypes, this::process );
086
087    errorIfProcessingOverAndInvalidTypesDetected( env );
088    return true;
089  }
090
091  private void process( @Nonnull final TypeElement element )
092    throws Exception
093  {
094    if ( !AnnotationsUtil.hasAnnotationOfType( element, Constants.AREZ_COMPONENT_CLASSNAME ) )
095    {
096      throw new ProcessorException( MemberChecks.must( Constants.PERSIST_TYPE_CLASSNAME,
097                                                       "be present on a type annotated with the " +
098                                                       MemberChecks.toSimpleName( Constants.AREZ_COMPONENT_CLASSNAME ) +
099                                                       " annotation" ),
100                                    element );
101    }
102
103    final AnnotationMirror annotation =
104      AnnotationsUtil.getAnnotationByType( element, Constants.PERSIST_TYPE_CLASSNAME );
105    final String name = extractPersistTypeName( element, annotation );
106    final boolean persistOnDispose = AnnotationsUtil.getAnnotationValueValue( annotation, "persistOnDispose" );
107    final String defaultStore = extractDefaultStore( element, annotation );
108    final List<ExecutableElement> methods =
109      ElementsUtil.getMethods( element, processingEnv.getElementUtils(), processingEnv.getTypeUtils() );
110    final Map<String, PropertyDescriptor> properties = new HashMap<>();
111    for ( final ExecutableElement method : methods )
112    {
113      final AnnotationMirror persistAnnotation =
114        AnnotationsUtil.findAnnotationByType( method, Constants.PERSIST_CLASSNAME );
115      if ( null != persistAnnotation )
116      {
117        final TypeMirror returnType = method.getReturnType();
118        if ( TypeKind.VOID == returnType.getKind() )
119        {
120          throw new ProcessorException( MemberChecks.must( Constants.PERSIST_CLASSNAME,
121                                                           "be present on the accessor method of the " +
122                                                           MemberChecks.toSimpleName( Constants.OBSERVABLE_CLASSNAME ) +
123                                                           " property" ),
124                                        method );
125        }
126
127        final String persistName = extractPersistName( method, persistAnnotation );
128        final String persistStore = extractStore( element, method, persistAnnotation, defaultStore );
129
130        final String setterName = "set" + firstCharacterToUpperCase( persistName );
131        final ExecutableElement setter = methods
132          .stream()
133          .filter( m -> TypeKind.VOID == m.getReturnType().getKind() &&
134                        setterName.equals( m.getSimpleName().toString() ) &&
135                        1 == m.getParameters().size() &&
136                        processingEnv.getTypeUtils().isSameType( m.getParameters().get( 0 ).asType(), returnType ) )
137          .findAny()
138          .orElse( null );
139        if ( null == setter )
140        {
141          throw new ProcessorException( MemberChecks.must( Constants.PERSIST_CLASSNAME,
142                                                           "be paired with a setter named " + setterName ),
143                                        method );
144        }
145
146        final PropertyDescriptor existing = properties.get( persistName );
147        if ( null != existing )
148        {
149          throw new ProcessorException( MemberChecks.must( Constants.PERSIST_CLASSNAME,
150                                                           "has the same name '" + persistName +
151                                                           "' as another persistent property declared by the name. " +
152                                                           "The other property is accessed by the method named " +
153                                                           existing.getGetter().getSimpleName() ),
154                                        method );
155        }
156        else
157        {
158          properties.put( persistName, new PropertyDescriptor( persistName, persistStore, method, setter ) );
159        }
160      }
161    }
162
163    if ( properties.isEmpty() )
164    {
165      throw new ProcessorException( MemberChecks.must( Constants.PERSIST_TYPE_CLASSNAME,
166                                                       "contain one or more " +
167                                                       MemberChecks.toSimpleName( Constants.OBSERVABLE_CLASSNAME ) +
168                                                       " properties annotated with " +
169                                                       MemberChecks.toSimpleName( Constants.PERSIST_CLASSNAME ) ),
170                                    element );
171    }
172
173    emitSidecar( new TypeDescriptor( name, persistOnDispose, element, new ArrayList<>( properties.values() ) ) );
174  }
175
176  @Nonnull
177  private String extractPersistTypeName( @Nonnull final TypeElement element,
178                                         @Nonnull final AnnotationMirror annotation )
179  {
180    final String declaredValue = AnnotationsUtil.getAnnotationValueValue( annotation, "name" );
181    if ( DEFAULT_SENTINEL.equals( declaredValue ) )
182    {
183      return element.getSimpleName().toString();
184    }
185    else if ( SourceVersion.isIdentifier( declaredValue ) )
186    {
187      return declaredValue;
188    }
189    else
190    {
191      throw new ProcessorException( MemberChecks.mustNot( Constants.PERSIST_TYPE_CLASSNAME,
192                                                          "specify a name parameter that is not a valid java identifier" ),
193                                    element,
194                                    annotation );
195    }
196  }
197
198  @Nonnull
199  private String extractDefaultStore( @Nonnull final TypeElement element, @Nonnull final AnnotationMirror annotation )
200  {
201    final String defaultStore = AnnotationsUtil.getAnnotationValueValue( annotation, "defaultStore" );
202    if ( !SourceVersion.isIdentifier( defaultStore ) )
203    {
204      throw new ProcessorException( MemberChecks.mustNot( Constants.PERSIST_TYPE_CLASSNAME,
205                                                          "specify a defaultStore parameter that is not a valid java identifier" ),
206                                    element,
207                                    annotation );
208    }
209    return defaultStore;
210  }
211
212  @Nonnull
213  private String extractPersistName( @Nonnull final ExecutableElement method,
214                                     @Nonnull final AnnotationMirror annotation )
215  {
216    final String declaredValue = AnnotationsUtil.getAnnotationValueValue( annotation, "name" );
217    final String value = getPropertyAccessorName( method, declaredValue );
218    if ( SourceVersion.isIdentifier( value ) )
219    {
220      return value;
221    }
222    else
223    {
224      throw new ProcessorException( MemberChecks.mustNot( Constants.PERSIST_CLASSNAME,
225                                                          "specify a name parameter that is not a valid java identifier" ),
226                                    method,
227                                    annotation );
228    }
229  }
230
231  @Nonnull
232  private String extractStore( @Nonnull final TypeElement typeElement,
233                               @Nonnull final ExecutableElement element,
234                               @Nonnull final AnnotationMirror annotation,
235                               @Nonnull final String defaultStore )
236  {
237    final String store = AnnotationsUtil.getAnnotationValueValue( annotation, "store" );
238    if ( DEFAULT_SENTINEL.equals( store ) )
239    {
240      return defaultStore;
241    }
242    else if ( SourceVersion.isIdentifier( store ) )
243    {
244      if ( defaultStore.equals( store ) &&
245           typeElement == element.getEnclosingElement() &&
246           ElementsUtil.isWarningNotSuppressed( element, Constants.WARNING_UNNECESSARY_STORE ) )
247      {
248        final String message =
249          MemberChecks.shouldNot( Constants.PERSIST_CLASSNAME,
250                                  "specify the store parameter when it is the same as the defaultStore " +
251                                  "parameter in the specified by the " +
252                                  MemberChecks.toSimpleName( Constants.PERSIST_TYPE_CLASSNAME ) +
253                                  " annotation on the enclosing type. " +
254                                  MemberChecks.suppressedBy( Constants.WARNING_UNNECESSARY_STORE ) );
255        processingEnv.getMessager().printMessage( Diagnostic.Kind.WARNING, message, element );
256      }
257      return store;
258    }
259    else
260
261    {
262      throw new ProcessorException( MemberChecks.mustNot( Constants.PERSIST_CLASSNAME,
263                                                          "specify a store parameter that is not a valid java identifier" ),
264                                    element,
265                                    annotation );
266    }
267  }
268
269  private void verifyPersistElements( @Nonnull final RoundEnvironment env,
270                                      @Nonnull final Set<? extends Element> elements )
271  {
272    for ( final Element element : elements )
273    {
274      assert ElementKind.METHOD == element.getKind();
275      if ( !AnnotationsUtil.hasAnnotationOfType( element, Constants.OBSERVABLE_CLASSNAME ) )
276      {
277        reportError( env,
278                     MemberChecks.must( Constants.PERSIST_CLASSNAME,
279                                        "be also be annotated with the " + Constants.OBSERVABLE_CLASSNAME +
280                                        " annotation" ),
281                     element );
282      }
283    }
284  }
285
286  private void emitSidecar( @Nonnull final TypeDescriptor type )
287    throws IOException
288  {
289    final String packageName = GeneratorUtil.getQualifiedPackageName( type.getElement() );
290    emitTypeSpec( packageName, SidecarGenerator.buildType( processingEnv, type ) );
291  }

293  @SuppressWarnings( "SameParameterValue" )
294  @Nonnull
295  private String getPropertyAccessorName( @Nonnull final ExecutableElement method, @Nonnull final String specifiedName )
296    throws ProcessorException
297  {
298    String name = deriveName( method, GETTER_PATTERN, specifiedName );
299    if ( null != name )
300    {
301      return name;
302    }
303    if ( method.getReturnType().getKind() == TypeKind.BOOLEAN )
304    {
305      name = deriveName( method, ISSER_PATTERN, specifiedName );
306      if ( null != name )
307      {
308        return name;
309      }
310    }
311    return method.getSimpleName().toString();
312  }
313
314  @Nullable
315  private String deriveName( @Nonnull final ExecutableElement method,
316                             @Nonnull final Pattern pattern,
317                             @Nonnull final String name )
318    throws ProcessorException
319  {
320    if ( DEFAULT_SENTINEL.equals( name ) )
321    {
322      final String methodName = method.getSimpleName().toString();
323      final Matcher matcher = pattern.matcher( methodName );
324      if ( matcher.find() )
325      {
326        final String candidate = matcher.group( 1 );
327        return firstCharacterToLowerCase( candidate );
328      }
329      else
330      {
331        return null;
332      }
333    }
334    else
335    {
336      return name;
337    }
338  }
339
340  @Nonnull
341  private String firstCharacterToLowerCase( @Nonnull final String name )
342  {
343    return Character.toLowerCase( name.charAt( 0 ) ) + name.substring( 1 );
344  }
345
346  @Nonnull
347  private String firstCharacterToUpperCase( @Nonnull final String name )
348  {
349    return Character.toUpperCase( name.charAt( 0 ) ) + name.substring( 1 );
350  }
