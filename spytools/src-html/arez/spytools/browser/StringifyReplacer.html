001package arez.spytools.browser;
002
003import akasha.lang.JsArray;
004import javax.annotation.Nonnull;
005import javax.annotation.Nullable;
006import jsinterop.annotations.JsPackage;
007import jsinterop.annotations.JsType;
008import jsinterop.base.Any;
009import jsinterop.base.Js;
010import jsinterop.base.JsArrayLike;
011import jsinterop.base.JsPropertyMap;
012
013/**
014 * Utility class that helps convert javascript values to strings for JSON.stringify method.
015 * This class is extracted so that downstream
016 */
017public class StringifyReplacer
018{
019  @Nonnull
020  private final JsArrayLike<Object> _array = new JsArray<>();
021
022  /**
023   * Return the transformed value for key-value pair.
024   *
025   * @param value the value to transform.
026   * @return the transformed value.
027   */
028  @Nullable
029  public Any handleValue( @Nullable final Object value )
030  {
031    if ( null == value )
032    {
033      return null;
034    }
035    else if ( Js.typeof( value ).equals( "function" ) )
036    {
037      return Js.asPropertyMap( value ).getAsAny( "name" );
038    }
039    else if ( !Js.typeof( value ).equals( "object" ) )
040    {
041      return Js.asAny( value );
042    }
043    else
044    {
045      final String v = String.valueOf( value );
046      if ( null == v )
047      {
048        // v may be null if value.toString() returns null which will occur in optimized code in some scenarios
049        return Js.asAny( "null" );
050      }
051      else if ( !v.startsWith( "[object " ) )
052      {
053        return Js.asAny( v );
054      }
055      else
056      {
057        for ( int i = 0; i < _array.getLength(); i++ )
058        {
059          if ( Js.isTripleEqual( value, _array.getAtAsAny( i ) ) )
060          {
061            return Js.asAny( "[Circular]" );
062          }
063        }
064        _array.setAt( _array.getLength(), value );
065
066        final String[] propertyNames = getPropertyNames( value );
067        final JsPropertyMap<Object> map = JsPropertyMap.of();
068        for ( final String propertyName : propertyNames )
069        {
070          map.set( propertyName, Js.asPropertyMap( value ).getAsAny( propertyName ) );
071        }
072        return Js.asAny( map );
073      }
074    }
075  }
076
077  /**
078   * Return the property names that should be extracted from object when converting object.
079   *
080   * @param object the object value.
081   * @return the property names.
082   */
083  @Nonnull
084  protected String[] getPropertyNames( @Nonnull final Object object )
085  {
086    final JsArray<String> names = JsObject.getOwnPropertyNames( object );
087    return names.asArray( new String[ names.length ] );
088  }
089
090  @JsType( isNative = true, name = "Object", namespace = JsPackage.GLOBAL )
091  private static class JsObject
092  {
093    @Nonnull
094    private static native JsArray<String> getOwnPropertyNames( @Nonnull Object obj );
095  }
096}
