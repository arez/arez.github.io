432  /**
433   * Convert array of parameters into something consumable by a human.
434   * Java objects use toString() while native javascript objects use <code>JSON.stringify()</code>.
435   *
436   * @param parameters the parameters to convert
437   * @return a string representation of the parameters
438   */
439  @Nonnull
440  protected String parametersToString( @Nonnull final Object[] parameters )
441  {
442    if ( 0 == parameters.length )
443    {
444      return "()";
445    }
446    else
447    {
448      final StringifyReplacer filter = getStringifyReplacer();
449      final StringBuilder sb = new StringBuilder();
450      sb.append( "(" );
451      boolean requireComma = false;
452      for ( final Object parameter : parameters )
453      {
454        if ( requireComma )
455        {
456          sb.append( ", " );
457        }
458        else
459        {
460          requireComma = true;
461        }
462        sb.append( Global.JSON.stringify( parameter, filter::handleValue ) );
463      }
464      sb.append( ")" );
465      return sb.toString();
466    }
467  }

472   *
473   * @return the stringify replacer callback function.

478    return new StringifyReplacer();

481  /**
482   * Log specified message with parameters
483   *
484   * @param delta   the nesting delta.
485   * @param message the message.
486   * @param styling the styling parameter. It is assumed that the message has a %c somewhere in it to identify the start of the styling.
487   */
488  protected void log( @Nonnull final SpyUtil.NestingDelta delta,
489                      @Nonnull final String message,
490                      @CssRules @Nonnull final String styling )
491  {
492    if ( SpyUtil.NestingDelta.INCREASE == delta )
493    {
494      DomGlobal.console.groupCollapsed( message, styling );
495    }
496    else if ( SpyUtil.NestingDelta.DECREASE == delta )
497    {
498      DomGlobal.console.log( message, styling );
499      DomGlobal.console.groupEnd();
500    }
501    else
502    {
503      DomGlobal.console.log( message, styling );
504    }
505  }
