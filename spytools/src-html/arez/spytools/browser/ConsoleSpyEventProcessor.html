389  {
390    final String message = ( e.isTracked() ? "Tracked " : "" ) +
391                           "Action Completed " +
392                           e.getName() +

399    log( d, "%c" + message, ACTION_COLOR );

402  /**
403   * Convert array of parameters into something consumable by a human.
404   * Java objects use toString() while native javascript objects use <code>JSON.stringify()</code>.
405   *
406   * @param parameters the parameters to convert
407   * @return a string representation of the parameters
408   */
409  @Nonnull
410  protected String parametersToString( @Nonnull final Object[] parameters )
411  {
412    if ( 0 == parameters.length )
413    {
414      return "()";
415    }
416    else
417    {
418      final StringifyReplacer filter = getStringifyReplacer();
419      final StringBuilder sb = new StringBuilder();
420      sb.append( "(" );
421      boolean requireComma = false;
422      for ( final Object parameter : parameters )
423      {
424        if ( requireComma )
425        {
426          sb.append( ", " );
427        }
428        else
429        {
430          requireComma = true;
431        }
432        sb.append( Global.JSON.stringify( parameter, filter::handleValue ) );
433      }
434      sb.append( ")" );
435      return sb.toString();
436    }
437  }

451  /**
452   * Log specified message with parameters
453   *
454   * @param delta   the nesting delta.
455   * @param message the message.
456   * @param styling the styling parameter. It is assumed that the message has a %c somewhere in it to identify the start of the styling.
457   */
458  protected void log( @Nonnull final SpyUtil.NestingDelta delta,
459                      @Nonnull final String message,
460                      @CssRules @Nonnull final String styling )
461  {
462    if ( SpyUtil.NestingDelta.INCREASE == delta )
463    {
464      DomGlobal.console.groupCollapsed( message, styling );
465    }
466    else if ( SpyUtil.NestingDelta.DECREASE == delta )
467    {
468      DomGlobal.console.log( message, styling );
469      DomGlobal.console.groupEnd();
470    }
471    else
472    {
473      DomGlobal.console.log( message, styling );
474    }
475  }
