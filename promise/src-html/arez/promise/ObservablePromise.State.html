001package arez.promise;
002
003import arez.Arez;
004import arez.annotations.Action;
005import arez.annotations.ArezComponent;
006import arez.annotations.Feature;
007import arez.annotations.Observable;
008import elemental2.promise.IThenable;
009import elemental2.promise.Promise;
010import java.util.Objects;
011import javax.annotation.Nonnull;
012import jsinterop.base.Js;
013import static org.realityforge.braincheck.Guards.*;
014
015/**
016 * An observable model that wraps a Promise and exposes observable state that track
017 * the state of the promise. The observable exposes the state of the promise as well
018 * as the value that it resolves to or the error it was rejected with as observable
019 * properties.
020 *
021 * <h1>A very simple example</h1>
022 * <pre>{@code
023 * import com.google.gwt.core.client.EntryPoint;
024 * import elemental2.dom.DomGlobal;
025 * import elemental2.dom.Response;
026 * import elemental2.promise.Promise;
027 * import arez.Arez;
028 * import arez.promise.ObservablePromise;
029 *
030 * public class Example
031 *   implements EntryPoint
032 * {
033 *   public void onModuleLoad()
034 *   {
035 *     final Promise<Response> promise = DomGlobal.fetch( "https://example.com/" );
036 *     final ObservablePromise<Response, Object> observablePromise = ObservablePromise.create( promise );
037 *     Arez.context().observer( () -> {
038 *       final String message = "Promise Status: " + observablePromise.getState();
039 *       DomGlobal.console.log( message );
040 *     } );
041 *   }
042 * }
043 * }</pre>
044 *
045 * @param <T> the type of the value that the promise will resolve to.
046 * @param <E> the type of the error if the promise is rejected.
047 */
048@ArezComponent( requireId = Feature.DISABLE )
049public abstract class ObservablePromise<T, E>
050{
051  /**
052   * The state of the promise.
053   */
054  public enum State
055  {
056    PENDING, FULFILLED, REJECTED
057  }
058
059  /**
060   * The underlying promise.
061   * This is not converted to a local variable to make it easy to debug scenarios from within the
062   * browsers DevTools.
063   */
064  @SuppressWarnings( "FieldCanBeLocal" )
065  private final Promise<T> _promise;
066  /**
067   * The state of the promise. Starts as {@link State#PENDING} and then transitions to either
068   * {@link State#FULFILLED} or {@link State#REJECTED}.
069   */
070  @Nonnull
071  private State _state;
072  /**
073   * The value that the promise resolved to. This is not valid unless the state is {@link State#FULFILLED}.
074   */
075  private T _value;
076  /**
077   * The error that the promise was rejected with. This is not valid unless the state is {@link State#REJECTED}.
078   */
079  private E _error;
080
081  /**
082   * Create the observable model that wraps specified promise.
083   *
084   * @param <T>     the type of the value that the promise will resolve to.
085   * @param <E>     the type of the error if the promise is rejected.
086   * @param promise the promise to wrap.
087   * @return the ObservablePromise
088   */
089  @Nonnull
090  public static <T, E> ObservablePromise<T, E> create( @Nonnull final Promise<T> promise )
091  {
092    return new Arez_ObservablePromise<>( promise );
093  }
094
095  ObservablePromise( @Nonnull final Promise<T> promise )
096  {
097    _state = State.PENDING;
098    _promise = Objects.requireNonNull( promise );
099    _promise.then( this::onFulfilled ).catch_( this::onRejected );
100  }
101
102  /**
103   * Return the promise state.
104   *
105   * @return the promise state.
106   */
107  @Observable
108  @Nonnull
109  public State getState()
110  {
111    return _state;
112  }
113
114  void setState( @Nonnull final State state )
115  {
116    _state = Objects.requireNonNull( state );
117  }
118
119  /**
120   * Return the value that the promise was resolved to.
121   * This should NOT be called if the state is not {@link State#FULFILLED} and will result in an invariant
122   * failure if invariants are enabled.
123   *
124   * @return the value that the promise was resolved to.
125   */
126  @Observable
127  public T getValue()
128  {
129    if ( Arez.shouldCheckApiInvariants() )
130    {
131      apiInvariant( () -> _state == State.FULFILLED,
132                    () -> "Arez-0165: ObservablePromise.getValue() called when the promise is not in " +
133                          "fulfilled state. State: " + _state + ", Promise: " + _promise );
134    }
135    return _value;
136  }
137
138  void setValue( final T value )
139  {
140    if ( Arez.shouldCheckInvariants() )
141    {
142      invariant( () -> _state == State.FULFILLED,
143                 () -> "Arez-0166: ObservablePromise.setValue() called when promise is in incorrect state. " +
144                       "State: " + _state + ", Promise: " + _promise );
145    }
146    _value = value;
147  }
148
149  /**
150   * Return the error that the promise was rejected with.
151   * This should NOT be called if the state is not {@link State#REJECTED} and will result in an invariant
152   * failure if invariants are enabled.
153   *
154   * @return the error that the promise was rejected with.
155   */
156  @Observable
157  public E getError()
158  {
159    if ( Arez.shouldCheckApiInvariants() )
160    {
161      apiInvariant( () -> _state == State.REJECTED,
162                    () -> "ObservablePromise.getError() called when the promise is not in " +
163                          "rejected state. State: " + _state + ", Promise: " + _promise );
164    }
165    return _error;
166  }
167
168  void setError( final E error )
169  {
170    if ( Arez.shouldCheckInvariants() )
171    {
172      invariant( () -> _state == State.REJECTED,
173                 () -> "ObservablePromise.setError() called when promise is in incorrect state. " +
174                       "State: " + _state + ", Promise: " + _promise );
175    }
176    _error = error;
177  }
178
179  @Action
180  @Nonnull
181  IThenable<T> onFulfilled( final T value )
182  {
183    setState( State.FULFILLED );
184    setValue( value );
185    return Promise.resolve( value );
186  }
187
188  @Action
189  IThenable<Object> onRejected( @Nonnull final Object error )
190  {
191    setState( State.REJECTED );
192    setError( Js.uncheckedCast( error ) );
193    return Promise.reject( error );
194  }
195}
