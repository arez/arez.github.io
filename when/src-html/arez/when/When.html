001package arez.when;
002
003import arez.Arez;
004import arez.Component;
005import arez.Observer;
006import arez.Priority;
007import arez.SafeFunction;
008import arez.SafeProcedure;
009import javax.annotation.Nonnull;
010import javax.annotation.Nullable;
011
012/**
013 * The utility class providing watcher methods.
014 */
015public final class When
016{
017  /**
018   * Id of next watcher to be created.
019   * This is only used if {@link Arez#areNamesEnabled()} returns true but no name has been supplied.
020   */
021  private static int c_nextWatcherId = 1;
022
023  private When()
024  {
025  }
026
027  /**
028   * Wait until a condition is true, then run effect. The effect is run in a read-only transaction.
029   * See {@link #when(String, boolean, SafeFunction, SafeProcedure)} for further details.
030   *
031   * @param condition The function that determines when the effect is run.
032   * @param effect    The procedure that is executed when the condition is true.
033   * @return the Observer representing the reactive component. The user can dispose the node if it is no longer required.
034   */
035  public static Observer when( @Nonnull final SafeFunction<Boolean> condition,
036                               @Nonnull final SafeProcedure effect )
037  {
038    return when( false, condition, effect );
039  }
040
041  /**
042   * Wait until a condition is true, then run effect.
043   * See {@link #when(String, boolean, SafeFunction, SafeProcedure)} for further details.
044   *
045   * @param mutation  true if the effect can mutate state, false otherwise.
046   * @param condition The function that determines when the effect is run.
047   * @param effect    The procedure that is executed when the condition is true.
048   * @return the Observer representing the reactive component. The user can dispose the node if it is no longer required.
049   */
050  public static Observer when( final boolean mutation,
051                               @Nonnull final SafeFunction<Boolean> condition,
052                               @Nonnull final SafeProcedure effect )
053  {
054    return when( null, mutation, condition, effect );
055  }
056
057  /**
058   * Wait until a condition is true, then run effect.
059   * See {@link #when(Component, String, boolean, SafeFunction, SafeProcedure, Priority, boolean)} for further details.
060   *
061   * @param name      the debug name (if any) used when naming the underlying Arez resources.
062   * @param mutation  true if the effect can mutate state, false otherwise.
063   * @param condition The function that determines when the effect is run.
064   * @param effect    The procedure that is executed when the condition is true.
065   * @return the Observer representing the reactive component. The user can dispose the node if it is no longer required.
066   */
067  public static Observer when( @Nullable final String name,
068                               final boolean mutation,
069                               @Nonnull final SafeFunction<Boolean> condition,
070                               @Nonnull final SafeProcedure effect )
071  {
072    return when( name, mutation, condition, effect, true );
073  }
074
075  /**
076   * Wait until a condition is true, then run effect.
077   * See {@link #when(Component, String, boolean, SafeFunction, SafeProcedure, Priority, boolean)} for further details.
078   *
079   * @param name           the debug name (if any) used when naming the underlying Arez resources.
080   * @param mutation       true if the effect can mutate state, false otherwise.
081   * @param condition      The function that determines when the effect is run.
082   * @param effect         The procedure that is executed when the condition is true.
083   * @param runImmediately true to invoke condition immediately, false to schedule reaction for next reaction cycle.
084   * @return the Observer representing the reactive component. The user can dispose the node if it is no longer required.
085   */
086  public static Observer when( @Nullable final String name,
087                               final boolean mutation,
088                               @Nonnull final SafeFunction<Boolean> condition,
089                               @Nonnull final SafeProcedure effect,
090                               final boolean runImmediately )
091  {
092    return when( name, mutation, condition, effect, Priority.NORMAL, runImmediately );
093  }
094
095  /**
096   * Wait until a condition is true, then run effect.
097   * See {@link #when(Component, String, boolean, SafeFunction, SafeProcedure, Priority, boolean)} for further details.
098   *
099   * @param name           the debug name (if any) used when naming the underlying Arez resources.
100   * @param mutation       true if the effect can mutate state, false otherwise.
101   * @param condition      The function that determines when the effect is run.
102   * @param effect         The procedure that is executed when the condition is true.
103   * @param priority       the priority of the observer.
104   * @param runImmediately true to invoke condition immediately, false to schedule reaction for next reaction cycle.
105   * @return the Observer representing the reactive component. The user can dispose the node if it is no longer required.
106   */
107  public static Observer when( @Nullable final String name,
108                               final boolean mutation,
109                               @Nonnull final SafeFunction<Boolean> condition,
110                               @Nonnull final SafeProcedure effect,
111                               @Nonnull final Priority priority,
112                               final boolean runImmediately )
113  {
114    return when( null, name, mutation, condition, effect, priority, runImmediately );
115  }
116
117  /**
118   * Wait until a condition is true, then run effect.
119   * The condition function is run in a read-only, tracking transaction and will be re-evaluated
120   * any time any of the observed elements are updated. The effect procedure is run in either a
121   * read-only or read-write, non-tracking transaction.
122   *
123   * @param component      the component containing when observer if any. Should be null if {@link Arez#areNativeComponentsEnabled()} returns false.
124   * @param name           the debug name (if any) used when naming the underlying Arez resources.
125   * @param mutation       true if the effect can mutate state, false otherwise.
126   * @param condition      The function that determines when the effect is run.
127   * @param effect         The procedure that is executed when the condition is true.
128   * @param priority       the priority of the observer.
129   * @param runImmediately true to invoke condition immediately, false to schedule reaction for next reaction cycle.
130   * @return the Observer representing the reactive component. The user can dispose the node if it is no longer required.
131   */
132  public static Observer when( @Nullable final Component component,
133                               @Nullable final String name,
134                               final boolean mutation,
135                               @Nonnull final SafeFunction<Boolean> condition,
136                               @Nonnull final SafeProcedure effect,
137                               @Nonnull final Priority priority,
138                               final boolean runImmediately )
139  {
140    return when( component, name, mutation, true, condition, effect, priority, runImmediately );
141  }
142
143  /**
144   * Wait until a condition is true, then run effect.
145   * The condition function is run in a read-only, tracking transaction and will be re-evaluated
146   * any time any of the observed elements are updated. The effect procedure is run in either a
147   * read-only or read-write, non-tracking transaction.
148   *
149   * @param component            the component containing when observer if any. Should be null if {@link Arez#areNativeComponentsEnabled()} returns false.
150   * @param name                 the debug name (if any) used when naming the underlying Arez resources.
151   * @param mutation             true if the effect can mutate state, false otherwise.
152   * @param verifyActionRequired true if the effect will add invariant checks to ensure reads or writes occur within
153   *                             the scope of the effect.
154   * @param condition            The function that determines when the effect is run.
155   * @param effect               The procedure that is executed when the condition is true.
156   * @param priority             the priority of the observer.
157   * @param runImmediately       true to invoke condition immediately, false to schedule reaction for next reaction cycle.
158   * @return the Observer representing the reactive component. The user can dispose the node if it is no longer required.
159   */
160  public static Observer when( @Nullable final Component component,
161                               @Nullable final String name,
162                               final boolean mutation,
163                               final boolean verifyActionRequired,
164                               @Nonnull final SafeFunction<Boolean> condition,
165                               @Nonnull final SafeProcedure effect,
166                               @Nonnull final Priority priority,
167                               final boolean runImmediately )
168  {
169    return new Watcher( Arez.areZonesEnabled() ? Arez.context() : null,
170                        component,
171                        generateNodeName( name ),
172                        mutation,
173                        verifyActionRequired,
174                        condition,
175                        effect,
176                        priority,
177                        runImmediately ).getWatcher();
178  }
179
180  /**
181   * Build name for node.
182   * If {@link Arez#areNamesEnabled()} returns false then this method will return null, otherwise the specified
183   * name will be returned or a name synthesized from the running number if no name is specified.
184   *
185   * @param name the name specified by the user.
186   * @return the name.
187   */
188  @Nullable
189  private static String generateNodeName( @Nullable final String name )
190  {
191    return Arez.areNamesEnabled() ?
192           null != name ? name : "When@" + c_nextWatcherId++ :
193           null;
194  }
195}
